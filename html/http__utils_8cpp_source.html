43 std::string url_decode(const std::string& text) {
44 std::string escaped;
45
46 for (std::string::const_iterator i = text.begin(), n = text.end(); i != n; ++i) {
47 std::string::value_type c = (*i);
48
49 if (c == '%') {
50 if (i[1] && i[2]) {
51 escaped += static_cast<char>(from_hex(i[1]) << 4 | from_hex(i[2]));
52 i += 2;
53 }
54 } else if (c == '+') {
55 escaped += ' ';
56 } else {
57 escaped += c;
58 }
59 }
60
61 return escaped;
62 }

64 std::string url_encode(const std::string& text) {
65 std::ostringstream escaped;
66 escaped.fill('0');
67 escaped << std::hex;
68
69 for (std::string::const_iterator i = text.begin(), n = text.end(); i != n; ++i) {
70 std::string::value_type c = (*i);
71
72 if (isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~') {
73 escaped << c;
74 } else {
75 escaped << std::uppercase;
76 escaped << '%' << std::setw(2) << static_cast<unsigned char>(c);
77 escaped << std::nouppercase;
78 }
79 }
80
81 return escaped.str();
82 }

120 std::string::size_type rpos = string.find_first_of(c, *pos);
121

132 std::string to_http_date(struct tm* tm) {
133 char buf[100];
134
135 if (tm == nullptr) {
136 time_t now = utils::system::time(nullptr);
137 tm = utils::system::gmtime(&now);
138 } else {
139 time_t time = utils::system::mktime(tm);
140 tm = utils::system::gmtime(&time);
141 }
142
143 (void) strftime(buf, sizeof buf, "%a, %d %b %Y %H:%M:%S %Z", tm);
144 errno = 0; // Errno can be modified by strftime even though there is no error
145
146 return std::string(buf);
147 }

174 bool ci_comp(const std::string& str1, const std::string& str2) {
175 return str1.size() == str2.size() && std::equal(str1.begin(), str1.end(), str2.begin(), [](auto a, auto b) {
176 return std::tolower(a) == std::tolower(b);
177 });

180 bool ci_contains(const std::string& str1, const std::string& str2) {
181 auto it = std::search(str1.begin(), str1.end(), str2.begin(), str2.end(), [](char ch1, char ch2) {
182 return std::toupper(ch1) == std::toupper(ch2);
183 });
184
185 return (it != str1.end());
