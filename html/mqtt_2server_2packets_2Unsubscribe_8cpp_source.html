37 std::size_t Unsubscribe::deserializeVP(iot::mqtt::MqttContext* mqttContext) {
38 std::size_t consumed = 0;
39
40 switch (state) {
41 case 0: // V-Header
42 consumed += packetIdentifier.deserialize(mqttContext);
43 if (!packetIdentifier.isComplete()) {
44 break;
45 }
46
47 state++;
48 [[fallthrough]];
49 case 1: // Payload
50 consumed += topic.deserialize(mqttContext);
51
52 if (!topic.isComplete()) {
53 break;
54 }
55
56 topics.push_back(topic);
57 topic.reset();
58
59 complete = getConsumed() + consumed >= this->getRemainingLength();
60
61 break;
62 }
63
64 return consumed;
65 }
