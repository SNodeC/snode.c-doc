148 void realListen(const std::function<void(const core::ProgressLog&)>& onError, unsigned int tries, double retryTimeoutScale) const {
149 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
150 new SocketAcceptor(
151 socketContextFactory,
152 onConnect,
153 onConnected,
154 onDisconnect,
155 [server = *this, onError, tries, retryTimeoutScale](const core::ProgressLog& progressLog) -> void {
156 onError(progressLog);
157
158 if (progressLog.getHasErrors() != 0 && server.getConfig().getRetry() &&
159 (server.getConfig().getRetryTries() == 0 || tries < server.getConfig().getRetryTries())) {
160 double relativeRetryTimeout = server.getConfig().getRetryLimit() > 0
161 ? std::min<double>(server.getConfig().getRetryTimeout() * retryTimeoutScale,
162 server.getConfig().getRetryLimit())
163 : server.getConfig().getRetryTimeout() * retryTimeoutScale;
164 relativeRetryTimeout -=
165 utils::Random::getInRange(-server.getConfig().getRetryJitter(), server.getConfig().getRetryJitter()) *
166 relativeRetryTimeout / 100.;
167
168 LOG(INFO) << "Retrying in " << relativeRetryTimeout << " seconds";
169
170 core::timer::Timer::singleshotTimer(
171 [server, onError, tries, retryTimeoutScale]() mutable -> void {
172 server.realListen(onError, tries + 1, retryTimeoutScale * server.getConfig().getRetryBase());
173 },
174 relativeRetryTimeout);
175 }
176 },
177 Super::config);
178 }
