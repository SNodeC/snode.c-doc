34/* (R0+R1), R2, R3, R4 are the different operations used in SHA1 */
35#define SHA1_R0(v, w, x, y, z, i) \
36 z += ((w & (x ^ y)) ^ y) + block[i] + 0x5a827999 + SHA1_ROL(v, 5); \
37 w = SHA1_ROL(w, 30);
38#define SHA1_R1(v, w, x, y, z, i) \
39 z += ((w & (x ^ y)) ^ y) + SHA1_BLK(i) + 0x5a827999 + SHA1_ROL(v, 5); \
40 w = SHA1_ROL(w, 30);
41#define SHA1_R2(v, w, x, y, z, i) \
42 z += (w ^ x ^ y) + SHA1_BLK(i) + 0x6ed9eba1 + SHA1_ROL(v, 5); \
43 w = SHA1_ROL(w, 30);
44#define SHA1_R3(v, w, x, y, z, i) \
45 z += (((w | x) & y) | (w & x)) + SHA1_BLK(i) + 0x8f1bbcdc + SHA1_ROL(v, 5); \
46 w = SHA1_ROL(w, 30);
47#define SHA1_R4(v, w, x, y, z, i) \
48 z += (w ^ x ^ y) + SHA1_BLK(i) + 0xca62c1d6 + SHA1_ROL(v, 5); \

60void SHA1::update(std::istream& is) {
61 std::string rest_of_buffer;
62 read(is, rest_of_buffer, static_cast<std::size_t>(static_cast<int>(BLOCK_BYTES) - static_cast<int>(buffer.size())));
63 buffer += rest_of_buffer;
64
65 while (is) {
66 uint32_t block[BLOCK_INTS];
67 buffer_to_block(buffer, block);
68 transform(block);
69 read(is, buffer, BLOCK_BYTES);
70 }
71}

77std::string SHA1::final() {
78 /* Total number of hashed bits */
79 uint64_t total_bits = (transforms * BLOCK_BYTES + buffer.size()) * 8;
80
81 /* Padding */
82 buffer += static_cast<std::string::value_type>(0x80);
83 std::size_t orig_size = buffer.size();
84 while (buffer.size() < BLOCK_BYTES) {
85 buffer += static_cast<char>(0x00);
86 }
87
88 uint32_t block[BLOCK_INTS];
89 buffer_to_block(buffer, block);
90
91 if (orig_size > BLOCK_BYTES - 8) {
92 transform(block);
93 for (unsigned int i = 0; i < BLOCK_INTS - 2; i++) {
94 block[i] = 0;
95 }
96 }
97
98 /* Append total_bits, split this uint64 into two uint32 */
99 block[BLOCK_INTS - 1] = static_cast<uint32_t>(total_bits & 0x00000000FFFFFFFF);
100 block[BLOCK_INTS - 2] = static_cast<uint32_t>((total_bits >> 32) & 0x00000000FFFFFFFF);
101 transform(block);
102
103 /* Hex std::string */
104 std::ostringstream result;
105 for (unsigned int i = 0; i < DIGEST_INTS; i++) {
106 result << std::hex << std::setfill('0') << std::setw(8);
107 result << (digest[i] & 0xffffffff);
108 }
109
110 /* Reset for next run */
111 reset();
112
113 return result.str();
114}

234void SHA1::buffer_to_block(const std::string& buffer, uint32_t block[BLOCK_BYTES]) {
235 /* Convert the std::string (byte buffer) to a uint32 array (MSB) */
236 for (unsigned int i = 0; i < BLOCK_INTS; i++) {
237 block[i] = static_cast<uint32_t>((buffer[4 * i + 3] & 0xff) | (buffer[4 * i + 2] & 0xff) << 8 | (buffer[4 * i + 1] & 0xff) << 16 |
238 (buffer[4 * i + 0] & 0xff) << 24);
239 }

249static std::vector<unsigned char> transform_to_binary(const std::string& string) {
250 std::vector<unsigned char> buf;
251
252 char hex_byte[3];
253 hex_byte[2] = 0;
254
255 for (std::size_t i = 0, j = 0; i < string.size(); i += 2, j += 1) {
256 hex_byte[0] = string.at(i);
257 hex_byte[1] = string.at(i + 1);
258 char* end_ptr = nullptr;
259 buf.push_back(static_cast<unsigned char>(strtoul(hex_byte, &end_ptr, 16)));
260 }
261
262 return buf;
263}
