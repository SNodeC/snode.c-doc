46 SocketAddress SocketAddress::setSunPath(const std::string& sunPath) {
47 if (sunPath.length() < sizeof(sockAddr.sun_path)) {
48 std::size_t len = sunPath.length();
49 std::memcpy(sockAddr.sun_path, sunPath.data(), len);
50 sockAddr.sun_path[len] = 0;
51 sockAddrLen = static_cast<socklen_t>(offsetof(sockaddr_un, sun_path) + len + 1);
52 } else {
53 errno = EINVAL;
54 throw core::socket::SocketAddress::BadSocketAddress(
55 "Unix-Domain error sun-path to long: Lenght is = " + std::to_string(sunPath.length()) +
56 ", should be: " + std::to_string(sizeof(sockAddr.sun_path) - 1),
57 core::socket::State::FATAL);
58 }
59
60 return *this;
61 }
