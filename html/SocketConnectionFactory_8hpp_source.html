39 template <typename PhysicalSocket, typename Config, typename SocketConnection>
40 bool SocketConnectionFactory<PhysicalSocket, Config, SocketConnection>::create(PhysicalSocket& physicalSocket,
41 const std::shared_ptr<Config>& config) {
42 SocketConnection* socketConnection = nullptr;
43
44 if (physicalSocket.isValid()) {
45 typename SocketAddress::SockAddr localSockAddr;
46 typename SocketAddress::SockAddr remoteSockAddr;
47 socklen_t localSockAddrLen = sizeof(typename SocketAddress::SockAddr);
48 socklen_t remoteSockAddrLen = sizeof(typename SocketAddress::SockAddr);
49
50 if (physicalSocket.getSockname(localSockAddr, localSockAddrLen) == 0 &&
51 physicalSocket.getPeername(remoteSockAddr, remoteSockAddrLen) == 0) {
52 physicalSocket.setDontClose();
53
54 socketConnection = new SocketConnection(physicalSocket,
55 SocketAddress(localSockAddr, localSockAddrLen),
56 SocketAddress(remoteSockAddr, remoteSockAddrLen),
57 onDisconnect,
58 config->getReadTimeout(),
59 config->getWriteTimeout(),
60 config->getReadBlockSize(),
61 config->getWriteBlockSize(),
62 config->getTerminateTimeout());
63 onConnect(socketConnection);
64 onConnected(socketConnection);
65 } else {
66 PLOG(TRACE) << "getsockname";
67 }
68 }
69
70 return socketConnection != nullptr;
71 }
