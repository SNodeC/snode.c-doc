56 ConfigPhysicalSocketClient::ConfigPhysicalSocketClient(ConfigInstance* instance)
57 : net::config::ConfigPhysicalSocket(instance) {
58 Super::add_flag_function(
59 reconnectOpt, //
60 "--reconnect{true}",
61 [this](int64_t) -> void {
62 if (!this->reconnectOpt->as<bool>()) {
63 this->reconnectTimeOpt->clear();
64 }
65 },
66 "Auto-reconnection in the event of a connection interruption",
67 "bool",
68 XSTR(RECONNECT),
69 CLI::IsMember({"true", "false"}));
70
71 Super::add_option(reconnectTimeOpt, //
72 "--reconnect-time",
73 "Duration after disconnect before reconnect",
74 "sec",
75 RECONNECT_TIME,
76 CLI::NonNegativeNumber);
77 reconnectTimeOpt->needs(reconnectOpt);
78
79 Super::add_option(connectTimeoutOpt, //
80 "--connect-timeout",
81 "Connect timeout",
82 "timeout",
83 CONNECT_TIMEOUT,
84 CLI::NonNegativeNumber);
85 }

92 if (reconnect) {
