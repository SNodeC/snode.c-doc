38 ConfigPhysicalSocketClient::ConfigPhysicalSocketClient(ConfigInstance* instance)
39 : Super(instance) {
40 reconnectOpt = addFlagFunction( //
41 "--reconnect{true}",
42 [this](int64_t) -> void {
43 if (!this->reconnectOpt->as<bool>()) {
44 this->reconnectTimeOpt->clear();
45 }
46 },
47 "Auto-reconnection in the event of a connection interruption",
48 "bool",
49 XSTR(RECONNECT),
50 CLI::IsMember({"true", "false"}));
51
52 reconnectTimeOpt = addOption( //
53 "--reconnect-time",
54 "Duration after disconnect before reconnect",
55 "sec",
56 RECONNECT_TIME,
57 CLI::NonNegativeNumber);
58 reconnectTimeOpt->needs(reconnectOpt);
59
60 connectTimeoutOpt = addOption( //
61 "--connect-timeout",
62 "Connect timeout",
63 "timeout",
64 CONNECT_TIMEOUT,
65 CLI::NonNegativeNumber);
66 }

68 ConfigPhysicalSocketClient& ConfigPhysicalSocketClient::setReconnect(bool reconnect) {
69 reconnectOpt //
70 ->default_val(reconnect ? "true" : "false")
71 ->clear();
72
73 if (reconnect) {
74 reconnectTimeOpt->remove_needs(reconnectOpt);
75 } else {
76 reconnectTimeOpt->needs(reconnectOpt);
77 }
78
79 return *this;
