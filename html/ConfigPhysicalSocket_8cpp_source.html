36 ConfigPhysicalSocket::ConfigPhysicalSocket(ConfigInstance* instance)
37 : ConfigSection(instance, "socket", "Configuration of socket behaviour for instance '" + instance->getInstanceName() + "'") {
38 add_socket_option(reuseAddressOpt, //
39 "--reuse-address",
40 SOL_SOCKET,
41 SO_REUSEADDR,
42 "Reuse socket address",
43 "bool",
44 REUSE_ADDRESS,
45 CLI::IsMember({"true", "false"}));
46 add_flag_function(
47 reconnectOpt, //
48 "--reconnect",
49 [this](int64_t) -> void {
50 if (!this->reconnectOpt->as<bool>()) {
51 this->reconnectTimeOpt->clear();
52 }
53 utils::ResetToDefault(this->reconnectOpt)(this->reconnectOpt->as<bool>());
54 },
55 "Automatically retry listen|connect",
56 "bool",
57 RECONNECT,
58 CLI::IsMember({"true", "false"}));
59
60 add_option(reconnectTimeOpt, //
61 "--reconnect-time",
62 "Duration after disconnect bevore reconnect",
63 "sec",
64 RECONNECT_TIME,
65 CLI::NonNegativeNumber);
66 reconnectTimeOpt->needs(reconnectOpt);
67
68 add_flag_function(
69 retryOpt, //
70 "--retry",
71 [this](int64_t) -> void {
72 if (!this->retryOpt->as<bool>()) {
73 this->retryTimeoutOpt->clear();
74 this->retryTriesOpt->clear();
75 this->retryBaseOpt->clear();
76 this->retryLimitOpt->clear();
77 }
78 utils::ResetToDefault(this->retryOpt)(this->retryOpt->as<bool>());
79 },
80 "Automatically retry listen|connect",
81 "bool",
82 RETRY,
83 CLI::IsMember({"true", "false"}));
84
85 add_option(retryTimeoutOpt, //
86 "--retry-timeout",
87 "Timeout of the retry timer",
88 "sec",
89 RETRY_TIMEOUT,
90 CLI::NonNegativeNumber);
91 retryTimeoutOpt->needs(retryOpt);
92
93 add_option(retryTriesOpt, //
94 "--retry-tries",
95 "Number of retry attempts bevore giving up (0 = unlimited)",
96 "tries",
97 RETRY_TRIES,
98 CLI::TypeValidator<unsigned int>());
99 retryTriesOpt->needs(retryOpt);
100
101 add_option(retryBaseOpt, //
102 "--retry-base",
103 "Base of exponential backoff",
104 "base",
105 RETRY_BASE,
106 CLI::PositiveNumber);
107 retryBaseOpt->needs(retryOpt);
108
109 add_option(retryJitterOpt, //
110 "--retry-jitter",
111 "Maximum jitter in percent to apply randomly to calculated retry timeout (0 to disable)",
112 "jitter",
113 RETRY_JITTER,
114 CLI::Range(0., 100.));
115 retryJitterOpt->needs(retryOpt);
116
117 add_option(retryLimitOpt, //
118 "--retry-limit",
119 "Upper limit in seconds of retry timeout",
120 "sec",
121 RETRY_LIMIT,
122 CLI::NonNegativeNumber);
123 retryLimitOpt->needs(retryOpt);
124 }

130 CLI::Option* ConfigPhysicalSocket::add_socket_option(CLI::Option*& opt,
131 const std::string& name,
132 int optLevel,
133 int optName,
134 const std::string& description,
135 const std::string& typeName,
136 const std::string& defaultValue,
137 const CLI::Validator& validator) {
138 return net::config::ConfigSection::add_flag_function(
139 opt,
140 name,
141 [this, &opt, optLevel, optName](int64_t) -> void {
142 if (opt->as<bool>()) {
143 socketOptionsMap.insert({optName, net::phy::PhysicalSocketOption(optLevel, optName, 1)});
144 } else {
145 socketOptionsMap.erase(optName);
146 }
147
148 (utils::ResetToDefault(opt))(opt->as<std::string>());
149 },
150 description,
151 typeName,
152 defaultValue,
153 validator);
154 }
