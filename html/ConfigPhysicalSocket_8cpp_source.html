39 ConfigPhysicalSocket::ConfigPhysicalSocket(ConfigInstance* instance)
40 : ConfigSection(instance, "socket", "Configuration of socket behavior") {
41 reuseAddressOpt = add_socket_option( //
42 "--reuse-address{true}",
43 SOL_SOCKET,
44 SO_REUSEADDR,
45 "Reuse socket address",
46 "bool",
47 XSTR(REUSE_ADDRESS),
48 CLI::IsMember({"true", "false"}));
49
50 retryOpt = addFlag( //
51 "--retry{true}",
52 "Automatically retry listen|connect",
53 "bool",
54 XSTR(RETRY),
55 CLI::IsMember({"true", "false"}));
56
57 retryOnFatalOpt = addFlagFunction( //
58 "--retry-on-fatal{true}",
59 [this](int64_t) -> void {
60 if (retryOnFatalOpt->as<bool>() && !retryOpt->as<bool>()) {
61 throw CLI::RequiresError(retryOnFatalOpt->get_name(), retryOpt->get_name().append("=true"));
62 }
63 },
64 "Retry also on fatal errors",
65 "bool",
66 XSTR(RETRY_ON_FATAL),
67 CLI::IsMember({"true", "false"}));
68 retryOnFatalOpt->needs(retryOpt);
69
70 retryTimeoutOpt = addOption( //
71 "--retry-timeout",
72 "Timeout of the retry timer",
73 "sec",
74 RETRY_TIMEOUT,
75 CLI::NonNegativeNumber);
76 retryTimeoutOpt->needs(retryOpt);
77
78 retryTriesOpt = addOption( //
79 "--retry-tries",
80 "Number of retry attempts before giving up (0 = unlimited)",
81 "tries",
82 RETRY_TRIES,
83 CLI::TypeValidator<unsigned int>());
84 retryTriesOpt->needs(retryOpt);
85
86 retryBaseOpt = addOption( //
87 "--retry-base",
88 "Base of exponential backoff",
89 "base",
90 RETRY_BASE,
91 CLI::PositiveNumber);
92 retryBaseOpt->needs(retryOpt);
93
94 retryJitterOpt = addOption( //
95 "--retry-jitter",
96 "Maximum jitter in percent to apply randomly to calculated retry timeout (0 to disable)",
97 "jitter",
98 RETRY_JITTER,
99 CLI::Range(0., 100.));
100 retryJitterOpt->needs(retryOpt);
101
102 retryLimitOpt = addOption( //
103 "--retry-limit",
104 "Upper limit in seconds of retry timeout (0 for infinite)",
105 "sec",
106 RETRY_LIMIT,
107 CLI::NonNegativeNumber);
108 retryLimitOpt->needs(retryOpt);
109 }

115 CLI::Option* ConfigPhysicalSocket::add_socket_option(const std::string& name,
116 int optLevel,
117 int optName,
118 const std::string& description,
119 const std::string& typeName,
120 const std::string& defaultValue,
121 const CLI::Validator& validator) {
122 return net::config::ConfigSection::addFlagFunction(
123 name,
124 [this, strippedName = name.substr(0, name.find('{')), optLevel, optName](int64_t) -> void {
125 try {
126 if (section->get_option(strippedName)->as<bool>()) {
127 socketOptionsMap.emplace(optName, net::phy::PhysicalSocketOption(optLevel, optName, 1));
128 } else {
129 socketOptionsMap.erase(optName);
130 }
131 } catch (CLI::OptionNotFound& err) {
132 LOG(ERROR) << err.what();
133 }
134 },
135 description,
136 typeName,
137 defaultValue,
138 validator);
139 }

183 ConfigPhysicalSocket& ConfigPhysicalSocket::setRetry(bool retry) {
184 retryOpt //
185 ->default_val(retry ? "true" : "false")
186 ->clear();
187
188 if (retry) {
189 retryLimitOpt->remove_needs(retryOpt);
190 retryJitterOpt->remove_needs(retryOpt);

192 retryTriesOpt->remove_needs(retryOpt);
193 retryTimeoutOpt->remove_needs(retryOpt);
194 retryOnFatalOpt->remove_needs(retryOpt);

196 retryLimitOpt->needs(retryOpt);
197 retryJitterOpt->needs(retryOpt);
198 retryBaseOpt->needs(retryOpt);
199 retryTriesOpt->needs(retryOpt);
200 retryTimeoutOpt->needs(retryOpt);
201 retryOnFatalOpt->needs(retryOpt);
202 }
203
204 return *this;
205 }
