57 void SocketWriter::doWrite() {
58 if (!writePuffer.empty()) {
59 const std::size_t writeLen = (writePuffer.size() < blockSize) ? writePuffer.size() : blockSize;
60 const ssize_t retWrite = write(writePuffer.data(), writeLen);
61
62 if (retWrite > 0) {
63 writePuffer.erase(writePuffer.begin(), writePuffer.begin() + retWrite);
64
65 if (writePuffer.capacity() > writePuffer.size() * 2) {
66 writePuffer.shrink_to_fit();
67 }
68
69 if (!isSuspended()) {
70 suspend();
71 }
72 span();
73 } else if ((errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) && isSuspended()) {
74 resume();
75 } else {
76 onStatus(errno);
77 disable();
78 }
79 } else {
80 if (!isSuspended()) {
81 suspend();
82 }
83
84 if (markShutdown) {
85 shutdownWrite(onShutdown);
86 } else if (source != nullptr) {
87 source->resume();
88 }
89 }
90 }

96 void SocketWriter::sendToPeer(const char* chunk, std::size_t chunkLen) {
97 if (!shutdownInProgress && !markShutdown) {
98 if (isEnabled()) {
99 if (writePuffer.empty()) {
100 resume();
101 }
102
103 writePuffer.insert(writePuffer.end(), chunk, chunk + chunkLen);
104
105 if (source != nullptr && writePuffer.size() > 5 * blockSize) {
106 source->suspend();
107 }
108 } else {
109 LOG(TRACE) << getName() << ": Send request while not enabled";
110 }
111 } else {
112 LOG(TRACE) << getName() << ": Send request while shutdown in progress";
113 }
114 }

116 bool SocketWriter::streamToPeer(core::pipe::Source* source) {
117 bool success = false;
118
119 if (!shutdownInProgress && !markShutdown) {
120 if (isEnabled()) {
121 success = source != nullptr;
122
123 if (success) {
124 LOG(TRACE) << getName() << ": Stream started";
125 } else {
126 LOG(TRACE) << getName() << ": Stream source is nullptr";
127 }
128 } else {
129 LOG(TRACE) << getName() << ": Stream request while not enabled";
130 }
131 } else {
132 LOG(TRACE) << getName() << ": Stream request while shutdown in progress";
133 }
134
135 this->source = source;
136
137 return success;
138 }

145 void SocketWriter::shutdownWrite(const std::function<void()>& onShutdown) {
146 if (!shutdownInProgress) {
147 SocketWriter::setTimeout(SocketWriter::terminateTimeout);
148
149 SocketWriter::onShutdown = onShutdown;
150 if (SocketWriter::writePuffer.empty()) {
151 doWriteShutdown(onShutdown);
152 shutdownInProgress = true;
153 } else {
154 SocketWriter::markShutdown = true;
155 LOG(TRACE) << getName() << ": Delay shutdown due to queued data";
156 }
157 }
