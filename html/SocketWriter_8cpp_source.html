64 void SocketWriter::doWrite() {
65 if (!writePuffer.empty()) {
66 const std::size_t writeLen = (writePuffer.size() < blockSize) ? writePuffer.size() : blockSize;
67 const ssize_t retWrite = write(writePuffer.data(), writeLen);
68
69 if (retWrite > 0) {
70 writePuffer.erase(writePuffer.begin(), writePuffer.begin() + retWrite);
71
72 if (writePuffer.capacity() > writePuffer.size() * 2) {
73 writePuffer.shrink_to_fit();
74 }
75
76 if (!isSuspended()) {
77 suspend();
78 }
79 span();
80 } else if ((errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) && isSuspended()) {
81 resume();
82 } else {
83 onStatus(errno);
84 disable();
85 }
86 } else {
87 if (!isSuspended()) {
88 suspend();
89 }
90
91 if (markShutdown) {
92 LOG(TRACE) << "SocketWriter: Do delayed shutdown";
93
94 shutdownWrite(onShutdown);
95 }
96 }
97 }

103 void SocketWriter::sendToPeer(const char* junk, std::size_t junkLen) {
104 if (isEnabled()) {
105 if (writePuffer.empty()) {
106 resume();
107 }
108
109 writePuffer.insert(writePuffer.end(), junk, junk + junkLen);
110 } else {
111 LOG(TRACE) << "SocketWriter: Send to peer while not enabled";

123 void SocketWriter::shutdownWrite(const std::function<void()>& onShutdown) {
124 if (!shutdownInProgress) {
125 SocketWriter::setTimeout(SocketWriter::terminateTimeout);
126
127 SocketWriter::onShutdown = onShutdown;
128 if (SocketWriter::writePuffer.empty()) {
129 doWriteShutdown(onShutdown);
130 shutdownInProgress = true;
131 } else {
132 SocketWriter::markShutdown = true;
133 LOG(TRACE) << "SocketWriter: Delay shutdown due to queued data";
134 }
135 }
