108 sigaction(SIGINT, &oldIntAct, nullptr);
109 sigaction(SIGTERM, &oldTermAct, nullptr);

116 TickStatus EventLoop::_tick(const utils::Timeval& tickTimeOut) {
117 TickStatus tickStatus = TickStatus::SUCCESS;
118
119 tickCounter++;
120
121 sigset_t newSet{};
122 sigaddset(&newSet, SIGINT);
123 sigaddset(&newSet, SIGTERM);
124 sigaddset(&newSet, SIGALRM);
125 sigaddset(&newSet, SIGHUP);
126
127 sigset_t oldSet{};
128 sigprocmask(SIG_BLOCK, &newSet, &oldSet);
129
130 if (eventLoopState == State::RUNNING || eventLoopState == State::STOPPING) {
131 tickStatus = eventMultiplexer.tick(tickTimeOut, oldSet);
132 }
133
134 sigprocmask(SIG_SETMASK, &oldSet, nullptr);
135
136 return tickStatus;
137 }

139 TickStatus EventLoop::tick(const utils::Timeval& timeOut) {
140 TickStatus tickStatus = TickStatus::TRACE;
141
142 if (eventLoopState == State::INITIALIZED) {
143 struct sigaction sact {};
144 sigemptyset(&sact.sa_mask);
145 sact.sa_flags = 0;
146 sact.sa_handler = SIG_IGN;
147
148 struct sigaction oldPipeAct {};
149 sigaction(SIGPIPE, &sact, &oldPipeAct);
150
151 tickStatus = EventLoop::instance()._tick(timeOut);
152
153 sigaction(SIGPIPE, &oldPipeAct, nullptr);
154 } else {
155 EventLoop::instance().eventMultiplexer.clearEventQueue();
156 free();
157
158 PLOG(FATAL) << "Core: not initialized: No events will be processed\nCall SNodeC::init(argc, argv) before SNodeC::tick().";
159 }
160
161 return tickStatus;
162 }

164 int EventLoop::start(const utils::Timeval& timeOut) {
165 struct sigaction sact {};
166 sigemptyset(&sact.sa_mask);
167 sact.sa_flags = 0;
168 sact.sa_handler = SIG_IGN;
169
170 struct sigaction oldPipeAct {};
171 sigaction(SIGPIPE, &sact, &oldPipeAct);
172
173 sact.sa_handler = EventLoop::stoponsig;
174
175 struct sigaction oldIntAct {};
176 sigaction(SIGINT, &sact, &oldIntAct);
177
178 struct sigaction oldTermAct {};
179 sigaction(SIGTERM, &sact, &oldTermAct);
180
181 struct sigaction oldAlarmAct {};
182 sigaction(SIGALRM, &sact, &oldAlarmAct);
183
184 struct sigaction oldHupAct {};
185 sigaction(SIGHUP, &sact, &oldHupAct);
186
187 if (eventLoopState == State::INITIALIZED && utils::Config::bootstrap()) {
188 eventLoopState = State::RUNNING;
189 core::TickStatus tickStatus = TickStatus::SUCCESS;
190
191 LOG(TRACE) << "Core::EventLoop: started";
192
193 do {
194 tickStatus = EventLoop::instance()._tick(timeOut);
195 } while ((tickStatus == TickStatus::SUCCESS || tickStatus == TickStatus::INTERRUPTED) && eventLoopState == State::RUNNING);
196
197 switch (tickStatus) {
198 case TickStatus::SUCCESS:
199 LOG(TRACE) << "Core::EventLoop: Stopped";
200 break;
201 case TickStatus::NOOBSERVER:
202 LOG(TRACE) << "Core::EventLoop: No Observer";
203 break;
204 case TickStatus::INTERRUPTED:
205 LOG(TRACE) << "Core::EventLoop: Interrupted";
206 break;
207 case TickStatus::TRACE:
208 PLOG(FATAL) << "Core::EventLoop: _tick()";
209 break;
210 }
211
212 } else {
213 EventLoop::instance().eventMultiplexer.clearEventQueue();
214 }
215
216 sigaction(SIGPIPE, &oldPipeAct, nullptr);
217 sigaction(SIGINT, &oldIntAct, nullptr);
218 sigaction(SIGTERM, &oldTermAct, nullptr);
219 sigaction(SIGALRM, &oldAlarmAct, nullptr);
220 sigaction(SIGHUP, &oldHupAct, nullptr);
221
222 free();
223
224 return stopsig;
225 }

231 void EventLoop::free() {
232 std::string signal = "SIG" + utils::system::sigabbrev_np(stopsig);
233
234 if (signal == "SIGUNKNOWN") {
235 signal = std::to_string(stopsig);
236 }
237
238 if (stopsig != 0) {
239 LOG(TRACE) << "Core: Sending signal " << signal << " to all DescriptorEventReceivers";
240
241 EventLoop::instance().eventMultiplexer.signal(stopsig);
242 }
243
244 eventLoopState = State::STOPPING;
245
246 utils::Timeval timeout = 2;
247
248 core::TickStatus tickStatus = TickStatus::SUCCESS;
249 do {
250 auto t1 = std::chrono::system_clock::now();
251 const utils::Timeval timeoutOp = timeout;
252
253 tickStatus = EventLoop::instance()._tick(timeoutOp);
254
255 auto t2 = std::chrono::system_clock::now();
256 const std::chrono::duration<double> seconds = t2 - t1;
257
258 timeout -= seconds.count();
259 } while (timeout > 0 && (tickStatus == TickStatus::SUCCESS || tickStatus == TickStatus::INTERRUPTED));
260
261 LOG(TRACE) << "Core: Terminate all stalled DescriptorEventReceivers";
262
263 EventLoop::instance().eventMultiplexer.terminate();
264
265 LOG(TRACE) << "Core: Close all libraries opened during runtime";
266
267 DynamicLoader::execDlCloseAll();
268
269 LOG(TRACE) << "Core:: Clean up the filesystem";
270
271 utils::Config::terminate();
272
273 LOG(TRACE) << "Core:: All resources released";
274
275 LOG(TRACE) << "SNode.C: Ended ... BYE";
276 }
