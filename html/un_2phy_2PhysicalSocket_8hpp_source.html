56 template <template <typename SocketAddress> typename PhysicalPeerSocket>
57 PhysicalSocket<PhysicalPeerSocket>::~PhysicalSocket() {
58 if (lockFd >= 0) {
59 if (std::remove(Super::getBindAddress().getSunPath().data()) == 0) {
60 LOG(TRACE) << "Remove sun path: " << Super::getBindAddress().getSunPath();
61 } else {
62 PLOG(TRACE) << "Remove sun path: " << Super::getBindAddress().getSunPath();
63 }
64
65 if (core::system::flock(lockFd, LOCK_UN) == 0) {
66 LOG(TRACE) << "Remove lock from file: " << Super::getBindAddress().getSunPath().append(".lock");
67 } else {
68 PLOG(TRACE) << "Remove lock from file: " << Super::getBindAddress().getSunPath().append(".lock");
69 }
70
71 if (std::remove(Super::bindAddress.getSunPath().append(".lock").data()) == 0) {
72 LOG(TRACE) << "Remove lock file: " << Super::getBindAddress().getSunPath().append(".lock");
73 } else {
74 PLOG(TRACE) << "Remove lock file: " << Super::getBindAddress().getSunPath().append(".lock");
75 }
76
77 core::system::close(lockFd);

82 template <template <typename SocketAddress> typename PhysicalPeerSocket>
83 int PhysicalSocket<PhysicalPeerSocket>::bind(SocketAddress& bindAddress) {
84 if (!bindAddress.getSunPath().empty()) {
85 if ((lockFd = open(bindAddress.getSunPath().append(".lock").data(), O_RDONLY | O_CREAT, 0600)) >= 0) {
86 LOG(TRACE) << "Opening lock file: " << bindAddress.getSunPath().append(".lock").data();
87 if (core::system::flock(lockFd, LOCK_EX | LOCK_NB) == 0) {
88 LOG(TRACE) << "Locking lock file: " << bindAddress.getSunPath().append(".lock").data();
89 if (std::filesystem::exists(bindAddress.getSunPath().data())) {
90 if (std::remove(bindAddress.getSunPath().data()) == 0) {
91 LOG(TRACE) << "Removed stalled sun_path: " << bindAddress.getSunPath().data();
92 } else {
93 PLOG(TRACE) << "Removed stalled sun path: " << bindAddress.getSunPath().data();
94 }
95 }
96 } else {
97 PLOG(TRACE) << "Locking lock file " << bindAddress.getSunPath().append(".lock").data();
98
99 core::system::close(lockFd);
100 lockFd = -1;
101 }
102 } else {
103 PLOG(TRACE) << "Opening lock file: " << bindAddress.getSunPath().append(".lock").data();
104 }
105 }
106
107 return Super::bind(bindAddress);
108 }
