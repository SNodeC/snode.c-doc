38 template <typename SubProtocolFactoryT>
39 class SubProtocolFactorySelector {
40 public:
41 SubProtocolFactorySelector(const SubProtocolFactorySelector&) = delete;
42 SubProtocolFactorySelector& operator=(const SubProtocolFactorySelector&) = delete;
43
44 using SubProtocolFactory = SubProtocolFactoryT;
45
46 enum class Role { SERVER, CLIENT };
47
48 protected:
49 SubProtocolFactorySelector() = default;
50 virtual ~SubProtocolFactorySelector() = default;
51
52 virtual SubProtocolFactory* load(const std::string& subProtocolName) = 0;
53
54 static SubProtocolFactory* load(const std::string& subProtocolName,
55 const std::string& subProtocolLibraryFile,
56 const std::string& subProtocolFactoryFunctionName) {
57 SubProtocolFactory* subProtocolFactory = nullptr;
58
59 void* handle = core::DynamicLoader::dlOpen(subProtocolLibraryFile);
60
61 if (handle != nullptr) {
62 SubProtocolFactory* (*getSubProtocolFactory)() =
63 reinterpret_cast<SubProtocolFactory* (*) ()>(core::DynamicLoader::dlSym(handle, subProtocolFactoryFunctionName));
64 if (getSubProtocolFactory != nullptr) {
65 subProtocolFactory = getSubProtocolFactory();
66 if (subProtocolFactory != nullptr) {
67 subProtocolFactory->setHandle(handle);
68 LOG(DEBUG) << "WebSocket: SubProtocolFactory created successfull: " << subProtocolName;
69 } else {
70 LOG(DEBUG) << "WebSocket: SubProtocolFactory not created: " << subProtocolName;
71 core::DynamicLoader::dlClose(handle);
72 }
73 } else {
74 LOG(DEBUG) << "WebSocket: Optaining function \"" << subProtocolFactoryFunctionName
75 << "\" in plugin failed: " << core::DynamicLoader::dlError();
76 core::DynamicLoader::dlClose(handle);
77 }
78 }
79
80 return subProtocolFactory;
81 }
82
83 public:
84 SubProtocolFactory* select(const std::string& subProtocolName) {
85 SubProtocolFactory* subProtocolFactory = nullptr;
86
87 if (subProtocolFactories.contains(subProtocolName)) {
88 subProtocolFactory = subProtocolFactories[subProtocolName];
89 }
90
91 return subProtocolFactory;
92 }
93
94 SubProtocolFactory* select(const std::string& subProtocolName, [[maybe_unused]] Role role) {
95 SubProtocolFactory* subProtocolFactory = select(subProtocolName);
96
97 if (subProtocolFactory == nullptr) {
98 if (linkedSubProtocolFactories.contains(subProtocolName)) {
99 SubProtocolFactory* (*plugin)() = linkedSubProtocolFactories[subProtocolName];
100 subProtocolFactory = plugin();
101 } else if (!onlyLinked) {
102 subProtocolFactory = load(subProtocolName);
103 }
104
105 if (subProtocolFactory != nullptr) {
106 subProtocolFactories.insert({subProtocolName, subProtocolFactory});
107 }
108 }
109
110 return subProtocolFactory;
111 }
112
113 void unload(SubProtocolFactory* subProtocolFactory) {
114 std::string name = subProtocolFactory->getName();
115
116 if (subProtocolFactories.contains(name)) {
117 subProtocolFactories.erase(name);
118
119 void* handle = subProtocolFactory->getHandle();
120 delete subProtocolFactory;
121
122 if (handle != nullptr) {
123 core::DynamicLoader::dlCloseDelayed(handle);
124 }
125 }
126 }
127
128 protected:
129 void allowDlOpen() {
130 onlyLinked = false;
131 }
132
133 template <typename SubProtocolFactory>
134 void link(const std::string& subProtocolName, SubProtocolFactory* (*subProtocolFactory)()) {
135 onlyLinked = true;
136 linkedSubProtocolFactories[subProtocolName] =
137 reinterpret_cast<SubProtocolFactorySelector::SubProtocolFactory* (*) ()>(subProtocolFactory);
138 }
139
140 private:
141 std::map<std::string, SubProtocolFactory*> subProtocolFactories;
142 std::map<std::string, SubProtocolFactory* (*) ()> linkedSubProtocolFactories;
143
144 bool onlyLinked = false;
145 };
