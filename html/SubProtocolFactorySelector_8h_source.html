53 SubProtocolFactory* load(const std::string& subProtocolName,
54 const std::string& subProtocolLibraryFile,
55 const std::string& subProtocolFactoryFunctionName) {
56 SubProtocolFactory* subProtocolFactory = nullptr;
57
58 void* handle = dlOpen(subProtocolLibraryFile, RTLD_LAZY | RTLD_GLOBAL);
59
60 if (handle != nullptr) {
61 SubProtocolFactory* (*getSubProtocolFactory)() =
62 reinterpret_cast<SubProtocolFactory* (*) ()>(core::DynamicLoader::dlSym(handle, subProtocolFactoryFunctionName));
63 if (getSubProtocolFactory != nullptr) {
64 subProtocolFactory = getSubProtocolFactory();
65 if (subProtocolFactory != nullptr) {
66 subProtocolFactory->setHandle(handle);
67 LOG(DEBUG) << "SubProtocolFactory created successfull: " << subProtocolName;
68 } else {
69 LOG(DEBUG) << "SubProtocolFactory not created: " << subProtocolName;
70 core::DynamicLoader::dlClose(handle);
71 }
72 } else {
73 LOG(DEBUG) << "Optaining function \"" << subProtocolFactoryFunctionName
74 << "\" in plugin failed: " << core::DynamicLoader::dlError();
75 core::DynamicLoader::dlClose(handle);
76 }
77 }
78
79 return subProtocolFactory;
80 }
81
82 public:
83 SubProtocolFactory* select(const std::string& subProtocolName) {
84 SubProtocolFactory* subProtocolFactory = nullptr;
85
86 if (subProtocolFactories.contains(subProtocolName)) {
87 subProtocolFactory = subProtocolFactories[subProtocolName];
88 }
89
90 return subProtocolFactory;
91 }
92
93 SubProtocolFactory* select(const std::string& subProtocolName, Role) {
94 SubProtocolFactory* subProtocolFactory = select(subProtocolName);
95
96 if (subProtocolFactory == nullptr) {
97 if (linkedSubProtocolFactories.contains(subProtocolName)) {
98 SubProtocolFactory* (*plugin)() = linkedSubProtocolFactories[subProtocolName];
99 subProtocolFactory = plugin();
100 } else if (!onlyLinked) {
101 subProtocolFactory = load(subProtocolName);
102 }
103
104 if (subProtocolFactory != nullptr) {
105 subProtocolFactories.insert({subProtocolName, subProtocolFactory});
106 }
107 }
108
109 return subProtocolFactory;
110 }
111
112 void unload(SubProtocolFactory* subProtocolFactory) {
113 std::string name = subProtocolFactory->getName();
114
115 if (subProtocolFactories.contains(name)) {
116 void* handle = subProtocolFactory->getHandle();
117 delete subProtocolFactory;
118
119 if (handle != nullptr) {
120 core::DynamicLoader::dlCloseDelayed(handle);
121 }
122
123 subProtocolFactories.erase(name);
124 }
125 }
126
127 protected:
128 void allowDlOpen() {
129 onlyLinked = false;
130 }

132 template <typename SubProtocolFactory>
133 void link(const std::string& subProtocolName, SubProtocolFactory* (*subProtocolFactory)()) {
134 onlyLinked = true;
135 linkedSubProtocolFactories[subProtocolName] =
136 reinterpret_cast<SubProtocolFactorySelector::SubProtocolFactory* (*) ()>(subProtocolFactory);
137 }
138
139 private:
140 std::map<std::string, SubProtocolFactory*> subProtocolFactories;

142
143 bool onlyLinked = false;
144 };
