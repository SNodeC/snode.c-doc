58 void Transmitter::send(bool end, uint8_t opCode, const char* message, std::size_t messageLength) const {
59 std::size_t messageOffset = 0;
60
61 do {
62 std::size_t sendFrameLength =
63 (messageLength - messageOffset <= WSMAXFRAMEPAYLOADLENGTH) ? messageLength - messageOffset : WSMAXFRAMEPAYLOADLENGTH;
64
65 bool fin = (sendFrameLength == messageLength - messageOffset) && end;
66
67 sendFrame(fin, opCode, message + messageOffset, sendFrameLength);
68
69 messageOffset += sendFrameLength;
70
71 opCode = 0; // continuation
72 } while (messageLength - messageOffset > 0);
73 }

75 void Transmitter::sendFrame(bool fin, uint8_t opCode, const char* payload, uint64_t payloadLength) const {
76 uint64_t length = 0;
77
78 if (payloadLength < 126) {
79 length = payloadLength;
80 } else if (payloadLength < 0x10000) {
81 length = 126;
82 } else {
83 length = 127;
84 }
85
86 char header[2];
87 header[0] = static_cast<char>((fin ? 0b10000000 : 0) | opCode);
88
89 header[1] = static_cast<char>((masking ? 0b10000000 : 0) | length);
90
91 sendFrameData(header, 2);
92
93 switch (length) {
94 case 126:
95 sendFrameData(static_cast<uint16_t>(payloadLength));
96 break;
97 case 127:
98 sendFrameData(payloadLength);
99 break;
100 }
101
102 union MaskingKey {
103 uint32_t keyAsValue;
104 char keyAsBytes[4];
105 };
106
107 MaskingKey maskingKeyAsArray = {.keyAsValue = distribution(randomDevice)};
108
109 if (masking) {
110 sendFrameData(htobe32(maskingKeyAsArray.keyAsValue));
111
112 for (uint64_t i = 0; i < payloadLength; i++) {
113 *(const_cast<char*>(payload) + i) = *(payload + i) ^ *(maskingKeyAsArray.keyAsBytes + i % 4);
114 }
115 }
116
117 sendFrameData(payload, payloadLength);
118
119 if (masking) {
120 for (uint64_t i = 0; i < payloadLength; i++) {
121 *(const_cast<char*>(payload) + i) = *(payload + i) ^ *(maskingKeyAsArray.keyAsBytes + i % 4);
122 }
123 }
124 }
