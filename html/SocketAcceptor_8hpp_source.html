57 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
58 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::initAcceptEvent() {
59 if (!config->getDisabled()) {
60 LOG(TRACE) << config->getInstanceName() << ": enabled";
61
62 core::eventreceiver::AcceptEventReceiver::setTimeout(config->getAcceptTimeout());
63
64 try {
65 State state = State::OK;
66
67 localAddress = config->Local::getSocketAddress();
68 physicalSocket = new PhysicalSocket();
69
70 if (physicalSocket->open(config->getSocketOptions(), PhysicalSocket::Flags::NONBLOCK) < 0) {
71 switch (errno) {
72 case EMFILE:
73 case ENFILE:
74 case ENOBUFS:
75 case ENOMEM:
76 state = State::ERROR;
77 PLOG(WARNING) << config->getInstanceName() << ": open '" << localAddress.toString() << "'";
78 break;
79 default:
80 state = State::FATAL;
81 PLOG(ERROR) << config->getInstanceName() << ": open '" << localAddress.toString() << "'";
82 break;
83 }
84 } else if (physicalSocket->bind(localAddress) < 0) {
85 switch (errno) {
86 case EADDRINUSE:
87 state = State::ERROR;
88 PLOG(WARNING) << config->getInstanceName() << ": bind '" << localAddress.toString() << "'";
89 break;
90 default:
91 state = State::FATAL;
92 PLOG(ERROR) << config->getInstanceName() << ": bind '" << localAddress.toString() << "'";
93 break;
94 }
95 } else if (physicalSocket->listen(config->getBacklog()) < 0) {
96 switch (errno) {
97 case EADDRINUSE:
98 state = State::ERROR;
99 PLOG(WARNING) << config->getInstanceName() << ": listen '" << localAddress.toString() << "'";
100 break;
101 default:
102 state = State::FATAL;
103 PLOG(ERROR) << config->getInstanceName() << ": listen '" << localAddress.toString() << "'";
104 break;
105 }
106 } else {
107 enable(physicalSocket->getFd());
108 LOG(TRACE) << config->getInstanceName() << ": listen '" << localAddress.toString() << "' success";
109 }
110
111 if (localAddress.useNext()) {
112 LOG(TRACE) << config->getInstanceName() << ": using next SocketAddress '"
113 << config->Local::getSocketAddress().toString() << "'";
114 new SocketAcceptor(socketContextFactory, onConnect, onConnected, onDisconnect, onStatus, config);
115 } else {
116 onStatus(localAddress, state);
117 }
118
119 if (!isEnabled()) {
120 destruct();
121 }
122 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
123 LOG(ERROR) << config->getInstanceName() << ": " << badSocketAddress.what();
124
125 onStatus(localAddress, badSocketAddress.getState());
126 destruct();
127 }
128 } else {
129 LOG(TRACE) << config->getInstanceName() << ": disabled";
130
131 onStatus(localAddress, core::socket::State::DISABLED);
132 destruct();
133 }
134 }

136 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
137 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::acceptEvent() {
138 int acceptsPerTick = config->getAcceptsPerTick();
139
140 do {
141 PhysicalSocket physicalClientSocket(physicalSocket->accept4(PhysicalSocket::Flags::NONBLOCK));
142 if (physicalClientSocket.isValid()) {
143 LOG(TRACE) << config->getInstanceName() << ": accept success '" << localAddress.toString() << "'";
144
145 SocketConnectionFactory(onConnect, onConnected, onDisconnect).create(physicalClientSocket, config);
146 } else if (errno != EINTR && errno != EAGAIN && errno != EWOULDBLOCK) {
147 PLOG(ERROR) << config->getInstanceName() << ": accept failed '" << localAddress.toString() << "'";
148 }
149 } while (--acceptsPerTick > 0);
150 }
