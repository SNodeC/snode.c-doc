36 template <typename SocketAddress, typename PhysicalSocket, typename Config>
37 SocketAddress getLocalSocketAddress(PhysicalSocket& physicalSocket, Config& config) {
38 typename SocketAddress::SockAddr localSockAddr;
39 typename SocketAddress::SockLen localSockAddrLen = sizeof(typename SocketAddress::SockAddr);
40
41 SocketAddress localPeerAddress;
42 if (physicalSocket.getSockName(localSockAddr, localSockAddrLen) == 0) {
43 try {
44 localPeerAddress = config->Local::getSocketAddress(localSockAddr, localSockAddrLen);
45 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << localPeerAddress.toString();
46 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
47 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << badSocketAddress.what();
48
49 try {
50 localPeerAddress = config->Local::Super::getSocketAddress(localSockAddr, localSockAddrLen);
51 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << localPeerAddress.toString();
52 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) { // cppcheck-suppress shadowVariable
53 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << badSocketAddress.what();
54 }
55 }
56 }
57
58 return localPeerAddress;
59 }

124 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
125 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::init() {
126 if (!config->getDisabled() && core::eventLoopState() == core::State::RUNNING) {
127 try {
128 LOG(TRACE) << config->getInstanceName() << ": Starting";
129
130 localAddress = config->Local::getSocketAddress();
131
132 core::socket::State state = core::socket::STATE_OK;
133
134 if (physicalServerSocket.open(config->getSocketOptions(), PhysicalServerSocket::Flags::NONBLOCK) < 0) {
135 switch (errno) {
136 case EMFILE:
137 case ENFILE:
138 case ENOBUFS:
139 case ENOMEM:
140 state = core::socket::STATE_ERROR;
141 PLOG(TRACE) << config->getInstanceName() << " open: '" << localAddress.toString() << "'";
142 break;
143 default:
144 state = core::socket::STATE_FATAL;
145 PLOG(TRACE) << config->getInstanceName() << " open: '" << localAddress.toString() << "'";
146 break;
147 }
148 } else if (physicalServerSocket.bind(localAddress) < 0) {
149 switch (errno) {
150 case EADDRINUSE:
151 state = core::socket::STATE_ERROR;
152 PLOG(TRACE) << config->getInstanceName() << " bind: '" << localAddress.toString() << "'";
153 break;
154 default:
155 state = core::socket::STATE_FATAL;
156 PLOG(TRACE) << config->getInstanceName() << " bind: '" << localAddress.toString() << "'";
157 break;
158 }
159 } else if (physicalServerSocket.listen(config->getBacklog()) < 0) {
160 switch (errno) {
161 case EADDRINUSE:
162 state = core::socket::STATE_ERROR;
163 PLOG(TRACE) << config->getInstanceName() << " listen: '" << localAddress.toString() << "'";
164 break;
165 default:
166 state = core::socket::STATE_FATAL;
167 PLOG(TRACE) << config->getInstanceName() << " listen: '" << localAddress.toString() << "'";
168 break;
169 }
170 } else {
171 if (enable(physicalServerSocket.getFd())) {
172 LOG(TRACE) << config->getInstanceName() << " enabled: '" << localAddress.toString() << "' success";
173 } else {
174 state = core::socket::STATE(core::socket::STATE_FATAL, ECANCELED, "SocketAcceptor not enabled");
175 LOG(TRACE) << config->getInstanceName() << " enabled: '" << localAddress.toString() << "' failed";
176 }
177 }
178
179 if (localAddress.useNext()) {
180 LOG(TRACE) << config->getInstanceName() << ": Using next SocketAddress '"
181 << config->Local::getSocketAddress().toString() << "'";
182
183 onStatus(localAddress, (state | core::socket::State::NO_RETRY));
184
185 useNextSocketAddress();
186 } else {
187 onStatus(localAddress, state);
188 }
189 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
190 LOG(TRACE) << config->getInstanceName() << ": " << badSocketAddress.what();
191
192 onStatus({}, core::socket::STATE(badSocketAddress.getState(), badSocketAddress.getErrnum(), badSocketAddress.what()));
193 }
194 } else {
195 LOG(TRACE) << config->getInstanceName() << ": Disabled";
196
197 onStatus({}, core::socket::STATE_DISABLED);
198 }
199
200 if (isEnabled()) {
201 core::eventreceiver::AcceptEventReceiver::setTimeout(config->getAcceptTimeout());
202 } else {
203 destruct();
204 }
205 }

207 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
208 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::acceptEvent() {
209 int acceptsPerTick = config->getAcceptsPerTick();
210
211 do {
212 PhysicalServerSocket connectedPhysicalSocket(physicalServerSocket.accept4(PhysicalServerSocket::Flags::NONBLOCK),
213 physicalServerSocket.getBindAddress());
214 if (connectedPhysicalSocket.isValid()) {
215 LOG(TRACE) << config->getInstanceName() << " accept: Success '" << connectedPhysicalSocket.getBindAddress().toString()
216 << "'";
217
218 SocketConnection* socketConnection =
219 new SocketConnection(config->getInstanceName(),
220 std::move(connectedPhysicalSocket),
221 onDisconnect,
222 localAddress.toString(false),
223 getLocalSocketAddress<SocketAddress>(connectedPhysicalSocket, config),
224 getRemoteSocketAddress<SocketAddress>(connectedPhysicalSocket, config),
225 config->getReadTimeout(),
226 config->getWriteTimeout(),
227 config->getReadBlockSize(),
228 config->getWriteBlockSize(),
229 config->getTerminateTimeout());
230
231 onConnect(socketConnection);
232 onConnected(socketConnection);
233 } else if (errno != EINTR && errno != EAGAIN && errno != EWOULDBLOCK) {
234 PLOG(TRACE) << config->getInstanceName() << " accept: Failed '" << physicalServerSocket.getBindAddress().toString() << "'";
235 }
236 } while (--acceptsPerTick > 0);
237 }
