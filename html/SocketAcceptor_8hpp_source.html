58 template <typename SocketAddress, typename PhysicalSocket, typename Config>
59 SocketAddress getLocalSocketAddress(PhysicalSocket& physicalSocket, Config& config) {
60 typename SocketAddress::SockAddr localSockAddr;
61 typename SocketAddress::SockLen localSockAddrLen = sizeof(typename SocketAddress::SockAddr);
62
63 SocketAddress localPeerAddress;
64 if (physicalSocket.getSockName(localSockAddr, localSockAddrLen) == 0) {
65 try {
66 localPeerAddress = config->Local::getSocketAddress(localSockAddr, localSockAddrLen);
67 LOG(TRACE) << config->getInstanceName() << " [" << physicalSocket.getFd() << "]" << std::setw(25)
68 << " PeerAddress (local): " << localPeerAddress.toString();
69 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
70 LOG(WARNING) << config->getInstanceName() << " [" << physicalSocket.getFd() << "]" << std::setw(25)
71 << " PeerAddress (local): " << badSocketAddress.what();
72 }
73 } else {
74 PLOG(WARNING) << config->getInstanceName() << " [" << physicalSocket.getFd() << "]" << std::setw(25)
75 << " PeerAddress (local) not retrievable";
76 }
77
78 return localPeerAddress;
79 }

142 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
143 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::init() {
144 if (!config->getDisabled() && core::eventLoopState() == core::State::RUNNING) {
145 try {
146 LOG(TRACE) << config->getInstanceName() << " Starting";
147
148 localAddress = config->Local::getSocketAddress();
149
150 core::socket::State state = core::socket::STATE_OK;
151
152 if (physicalServerSocket.open(config->getSocketOptions(), PhysicalServerSocket::Flags::NONBLOCK) < 0) {
153 switch (errno) {
154 case EMFILE:
155 case ENFILE:
156 case ENOBUFS:
157 case ENOMEM:
158 PLOG(DEBUG) << config->getInstanceName() << " open: '" << localAddress.toString() << "'";
159
160 state = core::socket::STATE_ERROR;
161 break;
162 default:
163 PLOG(DEBUG) << config->getInstanceName() << " open: '" << localAddress.toString() << "'";
164
165 state = core::socket::STATE_FATAL;
166 break;
167 }
168 } else if (physicalServerSocket.bind(localAddress) < 0) {
169 switch (errno) {
170 case EADDRINUSE:
171 PLOG(DEBUG) << config->getInstanceName() << " bind: '" << localAddress.toString() << "'";
172
173 state = core::socket::STATE_ERROR;
174 break;
175 default:
176 PLOG(DEBUG) << config->getInstanceName() << " bind: '" << localAddress.toString() << "'";
177
178 state = core::socket::STATE_FATAL;
179 break;
180 }
181 } else if (physicalServerSocket.listen(config->getBacklog()) < 0) {
182 switch (errno) {
183 case EADDRINUSE:
184 PLOG(DEBUG) << config->getInstanceName() << " listen: '" << localAddress.toString() << "'";
185
186 state = core::socket::STATE_ERROR;
187 break;
188 default:
189 PLOG(DEBUG) << config->getInstanceName() << " listen: '" << localAddress.toString() << "'";
190
191 state = core::socket::STATE_FATAL;
192 break;
193 }
194 } else {
195 if (enable(physicalServerSocket.getFd())) {
196 LOG(DEBUG) << config->getInstanceName() << " enabled: '" << localAddress.toString() << "' success";
197 } else {
198 LOG(DEBUG) << config->getInstanceName() << " enabled: '" << localAddress.toString() << "' failed";
199
200 state = core::socket::STATE(core::socket::STATE_FATAL, ECANCELED, "SocketAcceptor not enabled");
201 }
202 }
203
204 SocketAddress currentLocalAddress = localAddress;
205 if (localAddress.useNext()) {
206 onStatus(currentLocalAddress, (state | core::socket::State::NO_RETRY));
207
208 LOG(DEBUG) << config->getInstanceName() << " using next SocketAddress: '"
209 << config->Local::getSocketAddress().toString() << "'";
210
211 useNextSocketAddress();
212 } else {
213 onStatus(currentLocalAddress, state);
214 }
215 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
216 LOG(DEBUG) << config->getInstanceName() << " " << badSocketAddress.what();
217
218 onStatus({}, core::socket::STATE(badSocketAddress.getState(), badSocketAddress.getErrnum(), badSocketAddress.what()));
219 }
220 } else {
221 LOG(DEBUG) << config->getInstanceName() << " disabled";
222
223 onStatus({}, core::socket::STATE_DISABLED);
224 }
225
226 if (isEnabled()) {
227 core::eventreceiver::AcceptEventReceiver::setTimeout(config->getAcceptTimeout());
228 } else {
229 destruct();
230 }
231 }

233 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
234 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::acceptEvent() {
235 int acceptsPerTick = config->getAcceptsPerTick();
236
237 do {
238 PhysicalServerSocket connectedPhysicalSocket(physicalServerSocket.accept4(PhysicalServerSocket::Flags::NONBLOCK),
239 physicalServerSocket.getBindAddress());
240 if (connectedPhysicalSocket.isValid()) {
241 LOG(DEBUG) << config->getInstanceName() << " [" << connectedPhysicalSocket.getFd() << "] accept success: '"
242 << connectedPhysicalSocket.getBindAddress().toString() << "'";
243
244 SocketConnection* socketConnection =
245 new SocketConnection(config->getInstanceName(),
246 std::move(connectedPhysicalSocket),
247 onDisconnect,
248 localAddress.toString(false),
249 getLocalSocketAddress<SocketAddress>(connectedPhysicalSocket, config),
250 getRemoteSocketAddress<SocketAddress>(connectedPhysicalSocket, config),
251 config->getReadTimeout(),
252 config->getWriteTimeout(),
253 config->getReadBlockSize(),
254 config->getWriteBlockSize(),
255 config->getTerminateTimeout());
256
257 onConnect(socketConnection);
258 onConnected(socketConnection);
259 } else if (errno != EINTR && errno != EAGAIN && errno != EWOULDBLOCK) {
260 PLOG(WARNING) << config->getInstanceName() << " accept failed: '" << physicalServerSocket.getBindAddress().toString()
261 << "'";
262 }
263 } while (--acceptsPerTick > 0);
264 }
