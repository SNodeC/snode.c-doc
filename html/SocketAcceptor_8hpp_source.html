34 template <typename SocketAddress, typename PhysicalSocket, typename Config>
35 SocketAddress getLocalSocketAddress(PhysicalSocket& physicalSocket, Config& config) {
36 typename SocketAddress::SockAddr localSockAddr;
37 typename SocketAddress::SockLen localSockAddrLen = sizeof(typename SocketAddress::SockAddr);
38
39 SocketAddress localPeerAddress;
40 if (physicalSocket.getSockName(localSockAddr, localSockAddrLen) == 0) {
41 try {
42 localPeerAddress = config->Local::getSocketAddress(localSockAddr, localSockAddrLen);
43 LOG(TRACE) << config->getInstanceName() << " PeerAddress (local): " << localPeerAddress.toString();
44 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
45 LOG(TRACE) << config->getInstanceName() << " PeerAddress (local): " << badSocketAddress.what();
46
47 try {
48 localPeerAddress = config->Local::Super::getSocketAddress(localSockAddr, localSockAddrLen);
49 LOG(TRACE) << config->getInstanceName() << " PeerAddress (local): " << localPeerAddress.toString();
50 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) { // cppcheck-suppress shadowVariable
51 LOG(TRACE) << config->getInstanceName() << " PeerAddress (local): " << badSocketAddress.what();
52 }
53 }
54 }
55
56 return localPeerAddress;
57 }

134 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
135 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::init() {
136 if (!config->getDisabled()) {
137 try {
138 LOG(TRACE) << config->getInstanceName() << ": Starting";
139
140 SocketAddress localAddress(config->Local::getSocketAddress());
141
142 core::socket::State state = core::socket::STATE_OK;
143
144 if (physicalServerSocket.open(config->getSocketOptions(), PhysicalServerSocket::Flags::NONBLOCK) < 0) {
145 switch (errno) {
146 case EMFILE:
147 case ENFILE:
148 case ENOBUFS:
149 case ENOMEM:
150 state = core::socket::STATE_ERROR;
151 PLOG(TRACE) << config->getInstanceName() << " open: '" << localAddress.toString() << "'";
152 break;
153 default:
154 state = core::socket::STATE_FATAL;
155 PLOG(TRACE) << config->getInstanceName() << " open: '" << localAddress.toString() << "'";
156 break;
157 }
158 } else if (physicalServerSocket.bind(localAddress) < 0) {
159 switch (errno) {
160 case EADDRINUSE:
161 state = core::socket::STATE_ERROR;
162 PLOG(TRACE) << config->getInstanceName() << " bind: '" << localAddress.toString() << "'";
163 break;
164 default:
165 state = core::socket::STATE_FATAL;
166 PLOG(TRACE) << config->getInstanceName() << " bind: '" << localAddress.toString() << "'";
167 break;
168 }
169 } else if (physicalServerSocket.listen(config->getBacklog()) < 0) {
170 switch (errno) {
171 case EADDRINUSE:
172 state = core::socket::STATE_ERROR;
173 PLOG(TRACE) << config->getInstanceName() << " listen: '" << localAddress.toString() << "'";
174 break;
175 default:
176 state = core::socket::STATE_FATAL;
177 PLOG(TRACE) << config->getInstanceName() << " listen: '" << localAddress.toString() << "'";
178 break;
179 }
180 } else {
181 if (enable(physicalServerSocket.getFd())) {
182 LOG(TRACE) << config->getInstanceName() << " enabled: '" << localAddress.toString() << "' success";
183 } else {
184 LOG(TRACE) << config->getInstanceName() << " enabled: '" << localAddress.toString() << "' failed";
185 }
186 }
187
188 if (localAddress.useNext()) {
189 LOG(TRACE) << config->getInstanceName() << ": Using next SocketAddress '"
190 << config->Local::getSocketAddress().toString() << "'";
191
192 onStatus(localAddress, (state | core::socket::State::NO_RETRY));
193
194 useNextSocketAddress();
195 } else {
196 onStatus(localAddress, state);
197 }
198
199 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
200 LOG(TRACE) << config->getInstanceName() << ": " << badSocketAddress.what();
201
202 onStatus({}, core::socket::STATE(badSocketAddress.getState(), badSocketAddress.getErrnum(), badSocketAddress.what()));
203 }
204 } else {
205 LOG(TRACE) << config->getInstanceName() << ": Disabled";
206
207 onStatus({}, core::socket::STATE_DISABLED);
208 }
209
210 if (isEnabled()) {
211 core::eventreceiver::AcceptEventReceiver::setTimeout(config->getAcceptTimeout());
212 } else {
213 destruct();
214 }
215 }

217 template <typename PhysicalSocketServer, typename Config, template <typename PhysicalSocketServerT> typename SocketConnection>
218 void SocketAcceptor<PhysicalSocketServer, Config, SocketConnection>::acceptEvent() {
219 int acceptsPerTick = config->getAcceptsPerTick();
220
221 do {
222 PhysicalServerSocket connectedPhysicalSocket(physicalServerSocket.accept4(PhysicalServerSocket::Flags::NONBLOCK),
223 physicalServerSocket.getBindAddress());
224 if (connectedPhysicalSocket.isValid()) {
225 LOG(TRACE) << config->getInstanceName() << " accept: Success '" << connectedPhysicalSocket.getBindAddress().toString()
226 << "'";
227
228 SocketConnection* socketConnection =
229 new SocketConnection(config->getInstanceName(),
230 std::move(connectedPhysicalSocket),
231 onDisconnect,
232 config->Local::getSocketAddress().toString(false),
233 getLocalSocketAddress<SocketAddress>(connectedPhysicalSocket, config),
234 getRemoteSocketAddress<SocketAddress>(connectedPhysicalSocket, config),
235 config->getReadTimeout(),
236 config->getWriteTimeout(),
237 config->getReadBlockSize(),
238 config->getWriteBlockSize(),
239 config->getTerminateTimeout());
240
241 onConnect(socketConnection);
242 onConnected(socketConnection);
243 } else if (errno != EINTR && errno != EAGAIN && errno != EWOULDBLOCK) {
244 PLOG(TRACE) << config->getInstanceName() << " accept: Failed '" << physicalServerSocket.getBindAddress().toString() << "'";
245 }
246 } while (--acceptsPerTick > 0);
247 }
