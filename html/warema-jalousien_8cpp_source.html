80 int ret = system(arguments.c_str());

87
88 switch (ret) {
89 case 0:
90 res->status(200).send("OK: " + arguments);
91 break;
92 case 127:
93 res->status(404).send("ERROR not found: " + arguments);
94 break;
95 }
96 });
