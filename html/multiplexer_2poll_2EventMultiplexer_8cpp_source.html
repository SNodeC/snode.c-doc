48 void PollFdsManager::muxAdd(core::DescriptorEventReceiver* eventReceiver, short event) {
49 int fd = eventReceiver->getRegisteredFd();
50
51 if (!pollFdIndices.contains(fd)) {
52 pollfds[nextIndex].events = event;
53 pollfds[nextIndex].fd = fd;
54
55 pollFdIndices[fd].index = nextIndex;
56 pollFdIndices[fd].events = event;
57
58 ++nextIndex;
59
60 if (nextIndex == pollfds.size()) {
61 pollfds.resize(pollfds.size() * 2, {-1, 0, 0});
62 pollFdIndices.reserve(pollfds.size());
63 }
64 } else {
65 PollFdIndex& pollFdIndex = pollFdIndices[fd];
66
67 pollfds[pollFdIndex.index].events |= event;
68 pollFdIndex.events |= event;
69 }
70 }

72 void PollFdsManager::muxDel(int fd, short event) {
73 std::unordered_map<int, PollFdIndex>::iterator itPollFdIndex = pollFdIndices.find(fd);
74
75 PollFdIndex& pollFdIndex = itPollFdIndex->second;
76
77 pollfds[pollFdIndex.index].events &= static_cast<short>(~event); // tilde promotes to int
78 pollFdIndex.events &= static_cast<short>(~event); // tilde promotes to int
79
80 if (pollFdIndex.events == 0) {
81 pollfds[pollFdIndex.index].fd = -1; // Compress will keep track of that descriptor
82 pollFdIndices.erase(fd);
83
84 if (pollfds.size() > (pollFdIndices.size() * 2) + 1) {
85 compress();
86 }
87 }
88 }

101 void PollFdsManager::compress() {
102 (void) std::remove_if(pollfds.begin(), pollfds.end(), [](const pollfd& pollFd) -> bool {
103 return pollFd.fd < 0;
104 });
105
106 pollfds.resize(pollFdIndices.size() + 1, {-1, 0, 0});
107
108 pollFdIndices.reserve(pollFdIndices.size() + 1);
109
110 for (uint32_t i = 0; i < pollFdIndices.size(); i++) {
111 pollFdIndices[pollfds[i].fd].index = i;
112 }
113
114 nextIndex = pollFdIndices.size();
115 }
