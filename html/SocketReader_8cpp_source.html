53 std::size_t SocketReader::doRead() {
54 errno = 0;
55
56 if (size == 0) {
57 cursor = 0;
58
59 ssize_t retRead = 0;
60 if (!shutdownInProgress) {
61 const std::size_t readLen = blockSize - size;
62 retRead = read(readBuffer.data() + size, readLen);
63 }
64 if (retRead > 0) {
65 size += static_cast<std::size_t>(retRead);
66
67 if (!isSuspended()) {
68 suspend();
69 }
70 span();
71 } else if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
72 if (isSuspended()) {
73 resume();
74 }
75 } else {
76 const int errnum = errno;
77 onStatus(errnum);
78
79 disable();
80 }
81 } else {
82 span();
83 }
84
85 return size;
86 }

93 std::size_t SocketReader::readFromPeer(char* chunk, std::size_t chunkLen) {
94 const std::size_t maxReturn = std::min(chunkLen, size);
95
96 std::copy(readBuffer.data() + cursor, readBuffer.data() + cursor + maxReturn, chunk);
97
98 cursor += maxReturn;
99 size -= maxReturn;
100
101 return maxReturn;
102 }
