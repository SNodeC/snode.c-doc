52 std::size_t SocketReader::doRead() {
53 errno = 0;
54
55 if (size == 0) {
56 cursor = 0;
57
58 const std::size_t readLen = blockSize - size;
59 const ssize_t retRead = read(readBuffer.data() + size, readLen);
60
61 if (retRead > 0) {
62 size += static_cast<std::size_t>(retRead);
63
64 if (!isSuspended()) {
65 suspend();
66 }
67 span();
68 } else if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
69 if (isSuspended()) {
70 resume();
71 }
72 } else {
73 const int errnum = errno;
74 onStatus(errnum);
75 disable();
76 }
77 } else {
78 span();
79 }
80
81 return size;
82 }

89 std::size_t SocketReader::readFromPeer(char* junk, std::size_t junkLen) {
90 const std::size_t maxReturn = std::min(junkLen, size);
91
92 std::copy(readBuffer.data() + cursor, readBuffer.data() + cursor + maxReturn, junk);
93
94 cursor += maxReturn;
95 size -= maxReturn;
96
97 return maxReturn;
98 }
