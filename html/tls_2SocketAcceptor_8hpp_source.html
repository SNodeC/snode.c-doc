40 const std::function<void(SocketConnection*)>& onDisconnect,
41 const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
42 const std::shared_ptr<Config>& config)
43 : Super(
44 socketContextFactory,
45 [onConnect, this](SocketConnection* socketConnection) -> void { // onConnect
46 socketConnection->startSSL(this->masterSslCtx, Super::config->getInitTimeout(), Super::config->getShutdownTimeout());
47
48 onConnect(socketConnection);
49 },
50 [socketContextFactory, onConnected, this](SocketConnection* socketConnection) -> void { // on Connected

52
53 if (ssl != nullptr) {
54 SSL_set_accept_state(ssl);
55
56 socketConnection->doSSLHandshake(
57 [socketContextFactory, onConnected, socketConnection, instanceName = this->config->getInstanceName()]()
58 -> void { // onSuccess
59 LOG(TRACE) << instanceName << ": SSL/TLS initial handshake success";
60
61 onConnected(socketConnection);
62 socketConnection->connected(socketContextFactory);
63 },
64 [socketConnection, instanceName = this->config->getInstanceName()]() -> void { // onTimeout
65 LOG(TRACE) << instanceName << ": SSL/TLS initial handshake timed out";
66
67 socketConnection->close();
68 },
69 [socketConnection, instanceName = this->config->getInstanceName()](int sslErr) -> void { // onStatus
70 ssl_log(instanceName + ": SSL/TLS initial handshake failed", sslErr);
71
72 socketConnection->close();
73 });
74 } else {
75 ssl_log_error(this->config->getInstanceName() + ": SSL/TLS initialization failed");
76
77 socketConnection->close();
78 }
79 },
80 [onDisconnect, instanceName = config->getInstanceName()](SocketConnection* socketConnection) -> void { // onDisconnect
81 LOG(TRACE) << instanceName << ": SSL/TLS connection destroyed";
82
83 socketConnection->stopSSL();
84 onDisconnect(socketConnection);
85 },
86 onStatus,

109 template <typename PhysicalSocketServer, typename Config>
110 void SocketAcceptor<PhysicalSocketServer, Config>::initAcceptEvent() {
111 if (!config->getDisabled()) {
112 masterSslCtx = ssl_ctx_new(config);
113
114 if (masterSslCtx != nullptr) {
115 masterSslCtxSans = ssl_get_sans(masterSslCtx);
116
117 for (const std::string& san : masterSslCtxSans) {
118 LOG(TRACE) << config->getInstanceName() << ": SSL_CTX (M) for (san)'" << san << "' as master installed";
119 }
120
121 for (const auto& [domain, sniCertConf] : config->getSniCerts()) {
122 if (!sniCertConf.empty()) {
123 SSL_CTX* sniSslCtx = ssl_ctx_new(sniCertConf);
124
125 if (sniSslCtx != nullptr) {
126 if (!domain.empty()) {
127 if (sniSslCtxs.contains(domain)) {
128 ssl_ctx_free(sniSslCtxs[domain]);
129 }
130 sniSslCtxs.insert_or_assign(domain, sniSslCtx);
131
132 LOG(TRACE) << config->getInstanceName() << ": SSL_CTX (E) sni for (dom)'" << domain
133 << "' as server name indication (sni) installed";
134 }
135
136 for (const std::string& san : ssl_get_sans(sniSslCtx)) {
137 sniSslCtxs.insert({san, sniSslCtx});
138
139 LOG(TRACE) << config->getInstanceName() << ": SSL_CTX (S) sni for (san)'" << san
140 << "' as server name indication (sni) installed";
141 }
142 } else {
143 LOG(TRACE) << config->getInstanceName() << ": Can not create SNI_SSL_CTX for domain '" << domain << "'";
144 }
145 }
146 }
147 forceSni = config->getForceSni();
148
149 SSL_CTX_set_client_hello_cb(masterSslCtx, clientHelloCallback, this);
150
151 Super::initAcceptEvent();
152 } else {
153 LOG(ERROR) << config->getInstanceName() << ": Master SSL/TLS certificate activation failed";
154
155 Super::onStatus(Super::config->Local::getSocketAddress(), core::socket::State::ERROR);
156 Super::destruct();
157 }
158 } else {
159 Super::initAcceptEvent();
160 }
161 }

163 template <typename PhysicalSocketServer, typename Config>
164 SSL_CTX* SocketAcceptor<PhysicalSocketServer, Config>::getMasterSniCtx(const std::string& serverNameIndication) {
165 SSL_CTX* sniSslCtx = nullptr;
166
167 LOG(TRACE) << config->getInstanceName() << ": Search for sni='" << serverNameIndication << "' in master certificate";
168
169 std::set<std::string>::iterator masterSniIt =
170 std::find_if(masterSslCtxSans.begin(),
171 masterSslCtxSans.end(),
172 [&serverNameIndication, config = this->config](const std::string& sni) -> bool {
173 LOG(TRACE) << config->getInstanceName() << ": .. " << sni.c_str();
174 return match(sni.c_str(), serverNameIndication.c_str());
175 });
176 if (masterSniIt != masterSslCtxSans.end()) {
177 LOG(TRACE) << config->getInstanceName() << ": found '" << *masterSniIt << "'";
178 sniSslCtx = masterSslCtx;
179 } else {
180 LOG(TRACE) << config->getInstanceName() << ": not found";
181 }
182
183 return sniSslCtx;
184 }

186 template <typename PhysicalSocketServer, typename Config>
187 SSL_CTX* SocketAcceptor<PhysicalSocketServer, Config>::getPoolSniCtx(const std::string& serverNameIndication) {
188 SSL_CTX* sniCtx = nullptr;
189
190 LOG(TRACE) << config->getInstanceName() << ": Search for sni='" << serverNameIndication << "' in sni certificates";
191
192 std::map<std::string, SSL_CTX*>::iterator sniPairIt =
193 std::find_if(sniSslCtxs.begin(),
194 sniSslCtxs.end(),
195 [&serverNameIndication, config = this->config](const std::pair<std::string, SSL_CTX*>& sniPair) -> bool {
196 LOG(TRACE) << config->getInstanceName() << ": .. " << sniPair.first.c_str();
197 return match(sniPair.first.c_str(), serverNameIndication.c_str());
198 });
199
200 if (sniPairIt != sniSslCtxs.end()) {
201 LOG(TRACE) << config->getInstanceName() << ": found '" << sniPairIt->first << "'";
202 sniCtx = sniPairIt->second;
203 } else {
204 LOG(TRACE) << config->getInstanceName() << ": not found";
205 }
206
207 return sniCtx;

221 template <typename PhysicalSocketServer, typename Config>
222 int SocketAcceptor<PhysicalSocketServer, Config>::clientHelloCallback(SSL* ssl, int* al, void* arg) {
223 int ret = SSL_CLIENT_HELLO_SUCCESS;
224
225 SocketAcceptor* socketAcceptor = static_cast<SocketAcceptor*>(arg);
226
227 std::string serverNameIndication = ssl_get_servername_from_client_hello(ssl);
228
229 if (!serverNameIndication.empty()) {
230 SSL_CTX* sniSslCtx = socketAcceptor->getSniCtx(serverNameIndication);
231 if (sniSslCtx != nullptr) {
232 LOG(TRACE) << socketAcceptor->config->getInstanceName() << ": Setting sni certificate for '" << serverNameIndication << "'";
233 ssl_set_ssl_ctx(ssl, sniSslCtx);
234 } else if (socketAcceptor->forceSni) {
235 LOG(WARNING) << socketAcceptor->config->getInstanceName() << ": No sni certificate found for '" << serverNameIndication
236 << "' but forceSni set - terminating";
237 ret = SSL_CLIENT_HELLO_ERROR;
238 *al = SSL_AD_UNRECOGNIZED_NAME;
239 } else {
240 LOG(TRACE) << socketAcceptor->config->getInstanceName() << ": No sni certificate found for '" << serverNameIndication
241 << "' still using master certificate";
242 }
243 } else {
244 LOG(TRACE) << socketAcceptor->config->getInstanceName() << ": No sni certificate set - the client did not request one";
245 }
246
247 return ret;
248 }

250 template <typename PhysicalSocketServer, typename Config>
251 bool SocketAcceptor<PhysicalSocketServer, Config>::match(const char* first, const char* second) {
252 // If we reach at the end of both strings, we are done
253 if (*first == '\0' && *second == '\0')
254 return true;
255
256 // Make sure to eliminate consecutive '*'
257 if (*first == '*') {
258 while (*(first + 1) == '*')
259 first++;
260 }
261
262 // Make sure that the characters after '*' are present
263 // in second string. This function assumes that the
264 // first string will not contain two consecutive '*'
265 if (*first == '*' && *(first + 1) != '\0' && *second == '\0')
266 return false;
267
268 // If the first string contains '?', or current
269 // characters of both strings match
270 if (*first == '?' || *first == *second)
271 return match(first + 1, second + 1);
272
273 // If there is *, then there are two possibilities
274 // a) We consider current character of second string
275 // b) We ignore current character of second string.
276 if (*first == '*')
277 return match(first + 1, second) || match(first, second + 1);
278 return false;
279 }
