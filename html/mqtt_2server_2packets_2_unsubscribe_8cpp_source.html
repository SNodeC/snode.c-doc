34 std::size_t Unsubscribe::deserializeVP(iot::mqtt::MqttContext* mqttContext) {
35 std::size_t consumed = 0;
36
37 switch (state) {
38 case 0: // V-Header
39 consumed += packetIdentifier.deserialize(mqttContext);
40 if (!packetIdentifier.isComplete()) {
41 break;
42 }
43
44 state++;
45 [[fallthrough]];
46 case 1: // Payload
47 consumed += topic.deserialize(mqttContext);
48
49 if (!topic.isComplete()) {
50 break;
51 } else {
52 topics.push_back(topic);
53 topic.reset();
54
55 if (getConsumed() + consumed < this->getRemainingLength()) {
56 break;
57 }
58 }
59
60 complete = true;
61 break;
62 }
63
64 return consumed;
65 }
