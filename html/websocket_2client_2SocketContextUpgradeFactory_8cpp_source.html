47 SocketContextUpgrade* SocketContextUpgradeFactory::create(core::socket::stream::SocketConnection* socketConnection,
48 web::http::client::Request* request,
49 web::http::client::Response* response) {
50 SocketContextUpgrade* socketContext = nullptr;
51
52 if (response->header("sec-websocket-accept") == base64::serverWebSocketKey(request->header("Sec-WebSocket-Key"))) {
53 const std::string subProtocolName = response->header("sec-websocket-protocol");
54
55 socketContext = new SocketContextUpgrade(socketConnection, this);
56 const std::string selectedSubProtocolName = socketContext->loadSubProtocol(subProtocolName);
57
58 if (selectedSubProtocolName.empty()) {
59 delete socketContext;
60 socketContext = nullptr;
61 }
62 } else {
63 checkRefCount();
64 }
65
66 return socketContext;
67 }
