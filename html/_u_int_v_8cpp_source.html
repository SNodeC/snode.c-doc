34 std::size_t UIntV::deserialize(MqttContext* mqttContext) {
35 std::size_t consumed = 0;
36
37 do {
38 char byte;
39 consumed = mqttContext->recv(&byte, 1);
40
41 if (consumed > 0) {
42 value.push_back(byte);
43
44 if (value.size() > sizeof(uint32_t)) {
45 error = true;
46 } else {
47 complete = (byte & 0x80) == 0;
48 }
49 }
50 } while (consumed > 0 && !complete && !error);
51
52 return consumed;
53 }

55 uint32_t UIntV::operator=(const uint32_t& newValue) {
56 uint32_t remainingValue = newValue;
57 value.resize(0);
58
59 do {
60 uint8_t encodedByte = static_cast<uint8_t>(remainingValue % 0x80);
61
62 remainingValue /= 0x80;
63 if (remainingValue > 0) {
64 encodedByte |= 0x80;
65 }
66
67 value.push_back(static_cast<char>(encodedByte));
68 } while (remainingValue > 0);
69
70 return newValue;
71 }

73 UIntV::operator uint32_t() const {
74 uint32_t uint32Value = 0;
75 uint32_t multiplicator = 1;
76
77 for (std::size_t i = 0; i < value.size(); i++, multiplicator *= 0x80) {
78 uint32Value += static_cast<uint8_t>(value[i] & 0x7F) * multiplicator;
79 }
