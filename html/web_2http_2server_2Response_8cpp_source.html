104 Response& Response::set(const std::string& field, const std::string& value, bool overwrite) {
105 if (!value.empty()) {
106 if (overwrite) {
107 headers.insert_or_assign(field, value);
108 } else {
109 headers.insert({field, value});
110 }
111
112 if (web::http::ciEquals(field, "Connection")) {
113 if (web::http::ciContains(headers[field], "keep-alive")) {
114 connectionState = ConnectionState::Keep;
115 } else if (web::http::ciContains(headers[field], "close")) {
116 connectionState = ConnectionState::Close;
117 }
118 } else if (web::http::ciEquals(field, "Content-Length")) {
119 contentLength = std::stoul(value);
120 transferEncoding = TransferEncoding::Identity;
121 headers.erase("Transfer-Encoding");
122 } else if (web::http::ciEquals(field, "Transfer-Encoding")) {
123 if (web::http::ciContains(headers[field], "chunked")) {
124 transferEncoding = TransferEncoding::Chunked;
125 headers.erase("Content-Length");
126 }
127 if (web::http::ciContains(headers[field], "compressed")) {
128 }
129 if (web::http::ciContains(headers[field], "deflate")) {
130 }
131 if (web::http::ciContains(headers[field], "gzip")) {
132 }
133 } else if (web::http::ciEquals(field, "Content-Encoding")) {
134 if (web::http::ciContains(headers[field], "compressed")) {
135 }
136 if (web::http::ciContains(headers[field], "deflate")) {
137 }
138 if (web::http::ciContains(headers[field], "gzip")) {
139 }
140 if (web::http::ciContains(headers[field], "br")) {
141 }
142 }
143 } else {
144 headers.erase(field);
145 }

220 void Response::upgrade(const std::shared_ptr<Request>& request, const std::function<void(bool)>& status) {
221 bool success = false;
222
223 if (socketContext != nullptr) {
224 if (request != nullptr) {
225 if (web::http::ciContains(request->get("connection"), "Upgrade")) {
226 web::http::server::SocketContextUpgradeFactory* socketContextUpgradeFactory =
227 web::http::server::SocketContextUpgradeFactorySelector::instance()->select(*request, *this);
228
229 if (socketContextUpgradeFactory != nullptr) {
230 socketContextUpgrade = socketContextUpgradeFactory->create(socketContext->getSocketConnection());
231 if (socketContextUpgrade != nullptr) {
232 success = true;
233 } else {
234 set("Connection", "close").status(404);
235 }
236 } else {
237 set("Connection", "close").status(404);
238 }
239 } else {
240 set("Connection", "close").status(400);
241 }
242 } else {
243 set("Connection", "close").status(500);
244 }
245 }
246
247 status(success);
248 }

250 void Response::sendFile(const std::string& file, const std::function<void(int)>& callback) {
251 if (socketContext != nullptr) {
252 std::string absolutFileName = file;
253
254 if (std::filesystem::exists(absolutFileName)) {
255 std::error_code ec;
256 absolutFileName = std::filesystem::canonical(absolutFileName);
257
258 if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
259 core::file::FileReader::open(absolutFileName)->pipe(this, [this, &absolutFileName, &callback](int errnum) -> void {
260 callback(errnum);
261
262 if (errnum == 0) {
263 set("Content-Type", web::http::MimeTypes::contentType(absolutFileName), false);
264 set("Last-Modified", httputils::file_mod_http_date(absolutFileName), false);
265 if (httpMajor == 1) {
266 if (httpMinor == 1) {
267 set("Transfer-Encoding", "chunked");
268 } else {
269 set("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
270 }
271 }
272 }
273 });
274 } else {
275 errno = EEXIST;
276 callback(errno);
277 }
278 } else {
279 errno = ENOENT;
280 callback(errno);
281 }
282 }

289 Response& Response::sendHeader() {
290 if (socketContext != nullptr) {
291 socketContext->responseStarted();
292
293 socketContext->sendToPeer("HTTP/" + std::to_string(httpMajor)
294 .append(".")
295 .append(std::to_string(httpMinor))
296 .append(" ")
297 .append(std::to_string(statusCode))
298 .append(" ")
299 .append(StatusCode::reason(statusCode))
300 .append("\r\n"));
301 socketContext->sendToPeer("Date: " + httputils::to_http_date() + "\r\n");
302
303 set("X-Powered-By", "snode.c");
304
305 for (const auto& [field, value] : headers) {
306 socketContext->sendToPeer(std::string(field).append(": ").append(value).append("\r\n"));
307 }
308
309 for (const auto& [cookie, cookieValue] : cookies) {
310 const std::string cookieString = std::accumulate(
311 cookieValue.getOptions().begin(),
312 cookieValue.getOptions().end(),
313 cookie + "=" + cookieValue.getValue(),
314 [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
315 return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
316 });
317 socketContext->sendToPeer("Set-Cookie: " + cookieString + "\r\n");
318 }
319
320 socketContext->sendToPeer("\r\n");
321 }
322
323 return *this;
324 }

326 Response& Response::sendFragment(const char* chunk, std::size_t chunkLen) {
327 if (socketContext != nullptr) {
328 if (transferEncoding == TransferEncoding::Chunked) {
329 socketContext->sendToPeer(to_hex_str(chunkLen).append("\r\n"));
330 }
331
332 socketContext->sendToPeer(chunk, chunkLen);
333 contentSent += chunkLen;
334
335 if (transferEncoding == TransferEncoding::Chunked) {
336 socketContext->sendToPeer("\r\n");
337 contentLength += chunkLen;
338 }
339 }
340
341 return *this;
342 }

348 void Response::sendCompleted() {
349 if (transferEncoding == TransferEncoding::Chunked) {
350 sendFragment(""); // For transfer encoding chunked. Terminate the chunk sequence.
351
352 if (!trailer.empty()) {
353 for (auto& [field, value] : trailer) {
354 socketContext->sendToPeer(std::string(field).append(":").append(value).append("\r\n"));
355 }
356
357 socketContext->sendToPeer("\r\n");
358 }
359 }
360
361 if (socketContext != nullptr) {
362 socketContext->responseCompleted(contentSent == contentLength || (httpMajor == 1 && httpMinor == 0));
363
364 if (socketContextUpgrade != nullptr) {
365 socketContext->switchSocketContext(socketContextUpgrade);
366 socketContextUpgrade = nullptr;
367 }
368 }
