52 void Response::sendResponse(const char* junk, std::size_t junkLen) {
53 if (!headersSent && !sendHeaderInProgress) {
54 sendHeaderInProgress = true;
55 sendHeader();
56 sendHeaderInProgress = false;
57 headersSent = true;
58 }
59
60 // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDelete)
61 requestContext->sendToPeer(junk, junkLen);
62
63 if (headersSent) {
64 contentSent += junkLen;
65 }
66 }

126 Response& Response::set(const std::string& field, const std::string& value, bool overwrite) {
127 if (overwrite || headers.find(field) == headers.end()) {
128 headers.insert_or_assign(field, value);
129
130 if (field == "Content-Length") {
131 contentLength = std::stoul(value);
132 } else if (field == "Connection" && httputils::ci_contains(value, "close")) {
133 connectionState = ConnectionState::Close;
134 } else if (field == "Connection" && httputils::ci_contains(value, "keep-alive")) {
135 connectionState = ConnectionState::Keep;
136 }
137 }
138
139 return *this;
140 }

170 [[nodiscard]] bool Response::upgrade(Request& req) {
171 bool success = false;
172
173 if (httputils::ci_contains(req.get("connection"), "Upgrade")) {
174 web::http::server::SocketContextUpgradeFactory* socketContextUpgradeFactory =
175 web::http::server::SocketContextUpgradeFactorySelector::instance()->select(req, *this);
176
177 if (socketContextUpgradeFactory != nullptr) {
178 success = requestContext->switchSocketContext(socketContextUpgradeFactory);
179 } else {
180 set("Connection", "close").status(404);
181 }
182 } else {
183 set("Connection", "close").status(400);
184 }
185
186 return success;
187 }

189 void Response::sendFile(const std::string& file, const std::function<void(int errnum)>& onError) {
190 std::string absolutFileName = file;
191
192 if (std::filesystem::exists(absolutFileName)) {
193 std::error_code ec;
194 absolutFileName = std::filesystem::canonical(absolutFileName);
195
196 if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
197 source = core::file::FileReader::open(absolutFileName, *this, [this, &absolutFileName, onError](int err) -> void {
198 if (err == 0) {
199 headers.insert({{"Content-Type", web::http::MimeTypes::contentType(absolutFileName)},
200 {"Last-Modified", httputils::file_mod_http_date(absolutFileName)}});
201 headers.insert_or_assign("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
202 } else {
203 onError(err);
204 }
205 });
206
207 stream(source);
208 } else {
209 errno = EEXIST;
210 onError(errno);
211 }
212 } else {
213 errno = ENOENT;
214 onError(errno);
215 }
216 }

218 void Response::sendToPeerCompleted() {
219 if (contentSent == contentLength) {

221 } else if (contentSent > contentLength) {
222 requestContext->close();
223 }

225 delete requestContext;

228 void Response::sendHeader() {
229 sendResponse("HTTP/1.1 " + std::to_string(responseStatus) + " " + StatusCode::reason(responseStatus) + "\r\n");
230 sendResponse("Date: " + httputils::to_http_date() + "\r\n");
231
232 set({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
233
234 for (const auto& [field, value] : headers) {
235 sendResponse(std::string(field).append(": ").append(value).append("\r\n"));
236 }
237
238 for (const auto& [cookie, cookieValue] : cookies) { // cppcheck-suppress shadowFunction
239 const std::string cookieString =
240 std::accumulate(cookieValue.getOptions().begin(),
241 cookieValue.getOptions().end(),
242 cookie + "=" + cookieValue.getValue(),
243 [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
244 return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
245 });
246 sendResponse("Set-Cookie: " + cookieString + "\r\n");
247 }
248
249 sendResponse("\r\n");
250
251 if (headers.find("Content-Length") != headers.end()) {
252 contentLength = std::stoul(headers.find("Content-Length")->second);
253 } else {
254 contentLength = 0;
255 }
256 }
