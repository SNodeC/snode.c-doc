101 Response& Response::set(const std::string& field, const std::string& value, bool overwrite) {
102 if (!value.empty()) {
103 if (overwrite) {
104 headers.insert_or_assign(field, value);
105 } else {
106 headers.insert({field, value});
107 }
108
109 if (web::http::ciEquals(field, "Connection")) {
110 if (web::http::ciContains(headers[field], "keep-alive")) {
111 connectionState = ConnectionState::Keep;
112 } else if (web::http::ciContains(headers[field], "close")) {
113 connectionState = ConnectionState::Close;
114 }
115 } else if (web::http::ciEquals(field, "Content-Length")) {
116 contentLength = std::stoul(value);
117 transferEncoding = TransferEncoding::Identity;
118 headers.erase("Transfer-Encoding");
119 } else if (web::http::ciEquals(field, "Transfer-Encoding")) {
120 if (web::http::ciContains(headers[field], "chunked")) {
121 transferEncoding = TransferEncoding::Chunked;
122 headers.erase("Content-Length");
123 }
124 if (web::http::ciContains(headers[field], "compressed")) {
125 }
126 if (web::http::ciContains(headers[field], "deflate")) {
127 }
128 if (web::http::ciContains(headers[field], "gzip")) {
129 }
130 } else if (web::http::ciEquals(field, "Content-Encoding")) {
131 if (web::http::ciContains(headers[field], "compressed")) {
132 }
133 if (web::http::ciContains(headers[field], "deflate")) {
134 }
135 if (web::http::ciContains(headers[field], "gzip")) {
136 }
137 if (web::http::ciContains(headers[field], "br")) {

211 void Response::upgrade(const std::shared_ptr<Request>& request,
212 const std::function<void(bool)>& status,
213 const std::function<void()>& response) {
214 bool success = false;
215
216 if (socketContext != nullptr) {
217 if (request != nullptr) {
218 if (web::http::ciContains(request->get("connection"), "Upgrade")) {
219 web::http::server::SocketContextUpgradeFactory* socketContextUpgradeFactory =
220 web::http::server::SocketContextUpgradeFactorySelector::instance()->select(*request, *this);
221
222 if (socketContextUpgradeFactory != nullptr) {
223 socketContextUpgrade = socketContextUpgradeFactory->create(socketContext->getSocketConnection());
224 if (socketContextUpgrade != nullptr) {
225 success = true;
226 } else {
227 set("Connection", "close").status(404);
228 }
229 } else {
230 set("Connection", "close").status(404);
231 }
232 } else {
233 set("Connection", "close").status(400);
234 }
235 } else {
236 set("Connection", "close").status(500);
237 }
238 }
239
240 if (response) {
241 response();
242 } else {
243 end();
244 }
245
246 status(success);
247 }

249 void Response::sendFile(const std::string& file, const std::function<void(int)>& callback) {
250 if (socketContext != nullptr) {
251 std::string absolutFileName = file;
252
253 if (std::filesystem::exists(absolutFileName)) {
254 std::error_code ec;
255 absolutFileName = std::filesystem::canonical(absolutFileName);
256
257 if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
258 core::file::FileReader::open(absolutFileName)->pipe(this, [this, &absolutFileName, &callback](int errnum) -> void {
259 callback(errnum);
260
261 if (errnum == 0) {
262 set("Content-Type", web::http::MimeTypes::contentType(absolutFileName), false);
263 set("Last-Modified", httputils::file_mod_http_date(absolutFileName), false);
264 if (httpMajor == 1) {
265 if (httpMinor == 1) {
266 set("Transfer-Encoding", "chunked");
267 } else {
268 set("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
269 }
270 }
271 }
272 });
273 } else {
274 errno = EEXIST;
275 callback(errno);
276 }
277 } else {
278 errno = ENOENT;
279 callback(errno);
280 }
281 }

288 Response& Response::sendHeader() {
289 if (socketContext != nullptr) {
290 socketContext->responseStarted();
291
292 socketContext->sendToPeer("HTTP/" + std::to_string(httpMajor)
293 .append(".")
294 .append(std::to_string(httpMinor))
295 .append(" ")
296 .append(std::to_string(statusCode))
297 .append(" ")
298 .append(StatusCode::reason(statusCode))
299 .append("\r\n"));
300 socketContext->sendToPeer("Date: " + httputils::to_http_date() + "\r\n");
301
302 set("X-Powered-By", "snode.c");
303
304 for (const auto& [field, value] : headers) {
305 socketContext->sendToPeer(std::string(field).append(": ").append(value).append("\r\n"));
306 }
307
308 for (const auto& [cookie, cookieValue] : cookies) { // cppcheck-suppress shadowFunction
309 const std::string cookieString = std::accumulate(
310 cookieValue.getOptions().begin(),
311 cookieValue.getOptions().end(),
312 cookie + "=" + cookieValue.getValue(),
313 [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
314 return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
315 });
316 socketContext->sendToPeer("Set-Cookie: " + cookieString + "\r\n");
317 }
318
319 socketContext->sendToPeer("\r\n");
320 }
321
322 return *this;
323 }

325 Response& Response::sendFragment(const char* chunk, std::size_t chunkLen) {
326 if (socketContext != nullptr) {
327 if (transferEncoding == TransferEncoding::Chunked) {
328 socketContext->sendToPeer(to_hex_str(chunkLen).append("\r\n"));
329 }
330
331 socketContext->sendToPeer(chunk, chunkLen);
332 contentSent += chunkLen;
333
334 if (transferEncoding == TransferEncoding::Chunked) {
335 socketContext->sendToPeer("\r\n");
336 contentLength += chunkLen;
337 }
338 }
339
340 return *this;
341 }

347 void Response::sendCompleted() {
348 if (transferEncoding == TransferEncoding::Chunked) {
349 sendFragment(""); // For transfere encoding chunked. Terminate the chunk sequence.
350
351 if (!trailer.empty()) {
352 for (auto& [field, value] : trailer) {
353 socketContext->sendToPeer(std::string(field).append(":").append(value).append("\r\n"));
354 }
355
356 socketContext->sendToPeer("\r\n");
357 }
358 }
359
360 if (socketContext != nullptr) {
361 if (socketContextUpgrade != nullptr) {
362 socketContext->switchSocketContext(socketContextUpgrade);
363 }
364
365 socketContext->responseCompleted(contentSent == contentLength || (httpMajor == 1 && httpMinor == 0));
366 }
