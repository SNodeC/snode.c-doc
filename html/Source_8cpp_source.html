60 void Source::pipe(Sink* sink, const std::function<void(int)>& callback) {
61 if (sink != nullptr) {
62 if (isOpen()) {
63 callback(0);
64
65 this->sink = sink;
66 sink->pipe(this);
67 } else {
68 callback(errno);
69 stop();
70 }
71 } else {
