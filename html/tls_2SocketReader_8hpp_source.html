32 template <typename PhysicalSocket>
33 ssize_t SocketReader<PhysicalSocket>::read(char* junk, std::size_t junkLen) {
34 int sslShutdownState = SSL_get_shutdown(ssl);
35
36 int ret = SSL_read(ssl, junk, static_cast<int>(junkLen));
37
38 if (ret <= 0) {
39 int ssl_err = SSL_get_error(ssl, ret);
40
41 switch (ssl_err) {
42 case SSL_ERROR_NONE:
43 break;
44 case SSL_ERROR_WANT_READ:
45 ret = -1;
46 break;
47 case SSL_ERROR_WANT_WRITE: {
48 utils::PreserveErrno preserveErrno;
49
50 LOG(TRACE) << "SSL/TLS start renegotiation on read";
51 doSSLHandshake(
52 []() -> void {
53 LOG(TRACE) << "SSL/TLS renegotiation on read success";
54 },
55 []() -> void {
56 LOG(WARNING) << "SSL/TLS renegotiation on read timed out";
57 },
58 [](int ssl_err) -> void {
59 ssl_log("SSL/TLS renegotiation", ssl_err);
60 });
61 }
62 ret = -1;
63 break;
64 case SSL_ERROR_ZERO_RETURN: // received close_notify
65 SSL_set_shutdown(ssl, SSL_get_shutdown(ssl) | sslShutdownState);
66 doSSLShutdown();
67 errno = 0;
68 ret = 0;
69 break;
70 case SSL_ERROR_SYSCALL: {
71 utils::PreserveErrno preserveErrno;
72
73 SSL_set_shutdown(ssl, SSL_get_shutdown(ssl) | SSL_RECEIVED_SHUTDOWN);
74 VLOG(0) << "SSL/TLS: TCP-FIN without close_notify. Emulating SSL_RECEIVED_SHUTDOWN";
75 doSSLShutdown();
76 }
77 ret = -1;
78 break;
79 default:
80 ssl_log("SSL/TLS error read failed", ssl_err);
81 errno = EIO;
82 ret = -1;
83 break;
84 }
85 }
86
87 return ret;
88 }
