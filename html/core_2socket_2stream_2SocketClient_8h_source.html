152 void realConnect(const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
153 unsigned int tries,
154 double retryTimeoutScale) {
155 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
156 new SocketConnector(
157 socketContextFactory,
158 onConnect,
159 onConnected,
160 [client = *this, onStatus](SocketConnection* socketConnection) -> void {
161 client.onDisconnect(socketConnection);
162
163 if (client.getConfig().getReconnect()) {
164 double relativeReconnectTimeout = client.getConfig().getReconnectTime();
165
166 LOG(INFO) << "Reconnecting in " << relativeReconnectTimeout << " seconds";
167
168 core::timer::Timer::singleshotTimer(
169 [client, onStatus]() mutable -> void {
170 client.realConnect(onStatus, 0, 1);
171 },
172 relativeReconnectTimeout);
173 }
174 },
175 [client = *this, onStatus, tries, retryTimeoutScale](const SocketAddress& socketAddress,
176 core::socket::State state) -> void {
177 onStatus(socketAddress, state);
178
179 switch (state) {
180 case core::socket::State::ERROR:
181 if (client.getConfig().getRetry() &&
182 (client.getConfig().getRetryTries() == 0 || tries < client.getConfig().getRetryTries())) {
183 double relativeRetryTimeout =
184 client.getConfig().getRetryLimit() > 0
185 ? std::min<double>(client.getConfig().getRetryTimeout() * retryTimeoutScale,
186 client.getConfig().getRetryLimit())
187 : client.getConfig().getRetryTimeout() * retryTimeoutScale;
188 relativeRetryTimeout -= utils::Random::getInRange(-client.getConfig().getRetryJitter(),
189 client.getConfig().getRetryJitter()) *
190 relativeRetryTimeout / 100.;
191
192 LOG(INFO) << "Retrying in " << relativeRetryTimeout << " seconds";
193
194 core::timer::Timer::singleshotTimer(
195 [client, onStatus, tries, retryTimeoutScale]() mutable -> void {
196 client.realConnect(onStatus, tries + 1, retryTimeoutScale * client.getConfig().getRetryBase());
197 },
198 relativeRetryTimeout);
199 }
200 break;
201 case core::socket::State::OK:
202 case core::socket::State::DISABLED:
203 case core::socket::State::FATAL:
204 break;
205 }
206 },
207 Super::config);
208 }
