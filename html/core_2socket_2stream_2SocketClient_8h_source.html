164 private:
165 void realConnect(const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
166 unsigned int tries,
167 double retryTimeoutScale) const {
168 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
169 new SocketConnector(
170 socketContextFactory,
171 onConnect,
172 onConnected,
173 [config = this->config,
174 onConnect = this->onConnect,
175 onConnected = this->onConnected,
176 onDisconnect = this->onDisconnect,
177 socketContextFactory = this->socketContextFactory,
178 onStatus](SocketConnection* socketConnection) {
179 onDisconnect(socketConnection);
180
181 if (config->getReconnect() && core::eventLoopState() == core::State::RUNNING) {
182 double relativeReconnectTimeout = config->getReconnectTime();
183
184 LOG(INFO) << config->getInstanceName() << ": Reconnect in " << relativeReconnectTimeout << " seconds";
185
186 core::timer::Timer::singleshotTimer(
187 [config, onConnect, onConnected, onDisconnect, onStatus, socketContextFactory]() mutable {
188 if (config->getReconnect()) {
189 SocketClient(config, socketContextFactory, onConnect, onConnected, onDisconnect)
190 .realConnect(onStatus, 0, config->getRetryBase());
191 } else {
192 LOG(INFO) << config->getInstanceName() << ": Reconnect disabled during wait";
193 }
194 },
195 relativeReconnectTimeout);
196 }
197 },
198 [config = this->config,
199 onConnect = this->onConnect,
200 onConnected = this->onConnected,
201 onDisconnect = this->onDisconnect,
202 socketContextFactory = this->socketContextFactory,
203 onStatus,
204 tries,
205 retryTimeoutScale](const SocketAddress& socketAddress, core::socket::State state) mutable {
206 const bool retryFlag = (state & core::socket::State::NO_RETRY) == 0;
207 state &= ~core::socket::State::NO_RETRY;
208 onStatus(socketAddress, state);
209
210 if (retryFlag && config->getRetry() // Shall we potentially retry? In case are the ...
211 && (config->getRetryTries() == 0 || tries < config->getRetryTries()) // ... limits not reached and has an ...
212 && (state == core::socket::State::ERROR ||
213 (state == core::socket::State::FATAL && config->getRetryOnFatal()))) { // error occurred?
214 double relativeRetryTimeout =
215 config->getRetryLimit() > 0
216 ? std::min<double>(config->getRetryTimeout() * retryTimeoutScale, config->getRetryLimit())
217 : config->getRetryTimeout() * retryTimeoutScale;
218 relativeRetryTimeout -= utils::Random::getInRange(-config->getRetryJitter(), config->getRetryJitter()) *
219 relativeRetryTimeout / 100.;
220
221 LOG(INFO) << config->getInstanceName() << ": Retry connect in " << relativeRetryTimeout << " seconds";
222
223 core::timer::Timer::singleshotTimer(
224 [config,
225 onConnect,
226 onConnected,
227 onDisconnect,
228 onStatus,
229 tries,
230 retryTimeoutScale,
231 socketContextFactory]() mutable {
232 if (config->getRetry()) {
233 SocketClient(config, socketContextFactory, onConnect, onConnected, onDisconnect)
234 .realConnect(onStatus, tries + 1, retryTimeoutScale * config->getRetryBase());
235 } else {
236 LOG(INFO) << config->getInstanceName() << ": Retry connect disabled during wait";
237 }
238 },
239 relativeRetryTimeout);
240 }
241 },
242 Super::config);
243 }
244 }
