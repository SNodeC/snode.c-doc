194 const SocketClient& realConnect(const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
195 unsigned int tries,
196 double retryTimeoutScale) const {
197 LOG(DEBUG) << this->config->getInstanceName() << ": Initiating connect";
198
199 core::EventReceiver::atNextTick(
200 [config = this->config, sharedContext = this->sharedContext, onStatus, tries, retryTimeoutScale] {
201 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
202 auto autoConnectControl = sharedContext->autoConnectControl;
203 if (!autoConnectControl) {
204 autoConnectControl = std::make_shared<AutoConnectControl>();
205 sharedContext->autoConnectControl = autoConnectControl;
206 if (sharedContext->onAutoConnectControl) {
207 sharedContext->onAutoConnectControl(autoConnectControl);
208 }
209 }
210
211 new SocketConnector(
212 sharedContext->socketContextFactory,
213 sharedContext->onConnect,
214 sharedContext->onConnected,
215 [config, sharedContext, autoConnectControl, onDisconnect = sharedContext->onDisconnect, onStatus](
216 SocketConnection* socketConnection) {
217 onDisconnect(socketConnection);
218
219 if (config->getReconnect() && autoConnectControl->isReconnectEnabled() &&
220 core::eventLoopState() == core::State::RUNNING) {
221 double relativeReconnectTimeout = config->getReconnectTime();
222
223 LOG(INFO) << config->getInstanceName() << ": Reconnect in " << relativeReconnectTimeout << " seconds";
224
225 autoConnectControl->armReconnectTimer(
226 relativeReconnectTimeout, [config, sharedContext, autoConnectControl, /*generation,*/ onStatus]() {
227 if (!autoConnectControl->isReconnectEnabled()) {
228 return;
229 }
230 if (config->getReconnect()) {
231 SocketClient(config, sharedContext).realConnect(onStatus, 0, config->getRetryBase());
232 } else {
233 LOG(INFO) << config->getInstanceName() << ": Reconnect disabled during wait";
234 }
235 });
236 }
237 },
238 sharedContext->onInitState,
239 [config, sharedContext, autoConnectControl, onStatus, tries, retryTimeoutScale](
240 const SocketAddress& socketAddress, core::socket::State state) {
241 const bool retryFlag = (state & core::socket::State::NO_RETRY) == 0;
242 state &= ~core::socket::State::NO_RETRY;
243 onStatus(socketAddress, state);
244
245 if (retryFlag && config->getRetry() // Shall we potentially retry? In case are the ...
246 && autoConnectControl->isRetryEnabled() &&
247 (config->getRetryTries() == 0 ||
248 tries < config->getRetryTries()) // ... limits not reached and has an ...
249 && (state == core::socket::State::ERROR ||
250 (state == core::socket::State::FATAL && config->getRetryOnFatal()))) { // error occurred?
251 double relativeRetryTimeout =
252 config->getRetryLimit() > 0
253 ? std::min<double>(config->getRetryTimeout() * retryTimeoutScale, config->getRetryLimit())
254 : config->getRetryTimeout() * retryTimeoutScale;
255 relativeRetryTimeout -= utils::Random::getInRange(-config->getRetryJitter(), config->getRetryJitter()) *
256 relativeRetryTimeout / 100.;
257
258 LOG(INFO) << config->getInstanceName() << ": Retry connect in " << relativeRetryTimeout << " seconds";
259
260 autoConnectControl->armRetryTimer(
261 relativeRetryTimeout,
262 [config, sharedContext, autoConnectControl, /*generation,*/ onStatus, tries, retryTimeoutScale]() {
263 if (!autoConnectControl->isRetryEnabled()) {
264 return;
265 }
266 if (config->getRetry()) {
267 SocketClient(config, sharedContext)
268 .realConnect(onStatus, tries + 1, retryTimeoutScale * config->getRetryBase());
269 } else {
270 LOG(INFO) << config->getInstanceName() << ": Retry connect disabled during wait";
271 }
272 });
273 }
274 },
275 config);
276 }
277 });
278
279 return *this;
280 }
