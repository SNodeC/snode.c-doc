155 private:
156 void realConnect(const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
157 unsigned int tries,
158 double retryTimeoutScale) const {
159 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
160 new SocketConnector(
161 socketContextFactory,
162 onConnect,
163 onConnected,
164 [client = *this, onStatus](SocketConnection* socketConnection) -> void {
165 client.onDisconnect(socketConnection);
166
167 if (client.getConfig().getReconnect() && core::eventLoopState() == core::State::RUNNING) {
168 double relativeReconnectTimeout = client.getConfig().getReconnectTime();
169
170 LOG(INFO) << "Client OnDisconnect: " << client.getConfig().getInstanceName();
171 LOG(INFO) << " "
172 << " reconnecting in " << relativeReconnectTimeout << " seconds";
173
174 core::timer::Timer::singleshotTimer(
175 [client, onStatus]() -> void {
176 client.getConfig().Local::renew();
177 client.getConfig().Remote::renew();
178
179 client.realConnect(onStatus, 0, 1);
180 },
181 relativeReconnectTimeout);
182 }
183 },
184 [client = *this, onStatus, tries, retryTimeoutScale](const SocketAddress& socketAddress,
185 core::socket::State state) -> void {
186 bool retry = (state & core::socket::State::NO_RETRY) == 0 &&
187 (client.getConfig().getRetryTries() == 0 || tries < client.getConfig().getRetryTries());
188 state &= ~core::socket::State::NO_RETRY;
189
190 onStatus(socketAddress, state);
191
192 switch (state) {
193 case core::socket::State::OK:
194 [[fallthrough]];
195 case core::socket::State::DISABLED:
196 retry = false;
197 break;
198 case core::socket::State::ERROR:
199 retry = retry && client.getConfig().getRetry();
200 break;
201 case core::socket::State::FATAL:
202 retry = retry && client.getConfig().getRetry() && client.getConfig().getRetryOnFatal();
203 break;
204 }
205
206 if (retry) {
207 double relativeRetryTimeout = client.getConfig().getRetryLimit() > 0
208 ? std::min<double>(client.getConfig().getRetryTimeout() * retryTimeoutScale,
209 client.getConfig().getRetryLimit())
210 : client.getConfig().getRetryTimeout() * retryTimeoutScale;
211 relativeRetryTimeout -=
212 utils::Random::getInRange(-client.getConfig().getRetryJitter(), client.getConfig().getRetryJitter()) *
213 relativeRetryTimeout / 100.;
214
215 LOG(INFO) << "Client OnStatus: " << client.getConfig().getInstanceName();
216 LOG(INFO) << " "
217 << " retrying in " << relativeRetryTimeout << " seconds";
218
219 core::timer::Timer::singleshotTimer(
220 [client, onStatus, tries, retryTimeoutScale]() mutable -> void {
221 client.getConfig().Local::renew();
222 client.getConfig().Remote::renew();
223
224 client.realConnect(onStatus, tries + 1, retryTimeoutScale * client.getConfig().getRetryBase());
225 },
226 relativeRetryTimeout);
227 }
228 },
229 Super::config);
230 }
