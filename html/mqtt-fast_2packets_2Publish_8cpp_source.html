30 : iot::mqtt_fast::ControlPacket(MQTT_PUBLISH,
31 static_cast<uint8_t>((dup ? 0x04 : 0x00) | ((qoS << 1) & 0x06) | (retain ? 0x01 : 0x00)))
32 , packetIdentifier(packetIdentifier)
33 , topic(topic)
34 , message(message)
35 , dup(dup)
36 , qoS(qoS)
37 , retain(retain) {
38 // V-Header

40

49 Publish::Publish(iot::mqtt_fast::ControlPacketFactory& controlPacketFactory)
50 : iot::mqtt_fast::ControlPacket(controlPacketFactory) {
51 dup = (controlPacketFactory.getPacketFlags() & 0x08) != 0;
