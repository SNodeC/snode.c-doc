64 Request& Request::set(const std::string& field, const std::string& value, bool overwrite) {
65 if (overwrite) {
66 headers.insert_or_assign(field, value);
67 } else {
68 headers.insert({field, value});
69 }
70
71 if (field == "Content-Length") {
72 contentLength = std::stoul(value);
73 } else if (field == "Connection" && httputils::ci_contains(value, "close")) {
74 connectionState = ConnectionState::Close;
75 } else if (field == "Connection" && httputils::ci_contains(value, "keep-alive")) {
76 connectionState = ConnectionState::Keep;
77 }
78
79 return *this;

118 void Request::enqueue(const char* junk, std::size_t junkLen) {
119 if (!headersSent && !sendHeaderInProgress) {
120 sendHeaderInProgress = true;
121 sendHeader();
122 sendHeaderInProgress = false;
123 headersSent = true;
124 }
125
126 socketContext->sendToPeer(junk, junkLen);
127
128 if (headersSent) {
129 contentSent += junkLen;
130 if (contentSent == contentLength) {
131 socketContext->sendToPeerCompleted();
132 } else if (contentSent > contentLength) {
133 socketContext->close();
134 }
135 }
136 }

142 void Request::sendHeader() {
143 const std::string httpVersion = "HTTP/" + std::to_string(httpMajor) + "." + std::to_string(httpMinor);
144
145 std::string queryString;
146 if (!queries.empty()) {
147 queryString += "?";
148 for (auto& [key, value] : queries) {
149 queryString += httputils::url_encode(key) + "=" + httputils::url_encode(value) + "&";
150 }
151 queryString.pop_back();
152 }
153
154 enqueue(method + " " + url + queryString + " " + httpVersion + "\r\n");
155
156 if (!host.empty()) {
157 enqueue("Host: " + host + "\r\n");
158 }
159 enqueue("Date: " + httputils::to_http_date() + "\r\n");
160
161 headers.insert({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
162
163 for (const auto& [field, value] : headers) {
164 enqueue(std::string(field).append(":").append(value).append("\r\n"));
165 }
166
167 if (contentLength != 0) {
168 enqueue(std::string("Content-Length: ").append(std::to_string(contentLength)).append("\r\n"));
169 }
170
171 for (const auto& [name, value] : cookies) {
172 enqueue(std::string("Cookie:").append(name).append("=").append(value).append("\r\n"));
173 }
174
175 enqueue("\r\n");
176
177 if (headers.find("Content-Length") != headers.end()) {
178 contentLength = std::stoul(headers.find("Content-Length")->second);
179 } else {
180 contentLength = 0;
181 }
182 }

200 void Request::upgrade(const std::string& url, const std::string& protocols) {
201 this->url = url;
202
203 set("Connection", "Upgrade", true);
204 set("Upgrade", protocols);
205
206 // load upgrade context
207 // let upgrade context fill the request-header fields
208 // send the request to the server
209 // the response-code needs to check the response from the server and upgrade the context in case the server says OK
210
211 web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
212 web::http::client::SocketContextUpgradeFactorySelector::instance()->select(protocols, *this);
213
214 if (socketContextUpgradeFactory != nullptr) {
215 start();
216 socketContextUpgradeFactory->checkRefCount();
217 } else {
218 socketContext->close();
219 }
220 }
