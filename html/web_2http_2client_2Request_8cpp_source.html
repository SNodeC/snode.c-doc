121 Request& Request::set(const std::string& field, const std::string& value, bool overwrite) {
122 if (!value.empty()) {
123 if (overwrite) {
124 headers.insert_or_assign(field, value);
125 } else {
126 headers.insert({field, value});
127 }
128
129 if (field == "Content-Length") {
130 contentLength = std::stoul(value);
131 } else if (field == "Connection" && httputils::ci_contains(headers[field], "close")) {
132 connectionState = ConnectionState::Close;
133 } else if (field == "Connection" && httputils::ci_contains(headers[field], "keep-alive")) {
134 connectionState = ConnectionState::Keep;
135 }
136 } else {
137 headers.erase(field);
138 }
139
140 return *this;
141 }

221 void Request::upgrade(const std::shared_ptr<Response>& response, const std::function<void(bool success)>& status) {
222 core::socket::stream::SocketContext* newSocketContext = nullptr;
223
224 if (socketContext != nullptr) {
225 if (response != nullptr) {
226 if (httputils::ci_contains(response->get("connection"), "Upgrade")) {
227 web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
228 web::http::client::SocketContextUpgradeFactorySelector::instance()->select(*this, *response);
229
230 if (socketContextUpgradeFactory != nullptr) {
231 newSocketContext = socketContextUpgradeFactory->create(socketContext->getSocketConnection());
232
233 if (newSocketContext != nullptr) {
234 socketContext->switchSocketContext(newSocketContext);
235 } else {
236 LOG(DEBUG) << "HTTP: SocketContextUpgrade not created";
237 socketContext->close();
238 }
239 } else {
240 LOG(DEBUG) << "HTTP: SocketContextUpgradeFactory not existing";
241 socketContext->close();
242 }
243 } else {
244 LOG(DEBUG) << "HTTP: Response did not contain upgrade";
245 socketContext->close();
246 }
247 } else {
248 LOG(DEBUG) << "HTTP: Upgrade internal error: Response has gone away";
249 socketContext->close();
250 }
251 } else {
252 LOG(DEBUG) << "HTTP: Upgrade error: SocketContext has gone away";
253 }
254
255 status(newSocketContext != nullptr);
256 }

317 void Request::dispatchSendHeader() {
318 if (socketContext != nullptr) {
319 const std::string httpVersion = "HTTP/" + std::to_string(httpMajor) + "." + std::to_string(httpMinor);
320
321 std::string queryString;
322 if (!queries.empty()) {
323 queryString += "?";
324 for (auto& [key, value] : queries) {
325 queryString += httputils::url_encode(key) + "=" + httputils::url_encode(value) + "&";
326 }
327 queryString.pop_back();
328 }
329
330 socketContext->sendToPeer(method + " " + url + queryString + " " + httpVersion + "\r\n");
331 socketContext->sendToPeer("Date: " + httputils::to_http_date() + "\r\n");
332
333 for (const auto& [field, value] : headers) {
334 socketContext->sendToPeer(std::string(field).append(":").append(value).append("\r\n"));
335 }
336
337 for (const auto& [name, value] : cookies) {
338 socketContext->sendToPeer(std::string("Cookie:").append(name).append("=").append(value).append("\r\n"));
339 }
340
341 socketContext->sendToPeer("\r\n");
342 } else {
343 LOG(DEBUG) << "HTTP: dispatchSendHeader error: SocketContext has gone away";
344 }
345 }

356 void Request::dispatchSendFile(const std::string& file, const std::function<void(int errnum)>& onStatus) {
357 if (socketContext != nullptr) {
358 std::string absolutFileName = file;
359
360 if (std::filesystem::exists(absolutFileName)) {
361 std::error_code ec;
362 absolutFileName = std::filesystem::canonical(absolutFileName);
363
364 if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
365 core::file::FileReader::open(absolutFileName)
366 ->pipe(this, [this, &absolutFileName, &onStatus](core::pipe::Source* source, int errnum) -> void {
367 errno = errnum;
368 onStatus(errnum);
369
370 if (errnum == 0) {
371 set("Content-Type", web::http::MimeTypes::contentType(absolutFileName), false);
372 set("Last-Modified", httputils::file_mod_http_date(absolutFileName), false);
373 set("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
374 } else {
375 source->stop();
376
377 socketContext->requestSendError();
378 }
379 });
380 } else {
381 errno = EEXIST;
382 onStatus(errno);
383
384 socketContext->requestSendError();
385 }
386 } else {
387 errno = ENOENT;
388 onStatus(errno);
389
390 socketContext->requestSendError();
391 }
392 } else {
393 LOG(DEBUG) << "HTTP: Upgrade error: SocketContext has gone away";
394 }
395 }

397 void Request::dispatchUpgrade(const std::string& url, const std::string& protocols) {
398 if (socketContext != nullptr) {
399 this->url = url;
400
401 set("Connection", "Upgrade", true);
402 set("Upgrade", protocols, true);
403
404 web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
405 web::http::client::SocketContextUpgradeFactorySelector::instance()->select(protocols, *this);
406
407 if (socketContextUpgradeFactory != nullptr) {
408 socketContextUpgradeFactory->checkRefCount();
409
410 dispatchSendHeader();
411 } else {
412 socketContext->close();
413 }
414
415 dispatchCompleted();
416 }
417 }
