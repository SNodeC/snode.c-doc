136 Request& Request::set(const std::string& field, const std::string& value, bool overwrite) {
137 if (!value.empty()) {
138 if (overwrite) {
139 headers.insert_or_assign(field, value);
140 } else {
141 headers.insert({field, value});
142 }
143
144 if (web::http::ciEquals(field, "Connection")) {
145 if (web::http::ciContains(headers[field], "close")) {
146 connectionState = ConnectionState::Close;
147 } else if (web::http::ciContains(headers[field], "keep-alive")) {
148 connectionState = ConnectionState::Keep;
149 }
150 } else if (web::http::ciEquals(field, "Content-Length")) {
151 contentLength = std::stoul(value);
152 transferEncoding = TransferEncoding::Identity;
153 headers.erase("Transfer-Encoding");
154 } else if (web::http::ciEquals(field, "Transfer-Encoding")) {
155 if (web::http::ciContains(headers[field], "chunked")) {
156 transferEncoding = TransferEncoding::Chunked;
157 headers.erase("Content-Length");
158 }
159 if (web::http::ciContains(headers[field], "compressed")) {
160 }
161 if (web::http::ciContains(headers[field], "deflate")) {
162 }
163 if (web::http::ciContains(headers[field], "gzip")) {
164 }
165 } else if (web::http::ciEquals(field, "Content-Encoding")) {
166 if (web::http::ciContains(headers[field], "compressed")) {
167 }
168 if (web::http::ciContains(headers[field], "deflate")) {
169 }
170 if (web::http::ciContains(headers[field], "gzip")) {
171 }
172 if (web::http::ciContains(headers[field], "br")) {
173 }
174 }
175 } else {
176 headers.erase(field);
177 }
178
179 return *this;
180 }

297 bool Request::upgrade(const std::shared_ptr<Response>& response, const std::function<void(bool)>& status) {
298 bool upgraded = false;
299
300 core::socket::stream::SocketContext* newSocketContext = nullptr;
301
302 if (!masterRequest.expired()) {
303 if (response != nullptr) {
304 if (web::http::ciContains(response->get("connection"), "Upgrade")) {
305 web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
306 web::http::client::SocketContextUpgradeFactorySelector::instance()->select(*this, *response);
307
308 if (socketContextUpgradeFactory != nullptr) {
309 newSocketContext = socketContextUpgradeFactory->create(socketContext->getSocketConnection());
310 if (newSocketContext != nullptr) {
311 LOG(DEBUG) << "HTTP: SocketContextUpgrade created";
312 socketContext->switchSocketContext(newSocketContext);
313 newSocketContext = nullptr;
314 upgraded = true;
315 } else {
316 LOG(DEBUG) << "HTTP: SocketContextUpgrade not created";
317 socketContext->close();
318 }
319 } else {
320 LOG(DEBUG) << "HTTP: SocketContextUpgradeFactory not existing";
321 socketContext->close();
322 }
323 } else {
324 LOG(DEBUG) << "HTTP: Response did not contain upgrade";
325 socketContext->close();
326 }
327 } else {
328 LOG(DEBUG) << "HTTP: Upgrade internal error: Response has gone away";
329 socketContext->close();
330 }
331 } else {
332 LOG(DEBUG) << "HTTP: Upgrade error: SocketContext has gone away";
333 }
334
335 status(upgraded);
336
337 return upgraded;
338 }

402 bool Request::initiate() {
403 bool error = false;
404 bool atomar = true;
405
406 for (RequestCommand* requestCommand : requestCommands) {
407 if (!error) {
408 const bool atomarCommand = requestCommand->execute(this);
409 if (atomar) {
410 atomar = atomarCommand;
411 }
412
413 error = requestCommand->getError();
414 }
415
416 delete requestCommand;
417 }
418 requestCommands.clear();
419
420 if (atomar && (!error || contentLengthSent != 0)) {
421 requestDelivered();
422 }
423
424 return !error || contentLengthSent != 0;
425 }

427 bool Request::executeSendFile(const std::string& file, const std::function<void(int)>& onStatus) {
428 bool atomar = true;
429
430 std::string absolutFileName = file;
431
432 if (std::filesystem::exists(absolutFileName)) {
433 std::error_code ec;
434 absolutFileName = std::filesystem::canonical(absolutFileName);
435
436 if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
437 core::file::FileReader::open(absolutFileName)->pipe(this, [this, &atomar, &absolutFileName, &onStatus](int errnum) -> void {
438 errno = errnum;
439 onStatus(errnum);
440
441 if (errnum == 0) {
442 if (httpMajor == 1) {
443 atomar = false;
444
445 set("Content-Type", web::http::MimeTypes::contentType(absolutFileName), false);
446 set("Last-Modified", httputils::file_mod_http_date(absolutFileName), false);
447 if (httpMinor == 1 && contentLength == 0) {
448 set("Transfer-Encoding", "chunked");
449 } else {
450 set("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName) + contentLength));
451 }
452
453 executeSendHeader();
454 }
455 }
456 });
457 } else {
458 errno = EINVAL;
459 onStatus(errno);
460 }
461 } else {
462 errno = ENOENT;
463 onStatus(errno);
464 }
465
466 return atomar;
467 }

494 bool Request::executeSendHeader() {
495 const std::string httpVersion = "HTTP/" + std::to_string(httpMajor) + "." + std::to_string(httpMinor);
496
497 std::string queryString;
498 if (!queries.empty()) {
499 queryString += "?";
500 for (auto& [key, value] : queries) {
501 queryString += httputils::url_encode(key) + "=" + httputils::url_encode(value) + "&";
502 }
503 queryString.pop_back();
504 }
505
506 socketContext->sendToPeer(method + " " + url + queryString + " " + httpVersion + "\r\n");
507 socketContext->sendToPeer("Date: " + httputils::to_http_date() + "\r\n");
508
509 if (!headers.contains("Transfer-Encoding") && contentLength > 0) {
510 set("Content-Length", std::to_string(contentLength));
511 }
512
513 for (const auto& [field, value] : headers) {
514 socketContext->sendToPeer(std::string(field).append(":").append(value).append("\r\n"));
515 }
516
517 for (const auto& [name, value] : cookies) {
518 socketContext->sendToPeer(std::string("Cookie:").append(name).append("=").append(value).append("\r\n"));
519 }
520
521 socketContext->sendToPeer("\r\n");
522
523 return true;
524 }

526 bool Request::executeSendFragment(const char* chunk, std::size_t chunkLen) {
527 if (transferEncoding == TransferEncoding::Chunked) {
528 socketContext->sendToPeer(to_hex_str(chunkLen).append("\r\n"));
529 }
530
531 socketContext->sendToPeer(chunk, chunkLen);
532 contentLengthSent += chunkLen;
533
534 if (transferEncoding == TransferEncoding::Chunked) {
535 socketContext->sendToPeer("\r\n");
536 contentLength += chunkLen;
537 }
538
539 return true;

555 void Request::requestDelivered() {
556 if (!masterRequest.expired()) {
557 if (transferEncoding == TransferEncoding::Chunked) {
558 executeSendFragment("", 0); // For transfer encoding chunked. Terminate the chunk sequence.
559
560 if (!trailer.empty()) {
561 for (auto& [field, value] : trailer) {
562 socketContext->sendToPeer(std::string(field).append(":").append(value).append("\r\n"));
563 }
564 socketContext->sendToPeer("\r\n");
565 }
566 }
567
568 socketContext->requestDelivered(std::move(*this), contentLengthSent == contentLength);
569 }
