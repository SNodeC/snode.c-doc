47 template <typename PhysicalSocket>
48 std::size_t SocketReader<PhysicalSocket>::doRead() {
49 errno = 0;
50
51 if (size == 0) {
52 cursor = 0;
53
54 std::size_t readLen = blockSize - size;
55 ssize_t retRead = read(readBuffer.data() + size, readLen);
56
57 if (retRead > 0) {
58 size += static_cast<std::size_t>(retRead);
59
60 if (!isSuspended()) {
61 suspend();
62 }
63 span();
64 } else if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
65 if (isSuspended()) {
66 resume();
67 }
68 } else {
69 onStatus(errno);
70 disable();
71 }
72 } else {
73 span();
74 }
75
76 return size;
77 }

85 template <typename PhysicalSocket>
86 std::size_t SocketReader<PhysicalSocket>::readFromPeer(char* junk, std::size_t junkLen) {
87 std::size_t maxReturn = std::min(junkLen, size);
88
89 std::copy(readBuffer.data() + cursor, readBuffer.data() + cursor + maxReturn, junk);
90
91 cursor += maxReturn;
92 size -= maxReturn;
93
94 return maxReturn;
95 }
