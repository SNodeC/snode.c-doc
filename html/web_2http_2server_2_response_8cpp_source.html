48 void Response::enqueue(const char* junk, std::size_t junkLen) {
49 if (!headersSent && !sendHeaderInProgress) {
50 sendHeaderInProgress = true;
51 sendHeader();
52 sendHeaderInProgress = false;
53 headersSent = true;
54 }
55
56 // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDelete)
57 requestContext->sendToPeer(junk, junkLen);
58
59 if (headersSent) {
60 contentSent += junkLen;
61 if (contentSent == contentLength) {
62 requestContext->sendToPeerCompleted();
63 } else if (contentSent > contentLength) {
64 requestContext->close();
65 }
66 }
67 }

119 Response& Response::set(const std::string& field, const std::string& value, bool overwrite) {
120 if (overwrite || headers.find(field) == headers.end()) {
121 headers.insert_or_assign(field, value);
122
123 if (field == "Content-Length") {
124 contentLength = std::stoul(value);
125 } else if (field == "Connection" && httputils::ci_contains(value, "close")) {
126 connectionState = ConnectionState::Close;
127 } else if (field == "Connection" && httputils::ci_contains(value, "keep-alive")) {
128 connectionState = ConnectionState::Keep;
129 }
130 }
131
132 return *this;
133 }

163 bool Response::upgrade(Request& req) {
164 bool success = false;
165
166 if (httputils::ci_contains(req.get("connection"), "Upgrade")) {
167 web::http::server::SocketContextUpgradeFactory* socketContextUpgradeFactory =
168 web::http::server::SocketContextUpgradeFactorySelector::instance()->select(req, *this);
169
170 if (socketContextUpgradeFactory != nullptr) {
171 success = requestContext->switchSocketContext(socketContextUpgradeFactory) != nullptr;
172 } else {
173 set("Connection", "close").status(404);
174 }
175 } else {
176 set("Connection", "close").status(400);
177 }
178
179 return success;
180 }

182 void Response::sendFile(const std::string& file, const std::function<void(int err)>& onError) {
183 std::string absolutFileName = file;
184
185 if (std::filesystem::exists(absolutFileName)) {
186 std::error_code ec;
187 absolutFileName = std::filesystem::canonical(absolutFileName);
188
189 if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
190 core::file::FileReader::connect(absolutFileName, *this, [this, &absolutFileName, onError](int err) -> void {
191 if (err == 0) {
192 headers.insert({{"Content-Type", web::http::MimeTypes::contentType(absolutFileName)},
193 {"Last-Modified", httputils::file_mod_http_date(absolutFileName)}});
194 headers.insert_or_assign("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
195 } else {
196 onError(err);
197 }
198 });
199 } else {
200 errno = EEXIST;
201 onError(errno);
202 }
203 } else {
204 errno = ENOENT;
205 onError(errno);
206 }
207 }

209 void Response::sendHeader() {
210 enqueue("HTTP/1.1 " + std::to_string(responseStatus) + " " + StatusCode::reason(responseStatus) + "\r\n");
211 enqueue("Date: " + httputils::to_http_date() + "\r\n");
212
213 set({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
214
215 for (auto& [field, value] : headers) {
216 enqueue(field + ": " + value + "\r\n");
217 }
218
219 for (auto& [cookie, cookieValue] : cookies) {
220 std::string cookieString =
221 std::accumulate(cookieValue.getOptions().begin(),
222 cookieValue.getOptions().end(),
223 cookie + "=" + cookieValue.getValue(),
224 [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
225 return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
226 });
227 enqueue("Set-Cookie: " + cookieString + "\r\n");
228 }
229
230 enqueue("\r\n");
231
232 if (headers.find("Content-Length") != headers.end()) {
233 contentLength = std::stoul(headers.find("Content-Length")->second);
234 } else {
235 contentLength = 0;
236 }
237 }
