41 std::size_t Identity::read() {
42 std::size_t consumed = 0;
43
44 switch (state) {
45 case -1:
46 content.clear();
47 completed = false;
48 error = false;
49
50 state = 0;
51
52 [[fallthrough]];
53 case 0:
54 std::size_t ret = 0;
55 do {
56 ret = socketContext->readFromPeer(content.data() + contentLengthRead, contentLengthExpected - contentLengthRead);
57 contentLengthRead += ret;
58 consumed += ret;
59 completed = contentLengthExpected == contentLengthRead;
60 } while (ret > 0 && !completed);
61
62 if (completed) {
63 state = -1;
64 }
65
66 break;
67 }
68 return consumed;
69 }
