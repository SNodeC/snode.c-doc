58 VerboseRequest::VerboseRequest(Details details) {
59 use([details] MIDDLEWARE(req, res, next) {
60 LOG(DEBUG) << res->getSocketContext()->getSocketConnection()->getConnectionName()
61 << " Express VerboseMiddleware: " << req->method << " " << req->url << " " << req->httpVersion << "\n"
62 << httputils::toString(
63 req->method,
64 req->url,
65 req->httpVersion,
66 (details & Details::W_QUERIES) == Details::W_QUERIES ? req->queries : web::http::CiStringMap<std::string>(),
67 (details & Details::W_HEADERS) == Details::W_HEADERS ? req->headers : web::http::CiStringMap<std::string>(),
68 (details & Details::W_COOKIES) == Details::W_COOKIES ? req->cookies : web::http::CiStringMap<std::string>(),
69 (details & Details::W_CONTENT) == Details::W_CONTENT ? req->body : std::vector<char>());
70
71 next();
72 });
73 }

75 class VerboseRequest& VerboseRequest(VerboseRequest::Details details) {
76 static std::shared_ptr<class VerboseRequest> verboseRequest = nullptr;
77
78 if (verboseRequest == nullptr) {
79 verboseRequest = std::shared_ptr<class VerboseRequest>(new class VerboseRequest(details));
80 }
81
82 return *verboseRequest;
83 }
