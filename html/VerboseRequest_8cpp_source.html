41 static std::string hexDump(const std::vector<char>& bytes, int prefixLength = 0) {
42 std::stringstream hexStream;
43
44 uint8_t buff[17];
45 size_t i = 0;
46
47 hexStream << std::hex;
48
49 int currentPrefixLength = 0;
50
51 // Process every byte in the data.
52 for (i = 0; i < bytes.size(); i++) {
53 // Multiple of 16 means new line (with line offset).
54
55 if ((i % 16) == 0) {
56 // Just don't print ASCII for the zeroth line.
57 if (i != 0) {
58 hexStream << " " << buff << std::endl;
59 }
60
61 // Output the offset.
62 hexStream << Color::Code::FG_BLUE;
63 hexStream << std::setw(currentPrefixLength) << std::setfill(' ') << ""
64 << ": " << std::setw(4) << std::setfill('0') << static_cast<unsigned int>(i);
65 hexStream << Color::Code::FG_DEFAULT << " ";
66 currentPrefixLength = prefixLength;
67 }
68
69 // Now the hex code for the specific character.
70 hexStream << Color::Code::FG_GREEN;
71 hexStream << " " << std::setw(2) << std::setfill('0') << static_cast<unsigned int>(bytes[i]);
72 hexStream << Color::Code::FG_DEFAULT;
73
74 // And store a printable ASCII character for later.
75 if ((bytes[i] < 0x20) || (bytes[i] > 0x7e)) {
76 buff[i % 16] = '.';
77 } else {
78 buff[i % 16] = static_cast<uint8_t>(bytes[i]);
79 }
80 buff[(i % 16) + 1] = '\0';
81 }
82
83 hexStream << std::dec;
84
85 // Pad out last line if not exactly 16 characters.
86 while ((i % 16) != 0) {
87 hexStream << " ";
88 i++;
89 }
90
91 // And print the final ASCII bit.
92 hexStream << " " << buff;
93
94 return hexStream.str();
95 }

97 VerboseRequest::VerboseRequest(Details details) {
98 use([details] MIDDLEWARE(req, res, next) {
99 const int prefixLength = 9;
100 int keyLength = 0;
101
102 for (const auto& [key, value] : req->queries) {
103 keyLength = std::max(keyLength, static_cast<int>(key.size()));
104 }
105 for (const auto& [key, value] : req->headers) {
106 keyLength = std::max(keyLength, static_cast<int>(key.size()));
107 }
108 for (const auto& [key, value] : req->cookies) {
109 keyLength = std::max(keyLength, static_cast<int>(key.size()));
110 }
111
112 std::stringstream requestStream;
113
114 if ((details & Details::W_REQUEST) != 0) {
115 requestStream << std::setw(prefixLength) << "Request"
116 << ": " << std::setw(keyLength) << "Method"
117 << " : " << req->method << "\n";
118 requestStream << std::setw(prefixLength) << ""
119 << ": " << std::setw(keyLength) << "Url"
120 << " : " << req->url << "\n";
121 requestStream << std::setw(prefixLength) << ""
122 << ": " << std::setw(keyLength) << "Version"
123 << " : " << req->httpVersion << "\n";
124 }
125
126 std::string prefix;
127
128 if ((details & Details::W_QUERIES) != 0) {
129 prefix = "Queries";
130 for (const auto& [key, value] : req->queries) {
131 requestStream << std::setw(prefixLength) << prefix << ": " << std::setw(keyLength) << key << " : " << value << "\n";
132 prefix = "";
133 }
134 }
135
136 if ((details & Details::W_HEADERS) != 0) {
137 prefix = "Header";
138 for (const auto& [key, value] : req->headers) {
139 requestStream << std::setw(prefixLength) << prefix << ": " << std::setw(keyLength) << key << " : " << value << "\n";
140 prefix = "";
141 }
142 }
143
144 if ((details & Details::W_COOKIES) != 0) {
145 prefix = "Cookies";
146 for (const auto& [key, value] : req->cookies) {
147 requestStream << std::setw(prefixLength) << prefix << ": " << std::setw(keyLength) << key << " : " << value << "\n";
148 prefix = "";
149 }
150 }
151
152 if ((details & Details::W_CONTENT) != 0) {
153 if (!req->body.empty()) {
154 prefix = "Body";
155 requestStream << std::setw(prefixLength) << prefix << hexDump(req->body, prefixLength) << "\n";
156 }
157 }
158
159 std::string requestString = requestStream.str();
160 requestString.pop_back();
161 LOG(TRACE) << res->getSocketContext()->getSocketConnection()->getInstanceName() << " HTTP: '" << req->method << " " << req->url
162 << " " << req->httpVersion << "'\n"
163 << requestString;
164
165 next();
166 });
167 }
