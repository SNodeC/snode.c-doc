35 std::size_t ControlPacketFactory::construct() {
36 std::size_t consumed = 0;
37
38 switch (state) {
39 case 0: // Type - Flags
40 consumed = typeFlags.construct();
41 if (typeFlags.isCompleted()) {
42 state++;
43 } else {
44 error = typeFlags.isError();
45 }
46 break;
47 case 1: // Remaining length
48 consumed = remainingLength.construct();
49 if (remainingLength.isCompleted()) {
50 data.setLength(remainingLength.getValue());
51 completed = remainingLength.getValue() == 0;
52 state++;
53 } else {
54 error = remainingLength.isError();
55 }
56 break;
57 case 2: // Var-Header + Payload
58 consumed = data.construct();
59 if (data.isCompleted()) {
60 completed = true;
61 state++;
62 } else {
63 error = data.isError();
64 }
65 break;
66 }
67
68 return consumed;
69 }
