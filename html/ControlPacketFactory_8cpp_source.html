58 std::size_t ControlPacketFactory::construct() {
59 std::size_t consumed = 0;
60
61 switch (state) {
62 case 0: // Type - Flags
63 consumed = typeFlags.construct();
64 if (typeFlags.isComplete()) {
65 state++;
66 } else {
67 error = typeFlags.isError();
68 }
69 break;
70 case 1: // Remaining length
71 consumed = remainingLength.construct();
72 if (remainingLength.isComplete()) {
73 data.setLength(remainingLength.getValue());
74 completed = remainingLength.getValue() == 0;
75 state++;
76 } else {
77 error = remainingLength.isError();
78 }
79 break;
80 case 2: // Var-Header + Payload
81 consumed = data.construct();
82 if (data.isComplete()) {
83 completed = true;
84 state++;
85 } else {
86 error = data.isError();
87 }
88 break;
89 }
90
91 return consumed;
92 }
