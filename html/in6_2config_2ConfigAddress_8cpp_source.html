75 template <template <typename SocketAddress> typename ConfigAddressType>
76 ConfigAddress<ConfigAddressType>::ConfigAddress(net::config::ConfigInstance* instance,
77 const std::string& addressOptionName,
78 const std::string& addressOptionDescription)
79 : Super(instance, addressOptionName, addressOptionDescription) {
80 Super::add_option(hostOpt, //
81 "--host",
82 "Host name or IPv6 address",
83 "hostname|IPv6",
84 "::",
85 CLI::TypeValidator<std::string>());
86 Super::add_option(portOpt, //
87 "--port",
88 "Port number",
89 "port",
90 0,
91 CLI::Range(std::numeric_limits<uint16_t>::min(), std::numeric_limits<uint16_t>::max()));
92 Super::add_flag(numericOpt, //
93 "--numeric",
94 "Suppress host name lookup",
95 "bool",
96 XSTR(IPV6_NUMERIC),
97 CLI::IsMember({"true", "false"}));
98 Super::add_flag(numericReverseOpt,
99 "--numeric-reverse",
100 "Suppress reverse host name lookup",
101 "bool",
102 XSTR(IPV6_NUMERIC_REVERSE),
103 CLI::IsMember({"true", "false"}));
104 Super::add_flag(ipv4MappedOpt, //
105 "--ipv4-mapped",
106 "Resolve IPv4-mapped IPv6 addresses also",
107 "bool",
108 XSTR(IPV4_MAPPED),
109 CLI::IsMember({"true", "false"}));
110 }

112 template <template <typename SocketAddress> typename ConfigAddressType>
113 SocketAddress* ConfigAddress<ConfigAddressType>::init() {
114 return &(new SocketAddress(hostOpt->as<std::string>(), portOpt->as<uint16_t>()))
115 ->init({.aiFlags = (aiFlags & ~AI_V4MAPPED & ~AI_NUMERICHOST) | (ipv4MappedOpt->as<bool>() ? AI_V4MAPPED : 0) |
116 (numericOpt->as<bool>() ? AI_NUMERICHOST : 0),
117 .aiSockType = aiSockType,
118 .aiProtocol = aiProtocol});
119 }
