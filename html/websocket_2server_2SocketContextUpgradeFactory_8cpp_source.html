41 SocketContextUpgrade* SocketContextUpgradeFactory::create(core::socket::stream::SocketConnection* socketConnection,
42 web::http::server::Request* request,
43 web::http::server::Response* response) {
44 SocketContextUpgrade* socketContext = nullptr;
45
46 if (request->get("Sec-WebSocket-Version") == "13") {
47 std::string requestedSubProtocolNames = request->get("sec-websocket-protocol");
48
49 std::list<std::string> subProtocolNamesList;
50 do {
51 std::string subProtocolName;
52 std::tie(subProtocolName, requestedSubProtocolNames) = httputils::str_split(requestedSubProtocolNames, ',');
53 httputils::str_trimm(subProtocolName);
54 subProtocolNamesList.push_back(subProtocolName);
55 } while (!requestedSubProtocolNames.empty());
56
57 if (!subProtocolNamesList.empty()) {
58 std::string selectedSubProtocolName;
59
60 socketContext = new SocketContextUpgrade(socketConnection, this);
61 selectedSubProtocolName = socketContext->loadSubProtocol(subProtocolNamesList);
62
63 if (!selectedSubProtocolName.empty()) {
64 response->set("Upgrade", "websocket");
65 response->set("Connection", "Upgrade");
66 response->set("Sec-WebSocket-Protocol", selectedSubProtocolName);
67 response->set("Sec-WebSocket-Accept", base64::serverWebSocketKey(request->get("sec-websocket-key")));
68
69 response->status(101); // Switch Protocol
70 } else {
71 delete socketContext;
72 socketContext = nullptr;
73
74 response->set("Connection", "close");
75 response->status(400);
76 }
77 } else {
78 checkRefCount();
79
80 response->set("Connection", "close");
81 response->status(400);
82 }
83 } else {
84 checkRefCount();
85
86 response->set("Sec-WebSocket-Version", "13");
87 response->set("Connection", "close");
88 response->status(426);
89 }
90
91 return socketContext;
92 }
