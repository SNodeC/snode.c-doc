70 enum Parser::ParserState ResponseParser::parseStartLine(const std::string& line) {
71 enum Parser::ParserState parserState = Parser::ParserState::HEADER;
72
73 if (!line.empty()) {
74 std::string remaining;
75
76 std::tie(httpVersion, remaining) = httputils::str_split(line, ' ');
77
78 std::smatch httpVersionMatch;
79 if (!std::regex_match(httpVersion, httpVersionMatch, httpVersionRegex)) {
80 parserState = parsingError(400, "Wrong protocol version");
81 } else {
82 httpMajor = std::stoi(httpVersionMatch.str(1));
83 httpMinor = std::stoi(httpVersionMatch.str(2));
84
85 std::tie(statusCode, reason) = httputils::str_split(remaining, ' ');
86 if (StatusCode::contains(std::stoi(statusCode))) {
87 if (!reason.empty()) {
88 onResponse(httpVersion, statusCode, reason);
89 } else {
90 parserState = parsingError(400, "No reason phrase");
91 }
92 } else {
93 parserState = parsingError(400, "Unknown status code");
94 }
95 }
96 } else {
97 parserState = parsingError(400, "Response line empty");
98 }
99 return parserState;
100 }

102 enum Parser::ParserState ResponseParser::parseHeader() {
103 for (const auto& [headerFieldName, headerFieldValue] : Parser::headers) {
104 if (headerFieldName != "set-cookie") {
105 if (headerFieldName == "content-length") {
106 Parser::contentLength = static_cast<std::size_t>(std::stoi(headerFieldValue));
107 }
108 } else {
109 std::string cookiesLine = headerFieldValue;
110
111 while (!cookiesLine.empty()) {
112 std::string cookieLine;
113 std::tie(cookieLine, cookiesLine) = httputils::str_split(cookiesLine, ',');
114
115 std::string cookieOptions;
116 std::string cookie;
117 std::tie(cookie, cookieOptions) = httputils::str_split(cookieLine, ';');
118
119 std::string cookieName;
120 std::string cookieValue;
121 std::tie(cookieName, cookieValue) = httputils::str_split(cookie, '=');
122
123 httputils::str_trimm(cookieName);
124 httputils::str_trimm(cookieValue);
125
126 std::map<std::string, CookieOptions>::iterator cookieElement;
127 bool inserted = false;
128 std::tie(cookieElement, inserted) = cookies.insert({cookieName, CookieOptions(cookieValue)});
129
130 while (!cookieOptions.empty()) {
131 std::string option;
132 std::tie(option, cookieOptions) = httputils::str_split(cookieOptions, ';');
133
134 std::string optionName;
135 std::string optionValue;
136 std::tie(optionName, optionValue) = httputils::str_split(option, '=');
137
138 httputils::str_trimm(optionName);
139 httputils::str_trimm(optionValue);
140
141 cookieElement->second.setOption(optionName, optionValue);
142 }
143 }
144 }
145 }
146
147 Parser::headers.erase("set-cookie");
148
149 onHeader(Parser::headers, cookies);
150
151 enum Parser::ParserState parserState = Parser::ParserState::BODY;
152 if (contentLength == 0 && httpMinor == 1) {
153 parsingFinished();
154 parserState = ParserState::BEGIN;
155 }
156
157 return parserState;
158 }
