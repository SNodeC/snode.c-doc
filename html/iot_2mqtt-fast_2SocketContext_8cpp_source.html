139 std::size_t SocketContext::onReceivedFromPeer() {
140 std::size_t consumed = controlPacketFactory.construct();
141
142 if (controlPacketFactory.isError()) {
143 LOG(ERROR) << "SocketContext: Error during ControlPacket construction";
144 close();
145 } else if (controlPacketFactory.isComplete()) {
146 LOG(DEBUG) << "======================================================";
147 LOG(DEBUG) << "PacketType: " << static_cast<uint16_t>(controlPacketFactory.getPacketType());
148 LOG(DEBUG) << "PacketFlags: " << static_cast<uint16_t>(controlPacketFactory.getPacketFlags());
149 LOG(DEBUG) << "RemainingLength: " << static_cast<uint16_t>(controlPacketFactory.getRemainingLength());
150
151 printData(controlPacketFactory.getPacket().getValue());
152
153 switch (controlPacketFactory.getPacketType()) {
154 case MQTT_CONNECT:
155 onConnect(iot::mqtt_fast::packets::Connect(controlPacketFactory));
156 break;
157 case MQTT_CONNACK:
158 onConnack(iot::mqtt_fast::packets::Connack(controlPacketFactory));
159 break;
160 case MQTT_PUBLISH:
161 onPublish(iot::mqtt_fast::packets::Publish(controlPacketFactory));
162 break;
163 case MQTT_PUBACK:
164 onPuback(iot::mqtt_fast::packets::Puback(controlPacketFactory));
165 break;
166 case MQTT_PUBREC:
167 onPubrec(iot::mqtt_fast::packets::Pubrec(controlPacketFactory));
168 break;
169 case MQTT_PUBREL:
170 onPubrel(iot::mqtt_fast::packets::Pubrel(controlPacketFactory));
171 break;
172 case MQTT_PUBCOMP:
173 onPubcomp(iot::mqtt_fast::packets::Pubcomp(controlPacketFactory));
174 break;
175 case MQTT_SUBSCRIBE:
176 onSubscribe(iot::mqtt_fast::packets::Subscribe(controlPacketFactory));
177 break;
178 case MQTT_SUBACK:
179 onSuback(iot::mqtt_fast::packets::Suback(controlPacketFactory));
180 break;
181 case MQTT_UNSUBSCRIBE:
182 onUnsubscribe(iot::mqtt_fast::packets::Unsubscribe(controlPacketFactory));
183 break;
184 case MQTT_UNSUBACK:
185 onUnsuback(iot::mqtt_fast::packets::Unsuback(controlPacketFactory));
186 break;
187 case MQTT_PINGREQ:
188 onPingreq(iot::mqtt_fast::packets::Pingreq(controlPacketFactory));
189 break;
190 case MQTT_PINGRESP:
191 onPingresp(iot::mqtt_fast::packets::Pingresp(controlPacketFactory));
192 break;
193 case MQTT_DISCONNECT:
194 onDisconnect(iot::mqtt_fast::packets::Disconnect(controlPacketFactory));
195 break;
196 default:
197 close();
198 break;
199 }
200
201 controlPacketFactory.reset();
202 }
203
204 return consumed;
205 }

220 void SocketContext::printData(const std::vector<char>& data) const {
221 std::stringstream ss;
222
223 ss << "Data: ";
224 unsigned long i = 0;
225 for (char ch : data) {
226 if (i != 0 && i % 8 == 0 && i + 1 != data.size()) {
227 ss << std::endl;
228 ss << " ";
229 }
230 ++i;
231 ss << "0x" << std::hex << std::setfill('0') << std::setw(2) << static_cast<uint16_t>(static_cast<uint8_t>(ch))
232 << " "; // << " | ";
233 }
234
235 LOG(DEBUG) << ss.str();
236 }
