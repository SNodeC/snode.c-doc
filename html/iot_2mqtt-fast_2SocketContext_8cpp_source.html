163 std::size_t SocketContext::onReceivedFromPeer() {
164 const std::size_t consumed = controlPacketFactory.construct();
165
166 if (controlPacketFactory.isError()) {
167 LOG(ERROR) << "MQTT (fast): SocketContext: Error during ControlPacket construction";
168 close();
169 } else if (controlPacketFactory.isComplete()) {
170 LOG(DEBUG) << "MQTT (fast): ======================================================";
171 LOG(DEBUG) << "MQTT (fast): PacketType: " << static_cast<uint16_t>(controlPacketFactory.getPacketType());
172 LOG(DEBUG) << "MQTT (fast): PacketFlags: " << static_cast<uint16_t>(controlPacketFactory.getPacketFlags());
173 LOG(DEBUG) << "MQTT (fast): RemainingLength: " << static_cast<uint16_t>(controlPacketFactory.getRemainingLength());
174
175 printData(controlPacketFactory.getPacket().getValue());
176
177 switch (controlPacketFactory.getPacketType()) {
178 case MQTT_CONNECT:
179 onConnect(iot::mqtt_fast::packets::Connect(controlPacketFactory));
180 break;
181 case MQTT_CONNACK:
182 onConnack(iot::mqtt_fast::packets::Connack(controlPacketFactory));
183 break;
184 case MQTT_PUBLISH:
185 onPublish(iot::mqtt_fast::packets::Publish(controlPacketFactory));
186 break;
187 case MQTT_PUBACK:
188 onPuback(iot::mqtt_fast::packets::Puback(controlPacketFactory));
189 break;
190 case MQTT_PUBREC:
191 onPubrec(iot::mqtt_fast::packets::Pubrec(controlPacketFactory));
192 break;
193 case MQTT_PUBREL:
194 onPubrel(iot::mqtt_fast::packets::Pubrel(controlPacketFactory));
195 break;
196 case MQTT_PUBCOMP:
197 onPubcomp(iot::mqtt_fast::packets::Pubcomp(controlPacketFactory));
198 break;
199 case MQTT_SUBSCRIBE:
200 onSubscribe(iot::mqtt_fast::packets::Subscribe(controlPacketFactory));
201 break;
202 case MQTT_SUBACK:
203 onSuback(iot::mqtt_fast::packets::Suback(controlPacketFactory));
204 break;
205 case MQTT_UNSUBSCRIBE:
206 onUnsubscribe(iot::mqtt_fast::packets::Unsubscribe(controlPacketFactory));
207 break;
208 case MQTT_UNSUBACK:
209 onUnsuback(iot::mqtt_fast::packets::Unsuback(controlPacketFactory));
210 break;
211 case MQTT_PINGREQ:
212 onPingreq(iot::mqtt_fast::packets::Pingreq(controlPacketFactory));
213 break;
214 case MQTT_PINGRESP:
215 onPingresp(iot::mqtt_fast::packets::Pingresp(controlPacketFactory));
216 break;
217 case MQTT_DISCONNECT:
218 onDisconnect(iot::mqtt_fast::packets::Disconnect(controlPacketFactory));
219 break;
220 default:
221 close();
222 break;
223 }
224
225 controlPacketFactory.reset();
226 }
227
228 return consumed;
229 }

244 void SocketContext::printData(const std::vector<char>& data) {
245 std::stringstream ss;
246
247 ss << "Data: ";
248 unsigned long i = 0;
249 for (char const ch : data) {
250 if (i != 0 && i % 8 == 0 && i + 1 != data.size()) {
251 ss << std::endl;
252 ss << " ";
253 }
254 ++i;
255 ss << "0x" << std::hex << std::setfill('0') << std::setw(2) << static_cast<uint16_t>(static_cast<uint8_t>(ch))
256 << " "; // << " | ";
257 }
258
259 LOG(DEBUG) << ss.str();
260 }
