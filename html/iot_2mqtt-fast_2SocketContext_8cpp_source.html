162 std::size_t SocketContext::onReceivedFromPeer() {
163 const std::size_t consumed = controlPacketFactory.construct();
164
165 if (controlPacketFactory.isError()) {
166 LOG(ERROR) << "MQTT (fast): SocketContext: Error during ControlPacket construction";
167 close();
168 } else if (controlPacketFactory.isComplete()) {
169 LOG(DEBUG) << "MQTT (fast): ======================================================";
170 LOG(DEBUG) << "MQTT (fast): PacketType: " << static_cast<uint16_t>(controlPacketFactory.getPacketType());
171 LOG(DEBUG) << "MQTT (fast): PacketFlags: " << static_cast<uint16_t>(controlPacketFactory.getPacketFlags());
172 LOG(DEBUG) << "MQTT (fast): RemainingLength: " << static_cast<uint16_t>(controlPacketFactory.getRemainingLength());
173
174 printData(controlPacketFactory.getPacket().getValue());
175
176 switch (controlPacketFactory.getPacketType()) {
177 case MQTT_CONNECT:
178 onConnect(iot::mqtt_fast::packets::Connect(controlPacketFactory));
179 break;
180 case MQTT_CONNACK:
181 onConnack(iot::mqtt_fast::packets::Connack(controlPacketFactory));
182 break;
183 case MQTT_PUBLISH:
184 onPublish(iot::mqtt_fast::packets::Publish(controlPacketFactory));
185 break;
186 case MQTT_PUBACK:
187 onPuback(iot::mqtt_fast::packets::Puback(controlPacketFactory));
188 break;
189 case MQTT_PUBREC:
190 onPubrec(iot::mqtt_fast::packets::Pubrec(controlPacketFactory));
191 break;
192 case MQTT_PUBREL:
193 onPubrel(iot::mqtt_fast::packets::Pubrel(controlPacketFactory));
194 break;
195 case MQTT_PUBCOMP:
196 onPubcomp(iot::mqtt_fast::packets::Pubcomp(controlPacketFactory));
197 break;
198 case MQTT_SUBSCRIBE:
199 onSubscribe(iot::mqtt_fast::packets::Subscribe(controlPacketFactory));
200 break;
201 case MQTT_SUBACK:
202 onSuback(iot::mqtt_fast::packets::Suback(controlPacketFactory));
203 break;
204 case MQTT_UNSUBSCRIBE:
205 onUnsubscribe(iot::mqtt_fast::packets::Unsubscribe(controlPacketFactory));
206 break;
207 case MQTT_UNSUBACK:
208 onUnsuback(iot::mqtt_fast::packets::Unsuback(controlPacketFactory));
209 break;
210 case MQTT_PINGREQ:
211 onPingreq(iot::mqtt_fast::packets::Pingreq(controlPacketFactory));
212 break;
213 case MQTT_PINGRESP:
214 onPingresp(iot::mqtt_fast::packets::Pingresp(controlPacketFactory));
215 break;
216 case MQTT_DISCONNECT:
217 onDisconnect(iot::mqtt_fast::packets::Disconnect(controlPacketFactory));
218 break;
219 default:
220 close();
221 break;
222 }
223
224 controlPacketFactory.reset();
225 }
226
227 return consumed;
228 }

243 void SocketContext::printData(const std::vector<char>& data) {
244 std::stringstream ss;
245
246 ss << "Data: ";
247 unsigned long i = 0;
248 for (char const ch : data) {
249 if (i != 0 && i % 8 == 0 && i + 1 != data.size()) {
250 ss << std::endl;
251 ss << " ";
252 }
253 ++i;
254 ss << "0x" << std::hex << std::setfill('0') << std::setw(2) << static_cast<uint16_t>(static_cast<uint8_t>(ch))
255 << " "; // << " | ";
256 }
257
258 LOG(DEBUG) << ss.str();
259 }
