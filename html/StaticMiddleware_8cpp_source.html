34 StaticMiddleware::StaticMiddleware(const std::string& root)
35 : root(root) {
36 use(
37 [&stdHeaders = this->stdHeaders,
38 &stdCookies = this->stdCookies,
39 &forceClose = this->forceClose] MIDDLEWARE( // cppcheck-suppress assignBoolToPointer
40 req,
41 res,
42 next) {
43 if (req.method == "GET") {
44 if (forceClose) {
45 res.set("Connection", "Close");
46 } else {
47 res.set("Connection", "Keep-Alive");
48 }
49 res.set(stdHeaders);
50 for (auto& [value, options] : stdCookies) {
51 res.cookie(value, options.getValue(), options.getOptions());
52 }
53 next();
54 } else {
55 LOG(INFO) << "Express StaticMiddleware: Wrong method " << req.method;
56 res.set("Connection", "Close");
57 res.sendStatus(400);
58 }
59 },
60 [] MIDDLEWARE(req, res, next) {
61 if (req.url == "/") {
62 LOG(INFO) << "Express StaticMiddleware: REDIRECTING " + req.url + " -> " + "/index.html";
63 res.redirect(308, "/index.html");
64 } else {
65 next();
66 }
67 },
68 [&root = this->root] APPLICATION(req, res) {
69 LOG(INFO) << "Express StaticMiddleware:: GET " + req.url + " -> " + root + req.url;
70 res.sendFile(root + req.url, [&req, &res](int ret) -> void {
71 if (ret != 0) {
72 PLOG(INFO) << req.url;
73 res.status(404).end();
74 }
75 });
76 });
77 }

117 class StaticMiddleware& StaticMiddleware::instance(const std::string& root) {
118 // Keep all created static middlewares alive
119 static std::map<const std::string, std::shared_ptr<class StaticMiddleware>> staticMiddlewares;
120
121 if (!staticMiddlewares.contains(root)) {
122 staticMiddlewares[root] = std::shared_ptr<StaticMiddleware>(new StaticMiddleware(root));
123 }
124
125 return *staticMiddlewares[root];
126 }
