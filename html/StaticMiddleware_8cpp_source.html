32 StaticMiddleware::StaticMiddleware(const std::string& root)
33 : root(root) {
34 use(
35 [&stdHeaders = this->stdHeaders, &stdCookies = this->stdCookies, &connectionState = this->defaultConnectionState] MIDDLEWARE(
36 req, res, next) {
37 if (req->method == "GET") {
38 if (connectionState == web::http::ConnectionState::Close) {
39 res->set("Connection", "close");
40 } else if (connectionState == web::http::ConnectionState::Keep) {
41 res->set("Connection", "keep-alive");
42 }
43 res->set(stdHeaders);
44 for (auto& [value, options] : stdCookies) {
45 res->cookie(value, options.getValue(), options.getOptions());
46 }
47 next();
48 } else {
49 LOG(INFO) << "Express StaticMiddleware: Wrong method " << req->method;
50 if (connectionState == web::http::ConnectionState::Close) {
51 res->set("Connection", "close");
52 } else if (connectionState == web::http::ConnectionState::Keep) {
53 res->set("Connection", "keep-alive");
54 }
55 res->sendStatus(405);
56 }
57 },
58 [] MIDDLEWARE(req, res, next) {
59 if (req->url == "/") {
60 LOG(INFO) << "Express StaticMiddleware: REDIRECTING " + req->url + " -> " + "/index.html";
61 res->redirect(308, "/index.html");
62 } else {
63 next();
64 }
65 },
66 [&root = this->root] APPLICATION(req, res) {
67 res->sendFile(root + req->url, [&root, req, res](int ret) -> void {
68 PLOG(INFO) << "Express StaticMiddleware: GET " + req->url + " -> " + root + req->url;
69
70 if (ret != 0) {
71 res->status(404).end();
72 }
73 });
74 });
75 }

115 class StaticMiddleware& StaticMiddleware::instance(const std::string& root) {
116 // Keep all created static middlewares alive
117 static std::map<const std::string, std::shared_ptr<class StaticMiddleware>> staticMiddlewares;
118
119 if (!staticMiddlewares.contains(root)) {
120 staticMiddlewares[root] = std::shared_ptr<StaticMiddleware>(new StaticMiddleware(root));
121 }
122
123 return *staticMiddlewares[root];
124 }
