84 bool RetainTree::TopicLevel::retain(const Message& message, std::string topic, bool leafFound) {
85 if (leafFound) {
86 if (!message.getTopic().empty()) {
87 LOG(DEBUG) << "MQTT Broker: Retaining:";
88 LOG(DEBUG) << "MQTT Broker: Topic: " << message.getTopic();
89 LOG(DEBUG) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::stringToHexString(message.getMessage());
90 LOG(DEBUG) << "MQTT Broker: QoS: " << static_cast<uint16_t>(message.getQoS());
91 this->message = message;
92 }
93 } else {
94 const std::string::size_type slashPosition = topic.find('/');
95 const std::string topicLevel = topic.substr(0, slashPosition);
96
97 topic.erase(0, topicLevel.size() + 1);
98
99 if (subTopicLevels.insert({topicLevel, TopicLevel(broker)})
100 .first->second.retain(message, topic, slashPosition == std::string::npos)) {
101 subTopicLevels.erase(topicLevel);
102 }
103 }
104
105 return this->message.getMessage().empty() && subTopicLevels.empty();
106 }

112 void RetainTree::TopicLevel::appear(const std::string& clientId, std::string topic, uint8_t qoS, bool appeared) {
113 if (appeared) {
114 if (!message.getTopic().empty()) {
115 LOG(DEBUG) << "MQTT Broker: Retained Topic found:";
116 LOG(DEBUG) << "MQTT Broker: Topic: " << message.getTopic();
117 LOG(DEBUG) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::stringToHexString(message.getMessage());
118 LOG(DEBUG) << "MQTT Broker: QoS: " << static_cast<uint16_t>(message.getQoS());
119 LOG(DEBUG) << "MQTT Broker: Client:";
120 LOG(DEBUG) << "MQTT Broker: QoS: " << static_cast<uint16_t>(qoS);
121 LOG(DEBUG) << "MQTT Broker: distributing message ...";
122 broker->sendPublish(clientId, message, std::min(message.getQoS(), qoS), true);
123 LOG(DEBUG) << "MQTT Broker: ... completed!";
124 }
125 } else {
126 const std::string::size_type slashPosition = topic.find('/');
127 const std::string topicLevel = topic.substr(0, slashPosition);
128 const bool leafFound = slashPosition == std::string::npos;
129
130 topic.erase(0, topicLevel.size() + 1);
131
132 auto foundNode = subTopicLevels.find(topicLevel);
133 if (foundNode != subTopicLevels.end()) {
134 foundNode->second.appear(clientId, topic, qoS, leafFound);
135 } else if (topicLevel == "+") {
136 for (auto& [notUsed, topicTree] : subTopicLevels) {
137 topicTree.appear(clientId, topic, qoS, leafFound);
138 }
139 } else if (topicLevel == "#") {
140 appear(clientId, qoS);
141 }
142 }
143 }

145 void RetainTree::TopicLevel::appear(const std::string& clientId, uint8_t clientQoS) {
146 if (!message.getTopic().empty()) {
147 LOG(DEBUG) << "MQTT Broker: Retained Topic found:";
148 LOG(DEBUG) << "MQTT Broker: Topic: " << message.getTopic();
149 LOG(DEBUG) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::stringToHexString(message.getMessage());
150 LOG(DEBUG) << "MQTT Broker: QoS: " << static_cast<uint16_t>(message.getQoS());
151 LOG(DEBUG) << "MQTT Broker: Client:";
152 LOG(DEBUG) << "MQTT Broker: QoS: " << static_cast<uint16_t>(clientQoS);
153 LOG(DEBUG) << "MQTT Broker: distributing message ...";
154 broker->sendPublish(clientId, message, std::min(message.getQoS(), clientQoS), true);
155 LOG(DEBUG) << "MQTT Broker: ... completed!";
156 }
157
158 for (auto& [topicLevel, topicTree] : subTopicLevels) {
159 topicTree.appear(clientId, clientQoS);
160 }
161 }
