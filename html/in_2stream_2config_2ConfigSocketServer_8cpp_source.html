116 ConfigSocketServer& ConfigSocketServer::setReuseAddress(bool reuseAddress) {
117 if (reuseAddress) {
118 addSocketOption(SOL_SOCKET, SO_REUSEADDR, 1);
119 } else {
120 addSocketOption(SOL_SOCKET, SO_REUSEADDR, 0);
121 }
122
123 reuseAddressOpt //
124 ->default_val(reuseAddress ? "true" : "false")
125 ->clear();
126
127 return *this;
128 }
129
130 bool ConfigSocketServer::getReuseAddress() const {
131 return reuseAddressOpt->as<bool>();
