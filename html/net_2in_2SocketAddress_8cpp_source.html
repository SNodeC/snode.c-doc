59 SocketAddress::SocketAddress(const SockAddr& sockAddr, socklen_t sockAddrLen)
60 : net::SocketAddress<SockAddr>(sockAddr, sockAddrLen)
61 , socketAddrInfo(std::make_shared<SocketAddrInfo>()) {
62 char host[NI_MAXHOST];
63 char serv[NI_MAXSERV];
64 std::memset(host, 0, NI_MAXHOST);
65 std::memset(serv, 0, NI_MAXSERV);
66
67 int ret = core::system::getnameinfo(
68 reinterpret_cast<const sockaddr*>(&sockAddr), sizeof(sockAddr), nullptr, 0, serv, NI_MAXSERV, NI_NUMERICSERV);
69
70 if (ret == 0) {
71 ret = core::system::getnameinfo(
72 reinterpret_cast<const sockaddr*>(&sockAddr), sizeof(sockAddr), host, NI_MAXHOST, nullptr, 0, NI_NAMEREQD);
73 }
74 if (ret != 0) {
75 ret = core::system::getnameinfo(
76 reinterpret_cast<const sockaddr*>(&sockAddr), sizeof(sockAddr), host, NI_MAXHOST, nullptr, 0, NI_NUMERICHOST);
77 }
78
79 if (ret == 0) {
80 this->port = static_cast<uint16_t>(std::stoul(serv));
81 this->host = host;
82 } else {
83 this->host = gai_strerror(ret);
84 }
85 }

87 SocketAddress& SocketAddress::init() {
88 addrinfo hints{};
89
90 hints.ai_family = AF_INET;
91 hints.ai_socktype = aiSocktype;
92 hints.ai_protocol = aiProtocol;
93 hints.ai_flags = aiFlags | AI_ADDRCONFIG | AI_ALL;
94
95 int aiErrCode = 0;
96
97 if ((aiErrCode = socketAddrInfo->init(host, std::to_string(port), hints)) != 0) {
98 core::socket::State state = core::socket::STATE_OK;
99 switch (aiErrCode) {
100 case EAI_AGAIN:
101 case EAI_MEMORY:
102 state = core::socket::STATE_ERROR;
103 break;
104 default:
105 state = core::socket::STATE_FATAL;
106 break;
107 }
108 throw core::socket::SocketAddress::BadSocketAddress(state,
109 host + ": " +
110 (aiErrCode == EAI_SYSTEM ? strerror(errno) : gai_strerror(aiErrCode)),
111 (aiErrCode == EAI_SYSTEM ? errno : aiErrCode));
112 }
113
114 return *this;
115 }
