58 SocketAddress::SocketAddress(const SockAddr& sockAddr, SockLen sockAddrLen, bool numeric)
59 : net::SocketAddress<SockAddr>(sockAddr, sockAddrLen)
60 , socketAddrInfo(std::make_shared<SocketAddrInfo>()) {
61 char hostC[NI_MAXHOST];
62 char servC[NI_MAXSERV];
63 std::memset(hostC, 0, NI_MAXHOST);
64 std::memset(servC, 0, NI_MAXSERV);
65
66 const int aiErrCode = core::system::getnameinfo(reinterpret_cast<const sockaddr*>(&sockAddr),
67 sizeof(sockAddr),
68 hostC,
69 NI_MAXHOST,
70 servC,
71 NI_MAXSERV,
72 NI_NUMERICSERV | (numeric ? NI_NUMERICHOST : NI_NAMEREQD));
73 if (aiErrCode == 0) {
74 if (servC[0] != '\0') {
75 this->port = static_cast<uint16_t>(std::stoul(servC));
76 }
77
78 this->host = hostC;
79 this->canonName = hostC;
80 } else {
81 core::socket::State state = core::socket::STATE_OK;
82 switch (aiErrCode) {
83 case EAI_AGAIN:
84 case EAI_MEMORY:
85 state = core::socket::STATE_ERROR;
86 break;
87 default:
88 state = core::socket::STATE_FATAL;
89 break;
90 }
91
92 throw core::socket::SocketAddress::BadSocketAddress(
93 state, aiErrCode == EAI_SYSTEM ? strerror(errno) : gai_strerror(aiErrCode), aiErrCode == EAI_SYSTEM ? errno : aiErrCode);
94 }
95 }

97 SocketAddress& SocketAddress::init(const Hints& hints) {
98 addrinfo addrInfoHints{};
99
100 addrInfoHints.ai_family = Super::getAddressFamily();
101 addrInfoHints.ai_flags =
102 hints.aiFlags | AI_ADDRCONFIG | AI_CANONNAME /*| AI_CANONIDN*/; // AI_CANONIDN produces a still reachable memory leak
103 addrInfoHints.ai_socktype = hints.aiSockType;
104 addrInfoHints.ai_protocol = hints.aiProtocol;
105
106 const int aiErrCode = socketAddrInfo->resolve(host, std::to_string(port), addrInfoHints);
107 if (aiErrCode == 0) {
108 sockAddr = socketAddrInfo->getSockAddr();
109 canonName = socketAddrInfo->getCanonName();
110 } else {
111 core::socket::State state = core::socket::STATE_OK;
112 switch (aiErrCode) {
113 case EAI_AGAIN:
114 case EAI_MEMORY:
115 state = core::socket::STATE_ERROR;
116 break;
117 default:
118 state = core::socket::STATE_FATAL;
119 break;
120 }
121 throw core::socket::SocketAddress::BadSocketAddress(state,
122 host + ":" + std::to_string(port) + " - " +
123 (aiErrCode == EAI_SYSTEM ? strerror(errno) : gai_strerror(aiErrCode)),
124 (aiErrCode == EAI_SYSTEM ? errno : aiErrCode));
125 }
126
127 return *this;
128 }
