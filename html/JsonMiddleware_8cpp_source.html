35 JsonMiddleware::JsonMiddleware() {
36 use([] MIDDLEWARE(req, res, next) {
37 try {
38 // parse body string with json library
39 // store it as type json from nlohmann library
40 // nlohmann::json json = nlohmann::json::parse(req.body, req.body + req.contentLength);
41
42 req.body.push_back(0);
43 nlohmann::json json = nlohmann::json::parse(req.body);
44
45 // set all the json data as attributes in the request object
46 req.setAttribute<nlohmann::json>(json);
47 } catch ([[maybe_unused]] const nlohmann::detail::parse_error& error) {
48 // silently fail if body is not json
49 }
50
51 next();
52 });
53 }
