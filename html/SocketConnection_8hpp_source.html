185 template <typename PhysicalSocket,
186 template <typename PhysicalSocketT>
187 typename SocketReader,
188 template <typename PhysicalSocketT>
189 typename SocketWriter>
190 void SocketConnectionT<PhysicalSocket, SocketReader, SocketWriter>::onReceivedFromPeer(std::size_t available) {
191 std::size_t consumed = socketContext->onReceivedFromPeer();
192
193 if (newSocketContext != nullptr) { // Perform a pending SocketContextSwitch
194 disconnected();
195 socketContext = newSocketContext;
196 newSocketContext = nullptr;
197 connected(socketContext);
198 }
199
200 if (available != 0 && consumed == 0) {
201 close();
202 }
203 }
