74 std::size_t Receiver::readOpcode() {
75 char byte = 0;
76 std::size_t ret = readFrameData(&byte, 1);
77
78 if (ret > 0) {
79 uint8_t opCodeByte = static_cast<uint8_t>(byte);
80
81 fin = opCodeByte & 0b10000000;
82 opCode = opCodeByte & 0b00001111;
83
84 if (!continuation) {
85 onMessageStart(opCode);
86 parserState = ParserState::LENGTH;
87 } else if (opCode == 0) {
88 parserState = ParserState::LENGTH;
89 } else {
90 parserState = ParserState::ERROR;
91 errorState = 1002;
92 VLOG(0) << "Error opcode in continuation frame";
93 }
94 continuation = !fin;
95 }
96
97 return ret;
98 }

100 std::size_t Receiver::readLength() {
101 char byte = 0;
102 std::size_t ret = readFrameData(&byte, 1);
103
104 if (ret > 0) {
105 uint8_t lengthByte = static_cast<uint8_t>(byte);
106
107 masked = lengthByte & 0b10000000;
108 payLoadNumBytes = payLoadNumBytesLeft = lengthByte & 0b01111111;
109
110 if (payLoadNumBytes > 125) {
111 switch (payLoadNumBytes) {
112 case 126:
113 elengthNumBytes = elengthNumBytesLeft = 2;
114 break;
115 case 127:
116 elengthNumBytes = elengthNumBytesLeft = 8;
117 break;
118 }
119 parserState = ParserState::ELENGTH;
120 payLoadNumBytes = payLoadNumBytesLeft = 0;
121 } else {
122 if (masked) {
123 parserState = ParserState::MASKINGKEY;
124 } else if (payLoadNumBytes > 0) {
125 parserState = ParserState::PAYLOAD;
126 } else {
127 if (fin) {
128 onMessageEnd();
129 }
130 reset();
131 }
132 }
133 }
134
135 return ret;
136 }

138 std::size_t Receiver::readELength() {
139 std::size_t ret = readFrameData(elengthJunk, elengthNumBytesLeft);
140
141 for (std::size_t i = 0; i < ret; i++) {
142 payLoadNumBytes |= static_cast<uint64_t>(*reinterpret_cast<unsigned char*>(elengthJunk + i))
143 << (elengthNumBytes - elengthNumBytesLeft) * 8;
144
145 elengthNumBytesLeft--;
146 }
147
148 if (elengthNumBytesLeft == 0) {
149 switch (elengthNumBytes) {
150 case 2:
151 payLoadNumBytes = payLoadNumBytesLeft = be16toh(static_cast<uint16_t>(payLoadNumBytes));
152 break;
153 case 8:
154 payLoadNumBytes = payLoadNumBytesLeft = be64toh(payLoadNumBytes);
155 break;
156 }
157
158 if (payLoadNumBytes & static_cast<uint64_t>(0x01) << 63) {
159 parserState = ParserState::ERROR;
160 errorState = 1004;
161 } else if (masked) {
162 parserState = ParserState::MASKINGKEY;
163 } else {
164 parserState = ParserState::PAYLOAD;
165 }
166 }
167
168 return ret;
169 }

196 std::size_t Receiver::readPayload() {
197 std::size_t payloadJunkLeft = (MAX_PAYLOAD_JUNK_LEN <= payLoadNumBytesLeft) ? static_cast<std::size_t>(MAX_PAYLOAD_JUNK_LEN)
198 : static_cast<std::size_t>(payLoadNumBytesLeft);
199
200 std::size_t ret = readFrameData(payloadJunk, payloadJunkLeft);
201
202 if (ret > 0) {
203 std::size_t payloadJunkLen = static_cast<std::size_t>(ret);
204
205 if (masked) {
206 for (std::size_t i = 0; i < payloadJunkLen; i++) {
207 *(payloadJunk + i) =
208 *(payloadJunk + i) ^ *(maskingKeyAsArray.keyAsArray + (i + (payLoadNumBytes - payLoadNumBytesLeft)) % 4);
209 }
210 }
211
212 onMessageData(payloadJunk, payloadJunkLen);
213
214 payLoadNumBytesLeft -= payloadJunkLen;
215 }
216
217 if (payLoadNumBytesLeft == 0) {
218 if (fin) {
219 onMessageEnd();
220 }
221 reset();
222 }
223
224 return ret;
225 }

227 void Receiver::dumpFrame(char* frame, uint64_t frameLength) {
228 unsigned long modul = 4;

233 stringStream << std::setfill('0') << std::setw(2) << std::hex << static_cast<unsigned int>(static_cast<unsigned char>(frame[i]))
234 << " ";
235
236 if ((i + 1) % modul == 0 || i == frameLength) {
