61 template <typename PhysicalSocket>
62 SSL* SocketConnection<PhysicalSocket>::startSSL(SSL_CTX* ctx,
63 const utils::Timeval& sslInitTimeout,
64 const utils::Timeval& sslShutdownTimeout) {
65 this->sslInitTimeout = sslInitTimeout;
66 this->sslShutdownTimeout = sslShutdownTimeout;
67 if (ctx != nullptr) {
68 ssl = SSL_new(ctx);
69
70 if (ssl != nullptr) {
71 if (SSL_set_fd(ssl, PhysicalSocket::getFd()) == 1) {
72 SocketReader::ssl = ssl;
73 SocketWriter::ssl = ssl;
74 } else {
75 SSL_free(ssl);
76 ssl = nullptr;
77 }
78 }
79 }
80
81 return ssl;
82 }

171 template <typename PhysicalSocket>
172 void SocketConnection<PhysicalSocket>::doSSLShutdown() {
173 if (SSL_get_shutdown(ssl) == (SSL_SENT_SHUTDOWN | SSL_RECEIVED_SHUTDOWN)) {
174 LOG(TRACE) << "SSL_Shutdown COMPLETED: Close_notify sent and received";
175 if (SocketWriter::isEnabled()) {
176 SocketWriter::doWriteShutdown([this]([[maybe_unused]] int errnum) -> void {
177 if (errno != 0) {
178 PLOG(TRACE) << "SocketWriter::doWriteShutdown";
179 }
180 SocketWriter::disable();
181 });
182 }
183 } else {
184 LOG(TRACE) << "SSL_Shutdown WAITING: Close_notify received but not send";
185 }
186 }

188 template <typename PhysicalSocket>
189 void SocketConnection<PhysicalSocket>::doWriteShutdown(const std::function<void(int)>& onShutdown) {
190 if ((SSL_get_shutdown(ssl) & SSL_SENT_SHUTDOWN) == 0) {
191 doSSLShutdown(
192 [this, &onShutdown]() -> void { // thus send one
193 if (SSL_get_shutdown(ssl) == (SSL_SENT_SHUTDOWN | SSL_RECEIVED_SHUTDOWN)) {
194 LOG(TRACE) << "SSL_Shutdown COMPLETED: Close_notify sent and received";
195 SocketWriter::doWriteShutdown(onShutdown);
196 } else {
197 LOG(TRACE) << "SSL_Shutdown WAITING: Close_notify sent but not received";
198 }
199 },
200 [this]() -> void {
201 LOG(TRACE) << "SSL_shutdown: Handshake timed out";
202 SocketWriter::doWriteShutdown([this]([[maybe_unused]] int errnum) -> void {
203 if (errno != 0) {
204 PLOG(TRACE) << "SocketWriter::doWriteShutdown";
205 }
206 SocketConnection::close();
207 });
208 },
209 [this](int sslErr) -> void {
210 ssl_log("SSL_shutdown: Handshake failed", sslErr);
211 SocketWriter::doWriteShutdown([this]([[maybe_unused]] int errnum) -> void {
212 if (errno != 0) {
213 PLOG(TRACE) << "SocketWriter::doWriteShutdown";
214 }
215 SocketConnection::close();
216 });
217 },
218 sslShutdownTimeout);
219 }
