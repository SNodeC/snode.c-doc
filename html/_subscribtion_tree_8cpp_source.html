69 void SubscribtionTree::TopicLevel::appear(const std::string& clientId, const std::string& topic) {
70 if (qoSMap.contains(clientId)) {
71 broker->appear(clientId, topic, qoSMap[clientId]);
72 }
73
74 for (auto& [topicLevel, subscribtion] : topicLevels) {
75 subscribtion.appear(clientId, topic + (topic.empty() ? "" : "/") + topicLevel);
76 }

79 bool SubscribtionTree::TopicLevel::subscribe(const std::string& clientId, uint8_t qoS, std::string topic, bool leafFound) {
80 bool success = true;
81
82 if (leafFound) {
83 qoSMap[clientId] = qoS;
84 } else {
85 std::string::size_type slashPosition = topic.find('/');
86
87 std::string topicLevel = topic.substr(0, slashPosition);
88 bool leafFound = slashPosition == std::string::npos;
89
90 if ((topicLevel == "#" && !topic.ends_with('#'))) {
91 success = false;
92 } else {
93 topic.erase(0, topicLevel.size() + 1);
94
95 success = topicLevels.insert({topicLevel, SubscribtionTree::TopicLevel(broker)})
96 .first->second.subscribe(clientId, qoS, topic, leafFound);
97 }
98 }
99
100 return success;
101 }

103 void SubscribtionTree::TopicLevel::publish(Message& message, std::string topic, bool leafFound) {
104 if (leafFound) {
105 LOG(TRACE) << "Found match:";
106 LOG(TRACE) << " Topic: '" << message.getTopic() << "';";
107 LOG(TRACE) << " Message: '" << message.getMessage() << "' ";
108 LOG(TRACE) << "Distribute Publish for match ...";
109
110 for (auto& [clientId, clientQoS] : qoSMap) {
111 broker->sendPublish(clientId, message, clientQoS, false);
112 }
113
114 LOG(TRACE) << "... completed!";
115
116 auto nextHashLevel = topicLevels.find("#");
117 if (nextHashLevel != topicLevels.end()) {
118 LOG(TRACE) << "Found parent match:";
119 LOG(TRACE) << " Topic: '" << message.getTopic() << "'";
120 LOG(TRACE) << " Message: '" << message.getMessage() << "'";
121 LOG(TRACE) << "Distribute Publish for match ...";
122
123 for (auto& [clientId, clientQoS] : nextHashLevel->second.qoSMap) {
124 broker->sendPublish(clientId, message, clientQoS, false);
125 }
126
127 LOG(TRACE) << "... completed!";
128 }
129 } else {
130 std::string::size_type slashPosition = topic.find('/');
131
132 std::string topicLevel = topic.substr(0, slashPosition);
133 bool leafFound = slashPosition == std::string::npos;
134
135 topic.erase(0, topicLevel.size() + 1);
136
137 auto foundNode = topicLevels.find(topicLevel);
138 if (foundNode != topicLevels.end()) {
139 foundNode->second.publish(message, topic, leafFound);
140 }
141
142 foundNode = topicLevels.find("+");
143 if (foundNode != topicLevels.end()) {
144 foundNode->second.publish(message, topic, leafFound);
145 }
146
147 foundNode = topicLevels.find("#");
148 if (foundNode != topicLevels.end()) {
149 LOG(TRACE) << "Found match for topic filter: '.../" << topicLevel << "/#', topic: '" << message.getTopic()
150 << "', Message: '" << message.getMessage() << "'";
151 LOG(TRACE) << "Distribute Publish ...";
152 for (auto& [clientId, clientQoS] : foundNode->second.qoSMap) {
153 broker->sendPublish(clientId, message, clientQoS, false);
154 }
155 LOG(TRACE) << "... completed!";
156 }
157 }
158 }

160 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId, std::string topic, bool leafFound) {
161 if (leafFound) {
162 qoSMap.erase(clientId);
163 } else {
164 std::string::size_type slashPosition = topic.find('/');
165
166 std::string topicLevel = topic.substr(0, slashPosition);
167 bool leafFound = slashPosition == std::string::npos;
168
169 topic.erase(0, topicLevel.size() + 1);
170
171 if (topicLevels.contains(topicLevel) && topicLevels.find(topicLevel)->second.unsubscribe(clientId, topic, leafFound)) {
172 topicLevels.erase(topicLevel);
173 }
174 }
175
176 return qoSMap.empty() && topicLevels.empty();
177 }
178
179 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId) {
180 qoSMap.erase(clientId);
181
182 for (auto it = topicLevels.begin(); it != topicLevels.end();) {
183 if (it->second.unsubscribe(clientId)) {
184 it = topicLevels.erase(it);
185 } else {
186 ++it;
187 }
188 }
189
190 return qoSMap.empty() && topicLevels.empty();
191 }

207 SubscribtionTree::TopicLevel& SubscribtionTree::TopicLevel::fromJson(const nlohmann::json& json) {
208 qoSMap.clear();
209 topicLevels.clear();
210
211 if (json.contains("qos_map")) {
212 for (const auto& subscriber : json["qos_map"].items()) {
213 qoSMap.emplace(subscriber.key(), subscriber.value());
214 }
215 }
216
217 if (json.contains("topic_filter")) {
218 for (const auto& topicLevelItem : json["topic_filter"].items()) {
219 topicLevels.emplace(topicLevelItem.key(), TopicLevel(broker).fromJson(topicLevelItem.value()));
220 }
221 }
222
223 return *this;
224 }
