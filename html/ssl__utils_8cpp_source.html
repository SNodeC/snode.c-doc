51 static int password_callback(char* buf, int size, int, void* u) {
52 strncpy(buf, static_cast<char*>(u), static_cast<std::size_t>(size));
53 buf[size - 1] = '\0';
54
55 memset(u, 0, ::strlen(static_cast<char*>(u))); // garble Password
56 free(u);
57
58 return static_cast<int>(::strlen(buf));
59 }

61 static int verify_callback(int preverify_ok, X509_STORE_CTX* ctx) {
62 X509* curr_cert = X509_STORE_CTX_get_current_cert(ctx);
63 int depth = X509_STORE_CTX_get_error_depth(ctx);
64
65 char buf[256];
66 X509_NAME_oneline(X509_get_subject_name(curr_cert), buf, 256);
67
68 if (preverify_ok) {
69 LOG(TRACE) << "SSL/TLS verify ok at depth=" << depth << ": " << buf;
70 } else {
71 int err = X509_STORE_CTX_get_error(ctx);
72
73 LOG(WARNING) << "SSL/TLS verify error at depth=" << depth << ": verifyErr=" << err << ", "
74 << X509_verify_cert_error_string(err);
75
76 /*
77 * At this point, err contains the last verification error. We can use
78 * it for something special
79 */
80
81 switch (err) {
82 case X509_V_ERR_UNABLE_TO_GET_ISSUER_CERT:
83 X509_NAME_oneline(X509_get_issuer_name(curr_cert), buf, 256);
84 LOG(WARNING) << "no issuer certificate for issuer= " << buf;
85 break;
86 default:
87 break;
88 }
89 }
90
91 return preverify_ok;
92 }

121 SSL_CTX* ssl_ctx_new(const SslConfig& sslConfig);
122 SSL_CTX* ssl_ctx_new(const SslConfig& sslConfig) {
123 static int sslSessionCtxId = 1;
124
125 SSL_CTX* ctx = SSL_CTX_new(sslConfig.server ? TLS_server_method() : TLS_client_method());
126
127 if (ctx != nullptr) {
128#if OPENSSL_VERSION_NUMBER >= 0x30000000L
129 SSL_CTX_set_options(ctx, SSL_OP_IGNORE_UNEXPECTED_EOF);
130#endif
131 SSL_CTX_set_read_ahead(ctx, 1);
132
133 SSL_CTX_set_mode(ctx, SSL_MODE_ENABLE_PARTIAL_WRITE);
134 SSL_CTX_set_mode(ctx, SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
135
136 bool sslErr = false;
137
138 if (sslConfig.server) {
139 SSL_CTX_set_session_id_context(ctx, reinterpret_cast<const unsigned char*>(&sslSessionCtxId), sizeof(sslSessionCtxId));
140 sslSessionCtxId++;
141 }
142 if (!sslConfig.caFile.empty() || !sslConfig.caDir.empty()) {
143 if (SSL_CTX_load_verify_locations(ctx,
144 !sslConfig.caFile.empty() ? sslConfig.caFile.c_str() : nullptr,
145 !sslConfig.caDir.empty() ? sslConfig.caDir.c_str() : nullptr) == 0) {
146 ssl_log_error("Can not load CA certificate or non default CA directory: ca-cert-file '" + sslConfig.caFile +
147 "' : ca-cert-dir '" + sslConfig.caDir + "'");
148 sslErr = true;
149 } else {
150 LOG(TRACE) << "CA certificate loaded: ca-cert-file '" << sslConfig.caFile << "' : ca-cert-dir '"
151 << sslConfig.caDir + "'";
152 }
153 } else {
154 LOG(TRACE) << "Neither using ca-cert-file nor ca-cert-dir";
155 }
156 if (!sslErr && sslConfig.useDefaultCaDir) {
157 if (SSL_CTX_set_default_verify_paths(ctx) == 0) {
158 ssl_log_error("Can not load default CA directory");
159 sslErr = true;
160 } else {
161 ssl_log_info("Using default CA directory");
162 }
163 } else {
164 LOG(TRACE) << "Not using default CA directory";
165 }
166 if (!sslErr) {
167 if (sslConfig.useDefaultCaDir || !sslConfig.caFile.empty() || !sslConfig.caDir.empty()) {
168 SSL_CTX_set_verify_depth(ctx, 5);
169 SSL_CTX_set_verify(ctx, SSL_VERIFY_FLAGS, verify_callback);
170 }
171 if (!sslConfig.certChain.empty()) {
172 if (SSL_CTX_use_certificate_chain_file(ctx, sslConfig.certChain.c_str()) == 0) {
173 ssl_log_error("Can not load certificate chain: " + sslConfig.certChain);
174 sslErr = true;
175 } else if (!sslConfig.certChainKey.empty()) {
176 if (!sslConfig.password.empty()) {
177 SSL_CTX_set_default_passwd_cb(ctx, password_callback);
178 SSL_CTX_set_default_passwd_cb_userdata(ctx, ::strdup(sslConfig.password.c_str()));
179 }
180 if (SSL_CTX_use_PrivateKey_file(ctx, sslConfig.certChainKey.c_str(), SSL_FILETYPE_PEM) == 0) {
181 ssl_log_error("Can not load private key: " + sslConfig.certChainKey);
182 sslErr = true;
183 } else if (!SSL_CTX_check_private_key(ctx)) {
184 ssl_log_error("Private key not consistent with CA files: " + sslConfig.certChainKey);
185 sslErr = true;
186 } else {
187 LOG(TRACE) << "Private key loaded: " << sslConfig.certChainKey;
188 }
189 }
190 if (!sslErr) {
191 LOG(TRACE) << "Certificate chain loaded: " << sslConfig.certChain;
192 }
193 }
194 }
195 if (!sslErr) {
196 if (sslConfig.sslOptions != 0) {
197 SSL_CTX_set_options(ctx, sslConfig.sslOptions);
198 }
199 if (!sslConfig.cipherList.empty()) {
200 SSL_CTX_set_cipher_list(ctx, sslConfig.cipherList.c_str());
201 }
202 }
203 if (sslErr) {
204 SSL_CTX_free(ctx);
205 ctx = nullptr;
206 }
207 } else {
208 ssl_log_error("SSL CTX not created");
209 }
210
211 return ctx;
212 }

222 SSL_CTX* ssl_ctx_new(const std::map<std::string, std::variant<std::string, bool, ssl_option_t>>& sniCert) {
223 SslConfig sslConfig(true);
224
225 for (auto& [key, value] : sniCert) {
226 if (key == "CertChain") {
227 sslConfig.certChain = std::get<std::string>(value);
228 } else if (key == "CertKey") {
229 sslConfig.certChainKey = std::get<std::string>(value);
230 } else if (key == "CertKeyPassword") {
231 sslConfig.password = std::get<std::string>(value);
232 } else if (key == "CaCertFile") {
233 sslConfig.caFile = std::get<std::string>(value);
234 } else if (key == "CaCertDir") {
235 sslConfig.caDir = std::get<std::string>(value);
236 } else if (key == "UseDefaultCaDir") {
237 sslConfig.useDefaultCaDir = std::get<bool>(value);
238 } else if (key == "CipherList") {
239 sslConfig.cipherList = std::get<std::string>(value);
240 } else if (key == "SslOptions") {
241 sslConfig.sslOptions = std::get<ssl_option_t>(value);
242 }
243 }
244
245 return ssl_ctx_new(sslConfig);
246 }

248 SSL_CTX* ssl_set_ssl_ctx(SSL* ssl, SSL_CTX* sslCtx) {
249 SSL_CTX* newSslCtx = SSL_set_SSL_CTX(ssl, sslCtx);
250 SSL_clear_options(ssl, 0xFFFFFFFFL);
251 SSL_set_options(ssl, SSL_CTX_get_options(sslCtx));
252 SSL_set_verify(ssl, SSL_CTX_get_verify_mode(sslCtx), SSL_CTX_get_verify_callback(sslCtx));
253 SSL_set_verify_depth(ssl, SSL_CTX_get_verify_depth(sslCtx));
254 SSL_set_mode(ssl, SSL_CTX_get_mode(sslCtx));
255
256 return newSslCtx;

271 std::set<std::string> ssl_get_sans(SSL_CTX* sslCtx) {
272 std::set<std::string> sans;
273
274 if (sslCtx != nullptr) {
275 X509* x509 = SSL_CTX_get0_certificate(sslCtx);
276 if (x509 != nullptr) {
277 GENERAL_NAMES* subjectAltNames =
278 static_cast<GENERAL_NAMES*>(X509_get_ext_d2i(x509, NID_subject_alt_name, nullptr, nullptr));
279#ifdef __clang__
280#pragma clang diagnostic push
281#pragma clang diagnostic ignored "-Wused-but-marked-unused"
282#endif
283 int32_t altNameCount = sk_GENERAL_NAME_num(subjectAltNames);
284#ifdef __clang__
285#pragma clang diagnostic pop
286#endif
287 for (int32_t i = 0; i < altNameCount; ++i) {
288#ifdef __clang__
289#pragma clang diagnostic push
290#pragma clang diagnostic ignored "-Wused-but-marked-unused"
291#endif
292 GENERAL_NAME* generalName = sk_GENERAL_NAME_value(subjectAltNames, i);
293#ifdef __clang__
294#pragma clang diagnostic pop
295#endif
296 if (generalName->type == GEN_DNS || generalName->type == GEN_URI || generalName->type == GEN_EMAIL) {
297 std::string subjectAltName =
298 std::string(reinterpret_cast<const char*>(ASN1_STRING_get0_data(generalName->d.dNSName)),
299 static_cast<std::size_t>(ASN1_STRING_length(generalName->d.dNSName)));
300 sans.insert(subjectAltName);
301 }
302 }
303#ifdef __clang__
304#pragma clang diagnostic push
305#pragma clang diagnostic ignored "-Wused-but-marked-unused"
306#endif
307 sk_GENERAL_NAME_pop_free(subjectAltNames, GENERAL_NAME_free);
308#ifdef __clang__
309#pragma clang diagnostic pop
310#endif
311 }
312 }
313
314 return sans;
315 }

318 std::string ssl_get_servername_from_client_hello(SSL* ssl) {
319 const unsigned char* ext;
320 size_t ext_len;
321 size_t p = 0;
322 size_t server_name_list_len;
323 size_t server_name_len;
324
325 if (!SSL_client_hello_get0_ext(ssl, TLSEXT_TYPE_server_name, &ext, &ext_len)) {
326 return "";
327 }
328
329 /* length (2 bytes) + type (1) + length (2) + server name (1+) */
330 if (ext_len < 6) {
331 return "";
332 }
333
334 /* Fetch Server Name list length */
335 server_name_list_len = static_cast<size_t>((ext[p] << 8) + ext[p + 1]);
336 p += 2;
337 if (p + server_name_list_len != ext_len) {
338 return "";
339 }
340
341 /* Fetch Server Name Type */
342 if (ext[p] != TLSEXT_NAMETYPE_host_name) {
343 return "";
344 }
345 p++;
346
347 /* Fetch Server Name Length */
348 server_name_len = static_cast<size_t>((ext[p] << 8) + ext[p + 1]);
349 p += 2;
350 if (p + server_name_len != ext_len) {
351 return "";
352 }
353
354 /* ext_len >= 6 && p == 5 */
355
356 /* Finally fetch Server Name */
357
358 return std::string(reinterpret_cast<const char*>(ext + p), ext_len - p);
359 }

361 void ssl_log(const std::string& message, int sslErr) {
362 utils::PreserveErrno preserveErrno;
363
364 switch (sslErr) {
365 case SSL_ERROR_NONE:
366 [[fallthrough]];
367 case SSL_ERROR_ZERO_RETURN:
368 ssl_log_info(message);
369 break;
370 case SSL_ERROR_SYSCALL:
371 if (errno != 0) {
372 ssl_log_error(message);
373 } else {
374 ssl_log_info(message);
375 }
376 break;
377 default:
378 ssl_log_warning(message);
379 break;
380 }
381 }
