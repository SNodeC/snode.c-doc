43 static int password_callback(char* buf, int size, [[maybe_unused]] int a, void* u) {
44 strncpy(buf, static_cast<char*>(u), static_cast<std::size_t>(size));
45 buf[size - 1] = '\0';
46
47 memset(u, 0, ::strlen(static_cast<char*>(u))); // garble Password
48 free(u);
49
50 return static_cast<int>(std::strlen(buf));
51 }

53 static int verify_callback(int preverify_ok, X509_STORE_CTX* ctx) {
54 X509* curr_cert = X509_STORE_CTX_get_current_cert(ctx);
55 const int depth = X509_STORE_CTX_get_error_depth(ctx);
56
57 char buf[256];
58 X509_NAME_oneline(X509_get_subject_name(curr_cert), buf, 256);
59
60 if (preverify_ok != 0) {
61 LOG(TRACE) << " SSL/TLS: Verify success at depth=" << depth << ": " << buf;
62 } else {
63 const int err = X509_STORE_CTX_get_error(ctx);
64
65 LOG(TRACE) << " SSL/TLS: Verify error at depth=" << depth << ": verifyErr=" << err << ", "
66 << X509_verify_cert_error_string(err);
67
68 /*
69 * At this point, err contains the last verification error. We can use
70 * it for something special
71 */
72
73 X509_NAME_oneline(X509_get_issuer_name(curr_cert), buf, 256);
74 LOG(TRACE) << " SSL/TLS rejected: " << X509_verify_cert_error_string(err) << " '" << buf << "'";
75 }
76
77 return preverify_ok;
78 }

84 SSL_CTX* ssl_ctx_new(const SslConfig& sslConfig) {
85 static int sslSessionCtxId = 1;
86
87 SSL_CTX* ctx = SSL_CTX_new(sslConfig.server ? TLS_server_method() : TLS_client_method());
88
89 if (ctx != nullptr) {
90 SSL_CTX_set_read_ahead(ctx, 1);
91
92 SSL_CTX_set_mode(ctx, SSL_MODE_ENABLE_PARTIAL_WRITE);
93 SSL_CTX_set_mode(ctx, SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
94
95 bool sslErr = false;
96
97 if (sslConfig.server) {
98 SSL_CTX_set_session_id_context(ctx, reinterpret_cast<const unsigned char*>(&sslSessionCtxId), sizeof(sslSessionCtxId));
99 sslSessionCtxId++;
100 }
101 if (!sslConfig.caCert.empty() || !sslConfig.caCertDir.empty()) {
102 if (SSL_CTX_load_verify_locations(ctx,
103 !sslConfig.caCert.empty() ? sslConfig.caCert.c_str() : nullptr,
104 !sslConfig.caCertDir.empty() ? sslConfig.caCertDir.c_str() : nullptr) == 0) {
105 ssl_log_info(sslConfig.instanceName + " SSL/TLS: CA certificate error loading file '" + sslConfig.caCert + "', dir '" +
106 sslConfig.caCertDir + "'");
107 sslErr = true;
108 } else {
109 if (!sslConfig.caCert.empty()) {
110 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificate loaded";
111 LOG(TRACE) << " " << sslConfig.caCert;
112 } else {
113 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificate not loaded from a file";
114 }
115 if (!sslConfig.caCertDir.empty()) {
116 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificates load from";
117 LOG(TRACE) << " " << sslConfig.caCertDir;
118 } else {
119 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificates not loaded from a directory";
120 }
121 }
122 } else {
123 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificate not loaded from a file";
124 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificates not loaded from a directory";
125 }
126 if (!sslErr && sslConfig.caCertUseDefaultDir) {
127 if (SSL_CTX_set_default_verify_paths(ctx) == 0) {
128 ssl_log_info(sslConfig.instanceName + " SSL/TLS: CA certificates error load from default openssl CA directory");
129 sslErr = true;
130 } else {
131 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificates enabled load from default openssl CA directory";
132 }
133 } else {
134 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA certificates not loaded from default openssl CA directory";
135 }
136 if (!sslErr) {
137 SSL_CTX_set_verify_depth(ctx, 5);
138 SSL_CTX_set_verify(ctx,
139 (sslConfig.caCertAcceptUnknown ? SSL_VERIFY_NONE
140 : (!sslConfig.caCert.empty() || !sslConfig.caCertDir.empty() || sslConfig.caCertUseDefaultDir)
141 ? SSL_VERIFY_PEER | SSL_VERIFY_CLIENT_ONCE | SSL_VERIFY_FAIL_IF_NO_PEER_CERT
142 : 0),
143 verify_callback);
144 if ((SSL_CTX_get_verify_mode(ctx) & SSL_VERIFY_PEER) != 0) {
145 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: CA requested verify";
146 }
147 if (!sslConfig.cert.empty()) {
148 if (SSL_CTX_use_certificate_chain_file(ctx, sslConfig.cert.c_str()) == 0) {
149 ssl_log_info(sslConfig.instanceName + " SSL/TLS: Cert chain error loading from file '" + sslConfig.cert + "'");
150 sslErr = true;
151 } else if (!sslConfig.certKey.empty()) {
152 if (!sslConfig.password.empty()) {
153 SSL_CTX_set_default_passwd_cb(ctx, password_callback);
154 SSL_CTX_set_default_passwd_cb_userdata(ctx, ::strdup(sslConfig.password.c_str()));
155 }
156 if (SSL_CTX_use_PrivateKey_file(ctx, sslConfig.certKey.c_str(), SSL_FILETYPE_PEM) == 0) {
157 ssl_log_info(sslConfig.instanceName + " SSL/certKey: Cert chain key error loading file '" + sslConfig.certKey +
158 "'");
159 sslErr = true;
160 } else if (SSL_CTX_check_private_key(ctx) != 1) {
161 ssl_log_info(sslConfig.instanceName + " SSL/TLS: Cert chain key error");
162 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: Cert chain not loaded";
163 LOG(TRACE) << " " << sslConfig.certKey;
164 sslErr = true;
165 } else {
166 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: Cert chain key loaded";
167 LOG(TRACE) << " " << sslConfig.certKey;
168
169 LOG(TRACE) << sslConfig.instanceName << " SSL/TLS: Cert chain loaded";
170 LOG(TRACE) << " " << sslConfig.cert;
171 }
172 }
173 }
174 }
175 if (!sslErr) {
176 if (sslConfig.sslOptions != 0) {
177 SSL_CTX_set_options(ctx, sslConfig.sslOptions);
178 }
179 if (!sslConfig.cipherList.empty()) {
180 SSL_CTX_set_cipher_list(ctx, sslConfig.cipherList.c_str());
181 }
182 } else {
183 SSL_CTX_free(ctx);
184 ctx = nullptr;
185 }
186 }
187
188 return ctx;
189 }

191 std::map<std::string, SSL_CTX*> ssl_get_sans(SSL_CTX* sslCtx) {
192 std::map<std::string, SSL_CTX*> sans;
193
194 if (sslCtx != nullptr) {
195 X509* x509 = SSL_CTX_get0_certificate(sslCtx);
196 if (x509 != nullptr) {
197 GENERAL_NAMES* subjectAltNames =
198 static_cast<GENERAL_NAMES*>(X509_get_ext_d2i(x509, NID_subject_alt_name, nullptr, nullptr));
199#ifdef __GNUC__
200#pragma GCC diagnostic push
201#ifdef __has_warning
202#if __has_warning("-Wused-but-marked-unused")
203#pragma GCC diagnostic ignored "-Wused-but-marked-unused"
204#endif
205#endif
206#endif
207 const int32_t altNameCount = sk_GENERAL_NAME_num(subjectAltNames);
208#ifdef __GNUC__
209#pragma GCC diagnostic pop
210#endif
211 for (int32_t i = 0; i < altNameCount; ++i) {
212#ifdef __GNUC__
213#pragma GCC diagnostic push
214#ifdef __has_warning
215#if __has_warning("-Wused-but-marked-unused")
216#pragma GCC diagnostic ignored "-Wused-but-marked-unused"
217#endif
218#endif
219#endif
220 GENERAL_NAME* generalName = sk_GENERAL_NAME_value(subjectAltNames, i);
221#ifdef __GNUC__
222#pragma GCC diagnostic pop
223#endif
224 if (generalName->type == GEN_DNS || generalName->type == GEN_URI || generalName->type == GEN_EMAIL) {
225 const std::string subjectAltName =
226 std::string(reinterpret_cast<const char*>(ASN1_STRING_get0_data(generalName->d.dNSName)),
227 static_cast<std::size_t>(ASN1_STRING_length(generalName->d.dNSName)));
228 sans.insert({subjectAltName, sslCtx});
229 }
230 }
231#ifdef __GNUC__
232#pragma GCC diagnostic push
233#ifdef __has_warning
234#if __has_warning("-Wused-but-marked-unused")
235#pragma GCC diagnostic ignored "-Wused-but-marked-unused"
236#endif
237#endif
238#endif
239 sk_GENERAL_NAME_pop_free(subjectAltNames, GENERAL_NAME_free);
240#ifdef __GNUC__
241#pragma GCC diagnostic pop
242#endif
243 }
244 }
245
246 return sans;
247 }

255 SSL_CTX* ssl_set_ssl_ctx(SSL* ssl, SSL_CTX* sslCtx) {
256 SSL_CTX* newSslCtx = SSL_set_SSL_CTX(ssl, sslCtx);
257 SSL_clear_options(ssl, 0xFFFFFFFFL);
258 SSL_set_options(ssl, SSL_CTX_get_options(sslCtx));
259 SSL_set_verify(ssl, SSL_CTX_get_verify_mode(sslCtx), SSL_CTX_get_verify_callback(sslCtx));
260 SSL_set_verify_depth(ssl, SSL_CTX_get_verify_depth(sslCtx));
261 SSL_set_mode(ssl, SSL_CTX_get_mode(sslCtx));
262
263 return newSslCtx;

272 // From: https://www.bit-hive.com/documents/openssl-tutorial/
273 std::string ssl_get_servername_from_client_hello(SSL* ssl) {
274 const unsigned char* ext = nullptr;
275 size_t ext_len = 0;
276 size_t p = 0;
277 size_t server_name_list_len = 0;
278 size_t server_name_len = 0;
279
280 if (SSL_client_hello_get0_ext(ssl, TLSEXT_TYPE_server_name, &ext, &ext_len) == 0) {
281 return "";
282 }
283
284 /* length (2 bytes) + type (1) + length (2) + server name (1+) */
285 if (ext_len < 6) {
286 return "";
287 }
288
289 /* Fetch Server Name list length */
290 server_name_list_len = static_cast<size_t>((ext[p] << 8) + ext[p + 1]);
291 p += 2;
292 if (p + server_name_list_len != ext_len) {
293 return "";
294 }
295
296 /* Fetch Server Name Type */
297 if (ext[p] != TLSEXT_NAMETYPE_host_name) {
298 return "";
299 }
300 p++;
301
302 /* Fetch Server Name Length */
303 server_name_len = static_cast<size_t>((ext[p] << 8) + ext[p + 1]);
304 p += 2;
305 if (p + server_name_len != ext_len) {
306 return "";
307 }
308
309 /* ext_len >= 6 && p == 5 */
310
311 /* Finally fetch Server Name */
312
313 return std::string(reinterpret_cast<const char*>(ext + p), ext_len - p);
314 }

316 void ssl_log(const std::string& message, int sslErr) {
317 const utils::PreserveErrno preserveErrno;
318
319 switch (sslErr) {
320 case SSL_ERROR_NONE:
321 [[fallthrough]];
322 case SSL_ERROR_ZERO_RETURN:
323 ssl_log_info(message);
324 break;
325 case SSL_ERROR_SYSCALL:
326 if (errno != 0) {
327 ssl_log_error(message);
328 } else {
329 ssl_log_info(message);
330 }
331 break;
332 default:
333 ssl_log_warning(message);
334 break;
335 }

368 bool match(const char* first, const char* second) {
369 // If we reach at the end of both strings, we are done
370 if (*first == '\0' && *second == '\0') {
371 return true;
372 }
373
374 // Make sure to eliminate consecutive '*'
375 if (*first == '*') {
376 while (*(first + 1) == '*') {
377 first++;
378 }
379 }
380
381 // Make sure that the characters after '*' are present
382 // in second string. This function assumes that the
383 // first string will not contain two consecutive '*'
384 if (*first == '*' && *(first + 1) != '\0' && *second == '\0') {
385 return false;
386 }
387
388 // If the first string contains '?', or current
389 // characters of both strings match
390 if (*first == '?' || *first == *second) {
391 return match(first + 1, second + 1);
392 }
393
394 // If there is *, then there are two possibilities
395 // a) We consider current character of second string
396 // b) We ignore current character of second string.
397 if (*first == '*') {
398 return match(first + 1, second) || match(first, second + 1);
399 }
400
401 return false;
402 }
