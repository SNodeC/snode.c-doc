37 template <typename PhysicalClientSocket, typename Config>
38 core::socket::stream::tls::SocketConnector<PhysicalClientSocket, Config>::SocketConnector(
39 const std::shared_ptr<SocketContextFactory>& socketContextFactory,
40 const std::function<void(SocketConnection*)>& onConnect,
41 const std::function<void(SocketConnection*)>& onConnected,
42 const std::function<void(SocketConnection*)>& onDisconnect,
43 const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
44 const std::shared_ptr<Config>& config)
45 : Super(
46 socketContextFactory,
47 [onConnect, this](SocketConnection* socketConnection) -> void { // onConnect
48 onConnect(socketConnection);
49
50 SSL* ssl = socketConnection->startSSL(socketConnection->getFd(),
51 this->config->getSslCtx(),
52 Super::config->getInitTimeout(),
53 Super::config->getShutdownTimeout());
54
55 if (ssl != nullptr) {
56 SSL_set_connect_state(ssl);
57 ssl_set_sni(ssl, Super::config->getSni());
58 }
59 },
60 [socketContextFactory, onConnected, this](SocketConnection* socketConnection) -> void { // onConnected
61 if (socketConnection->doSSLHandshake(
62 [socketContextFactory, onConnected, socketConnection, instanceName = Super::config->getInstanceName()]()
63 -> void { // onSuccess
64 LOG(TRACE) << instanceName << " SSL/TLS: Initial handshake success";
65
66 onConnected(socketConnection);
67 socketConnection->connectSocketContext(socketContextFactory);
68 },
69 [socketConnection, instanceName = Super::config->getInstanceName()]() -> void { // onTimeout
70 LOG(TRACE) << instanceName << " SSL/TLS: Initial handshake timed out";
71
72 socketConnection->close();
73 },
74 [socketConnection, instanceName = Super::config->getInstanceName()](int sslErr) -> void { // onError
75 ssl_log(instanceName + " SSL/TLS: Initial handshake failed", sslErr);
76
77 socketConnection->close();
78 })) {
79 LOG(TRACE) << Super::config->getInstanceName() << " SSL/TLS: Initial handshake running";
80 } else {
81 ssl_log_error(Super::config->getInstanceName() + " SSL/TLS: Initialization failed");
82
83 socketConnection->close();
84 }
85 },
86 [onDisconnect, instanceName = config->getInstanceName()](SocketConnection* socketConnection) -> void { // onDisconnect
87 socketConnection->stopSSL();
88 onDisconnect(socketConnection);
89 },
90 onStatus,
91 config) {
92 }

109 template <typename PhysicalSocketClient, typename Config>
110 void SocketConnector<PhysicalSocketClient, Config>::init() {
111 if (!config->getDisabled()) {
112 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: SSL_CTX creating ...";
113
114 const SSL_CTX* ctx = config->getSslCtx();
115
116 if (ctx != nullptr) {
117 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: SSL_CTX created";
118 Super::init();
119 } else {
120 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: SSL_CTX creation failed";
121
122 Super::onStatus(config->Remote::getSocketAddress(), core::socket::STATE_FATAL);
123 Super::destruct();
124 }
125 } else {
126 Super::init();
127 }
128 }
