59 template <typename PhysicalClientSocket, typename Config>
60 SocketConnector<PhysicalClientSocket, Config>::SocketConnector(
61 const std::shared_ptr<SocketContextFactory>& socketContextFactory,
62 const std::function<void(SocketConnection*)>& onConnect,
63 const std::function<void(SocketConnection*)>& onConnected,
64 const std::function<void(SocketConnection*)>& onDisconnect,
65 const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
66 const std::shared_ptr<Config>& config)
67 : Super(
68 socketContextFactory,
69 [onConnect, this](SocketConnection* socketConnection) { // onConnect
70 onConnect(socketConnection);
71
72 SSL* ssl = socketConnection->startSSL(socketConnection->getFd(),
73 Super::config->getSslCtx(),
74 Super::config->getInitTimeout(),
75 Super::config->getShutdownTimeout(),
76 !Super::config->getNoCloseNotifyIsEOF());
77 if (ssl != nullptr) {
78 SSL_set_connect_state(ssl);
79 SSL_set_ex_data(ssl, 1, Super::config.get());
80
81 ssl_set_sni(ssl, Super::config->getSni());
82 }
83 },
84 [socketContextFactory, onConnected](SocketConnection* socketConnection) { // onConnected
85 LOG(TRACE) << socketConnection->getConnectionName() << " SSL/TLS: Start handshake";
86 if (!socketConnection->doSSLHandshake(
87 [socketContextFactory,
88 onConnected,
89 socketConnection]() { // onSuccess
90 LOG(DEBUG) << socketConnection->getConnectionName() << " SSL/TLS: Handshake success";
91
92 onConnected(socketConnection);
93
94 socketConnection->connectSocketContext(socketContextFactory);
95
96 },
97 [socketConnection]() { // onTimeout
98 LOG(ERROR) << socketConnection->getConnectionName() << " SSL/TLS: Handshake timed out";
99
100 socketConnection->close();
101 },
102 [socketConnection](int sslErr) { // onError
103 ssl_log(socketConnection->getConnectionName() + " SSL/TLS: Handshake failed", sslErr);
104
105 socketConnection->close();
106 })) {
107 LOG(ERROR) << socketConnection->getConnectionName() + " SSL/TLS: Handshake failed";
108
109 socketConnection->close();
110 }
111 },
112 [onDisconnect](SocketConnection* socketConnection) { // onDisconnect
113 socketConnection->stopSSL();
114 onDisconnect(socketConnection);
115 },
116 onStatus,
117 config) {
118 }

130 template <typename PhysicalSocketClient, typename Config>
131 void SocketConnector<PhysicalSocketClient, Config>::init() {
132 if (core::eventLoopState() == core::State::RUNNING && !config->getDisabled()) {
133 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: SSL_CTX creating ...";
134
135 if (config->getSslCtx() != nullptr) {
136 LOG(DEBUG) << config->getInstanceName() << " SSL/TLS: SSL_CTX created";
137
138 Super::init();
139 } else {
140 LOG(ERROR) << config->getInstanceName() << " SSL/TLS: SSL_CTX creation failed";
141
142 Super::onStatus(config->Remote::getSocketAddress(), core::socket::STATE_FATAL);
143 Super::destruct();
144 }
145 } else {
146 Super::init();
147 }
148 }
