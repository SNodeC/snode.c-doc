62 std::size_t UIntV::deserialize(MqttContext* mqttContext) {
63 std::size_t consumed = 0;
64
65 do {
66 char byte = 0;
67 consumed = mqttContext->recv(&byte, 1);
68
69 if (consumed > 0) {
70 value.push_back(byte);
71
72 if (value.size() > sizeof(uint32_t)) {
73 error = true;
74 } else {
75 complete = (byte & 0x80) == 0;
76 }
77 }
78 } while (consumed > 0 && !complete && !error);
79
80 return consumed;
81 }

83 UIntV& UIntV::operator=(const uint32_t& newValue) {
84 uint32_t remainingValue = newValue;
85 value.clear();
86
87 do {
88 char encodedByte = static_cast<char>(remainingValue % 0x80);
89
90 remainingValue /= 0x80;
91 if (remainingValue > 0) {
92 encodedByte |= static_cast<char>(0x80);
93 }
94
95 value.push_back(encodedByte);
96 } while (remainingValue > 0);
97
98 return *this;
99 }

101 UIntV::operator uint32_t() const {
102 uint32_t uint32Value = 0;
103 uint32_t multiplicator = 1;
104
105 for (std::size_t i = 0; i < value.size(); i++, multiplicator *= 0x80) {
106 uint32Value += static_cast<uint8_t>(value[i] & 0x7F) * multiplicator;
107 }
108
109 return uint32Value;
110 }
