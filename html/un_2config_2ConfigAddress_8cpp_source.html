61 template <template <typename SocketAddress> typename ConfigAddressType>
62 ConfigAddress<ConfigAddressType>::ConfigAddress(net::config::ConfigInstance* instance,
63 const std::string& addressOptionName,
64 const std::string& addressOptionDescription)
65 : Super(instance, addressOptionName, addressOptionDescription) {
66 sunPathOpt = Super::addOption( //
67 "--sun-path",
68 "Unix domain bind path",
69 "filename:FILE",
70 std::string('\0' + instance->getInstanceName() + std::to_string(getpid()) + "_" + utils::Uuid::getUuid()));
71 }

73 template <template <typename SocketAddress> typename ConfigAddressType>
74 SocketAddress* ConfigAddress<ConfigAddressType>::init() {
75 SocketAddress* socketAddress = new SocketAddress(sunPathOpt->as<std::string>());
76
77 try {
78 socketAddress->init();
79 } catch (const core::socket::SocketAddress::BadSocketAddress&) {
80 delete socketAddress;
81 socketAddress = nullptr;
82
83 throw;
84 }
85
86 return socketAddress;
87 }
