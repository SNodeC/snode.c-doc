73 void DescriptorEventPublisher::releaseDisabledEvents(const utils::Timeval& currentTime) {
74 if (observedEventReceiversDirty) {
75 observedEventReceiversDirty = false;
76
77 for (auto& [fd, observedEventReceiverList] : observedEventReceivers) {
78 DescriptorEventReceiver* beforeFirst = observedEventReceiverList.front();
79 if (std::erase_if(observedEventReceiverList, [](DescriptorEventReceiver* descriptorEventReceiver) -> bool {
80 bool isDisabled = !descriptorEventReceiver->isEnabled();
81 if (isDisabled) {
82 descriptorEventReceiver->setDisabled();
83 if (!descriptorEventReceiver->isObserved()) {
84 descriptorEventReceiver->unobservedEvent();
85 }
86 }
87 return isDisabled;
88 }) > 0) {
89 if (observedEventReceiverList.empty()) {
90 muxDel(fd);
91 } else {
92 DescriptorEventReceiver* afterFirst = observedEventReceiverList.front();
93 if (beforeFirst != afterFirst) {
94 afterFirst->triggered(currentTime);
95 if (!afterFirst->isSuspended()) {
96 muxOn(afterFirst);
97 } else {
98 muxOff(afterFirst);
99 }
100 }
101 }
102 }
103 }
104
105 std::erase_if(observedEventReceivers, [](auto& observedEventReceiversEntry) -> bool {
106 return observedEventReceiversEntry.second.empty();
107 });
108 }
109 }

125 utils::Timeval DescriptorEventPublisher::getNextTimeout(const utils::Timeval& currentTime) const {
126 utils::Timeval nextTimeout = DescriptorEventReceiver::TIMEOUT::MAX;
127
128 if (!observedEventReceiversDirty) {
129 for (auto& [fd, eventReceivers] : observedEventReceivers) {
130 nextTimeout = std::min(eventReceivers.front()->getTimeout(currentTime), nextTimeout);
131 }
132 } else {
133 nextTimeout = 0;
134 }
135
136 return nextTimeout;
137 }
