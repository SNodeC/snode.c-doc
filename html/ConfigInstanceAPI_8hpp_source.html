59 template <typename T>
60 std::shared_ptr<utils::AppWithPtr<T>> Instance(const std::string& name, const std::string& description, T* section) {
61 std::shared_ptr<utils::AppWithPtr<T>> instanceSc = std::make_shared<utils::AppWithPtr<T>>(description, name, section);
62
63 instanceSc->option_defaults()->take_last();
64 instanceSc->formatter(utils::Config::sectionFormatter);
65 instanceSc->option_defaults()->group(instanceSc->get_formatter()->get_label("Nonpersistent Options"));
66
67 return instanceSc;
68 }

70 template <typename SectionType>
71 SectionType* ConfigInstance::getSection() const {
72 utils::AppWithPtr<SectionType>* sectionApp =
73 dynamic_cast<utils::AppWithPtr<SectionType>*>(instanceSc->get_subcommand_no_throw(std::string(SectionType::name)));
74
75 return sectionApp != nullptr ? sectionApp->getPtr() : nullptr;
76 }

78 template <typename ConcreteConfigSection, typename... Args>
79 ConcreteConfigSection* ConfigInstance::addSection(Args&&... args) {
80 return dynamic_cast<ConcreteConfigSection*>(addSection(std::make_shared<ConcreteConfigSection>(this, std::forward<Args>(args)...)));
81 }
