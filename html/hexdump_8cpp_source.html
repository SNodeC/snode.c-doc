35 // From: https://gist.github.com/shreyasbharath/32a8092666303a916e24a81b18af146b
36 std::string hexDump(const std::vector<char>& bytes, int prefixLength) {
37 std::stringstream hexStream;
38
39 if (!bytes.empty()) {
40 uint8_t buff[17];
41 size_t i = 0;
42
43 hexStream << std::hex;
44
45 int currentPrefixLength = 0;
46
47 // Process every byte in the data.
48 for (i = 0; i < bytes.size(); i++) {
49 // Multiple of 16 means new line (with line offset).
50
51 if ((i % 16) == 0) {
52 // Just don't print ASCII for the zeroth line.
53 if (i != 0) {
54 hexStream << " " << buff << std::endl;
55 }
56
57 // Output the offset.
58 hexStream << Color::Code::FG_BLUE;
59 hexStream << std::setw(currentPrefixLength) << std::setfill(' ') << ""
60 << ": " << std::setw(8) << std::setfill('0') << static_cast<unsigned int>(i);
61 hexStream << Color::Code::FG_DEFAULT << " ";
62 currentPrefixLength = prefixLength;
63 }
64
65 // Now the hex code for the specific character.
66 hexStream << Color::Code::FG_GREEN;
67 hexStream << " " << std::setw(2) << std::setfill('0') << static_cast<unsigned int>(static_cast<unsigned char>(bytes[i]));
68 hexStream << Color::Code::FG_DEFAULT;
69
70 // And store a printable ASCII character for later.
71 if ((bytes[i] < 0x20) || (bytes[i] > 0x7e)) {
72 buff[i % 16] = '.';
73 } else {
74 buff[i % 16] = static_cast<uint8_t>(bytes[i]);
75 }
76 buff[(i % 16) + 1] = '\0';
77 }
78
79 hexStream << std::dec;
80
81 // Pad out last line if not exactly 16 characters.
82 while ((i % 16) != 0) {
83 hexStream << " ";
84 i++;
85 }
86
87 // And print the final ASCII bit.
88 hexStream << " " << buff;
89 }
90
91 return hexStream.str();
92 }
