45 void PipeSink::readEvent() {
46 // NOLINTNEXTLINE(cppcoreguidelines-avoid-c-arrays, hicpp-avoid-c-arrays, modernize-avoid-c-arrays)
47 static char junk[MAX_READ_JUNKSIZE];
48
49 ssize_t ret = core::system::read(getRegisteredFd(), junk, MAX_READ_JUNKSIZE);
50
51 if (ret > 0) {
52 if (onData) {
53 onData(junk, static_cast<std::size_t>(ret));
54 }
55 } else {
56 ReadEventReceiver::disable();
57
58 if (ret == 0) {
59 if (onEof) {
60 onEof();
61 }
62 } else {
63 if (onError) {
64 onError(errno);
65 }
66 }
67 }
68 }
