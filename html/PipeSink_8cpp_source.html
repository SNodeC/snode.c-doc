71 void PipeSink::readEvent() {
72 // NOLINTNEXTLINE(cppcoreguidelines-avoid-c-arrays, hicpp-avoid-c-arrays, modernize-avoid-c-arrays)
73 static char chunk[MAX_READ_JUNKSIZE];
74
75 const ssize_t ret = core::system::read(getRegisteredFd(), chunk, MAX_READ_JUNKSIZE);
76
77 if (ret > 0) {
78 if (onData) {
79 onData(chunk, static_cast<std::size_t>(ret));
80 }
81 } else {
82 ReadEventReceiver::disable();
83
84 if (ret == 0) {
85 if (onEof) {
86 onEof();
87 }
88 } else {
89 if (onError) {
90 onError(errno);
91 }
92 }
93 }
94 }
