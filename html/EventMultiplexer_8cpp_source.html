69 TickStatus EventMultiplexer::tick(const utils::Timeval& tickTimeOut, const sigset_t& sigMask) {
70 utils::Timeval currentTime = utils::Timeval::currentTime();
71
72 int activeDescriptorCount = 0;
73
74 TickStatus tickStatus = waitForEvents(tickTimeOut, currentTime, sigMask, activeDescriptorCount);
75
76 if (tickStatus == TickStatus::SUCCESS) {
77 spanActiveEvents(currentTime, activeDescriptorCount);
78 executeEventQueue(currentTime);
79 checkTimedOutEvents(currentTime);
80 releaseExpiredResources(currentTime);
81 }
82
83 return tickStatus;
84 }

105 TickStatus EventMultiplexer::waitForEvents(const utils::Timeval& tickTimeOut,
106 const utils::Timeval& currentTime,
107 const sigset_t& sigMask,
108 int& activeDescriptorCount) {
109 TickStatus tickStatus = TickStatus::SUCCESS;
110
111 if (observedEventReceiverCount() > 0 || !timerEventPublisher->empty() || !eventQueue.empty()) {
112 utils::Timeval nextTimeout = std::min(getNextTimeout(currentTime), tickTimeOut);
113
114 activeDescriptorCount = monitorDescriptors(nextTimeout, sigMask);
115
116 if (activeDescriptorCount < 0 && errno != EINTR) {
117 tickStatus = TickStatus::TRACE;
118 } else if (errno == EINTR) {
119 tickStatus = TickStatus::INTERRUPTED;
120 }
121 } else {
122 tickStatus = TickStatus::NOOBSERVER;
123 }
124
125 return tickStatus;
126 }

151 utils::Timeval EventMultiplexer::getNextTimeout(const utils::Timeval& currentTime) {
152 utils::Timeval nextTimeout = DescriptorEventReceiver::TIMEOUT::MAX;
153
154 if (eventQueue.empty()) {
155 for (DescriptorEventPublisher* const descriptorEventPublisher : descriptorEventPublishers) {
156 nextTimeout = std::min(descriptorEventPublisher->getNextTimeout(currentTime), nextTimeout);
157 }
158 nextTimeout = std::min(timerEventPublisher->getNextTimeout(currentTime), nextTimeout);
159 nextTimeout = std::max(nextTimeout, utils::Timeval()); // In case nextTimeout is negative
160 } else {
161 nextTimeout = 0;
162 }
163
164 return nextTimeout;
165 }

167 int EventMultiplexer::observedEventReceiverCount() {
168 return std::accumulate(descriptorEventPublishers.begin(),
169 descriptorEventPublishers.end(),
