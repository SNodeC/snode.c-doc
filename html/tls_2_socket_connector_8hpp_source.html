34 template <typename PhysicalClientSocket, typename Config>
35 core::socket::stream::tls::SocketConnector<PhysicalClientSocket, Config>::SocketConnector(
36 const std::shared_ptr<SocketContextFactory>& socketContextFactory,
37 const std::function<void(SocketConnection*)>& onConnect,
38 const std::function<void(SocketConnection*)>& onConnected,
39 const std::function<void(SocketConnection*)>& onDisconnect,
40 const std::function<void(const core::ProgressLog&)>& onError,
41 const std::shared_ptr<Config>& config)
42 : Super(
43 socketContextFactory,
44 [onConnect, this](SocketConnection* socketConnection) -> void { // onConnect
45 socketConnection->startSSL(this->ctx, this->config->getInitTimeout(), this->config->getShutdownTimeout());
46 onConnect(socketConnection);
47 },
48 [onConnected, this](SocketConnection* socketConnection) -> void { // onConnect
49 SSL* ssl = socketConnection->getSSL();
50
51 if (ssl != nullptr) {
52 SSL_set_connect_state(ssl);
53 ssl_set_sni(ssl, this->config);
54
55 socketConnection->doSSLHandshake(
56 [onConnected, socketConnection]() -> void { // onSuccess
57 LOG(INFO) << "SSL/TLS initial handshake success";
58 onConnected(socketConnection);
59 socketConnection->onConnected();
60 },
61 [onError = this->onError, progressLog = Super::progressLog, config = this->config]() -> void { // onTimeout
62 LOG(WARNING) << "SSL/TLS initial handshake timed out";
63 },
64 [onError = this->onError, progressLog = Super::progressLog, config = this->config](
65 int sslErr) -> void { // onError
66 ssl_log("SSL/TLS initial handshake failed", sslErr);
67 });
68 } else {
69 socketConnection->close();
70 ssl_log_error("SSL/TLS initialization failed");
71 this->onError(*Super::progressLog.get());
72 }
73 },
74 [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
75 socketConnection->stopSSL();
76 onDisconnect(socketConnection);
77 },
78 onError,
79 config) {
80 }
