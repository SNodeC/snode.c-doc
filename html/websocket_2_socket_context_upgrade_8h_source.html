116 void sendClose(uint16_t statusCode = 1000, const char* reason = nullptr, std::size_t reasonLength = 0) override {
117 std::size_t closePayloadLength = reasonLength + 2;
118 char* closePayload = new char[closePayloadLength];
119
120 *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
121
122 if (reasonLength > 0) {
123 memcpy(closePayload + 2, reason, reasonLength);
124 }
125
126 sendClose(closePayload, closePayloadLength);
127
128 delete[] closePayload;
129 }

131 void sendClose(const char* message, std::size_t messageLength) override {
132 if (!closeSent) {
133 LOG(INFO) << "Sending close to peer";
134
135 sendMessage(8, message, messageLength);
136 shutdownWrite();
137
138 setTimeout(CLOSE_SOCKET_TIMEOUT);
139
140 closeSent = true;
141 }

148 /* WSReceiver */
149 void onMessageStart(int opCode) override {
150 receivedOpCode = opCode;
151
152 switch (opCode) {
153 case SubProtocolContext::OpCode::CLOSE:
154 [[fallthrough]];
155 case SubProtocolContext::OpCode::PING:
156 [[fallthrough]];
157 case SubProtocolContext::OpCode::PONG:
158 break;
159 default:
160 subProtocol->onMessageStart(opCode);
161 break;
162 }
163 }
164
165 void onMessageData(const char* junk, uint64_t junkLen) override {
166 switch (receivedOpCode) {
167 case SubProtocolContext::OpCode::CLOSE:
168 [[fallthrough]];
169 case SubProtocolContext::OpCode::PING:
170 [[fallthrough]];
171 case SubProtocolContext::OpCode::PONG:
172 pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
173 break;
174 default:
175 std::size_t junkOffset = 0;
176
177 do {
178 std::size_t sendJunkLen =
179 (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
180 subProtocol->onMessageData(junk + junkOffset, sendJunkLen);
181 junkOffset += sendJunkLen;
182 } while (junkLen - junkOffset > 0);
183 break;
184 }
185 }
186
187 void onMessageEnd() override {
188 switch (receivedOpCode) {
189 case SubProtocolContext::OpCode::CLOSE:
190 if (closeSent) { // active close
191 closeSent = false;
192 LOG(INFO) << "Close confirmed from peer";
193 } else { // passive close
194 LOG(INFO) << "Close request received - replying with close";
195 sendClose(pongCloseData.data(), pongCloseData.length());
196 pongCloseData.clear();
197 }
198 break;
199 case SubProtocolContext::OpCode::PING:
200 sendPong(pongCloseData.data(), pongCloseData.length());
201 pongCloseData.clear();
202 break;
203 case SubProtocolContext::OpCode::PONG:
204 subProtocol->onPongReceived();
205 break;
206 default:
207 subProtocol->onMessageEnd();
208 break;
209 }
210 }
211
212 void onMessageError(uint16_t errnum) override {
213 subProtocol->onMessageError(errnum);
