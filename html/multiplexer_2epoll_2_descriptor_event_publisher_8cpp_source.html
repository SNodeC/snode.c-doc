40 void DescriptorEventPublisher::EPollEvents::muxAdd(core::DescriptorEventReceiver* eventReceiver) {
41 utils::PreserveErrno preserveErrno;
42
43 epoll_event ePollEvent{};
44
45 ePollEvent.data.ptr = eventReceiver;
46 ePollEvent.events = events;
47
48 if (core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, eventReceiver->getRegisteredFd(), &ePollEvent) == 0) {
49 interestCount++;
50
51 if (interestCount >= ePollEvents.size()) {
52 ePollEvents.resize(ePollEvents.size() * 2);
53 }
54 } else if (errno == EEXIST) {
55 muxMod(eventReceiver->getRegisteredFd(), events, eventReceiver);
56 }
57 }

59 void DescriptorEventPublisher::EPollEvents::muxDel(int fd) {
60 utils::PreserveErrno preserveErrno;
61
62 if (core::system::epoll_ctl(epfd, EPOLL_CTL_DEL, fd, nullptr) == 0 || errno == EBADF) {
63 interestCount--;
64
65 if (ePollEvents.size() > (interestCount * 2) + 1) {
66 ePollEvents.resize(ePollEvents.size() / 2);
67 ePollEvents.shrink_to_fit();
68 }
69 }
70 }

125 int DescriptorEventPublisher::spanActiveEvents() {
126 int count = core::system::epoll_wait(ePollEvents.getEPFd(), ePollEvents.getEvents(), ePollEvents.getInterestCount(), 0);
127
128 for (int i = 0; i < count; i++) {
129 epoll_event& ev = ePollEvents.getEvents()[i];
130 core::DescriptorEventReceiver* eventReceiver = static_cast<core::DescriptorEventReceiver*>(ev.data.ptr);
131 if (eventReceiver != nullptr && (ev.events & revents) != 0) {
132 eventCounter++;
133 eventReceiver->span();
134 }
135 }
136
137 return count;
138 }
