336 template <typename ValueTypeT>
337 CLI::Option* SubCommand::setDefaultValue(CLI::Option* option, const ValueTypeT& value, bool clear) const {
338 try {
339 option->default_val(value);
340
341 if (clear) {
342 option->clear();
343 }
344 } catch (const CLI::ParseError&) {
345 option = nullptr;
346 }
347
348 return option;
349 }
