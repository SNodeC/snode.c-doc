59 std::size_t Subscribe::deserializeVP(iot::mqtt::MqttContext* mqttContext) {
60 std::size_t consumed = 0;
61
62 switch (state) {
63 case 0: // V-Header
64 consumed += packetIdentifier.deserialize(mqttContext);
65
66 if (!packetIdentifier.isComplete()) {
67 break;
68 }
69
70 state++;
71 [[fallthrough]];
72 case 1: // Payload
73 consumed += topic.deserialize(mqttContext);
74
75 if (!topic.isComplete()) {
76 break;
77 }
78
79 state++;
80 [[fallthrough]];
81 case 2:
82 consumed += qoS.deserialize(mqttContext);
83
84 if (!qoS.isComplete()) {
85 break;
86 }
87 topics.emplace_back(topic, qoS);
88 topic.reset();
89 qoS.reset();
90
91 if (getConsumed() + consumed < this->getRemainingLength()) {
92 state = 1;
93 }
94
95 complete = getConsumed() + consumed >= this->getRemainingLength();
96
97 break;
98 }
99
100 return consumed;
101 }
