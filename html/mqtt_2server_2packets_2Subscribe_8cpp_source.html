34 std::size_t Subscribe::deserializeVP(iot::mqtt::MqttContext* mqttContext) {
35 std::size_t consumed = 0;
36
37 switch (state) {
38 case 0: // V-Header
39 consumed += packetIdentifier.deserialize(mqttContext);
40
41 if (!packetIdentifier.isComplete()) {
42 break;
43 }
44
45 state++;
46 [[fallthrough]];
47 case 1: // Payload
48 consumed += topic.deserialize(mqttContext);
49
50 if (!topic.isComplete()) {
51 break;
52 }
53
54 state++;
55 [[fallthrough]];
56 case 2:
57 consumed += qoS.deserialize(mqttContext);
58
59 if (!qoS.isComplete()) {
60 break;
61 } else {
62 topics.push_back(Topic(topic, qoS));
63 topic.reset();
64 qoS.reset();
65
66 if (getConsumed() + consumed < this->getRemainingLength()) {
67 state = 1;
68 break;
69 }
70 }
71
72 complete = true;
73 break;
74 }
75
76 return consumed;
77 }
