144 template <template <typename SocketAddress> typename ConfigAddressType>
145 SocketAddress* ConfigAddress<ConfigAddressType>::init() {
146 SocketAddress* socketAddress = new SocketAddress(hostOpt->as<std::string>(), portOpt->as<uint16_t>());
147
148 try {
149 socketAddress->init({.aiFlags = (aiFlags & ~AI_NUMERICHOST) | (numericOpt->as<bool>() ? AI_NUMERICHOST : 0),
150 .aiSockType = aiSockType,
151 .aiProtocol = aiProtocol});
152 } catch ([[maybe_unused]] const core::socket::SocketAddress::BadSocketAddress& badSocketAddress) {
153 delete socketAddress;
154 socketAddress = nullptr;
155
156 throw;
157 }
158
159 return socketAddress;
160 }
