59 std::size_t Suback::deserializeVP(iot::mqtt::MqttContext* mqttContext) {
60 std::size_t consumed = 0;
61
62 switch (state) {
63 case 0: // V-Header
64 consumed += packetIdentifier.deserialize(mqttContext);
65
66 if (!packetIdentifier.isComplete()) {
67 break;
68 }
69
70 state++;
71 [[fallthrough]];
72 case 1: // Payload
73 consumed += returnCode.deserialize(mqttContext);
74
75 if (!returnCode.isComplete()) {
76 break;
77 }
78 returnCodes.push_back(returnCode);
79 returnCode.reset();
80
81 if (getConsumed() + consumed < this->getRemainingLength()) {
82 break;
83 }
84
85 complete = true;
86 break;
87 }
88
89 return consumed;
90 }
