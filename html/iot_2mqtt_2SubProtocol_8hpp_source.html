66 template <typename WSSubProtocolRole>
67 std::size_t SubProtocol<WSSubProtocolRole>::recv(char* junk, std::size_t junklen) {
68 std::size_t maxReturn = std::min(junklen, size);
69
70 std::copy(buffer.data() + cursor, buffer.data() + cursor + maxReturn, junk);
71
72 cursor += maxReturn;
73 size -= maxReturn;
74
75 return maxReturn;
76 }

109 template <typename WSSubProtocolRole>
110 void SubProtocol<WSSubProtocolRole>::onMessageData(const char* junk, std::size_t junkLen) {
111 data += std::string(junk, junkLen);
112
113 std::stringstream ss;
114
115 ss << "WebSocket: Frame data: ";
116 unsigned long i = 0;
117 for (char ch : std::string(junk, junkLen)) {
118 if (i != 0 && i % 8 == 0 && i != data.size()) {
119 ss << std::endl;
120 ss << " ";
121 }
122 ++i;
123 ss << "0x" << std::hex << std::setfill('0') << std::setw(2) << static_cast<uint16_t>(static_cast<uint8_t>(ch))
124 << " "; // << " | ";
125 }
126
127 LOG(TRACE) << ss.str();
128 }

130 template <typename WSSubProtocolRole>
131 void SubProtocol<WSSubProtocolRole>::onMessageEnd() {
132 LOG(TRACE) << "WebSocket: Message END";
133
134 buffer.insert(buffer.end(), data.begin(), data.end());
135 size += data.size();
136 data.clear();
137
138 if (buffer.size() > 0) {
139 onData.span();
140 }
141 }
