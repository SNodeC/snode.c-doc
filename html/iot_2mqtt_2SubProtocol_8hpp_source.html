53 template <typename WSSubProtocolRole>
54 std::size_t SubProtocol<WSSubProtocolRole>::recv(char* chunk, std::size_t chunklen) {
55 std::size_t maxReturn = std::min(chunklen, size);
56
57 std::copy(buffer.data() + cursor, buffer.data() + cursor + maxReturn, chunk);
58
59 cursor += maxReturn;
60 size -= maxReturn;
61
62 return maxReturn;
63 }
