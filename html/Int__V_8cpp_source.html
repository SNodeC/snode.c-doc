31 std::size_t Int_V::construct() {
32 std::size_t consumed = 0;
33 std::size_t ret = 0;
34
35 do {
36 char byte;
37 ret = read(&byte, 1);
38
39 if (ret > 0) {
40 value += static_cast<uint32_t>((byte & 0x7F) * multiplier);
41 if (multiplier > 0x80 * 0x80 * 0x80) {
42 error = true;
43 } else {
44 multiplier *= 0x80;
45 completed = (byte & 0x80) == 0;
46 }
47 consumed += ret;
48 }
49 } while (ret > 0 && !completed && !error);
50
51 return consumed;
52 }
