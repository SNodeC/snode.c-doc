54 std::size_t Int_V::construct() {
55 std::size_t consumed = 0;
56 std::size_t ret = 0;
57
58 do {
59 char byte = 0;
60 ret = read(&byte, 1);
61
62 if (ret > 0) {
63 value += static_cast<uint32_t>((byte & 0x7F) * multiplier);
64 if (multiplier > 0x80 * 0x80 * 0x80) {
65 error = true;
66 } else {
67 multiplier *= 0x80;
68 completed = (byte & 0x80) == 0;
69 }
70 consumed += ret;
71 }
72 } while (ret > 0 && !completed && !error);
73
74 return consumed;
75 }
