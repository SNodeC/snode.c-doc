32 std::size_t Int_V::construct() {
33 std::size_t consumed = 0;
34 std::size_t ret = 0;
35
36 do {
37 char byte = 0;
38 ret = read(&byte, 1);
39
40 if (ret > 0) {
41 value += static_cast<uint32_t>((byte & 0x7F) * multiplier);
42 if (multiplier > 0x80 * 0x80 * 0x80) {
43 error = true;
44 } else {
45 multiplier *= 0x80;
46 completed = (byte & 0x80) == 0;
47 }
48 consumed += ret;
49 }
50 } while (ret > 0 && !completed && !error);
51
52 return consumed;
53 }
