44 void Session::sendPublish(Message& message, uint8_t qoS, bool retain) {
45 LOG(DEBUG) << "MQTT Broker: TopicName: " << message.getTopic();
46 LOG(DEBUG) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::toHexString(message.getMessage());
47 LOG(DEBUG) << "MQTT Broker: QoS: " << static_cast<uint16_t>(std::min(qoS, message.getQoS()));
48
49 if (isActive()) {
50 if ((mqtt->getReflect() || mqtt->getClientId() != message.getOriginClientId())) {
51 mqtt->sendPublish(message.getTopic(),
52 message.getMessage(),
53 std::min(message.getQoS(), qoS),
54 !mqtt->getReflect() ? message.getOriginRetain() || retain : retain);
55 } else {
56 LOG(DEBUG) << "MQTT Broker: Suppress reflection to origin to avoid message looping";
57 LOG(DEBUG) << " Origin: " << message.getOriginClientId();
58 }
59 } else {
60 if (message.getQoS() == 0) {
61 messageQueue.clear();
62 }
63
64 message.setQoS(std::min(message.getQoS(), qoS));
65 messageQueue.emplace_back(message);
66 }
67 }

69 void Session::publishQueued() {
70 LOG(DEBUG) << "MQTT Broker: send queued messages ...";
71 for (iot::mqtt::server::broker::Message& message : messageQueue) {
72 sendPublish(message, message.getQoS(), false);
73 }
74 LOG(DEBUG) << "MQTT Broker: ... done";
75
76 messageQueue.clear();
77 }
