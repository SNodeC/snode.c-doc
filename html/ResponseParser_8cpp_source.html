56 Parser::ParserState ResponseParser::parseStartLine(const std::string& line) {
57 Parser::ParserState parserState = Parser::ParserState::HEADER;
58
59 if (!line.empty()) {
60 std::string remaining;
61
62 std::tie(response.httpVersion, remaining) = httputils::str_split(line, ' ');
63
64 std::smatch httpVersionMatch;
65 if (!std::regex_match(response.httpVersion, httpVersionMatch, httpVersionRegex)) {
66 parserState = parsingError(400, "Wrong protocol version");
67 } else {
68 response.httpMajor = std::stoi(httpVersionMatch.str(1));
69 response.httpMinor = std::stoi(httpVersionMatch.str(2));
70
71 std::tie(response.statusCode, response.reason) = httputils::str_split(remaining, ' ');
72 if (StatusCode::contains(std::stoi(response.statusCode))) {
73 if (response.reason.empty()) {
74 parserState = parsingError(400, "No reason phrase");
75 }
76 } else {
77 parserState = parsingError(400, "Unknown status code");
78 }
79 }
80 } else {
81 parserState = parsingError(400, "Response line empty");
82 }
83 return parserState;
84 }

86 Parser::ParserState ResponseParser::parseHeader() {
87 for (const auto& [headerFieldName, headerFieldValue] : Parser::headers) {
88 if (headerFieldName != "set-cookie") {
89 if (headerFieldName == "content-length") {
90 Parser::contentLength = static_cast<std::size_t>(std::stoi(headerFieldValue));
91 } else if (headerFieldName == "connection" && httputils::ci_contains(headerFieldValue, "close")) {
92 response.connectionState = ConnectionState::Close;
93 } else if (headerFieldName == "connection" && httputils::ci_contains(headerFieldValue, "keep-alive")) {
94 response.connectionState = ConnectionState::Keep;
95 }
96 } else {
97 std::string cookiesLine = headerFieldValue;
98
99 while (!cookiesLine.empty()) {
100 std::string cookieLine;
101 std::tie(cookieLine, cookiesLine) = httputils::str_split(cookiesLine, ',');
102
103 std::string cookieOptions;
104 std::string cookie;
105 std::tie(cookie, cookieOptions) = httputils::str_split(cookieLine, ';');
106
107 std::string cookieName;
108 std::string cookieValue;
109 std::tie(cookieName, cookieValue) = httputils::str_split(cookie, '=');
110
111 httputils::str_trimm(cookieName);
112 httputils::str_trimm(cookieValue);
113
114 std::map<std::string, CookieOptions>::iterator cookieElement;
115 bool inserted = false;
116 std::tie(cookieElement, inserted) = response.cookies.insert({cookieName, CookieOptions(cookieValue)});
117
118 while (!cookieOptions.empty()) {
119 std::string option;
120 std::tie(option, cookieOptions) = httputils::str_split(cookieOptions, ';');
121
122 std::string optionName;
123 std::string optionValue;
124 std::tie(optionName, optionValue) = httputils::str_split(option, '=');
125
126 httputils::str_trimm(optionName);
127 httputils::str_trimm(optionValue);
128
129 cookieElement->second.setOption(optionName, optionValue);
130 }
131 }
132 }
133 }
134
135 headers.erase("set-cookie");
136
137 response.headers = std::move(Parser::headers);
138
139 ParserState parserState = Parser::ParserState::BODY;
140 if (contentLength == 0 && response.httpMinor == 1) {
141 parserState = parsingFinished();
142 } else {
143 httpMajor = response.httpMajor;
144 httpMinor = response.httpMinor;
145 }
146
147 return parserState;
148 }
