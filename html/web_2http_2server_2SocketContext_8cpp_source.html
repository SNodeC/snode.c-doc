69 void SocketContext::deliverRequest(web::http::server::Request& request) {
70 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request parsed " << request.method << " " << request.url;
71
72 currentRequest = std::make_shared<Request>(std::move(request));
73
74 masterResponse->init();
75 masterResponse->httpMajor = currentRequest->httpMajor;
76 masterResponse->httpMinor = currentRequest->httpMinor;
77 const std::string connection = currentRequest->get("Connection");
78 if (!connection.empty()) {
79 masterResponse->set("Connection", connection);
80 }
81
82 onRequestReady(currentRequest, masterResponse);
83 }

89 void SocketContext::responseCompleted(bool success) {
90 if (success) {
91 httpClose = masterResponse->connectionState == ConnectionState::Close ||
92 (masterResponse->connectionState == ConnectionState::Default &&
93 ((masterResponse->httpMajor == 0 && masterResponse->httpMinor == 9) ||
94 (masterResponse->httpMajor == 1 && masterResponse->httpMinor == 0)));
95
96 requestCompleted();
97 } else {
98 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response wrong content length";
99
100 shutdownWrite();
101 }
102 }

104 void SocketContext::requestCompleted() {
105 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request completed";
106
107 if (httpClose) {
108 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Connection = Close";
109
110 shutdownWrite();
111 } else {
112 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Connection = Keep-Alive";
113
114 core::EventReceiver::atNextTick([this, response = static_cast<std::weak_ptr<Response>>(this->masterResponse)]() -> void {
115 if (!response.expired()) {
116 if (!pendingRequests.empty()) {
117 deliverRequest(pendingRequests.front());
118 pendingRequests.pop_front();
119 } else {
120 currentRequest = nullptr;
121 }
122 }
123 });
124 }
