95 void SocketContext::deliverRequest() {
96 if (!pendingRequests.empty()) {
97 const std::shared_ptr<Request>& pendingRequest = pendingRequests.front();
98
99 if (pendingRequest->status == 0) {
100 LOG(INFO) << getSocketConnection()->getConnectionName() << " HTTP: Request deliver: " << pendingRequest->method << " "
101 << pendingRequest->url << " HTTP/" << pendingRequest->httpMajor << "." << pendingRequest->httpMinor;
102
103 masterResponse->init();
104 masterResponse->requestMethod = pendingRequest->method;
105 masterResponse->httpMajor = pendingRequest->httpMajor;
106 masterResponse->httpMinor = pendingRequest->httpMinor;
107
108 const std::string connection = pendingRequest->get("Connection");
109 if (!connection.empty()) {
110 masterResponse->set("Connection", connection);
111 }
112
113 onRequestReady(pendingRequest, masterResponse);
114 } else {
115 masterResponse->init();
116 masterResponse->httpMajor = 1;
117 masterResponse->httpMinor = 1;
118
119 masterResponse->status(pendingRequest->status).send(pendingRequest->reason);
120 }
121 } else {
122 LOG(INFO) << getSocketConnection()->getConnectionName() << " HTTP: No more pending request";
123 }
124 }

126 void SocketContext::responseStarted(const Response& response) {
127 if (!pendingRequests.empty()) {
128 const std::shared_ptr<Request>& pendingRequest = pendingRequests.front();
129
130 serverSentEvent = web::http::ciContains(pendingRequest->get("Accept"), "text/event-stream");
131
132 if (serverSentEvent) {
133 getSocketConnection()->setReadTimeout(0);
134 }
135
136 LOG(INFO) << getSocketConnection()->getConnectionName() << " HTTP: Response start for request: " << pendingRequest->method
137 << " " << pendingRequest->url << " HTTP/" << pendingRequest->httpMajor << "." << pendingRequest->httpMinor;
138 LOG(INFO) << getSocketConnection()->getConnectionName() << " "
139 << "HTTP/" + std::to_string(response.httpMajor)
140 .append(".")
141 .append(std::to_string(response.httpMinor))
142 .append(" ")
143 .append(std::to_string(response.statusCode))
144 .append(" ")
145 .append(StatusCode::reason(response.statusCode));
146 }
147 }

160 void SocketContext::requestCompleted(const Response& response) {
161 const std::shared_ptr<Request> request = std::move(pendingRequests.front());
162 pendingRequests.pop_front();
163
164 LOG(INFO) << getSocketConnection()->getConnectionName() << " HTTP: Response completed for request: " << request->method << " "
165 << request->url << " HTTP/" << request->httpMajor << "." << request->httpMinor;
166 LOG(INFO) << getSocketConnection()->getConnectionName() << " "
167 << "HTTP/" + std::to_string(response.httpMajor)
168 .append(".")
169 .append(std::to_string(response.httpMinor))
170 .append(" ")
171 .append(std::to_string(response.statusCode))
172 .append(" ")
173 .append(StatusCode::reason(response.statusCode));
174
175 httpClose = response.connectionState == ConnectionState::Close ||
176 (response.connectionState == ConnectionState::Default &&
177 ((response.httpMajor == 0 && response.httpMinor == 9) || (response.httpMajor == 1 && response.httpMinor == 0)));
178
179 if (httpClose) {
180 LOG(DEBUG) << getSocketConnection()->getConnectionName() << " HTTP: Connection = Close";
181
182 shutdownWrite(true);
183 } else {
184 LOG(DEBUG) << getSocketConnection()->getConnectionName() << " HTTP: Connection = Keep-Alive";
185
186 if (!pendingRequests.empty()) {
187 core::EventReceiver::atNextTick([this, response = std::weak_ptr<Response>(masterResponse)]() {
188 if (!response.expired()) {
189 deliverRequest();
190 }
191 });
192 }
193 }
