35 void* DynamicLoader::dlRegisterHandle(void* handle, const std::string& libFile) {
36 if (handle != nullptr) {
37 if (!dlOpenedLibraries.contains(handle)) {
38 dlOpenedLibraries[handle].fileName = libFile;
39 dlOpenedLibraries[handle].handle = handle;
40 }
41 dlOpenedLibraries[handle].refCount++;
42 LOG(TRACE) << "dlOpen: '" << libFile << "' success: handle=0x" << std::hex << handle;
43 } else {
44 LOG(WARNING) << "dlOpen: " << DynamicLoader::dlError();
45 }
46
47 return handle;
48 }

50 void DynamicLoader::dlCloseDelayed(void* handle) {
51 if (handle != nullptr) {
52 if (dlOpenedLibraries.contains(handle)) {
53 if (std::find(closeHandles.begin(), closeHandles.end(), handle) == closeHandles.end()) {
54 LOG(TRACE) << "dlCloseDelayed file = " << dlOpenedLibraries[handle].fileName << ": registered";
55
56 closeHandles.insert(handle);
57 } else {
58 LOG(WARNING) << "dlCloseDelayed file = " << dlOpenedLibraries[handle].fileName
59 << ": already registered for dlCloseDelayed";
60 }
61 } else {
62 LOG(WARNING) << "dlCloseDelayed handle = " << handle << ": not opened using dlOpen";
63 }
64 } else {
65 LOG(WARNING) << "dlCloseDelayed handle: nullptr";
66 }
67 }

69 int DynamicLoader::dlClose(void* handle) {
70 int ret = 0;
71
72 if (handle != nullptr) {
73 if (!closeHandles.contains(handle)) {
74 if (dlOpenedLibraries.contains(handle)) {
75 LOG(TRACE) << "dlClose file = " << dlOpenedLibraries[handle].fileName << ": registered";
76 ret = dlClose(dlOpenedLibraries[handle]);
77
78 dlOpenedLibraries.erase(handle);
79 } else {
80 LOG(WARNING) << "dlCloseDelayed handle = " << handle << ": not opened using dlOpen";
81 }
82 } else {
83 LOG(WARNING) << "dlClose file = " << dlOpenedLibraries[handle].fileName << ": already registered for dlCloseDelayed";
84 }
85 } else {
86 LOG(WARNING) << "dlClose handle: nullptr";
87 }
88
89 return ret;
90 }

92 int DynamicLoader::dlClose(Library& library) {
93 int ret = 0;
94
95 while (library.refCount-- > 0 && ret == 0) {
96 ret = execDlClose(library);
97
98 if (ret != 0) {
99 LOG(WARNING) << "dlClose: " << DynamicLoader::dlError();
100 } else {
101 LOG(TRACE) << "dlClose file = " << library.fileName << ": closed";
102 }
103 }
104
105 return ret;
106 }

124 void DynamicLoader::execDlCloseDeleyed() {
125 for (void* handle : closeHandles) {
126 Library& library = dlOpenedLibraries[handle];
127
128 if (execDlClose(library) != 0) {
129 LOG(WARNING) << "execDlCloseDeleyed file = " << DynamicLoader::dlError();
130 } else {
131 LOG(TRACE) << "execDlCloseDeleyed file = " << library.fileName << ": closed";
132 }
133
134 dlOpenedLibraries.erase(handle);
135 }
136
137 closeHandles.clear();
138 }
139
140 void DynamicLoader::execDlCloseAll() {
141 execDlCloseDeleyed();
142
143 for (auto& [handle, library] : dlOpenedLibraries) {
144 int ret = dlClose(library);
145
146 if (ret != 0) {
147 LOG(WARNING) << "execDlCloseAll file = " << library.fileName << ": " << DynamicLoader::dlError();
148 } else {
149 LOG(TRACE) << "execDlCloseAll file = " << library.fileName << ": closed";
150 }
151 }
152
153 dlOpenedLibraries.clear();
154 }
