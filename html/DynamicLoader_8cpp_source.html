36 void* DynamicLoader::dlRegisterHandle(void* handle, const std::string& libFile) {
37 if (handle != nullptr) {
38 if (!dlOpenedLibraries.contains(handle)) {
39 dlOpenedLibraries[handle].fileName = libFile;
40 dlOpenedLibraries[handle].handle = handle;
41 }
42 LOG(TRACE) << "DynLoader: dlOpen: success";
43 LOG(TRACE) << " " << libFile;
44 } else {
45 LOG(TRACE) << "DynLoader: dlOpen: " << DynamicLoader::dlError();
46 LOG(TRACE) << " " << libFile;
47 }
48
49 return handle;
50 }

52 void DynamicLoader::dlCloseDelayed(void* handle) {
53 if (handle != nullptr) {
54 if (dlOpenedLibraries.contains(handle)) {
55 if (std::find(closeHandles.begin(), closeHandles.end(), handle) == closeHandles.end()) {
56 LOG(TRACE) << "DynLoader: dlCloseDelayed: Registered";
57 LOG(TRACE) << " " << dlOpenedLibraries[handle].fileName;
58
59 closeHandles.push_back(handle);
60 } else {
61 LOG(TRACE) << "DynLoader: dlCloseDelayed: Already registered";
62 LOG(TRACE) << " " << dlOpenedLibraries[handle].fileName;
63 }
64 } else {
65 LOG(TRACE) << "DynLoader: dlCloseDelayed: Not opened using dlOpen";
66 LOG(TRACE) << " " << handle;
67 }
68 } else {
69 LOG(TRACE) << "DynLoader: dlCloseDelayed: Handle is nullptr";
70 }
71 }

73 int DynamicLoader::dlClose(void* handle) {
74 int ret = 0;
75
76 if (handle != nullptr) {
77 if (dlOpenedLibraries.contains(handle)) {
78 ret = dlClose(dlOpenedLibraries[handle]);
79
80 dlOpenedLibraries.erase(handle);
81 } else {
82 LOG(TRACE) << "DynLoader: dlCloseDelayed: Not opened using dlOpen";
83 LOG(TRACE) << " " << handle;
84 }
85 } else {
86 LOG(TRACE) << "DynLoader: dlClose handle: nullptr";
87 }
88
89 return ret;
90 }

104 int DynamicLoader::dlClose(const Library& library) {
105 int ret = 0;
106 ret = realExecDlClose(library);
107
108 if (ret != 0) {
109 LOG(TRACE) << "DynLoader: dlClose library: " << DynamicLoader::dlError();
110 } else {
111 LOG(TRACE) << "DynLoader: dlClose library";
112 }
113 LOG(TRACE) << " " << library.fileName;
114
115 return ret;

118 void DynamicLoader::execDlCloseDeleyed() {
119 if (!closeHandles.empty()) {
120 LOG(TRACE) << "DynLoader: execDlCloseDeleyed ...";
121
122 for (void* handle : closeHandles) {
123 dlClose(dlOpenedLibraries[handle]);
124 dlOpenedLibraries.erase(handle);
125 }
126
127 closeHandles.clear();
128
129 LOG(TRACE) << "DynLoader: execDlCloseDeleyed done";
130 }
131 }

133 void DynamicLoader::execDlCloseAll() {
134 LOG(TRACE) << "DynLoader: execDlCloseAll ...";
135
136 for (auto& [handle, library] : dlOpenedLibraries) {
137 dlClose(library);
138 }
139
140 dlOpenedLibraries.clear();
141 closeHandles.clear();
142
143 LOG(TRACE) << "DynLoader: execDlCloseAll done";
144 }
