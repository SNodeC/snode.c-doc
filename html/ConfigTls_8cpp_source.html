66 "bool",
67 "false",
68 CLI::IsMember({"true", "false"}));

73 "bool",
74 "false",
75 CLI::IsMember({"true", "false"}));

103 CLI::PositiveNumber);
104
105 noCloseNotifyIsEOFOpt = addFlag( //
106 "--no-close-notify-is-eof{true}",
107 "Do not interpret a SSL/TLS close_notify alert as EOF",
