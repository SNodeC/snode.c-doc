74 template <typename SubProtocol, typename Request, typename Response>
75 void web::websocket::SocketContextUpgrade<SubProtocol, Request, Response>::sendClose(uint16_t statusCode,
76 const char* reason,
77 std::size_t reasonLength) {
78 std::size_t closePayloadLength = reasonLength + 2;
79 char* closePayload = new char[closePayloadLength];
80
81 *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode); // cppcheck-suppress uninitdata
82
83 if (reasonLength > 0) {
84 memcpy(closePayload + 2, reason, reasonLength);
85 }
86
87 sendClose(closePayload, closePayloadLength);
88
89 delete[] closePayload;
90 }

111 void web::websocket::SocketContextUpgrade<SubProtocol, Request, Response>::onMessageStart(int opCode) {
112 receivedOpCode = opCode;
113
114 switch (opCode) {
115 case SubProtocolContext::OpCode::CLOSE:
116 [[fallthrough]];
117 case SubProtocolContext::OpCode::PING:

127 template <typename SubProtocol, typename Request, typename Response>
128 void web::websocket::SocketContextUpgrade<SubProtocol, Request, Response>::onMessageData(const char* chunk, uint64_t chunkLen) {
129 switch (receivedOpCode) {
130 case SubProtocolContext::OpCode::CLOSE:
131 [[fallthrough]];
132 case SubProtocolContext::OpCode::PING:
133 [[fallthrough]];
134 case SubProtocolContext::OpCode::PONG:
135 pongCloseData += std::string(chunk, static_cast<std::size_t>(chunkLen));
136 break;
137 default:
138 std::size_t chunkOffset = 0;
139
140 do {
141 std::size_t sendChunkLen =
142 (chunkLen - chunkOffset <= SIZE_MAX) ? static_cast<std::size_t>(chunkLen - chunkOffset) : SIZE_MAX;
143 subProtocol->onMessageData(chunk + chunkOffset, sendChunkLen);
144 chunkOffset += sendChunkLen;
145 } while (chunkLen - chunkOffset > 0);
146 break;
147 }
148 }

150 template <typename SubProtocol, typename Request, typename Response>
151 void web::websocket::SocketContextUpgrade<SubProtocol, Request, Response>::onMessageEnd() {
152 switch (receivedOpCode) {
153 case SubProtocolContext::OpCode::CLOSE:
154 if (closeSent) { // active close
155 closeSent = false;
156 LOG(DEBUG) << this->getSocketConnection()->getInstanceName() << " WebSocket: Close confirmed from peer";
157 } else { // passive close
158 LOG(DEBUG) << this->getSocketConnection()->getInstanceName()
159 << " WebSocket: Close request received - replying with close";
160 sendClose(pongCloseData.data(), pongCloseData.length());
161 pongCloseData.clear();
162 shutdownWrite();
163 }
164 break;
165 case SubProtocolContext::OpCode::PING:
166 sendPong(pongCloseData.data(), pongCloseData.length());
167 pongCloseData.clear();
168 break;
169 case SubProtocolContext::OpCode::PONG:
170 subProtocol->onPongReceived();
171 break;
172 default:
173 subProtocol->onMessageEnd();
174 break;
175 }
176 }

238 void web::websocket::SocketContextUpgrade<SubProtocol, Request, Response>::sendFrameData(const char* frame,
239 uint64_t frameLength) const {
240 if (!closeSent) {
241 uint64_t frameOffset = 0;
242
243 do {
244 std::size_t sendChunkLen =
245 (frameLength - frameOffset <= SIZE_MAX) ? static_cast<std::size_t>(frameLength - frameOffset) : SIZE_MAX;
246 sendToPeer(frame + frameOffset, sendChunkLen);
247 frameOffset += sendChunkLen;
248 } while (frameLength - frameOffset > 0);

257 template <typename SubProtocol, typename Request, typename Response>
258 void web::websocket::SocketContextUpgrade<SubProtocol, Request, Response>::dumpFrame(char* frame, uint64_t frameLength) {
259 const unsigned int modul = 4;
260
261 std::stringstream stringStream;
262
263 for (std::size_t i = 0; i < frameLength; i++) {
264 stringStream << std::setfill('0') << std::setw(2) << std::hex << static_cast<unsigned int>(static_cast<unsigned char>(frame[i]))
265 << " ";
266
267 if ((i + 1) % modul == 0 || i == frameLength) {
268 LOG(TRACE) << "WebSocket: Frame = " << stringStream.str();
269 stringStream.str("");
270 }
271 }
272 }
