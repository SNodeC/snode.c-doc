491 bool Config::bootstrap() {
492 prefixMap.clear();
493
494 app.final_callback([](void) -> void {
495 if (daemonizeOpt->as<bool>() && app["--write-config"]->count() == 0 && app["--commandline"]->count() == 0 &&
496 app["--commandline-configured"]->count() == 0 && app["--commandline-full"]->count() == 0 &&
497 app["--show-config"]->count() == 0) {
498 std::cout << "Running as daemon" << std::endl;
499
500 try {
501 utils::Daemon::startDaemon(
502 pidDirectory + "/" + applicationName + ".pid", userNameOpt->as<std::string>(), groupNameOpt->as<std::string>());
503 } catch (const DaemonizeError& e) {
504 std::cout << "Daemonize error: " << e.what() << " ... exiting" << std::endl;
505 std::exit(EXIT_FAILURE);
506 } catch (const DaemonizeFailure& e) {
507 std::cout << "Daemonize failure: " << e.what() << " ... exiting" << std::endl;
508 std::exit(EXIT_FAILURE);
509 } catch (const DaemonizeSuccess&) {
510 std::exit(EXIT_SUCCESS);
511 }
512
513 logger::Logger::quiet();
514
515 std::string logFile = logFileOpt->as<std::string>();
516 if (!logFile.empty()) {
517 logger::Logger::logToFile(logFile);
518 }
519 } else if (app["--enforce-log-file"]->as<bool>()) {
520 std::string logFile = logFileOpt->as<std::string>();
521 std::cout << "Writing logs to file " << logFile << std::endl;
522
523 logger::Logger::logToFile(logFile);
524 }
525 });
526
527 return parse2();
528 }

530 static void createCommandLineOptions(std::stringstream& out, CLI::App* app, CLI::CallForCommandline::Mode mode) {
531 for (const CLI::Option* option : app->get_options()) {
532 if (option->get_configurable()) {
533 std::string value;
534
535 if (option->reduced_results().size() > 0 && ((option->get_required() || mode == CLI::CallForCommandline::Mode::ALL) ||
536 mode == CLI::CallForCommandline::Mode::CONFIGURED)) {
537 value = option->reduced_results()[0];
538 } else if (!option->get_default_str().empty()) {
539 value = (mode == CLI::CallForCommandline::Mode::ALL || option->get_required()) ? option->get_default_str() : "";
540 } else if (option->get_required()) {
541 value = "<REQUIRED>";
542 } else {
543 value = mode == CLI::CallForCommandline::Mode::ALL ? "\"\"" : "";
544 }
545
546 if (value.empty() && option->count() > 0 &&
547 (mode == CLI::CallForCommandline::Mode::ALL || mode == CLI::CallForCommandline::Mode::CONFIGURED)) {
548 value = "\"\"";
549 }
550
551 if (!value.empty()) {
552 out << "--" << option->get_single_name() << ((option->get_items_expected_max() == 0) ? "=" : " ") << value << " ";
553 }
554 }
555 }
556 }

571 static std::string createCommandLineSubcommands(CLI::App* app, CLI::CallForCommandline::Mode mode) {
572 std::stringstream out;

575 if (!subcommand->get_name().empty()) {

585 static void createCommandLineTemplate(std::stringstream& out, CLI::App* app, CLI::CallForCommandline::Mode mode) {
586 std::string outString;
587
588 outString = createCommandLineOptions(app, mode);
589 if (!outString.empty()) {
590 outString += " ";
591 }
592
593 outString += createCommandLineSubcommands(app, mode);
594 if (!outString.empty()) {
595 out << app->get_name() << " " << outString;
596 }
597 }
598
599 static std::string createCommandLineTemplate(CLI::App* app, CLI::CallForCommandline::Mode mode) {
600 std::stringstream out;
601
602 createCommandLineTemplate(out, app, mode);
603
604 std::string outString = out.str();
605 while (app->get_parent() != nullptr) {
606 app = app->get_parent();
607 std::string parentOptions = createCommandLineOptions(app, mode);
608 outString = app->get_name() + " " + (!parentOptions.empty() ? parentOptions + " " : "") + outString;
609 }
610
611 if (outString.empty()) {
612 outString = Config::getApplicationName();
613 }
614
615 return outString;
616 }

670 bool Config::parse2() {
671 bool ret = false;
672
673 try {
674 try {
675 try {
676 app.parse(argc, argv);
677 if (app["--show-config"]->count() > 0) {
678 throw CLI::CallForShowConfig();
679 } else if (app["--write-config"]->count() > 0) {
680 throw CLI::CallForWriteConfig(app["--write-config"]->as<std::string>());
681 } else if (app["--commandline"]->count() > 0) {
682 throw CLI::CallForCommandline(&app, CLI::CallForCommandline::Mode::REQUIRED);
683 } else if (app["--commandline-configured"]->count() > 0) {
684 throw CLI::CallForCommandline(&app, CLI::CallForCommandline::Mode::CONFIGURED);
685 } else if (app["--commandline-full"]->count() > 0) {
686 throw CLI::CallForCommandline(&app, CLI::CallForCommandline::Mode::ALL);
687 }
688 ret = true;
689 } catch (const CLI::ParseError&) {
690 if (app["--show-config"]->count() > 0) {
691 throw CLI::CallForShowConfig();
692 } else if (app["--write-config"]->count() > 0) {
693 throw CLI::CallForWriteConfig(app["--write-config"]->as<std::string>());
694 } else if (app["--commandline"]->count() > 0) {
695 throw CLI::CallForCommandline(&app, CLI::CallForCommandline::Mode::REQUIRED);
696 } else if (app["--commandline-configured"]->count() > 0) {
697 throw CLI::CallForCommandline(&app, CLI::CallForCommandline::Mode::CONFIGURED);
698 } else if (app["--commandline-full"]->count() > 0) {
699 throw CLI::CallForCommandline(&app, CLI::CallForCommandline::Mode::ALL);
700 }
701 throw;
702 }
703 } catch (const CLI::CallForHelp&) {
704 std::cout << app.help() << std::endl;
705 } catch (const CLI::CallForAllHelp&) {
706 std::cout << app.help("", CLI::AppFormatMode::All) << std::endl;
707 } catch (const CLI::CallForVersion&) {
708 std::cout << "SNode.C-Version: " << app.version() << std::endl << std::endl;
709 } catch (const CLI::CallForCommandline& e) {
710 std::cout << e.what();
711 if (e.getMode() == CLI::CallForCommandline::Mode::REQUIRED) {
712 std::cout << "* Required options show eigher their configured value or <REQUIRED>" << std::endl;
713 } else if (e.getMode() == CLI::CallForCommandline::Mode::CONFIGURED) {
714 std::cout << "* Required but not yet configured options show <REQUIRED> as value" << std::endl;
715 std::cout << "* Configured options show their configured value" << std::endl;
716 } else if (e.getMode() == CLI::CallForCommandline::Mode::ALL) {
717 std::cout << "* Required but not yet configured options show <REQUIRED> as value " << std::endl;
718 std::cout << "* Remaining options show either their default or configured value" << std::endl;
719 }
720 std::cout << "* Options marked as <REQUIRED> need to be configured for a successfull bootstrap" << std::endl;
721 std::cout << std::endl
722 << Color::Code::FG_GREEN << "command@line" << Color::Code::FG_DEFAULT << ":" << Color::Code::FG_BLUE << "~/> "
723 << Color::Code::FG_DEFAULT << createCommandLineTemplate(e.getApp(), e.getMode()) << std::endl;
724 std::cout << std::endl << app.get_footer() << std::endl;
725 } catch (const CLI::CallForShowConfig& e) {
726 std::cout << e.what() << std::endl;
727 try {
728 std::cout << app.config_to_str(true, true);
729 } catch (const CLI::ParseError& e1) {
730 std::cout << "Error parsing config file: " << e1.get_name() << " " << e1.what() << std::endl;
731 throw;
732 }
733 } catch (const CLI::CallForWriteConfig& e) {
734 std::cout << e.what() << std::endl;
735 std::ofstream confFile(e.getConfigFile());
736 if (confFile.is_open()) {
737 confFile << app.config_to_str(true, true);
738 confFile.close();
739 } else {
740 std::cout << "Error writing config file" << std::endl;
741 }
742 std::cout << std::endl << app.get_footer() << std::endl;
743 } catch (const CLI::ConversionError& e) {
744 std::cout << "Command line conversion error: " << e.what() << std::endl;
745 throw;
746 } catch (const CLI::ArgumentMismatch& e) {
747 std::cout << "Command line error: Argument for " << e.what() << std::endl;
748 throw;
749 } catch (const CLI::ConfigError& e) {
750 std::cout << "Config file (INI) parse error: " << e.get_name() << e.what() << std::endl;
751 std::cout << "Hint: Rewrite the config file by appending -w [config file] to your command line." << std::endl;
752 throw;
753 } catch (const CLI::ParseError& e) {
754 std::string what = e.what();
755 if (what.find("[Option Group: ]") != std::string::npos) { // If CLI11 throws that error it means for us there are
756 // unconfigured anonymous instances
757 std::cout << "Bootstrap error: Anonymous instance(s) not configured in source code!" << std::endl;
758 } else {
759 std::cout << "Command line error: " << what << std::endl;
760 }
761 throw;
762 }
763 } catch (const CLI::Success& e) {
764 std::cout << "Success: " << e.get_name() << " " << e.what() << std::endl;
765 } catch (const CLI::ParseError&) {
766 std::cout << "Append -h, --help, or --help-all to your command line for more information." << std::endl;
767 std::cout << std::endl << app.get_footer() << std::endl;
768 } catch (const CLI::Error& e) {
769 std::cout << "Error: " << e.get_name() << " " << e.what() << std::endl;
770 std::cout << "Append -h, --help, or --help-all to your command line for more information." << std::endl;
771 std::cout << std::endl << app.get_footer() << std::endl;
772 }
773
774 return ret;
775 }
