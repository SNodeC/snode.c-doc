49 std::size_t Parser::parse() {
50 std::size_t ret = 0;
51 std::size_t consumed = 0;
52
53 do {
54 switch (parserState) {
55 case ParserState::BEGIN:
56 reset();
57 begin();
58 parserState = ParserState::FIRSTLINE;
59 [[fallthrough]];
60 case ParserState::FIRSTLINE:
61 ret = readStartLine();
62 break;
63 case ParserState::HEADER:
64 ret = readHeaderLine();
65 break;
66 case ParserState::BODY:
67 ret = readContent();
68 break;
69 case ParserState::ERROR:
70 break;
71 }
72 consumed += ret;
73 } while (ret > 0 && parserState != ParserState::BEGIN && parserState != ParserState::ERROR);
74 return consumed;
75 }

101 std::size_t Parser::readHeaderLine() {
102 std::size_t consumed = 0;
103 std::size_t ret = 0;
104
105 do {
106 char ch = 0;
107 ret = socketContext->readFromPeer(&ch, 1);
108
109 if (ret > 0) {
110 if (ch == '\r' || ch == '\n') {
111 consumed++;
112 if (ch == '\n') {
113 if (EOL) {
114 splitHeaderLine(line);
115 line.clear();
116 if (parserState != ParserState::ERROR) {
117 parserState = parseHeader();
118 }
119 EOL = false;
120 } else if (line.empty()) {
121 if (parserState != ParserState::ERROR) {
122 parserState = parseHeader();
123 }
124 } else {
125 EOL = true;
126 }
127 }
128 } else if (EOL) {
129 if (std::isblank(ch)) {
130 if ((hTTPCompliance & HTTPCompliance::RFC7230) == HTTPCompliance::RFC7230) {
131 parserState = parsingError(400, "Header Folding");
132 } else {
133 line += ch;
134 consumed++;
135 }
136 } else {
137 splitHeaderLine(line);
138 line.clear();
139 line += ch;
140 consumed++;
141 }
142 EOL = false;
143 } else {
144 line += ch;
145 consumed++;
146 }
147 }
148 } while (ret > 0 && parserState == ParserState::HEADER);
149
150 return consumed;
151 }

153 void Parser::splitHeaderLine(const std::string& line) {
154 if (!line.empty()) {
155 std::string headerFieldName;
156 std::string headerFieldValue;
157 std::tie(headerFieldName, headerFieldValue) = httputils::str_split(line, ':');
158
159 if (headerFieldName.empty()) {
160 parserState = parsingError(400, "Header-field empty");
161 } else if (std::isblank(headerFieldName.back()) || std::isblank(headerFieldName.front())) {
162 parserState = parsingError(400, "White space before or after header-field");
163 } else if (headerFieldValue.empty()) {
164 parserState = parsingError(400, "Header-value of field \"" + headerFieldName + "\" empty");
165 } else {
166 httputils::to_lower(headerFieldName);
167 httputils::str_trimm(headerFieldValue);
168
169 if (headers.find(headerFieldName) == headers.end()) {
170 headers.insert({headerFieldName, headerFieldValue});
171 } else {
172 headers[headerFieldName] += "," + headerFieldValue;
173 }
174 }
175 } else {
176 parserState = parsingError(400, "Header-line empty");
177 }
178 }

180 std::size_t Parser::readContent() {
181 static char contentJunk[MAX_CONTENT_JUNK_LEN];
182
183 std::size_t consumed = 0;
184
185 if (httpMinor == 0 && contentLength == 0) {
186 consumed = socketContext->readFromPeer(contentJunk, MAX_CONTENT_JUNK_LEN);
187
188 std::size_t contentJunkLen = static_cast<std::size_t>(consumed);
189 if (contentJunkLen > 0) {
190 content.insert(content.end(), contentJunk, contentJunk + contentJunkLen);
191 } else {
192 parserState = parseContent(content);
193 }
194 } else if (httpMinor == 1) {
195 std::size_t contentJunkLenLeft =
196 (contentLength - contentRead < MAX_CONTENT_JUNK_LEN) ? contentLength - contentRead : MAX_CONTENT_JUNK_LEN;
197
198 consumed = socketContext->readFromPeer(contentJunk, contentJunkLenLeft);
199
200 std::size_t contentJunkLen = static_cast<std::size_t>(consumed);
201 if (contentJunkLen > 0) {
202 if (contentRead + contentJunkLen <= contentLength) {
203 content.insert(content.end(), contentJunk, contentJunk + contentJunkLen);
204
205 contentRead += contentJunkLen;
206 if (contentRead == contentLength) {
207 parserState = parseContent(content);
208 }
209 } else {
210 parserState = parsingError(400, "Content to long");
211 }
212 }
213 }
214
215 return consumed;
216 }
