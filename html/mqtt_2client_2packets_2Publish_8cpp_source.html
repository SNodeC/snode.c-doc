40 std::size_t Publish::deserializeVP(iot::mqtt::MqttContext* mqttContext) {
41 std::size_t consumed = 0;
42
43 switch (state) {
44 case 0: // V-Header
45 consumed += topic.deserialize(mqttContext);
46 if (!topic.isComplete()) {
47 break;
48 }
49
50 state++;
51 [[fallthrough]];
52 case 1:
53 if (qoS > 0) {
54 consumed += packetIdentifier.deserialize(mqttContext);
55 if (!packetIdentifier.isComplete()) {
56 break;
57 }
58 }
59
60 message.setSize(static_cast<uint16_t>(getRemainingLength() - getConsumed() - consumed));
61
62 state++;
63 [[fallthrough]];
64 case 2: // Payload
65 consumed += message.deserialize(mqttContext);
66 if (!message.isComplete()) {
67 break;
68 }
69
70 complete = true;
71 break;
72 }
73
74 return consumed;
75 }
