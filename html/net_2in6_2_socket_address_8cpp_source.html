58 SocketAddress::SocketAddress(const SockAddr& sockAddr, socklen_t sockAddrLen)
59 : net::SocketAddress<SocketAddress::SockAddr>(sockAddr, sockAddrLen)
60 , socketAddrInfo(std::make_shared<SocketAddrInfo>()) {
61 char host[NI_MAXHOST];
62 char serv[NI_MAXSERV];
63 std::memset(host, 0, NI_MAXHOST);
64 std::memset(serv, 0, NI_MAXSERV);
65
66 int ret = core::system::getnameinfo(
67 reinterpret_cast<const sockaddr*>(&sockAddr), sizeof(sockAddr), nullptr, 0, serv, NI_MAXSERV, NI_NUMERICSERV);
68
69 if (ret == 0) {
70 this->port = static_cast<uint16_t>(std::stoul(serv));
71
72 ret = core::system::getnameinfo(
73 reinterpret_cast<const sockaddr*>(&sockAddr), sizeof(sockAddr), host, NI_MAXHOST, nullptr, 0, NI_NAMEREQD);
74 }
75 if (ret != 0) {
76 this->host = host;
77
78 ret = core::system::getnameinfo(
79 reinterpret_cast<const sockaddr*>(&sockAddr), sizeof(sockAddr), host, NI_MAXHOST, nullptr, 0, NI_NUMERICHOST);
80 }
81 if (ret == 0) {
82 this->host = host;
83 } else {
84 this->host = gai_strerror(ret);
85 }
86 }

88 SocketAddress& SocketAddress::init() {
89 addrinfo hints{};
90
91 hints.ai_family = AF_INET6;
92 hints.ai_socktype = aiSocktype;
93 hints.ai_protocol = aiProtocol;
94 hints.ai_flags = aiFlags | AI_ADDRCONFIG | AI_ALL;
95
96 int aiErrCode = 0;
97
98 if ((aiErrCode = socketAddrInfo->init(host, std::to_string(port), hints)) == 0) {
99 const sockaddr* baseSockAddr = socketAddrInfo->getSockAddr();
100 if (baseSockAddr != nullptr) {
101 sockAddr = *reinterpret_cast<const SockAddr*>(socketAddrInfo->getSockAddr());
102 } else {
103 throw core::socket::SocketAddress::BadSocketAddress(
104 "IPv4 getaddrinfo for '" + host + "': " + (aiErrCode == EAI_SYSTEM ? strerror(errno) : gai_strerror(aiErrCode)),
105 aiErrCode == EAI_SYSTEM ? errno : EINVAL);
106 }
107 } else {
108 throw core::socket::SocketAddress::BadSocketAddress(
109 "IPv4 init for '" + host + "': " + (aiErrCode == EAI_SYSTEM ? strerror(errno) : gai_strerror(aiErrCode)),
110 aiErrCode == EAI_SYSTEM ? errno : EINVAL);
111 }
112
113 return *this;
114 }
