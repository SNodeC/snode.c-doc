48 bool SubscribtionTree::subscribe(const std::string& topic, const std::string& clientId, uint8_t qoS) {
49 bool success = false;
50
51 auto hashCount = std::ranges::count(topic, '#');
52 if (!topic.empty() && (hashCount == 0 || (hashCount == 1 && topic.ends_with('#')))) {
53 head.subscribe(clientId, qoS, topic);
54
55 success = true;
56 } else {
57 LOG(DEBUG) << "MQTT Broker: Subscribe: Wrong '#' placement: " << topic;
58 }
59
60 return success;
61 }

100 bool SubscribtionTree::TopicLevel::subscribe(const std::string& clientId, uint8_t qoS, std::string topic) {
101 if (topic.empty()) {
102 LOG(DEBUG) << "MQTT Broker: Subscribe";
103 LOG(DEBUG) << "MQTT Broker: ClientId: " << clientId;
104
105 clientIds[clientId] = qoS;
106 } else {
107 const std::string topicLevel = topic.substr(0, topic.find('/'));
108
109 topic.erase(0, topicLevel.size() + 1);
110
111 const auto& [it, inserted] = topicLevels.insert({topicLevel, SubscribtionTree::TopicLevel(broker, topicLevel)});
112
113 if (!it->second.subscribe(clientId, qoS, topic)) {
114 LOG(DEBUG) << "MQTT Broker: Erase topic: " << topicLevel << " /" << topic;
115
116 topicLevels.erase(it);
117 } else {
118 LOG(DEBUG) << "MQTT Broker: Topic: " << topicLevel << " /" << topic;
119 }
120 }
121
122 return !topicLevels.empty() || !clientIds.empty();
123 }

125 void SubscribtionTree::TopicLevel::publish(Message& message, std::string topic) {
126 if (topic.empty()) {
127 LOG(DEBUG) << "MQTT Broker: Found match:";
128 LOG(DEBUG) << "MQTT Broker: Topic: '" << message.getTopic() << "';";
129 LOG(DEBUG) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::toHexString(message.getMessage());
130 LOG(DEBUG) << "MQTT Broker: Distribute Publish for match ...";
131
132 for (auto& [clientId, clientQoS] : clientIds) {
133 broker->sendPublish(clientId, message, clientQoS, false);
134 }
135
136 LOG(DEBUG) << "MQTT Broker: ... completed!";
137
138 const auto nextHashLevel = topicLevels.find("#");
139 if (nextHashLevel != topicLevels.end()) {
140 LOG(DEBUG) << "MQTT Broker: Found parent match:";
141 LOG(DEBUG) << "MQTT Broker: Topic: '" << message.getTopic() << "'";
142 LOG(DEBUG) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::toHexString(message.getMessage());
143 LOG(DEBUG) << "MQTT Broker: Distribute Publish for match ...";
144
145 for (auto& [clientId, clientQoS] : nextHashLevel->second.clientIds) {
146 broker->sendPublish(clientId, message, clientQoS, false);
147 }
148
149 LOG(DEBUG) << "MQTT Broker: ... completed!";
150 }
151 } else {
152 const std::string topicLevel = topic.substr(0, topic.find('/'));
153
154 topic.erase(0, topicLevel.size() + 1);
155
156 auto foundNode = topicLevels.find(topicLevel);
157 if (foundNode != topicLevels.end()) {
158 foundNode->second.publish(message, topic);
159 }
160
161 foundNode = topicLevels.find("+");
162 if (foundNode != topicLevels.end()) {
163 foundNode->second.publish(message, topic);
164 }
165
166 foundNode = topicLevels.find("#");
167 if (foundNode != topicLevels.end()) {
168 LOG(DEBUG) << "MQTT Broker: Found match:";
169 LOG(DEBUG) << "MQTT Broker: Topic: '" << message.getTopic() << "';";
170 LOG(DEBUG) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::toHexString(message.getMessage());
171 LOG(DEBUG) << "MQTT Broker: Distribute Publish for match ...";
172
173 for (auto& [clientId, clientQoS] : foundNode->second.clientIds) {
174 broker->sendPublish(clientId, message, clientQoS, false);
175 }
176
177 LOG(DEBUG) << "MQTT Broker: ... completed!";
178 }
179 }

182 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId, std::string topic) {
183 if (topic.empty()) {
184 if (clientIds.erase(clientId) != 0) {
185 LOG(DEBUG) << "MQTT Broker: Unsubscribe";
186 LOG(DEBUG) << "MQTT Broker: ClientId: " << clientId;
187 LOG(DEBUG) << "MQTT Broker: Topic: " << topicLevel;
188 }
189 } else {
190 const std::string topicLevel = topic.substr(0, topic.find('/'));
191
192 auto&& it = topicLevels.find(topicLevel);
193 if (it != topicLevels.end()) {
194 topic.erase(0, topicLevel.size() + 1);
195
196 if (it->second.unsubscribe(clientId, topic)) {
197 LOG(DEBUG) << "MQTT Broker: Erase Topic: " << it->first;
198
199 topicLevels.erase(it);
200 }
201 }
202 }
203
204 return clientIds.empty() && topicLevels.empty();
205 }

207 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId) {
208 if (clientIds.erase(clientId) != 0) {
209 LOG(DEBUG) << "MQTT Broker: Unsubscribe";
210 LOG(DEBUG) << "MQTT Broker: ClientId: " << clientId;
211 LOG(DEBUG) << "MQTT Broker: Topic: " << topicLevel;
212 }
213
214 for (auto it = topicLevels.begin(); it != topicLevels.end();) {
215 if (it->second.unsubscribe(clientId)) {
216 LOG(DEBUG) << "MQTT Broker: Erase Topic: " << it->first;
217
218 it = topicLevels.erase(it);
219 } else {
220 ++it;
221 }
222 }
223
224 return clientIds.empty() && topicLevels.empty();
225 }

231 json["topic_filter"][topicLevelName] = topicLevel.toJson();

241 SubscribtionTree::TopicLevel& SubscribtionTree::TopicLevel::fromJson(const nlohmann::json& json) {
242 clientIds.clear();
243 topicLevels.clear();
244
245 if (json.contains("qos_map")) {
246 for (const auto& subscriber : json["qos_map"].items()) {
247 clientIds.emplace(subscriber.key(), subscriber.value());
248 }
249 }
250
251 if (json.contains("topic_filter")) {
252 for (const auto& topicLevelItem : json["topic_filter"].items()) {
253 topicLevels.emplace(topicLevelItem.key(), TopicLevel(broker, topicLevelItem.key()).fromJson(topicLevelItem.value()));
254 }
255 }
256
257 return *this;
258 }
