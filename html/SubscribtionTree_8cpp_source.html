70 bool SubscribtionTree::subscribe(const std::string& topic, const std::string& clientId, uint8_t qoS) {
71 bool success = false;
72
73 auto hashCount = std::ranges::count(topic, '#');
74 if (!topic.empty() && (hashCount == 0 || (hashCount == 1 && topic.ends_with('#')))) {
75 head.subscribe(clientId, qoS, topic);
76
77 success = true;
78 } else {
79 LOG(ERROR) << "MQTT Broker: Subscribe: Wrong '#' placement: " << topic;
80 }
81
82 return success;
83 }

132 void SubscribtionTree::TopicLevel::appear(const std::string& clientId, const std::string& topic) {
133 if (clientIds.contains(clientId)) {
134 broker->appear(clientId, topic, clientIds[clientId]);
135 }
136
137 for (auto& [topicLevel, subscribtion] : topicLevels) {
138 subscribtion.appear(clientId, std::string(topic).append(topic.empty() ? "" : "/").append(topicLevel));

142 bool SubscribtionTree::TopicLevel::subscribe(const std::string& clientId, uint8_t qoS, std::string topic) {
143 if (topic.empty()) {
144 LOG(INFO) << "MQTT Broker: Subscribe";
145 LOG(INFO) << "MQTT Broker: ClientId: " << clientId;
146
147 clientIds[clientId] = qoS;
148 } else {
149 const std::string topicLevel = topic.substr(0, topic.find('/'));
150
151 topic.erase(0, topicLevel.size() + 1);
152
153 const auto& [it, inserted] = topicLevels.insert({topicLevel, SubscribtionTree::TopicLevel(broker, topicLevel)});
154
155 if (!it->second.subscribe(clientId, qoS, topic)) {
156 LOG(DEBUG) << "MQTT Broker: Erase topic: " << topicLevel << " /" << topic;
157
158 topicLevels.erase(it);
159 } else {
160 LOG(INFO) << "MQTT Broker: Topic: " << topicLevel << " /" << topic;
161 }
162 }
163
164 return !topicLevels.empty() || !clientIds.empty();
165 }

167 void SubscribtionTree::TopicLevel::publish(Message& message, std::string topic) {
168 if (topic.empty()) {
169 LOG(INFO) << "MQTT Broker: Found match:";
170 LOG(INFO) << "MQTT Broker: Topic: '" << message.getTopic() << "';";
171 LOG(INFO) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::toHexString(message.getMessage());
172
173 LOG(INFO) << "MQTT Broker: Distribute PUBLISH for match ...";
174 for (auto& [clientId, clientQoS] : clientIds) {
175 broker->sendPublish(clientId, message, clientQoS, false);
176 }
177 LOG(INFO) << "MQTT Broker: ... distributing PUBLISH for match completed";
178
179 const auto nextHashLevel = topicLevels.find("#");
180 if (nextHashLevel != topicLevels.end()) {
181 LOG(INFO) << "MQTT Broker: Found parent match:";
182 LOG(INFO) << "MQTT Broker: Topic: '" << message.getTopic() << "'";
183 LOG(INFO) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::toHexString(message.getMessage());
184
185 LOG(INFO) << "MQTT Broker: Distribute PUBLISH for match ...";
186 for (auto& [clientId, clientQoS] : nextHashLevel->second.clientIds) {
187 broker->sendPublish(clientId, message, clientQoS, false);
188 }
189 LOG(INFO) << "MQTT Broker: ... distributing PUBLISH for match completed";
190 }
191 } else {
192 const std::string topicLevel = topic.substr(0, topic.find('/'));
193
194 topic.erase(0, topicLevel.size() + 1);
195
196 auto foundNode = topicLevels.find(topicLevel);
197 if (foundNode != topicLevels.end()) {
198 foundNode->second.publish(message, topic);
199 }
200
201 foundNode = topicLevels.find("+");
202 if (foundNode != topicLevels.end()) {
203 foundNode->second.publish(message, topic);
204 }
205
206 foundNode = topicLevels.find("#");
207 if (foundNode != topicLevels.end()) {
208 LOG(INFO) << "MQTT Broker: Found match:";
209 LOG(INFO) << "MQTT Broker: Topic: '" << message.getTopic() << "'";
210 LOG(INFO) << "MQTT Broker: Message:\n" << iot::mqtt::Mqtt::toHexString(message.getMessage());
211
212 LOG(INFO) << "MQTT Broker: Distribute PUBLISH for match '" << message.getTopic() << "' ...";
213 for (auto& [clientId, clientQoS] : foundNode->second.clientIds) {
214 broker->sendPublish(clientId, message, clientQoS, false);
215 }
216 LOG(INFO) << "MQTT Broker: ... distributing PUBLISH for match completed";
217 }
218 }

221 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId, std::string topic) {
222 if (topic.empty()) {
223 if (clientIds.erase(clientId) != 0) {
224 LOG(INFO) << "MQTT Broker: Unsubscribe";
225 LOG(INFO) << "MQTT Broker: ClientId: " << clientId;
226 LOG(INFO) << "MQTT Broker: Topic: " << topicLevel;
227 }
228 } else {
229 const std::string topicLevel = topic.substr(0, topic.find('/'));
230
231 auto&& it = topicLevels.find(topicLevel);
232 if (it != topicLevels.end()) {
233 topic.erase(0, topicLevel.size() + 1);
234
235 if (it->second.unsubscribe(clientId, topic)) {
236 LOG(DEBUG) << "MQTT Broker: Erase Topic: " << it->first;
237
238 topicLevels.erase(it);
239 }
240 }
241 }
242
243 return clientIds.empty() && topicLevels.empty();
244 }

246 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId) {
247 if (clientIds.erase(clientId) != 0) {
248 LOG(INFO) << "MQTT Broker: Unsubscribe";
249 LOG(INFO) << "MQTT Broker: ClientId: " << clientId;
250 LOG(INFO) << "MQTT Broker: Topic: " << topicLevel;
251 }
252
253 for (auto it = topicLevels.begin(); it != topicLevels.end();) {
254 if (it->second.unsubscribe(clientId)) {
255 LOG(DEBUG) << "MQTT Broker: Erase Topic: " << it->first;
256
257 it = topicLevels.erase(it);
258 } else {
259 ++it;
260 }
261 }
262
263 return clientIds.empty() && topicLevels.empty();
264 }

284 std::list<std::string> SubscribtionTree::TopicLevel::getSubscriptions(const std::string& absoluteTopicLevel,
285 const std::string& clientId) const {
286 std::list<std::string> topicLevelList;
287
288 for (const auto& [topicLevelName, nextTopicLevel] : topicLevels) {
289 const std::string currentAbsoluteTopicLevel = absoluteTopicLevel + topicLevelName;
290
291 if (nextTopicLevel.clientIds.contains(clientId)) {
292 topicLevelList.push_back(currentAbsoluteTopicLevel);
293 }
294
295 topicLevelList.splice(topicLevelList.end(), nextTopicLevel.getSubscriptions(currentAbsoluteTopicLevel + "/", clientId));
296 }
297
298 return topicLevelList;
299 }

301 SubscribtionTree::TopicLevel& SubscribtionTree::TopicLevel::fromJson(const nlohmann::json& json) {
302 clientIds.clear();
303 topicLevels.clear();
304
305 if (json.contains("qos_map")) {
306 for (const auto& subscriber : json["qos_map"].items()) {
307 clientIds.emplace(subscriber.key(), subscriber.value());
308 }
309 }
310
311 if (json.contains("topic_filter")) {
312 for (const auto& topicLevelItem : json["topic_filter"].items()) {
313 topicLevels.emplace(topicLevelItem.key(), TopicLevel(broker, topicLevelItem.key()).fromJson(topicLevelItem.value()));
314 }
315 }
316
317 return *this;
318 }
