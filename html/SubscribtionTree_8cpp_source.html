70 void SubscribtionTree::TopicLevel::appear(const std::string& clientId, const std::string& topic) {
71 if (clientIds.contains(clientId)) {
72 broker->appear(clientId, topic, clientIds[clientId]);
73 }
74
75 for (auto& [topicLevel, subscribtion] : topicLevels) {

80 bool SubscribtionTree::TopicLevel::subscribe(const std::string& clientId, uint8_t qoS, std::string topic, bool leafFound) {
81 bool success = true;
82
83 if (leafFound) {
84 clientIds[clientId] = qoS;
85 } else {
86 std::string::size_type slashPosition = topic.find('/');
87
88 std::string topicLevel = topic.substr(0, slashPosition);
89 bool leafFound = slashPosition == std::string::npos;
90
91 if ((topicLevel == "#" && !topic.ends_with('#'))) {
92 success = false;
93 } else {
94 topic.erase(0, topicLevel.size() + 1);
95
96 success = topicLevels.insert({topicLevel, SubscribtionTree::TopicLevel(broker)})
97 .first->second.subscribe(clientId, qoS, topic, leafFound);
98 }
99 }
100
101 return success;
102 }

104 void SubscribtionTree::TopicLevel::publish(Message& message, std::string topic, bool leafFound) {
105 if (leafFound) {
106 LOG(DEBUG) << "Found match:";
107 LOG(DEBUG) << " Topic: '" << message.getTopic() << "';";
108 LOG(DEBUG) << " Message:\n" << iot::mqtt::Mqtt::stringToHexString(message.getMessage());
109 LOG(DEBUG) << "Distribute Publish for match ...";
110
111 for (auto& [clientId, clientQoS] : clientIds) {
112 broker->sendPublish(clientId, message, clientQoS, false);
113 }
114
115 LOG(DEBUG) << "... completed!";
116
117 auto nextHashLevel = topicLevels.find("#");
118 if (nextHashLevel != topicLevels.end()) {
119 LOG(DEBUG) << "Found parent match:";
120 LOG(DEBUG) << " Topic: '" << message.getTopic() << "'";
121 LOG(DEBUG) << " Message:\n" << iot::mqtt::Mqtt::stringToHexString(message.getMessage());
122 LOG(DEBUG) << "Distribute Publish for match ...";
123
124 for (auto& [clientId, clientQoS] : nextHashLevel->second.clientIds) {
125 broker->sendPublish(clientId, message, clientQoS, false);
126 }
127
128 LOG(DEBUG) << "... completed!";
129 }
130 } else {
131 std::string::size_type slashPosition = topic.find('/');
132
133 std::string topicLevel = topic.substr(0, slashPosition);
134 bool leafFound = slashPosition == std::string::npos;
135
136 topic.erase(0, topicLevel.size() + 1);
137
138 auto foundNode = topicLevels.find(topicLevel);
139 if (foundNode != topicLevels.end()) {
140 foundNode->second.publish(message, topic, leafFound);
141 }
142
143 foundNode = topicLevels.find("+");
144 if (foundNode != topicLevels.end()) {
145 foundNode->second.publish(message, topic, leafFound);
146 }
147
148 foundNode = topicLevels.find("#");
149 if (foundNode != topicLevels.end()) {
150 LOG(DEBUG) << "Found match:";
151 LOG(DEBUG) << " Topic: '" << message.getTopic() << "';";
152 LOG(DEBUG) << " Message:\n" << iot::mqtt::Mqtt::stringToHexString(message.getMessage());
153 LOG(DEBUG) << "Distribute Publish for match ...";
154
155 for (auto& [clientId, clientQoS] : foundNode->second.clientIds) {
156 broker->sendPublish(clientId, message, clientQoS, false);
157 }
158
159 LOG(DEBUG) << "... completed!";
160 }
161 }

164 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId, std::string topic, bool leafFound) {
165 if (leafFound) {
166 clientIds.erase(clientId);
167 } else {
168 std::string::size_type slashPosition = topic.find('/');
169
170 std::string topicLevel = topic.substr(0, slashPosition);
171 bool leafFound = slashPosition == std::string::npos;
172
173 topic.erase(0, topicLevel.size() + 1);
174
175 if (topicLevels.contains(topicLevel) && topicLevels.find(topicLevel)->second.unsubscribe(clientId, topic, leafFound)) {
176 topicLevels.erase(topicLevel);
177 }
178 }
179
180 return clientIds.empty() && topicLevels.empty();
181 }

183 bool SubscribtionTree::TopicLevel::unsubscribe(const std::string& clientId) {
184 clientIds.erase(clientId);
185
186 for (auto it = topicLevels.begin(); it != topicLevels.end();) {
187 if (it->second.unsubscribe(clientId)) {
188 it = topicLevels.erase(it);
189 } else {
190 ++it;
191 }
192 }
193
194 return clientIds.empty() && topicLevels.empty();
195 }

211 SubscribtionTree::TopicLevel& SubscribtionTree::TopicLevel::fromJson(const nlohmann::json& json) {
212 clientIds.clear();
213 topicLevels.clear();
214
215 if (json.contains("qos_map")) {
216 for (const auto& subscriber : json["qos_map"].items()) {
217 clientIds.emplace(subscriber.key(), subscriber.value());
218 }
219 }
220
221 if (json.contains("topic_filter")) {
222 for (const auto& topicLevelItem : json["topic_filter"].items()) {
223 topicLevels.emplace(topicLevelItem.key(), TopicLevel(broker).fromJson(topicLevelItem.value()));
224 }
225 }
226
227 return *this;
228 }
