36 bool ApplicationDispatcher::dispatch(express::Controller& controller,
37 const std::string& parentMountPath,
38 const express::MountPoint& mountPoint) {
39 bool dispatched = false;
40
41 if ((controller.getFlags() & Controller::NEXT) == 0) {
42 std::string absoluteMountPath = path_concat(parentMountPath, mountPoint.relativeMountPath);
43 if (((controller.getRequest()->path.rfind(absoluteMountPath, 0) == 0 ||
44 controller.getRequest()->url.rfind(absoluteMountPath, 0) == 0) &&
45 mountPoint.method == "use") ||
46 ((absoluteMountPath == controller.getRequest()->url || absoluteMountPath == controller.getRequest()->path ||
47 checkForUrlMatch(absoluteMountPath, controller.getRequest()->url)) &&
48 (controller.getRequest()->method == mountPoint.method || mountPoint.method == "all"))) {
49 dispatched = true;
50
51 if (hasResult(absoluteMountPath)) {
52 setParams(absoluteMountPath, *controller.getRequest());
53 }
54
55 lambda(*controller.getRequest(), *controller.getResponse());
56 }
57 }
58
59 return dispatched;
60 }
