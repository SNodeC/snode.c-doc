50 ConfigSection::ConfigSection(ConfigInstance* instance, const std::string& name, const std::string& description)
51 : instance(instance) {
52 CLI::App* existingSection = instance->getSection(name);
53 section = existingSection != nullptr ? existingSection
54 : instance //
55 ->addSection(name, description + " for instance '" + instance->getInstanceName() + "'")
56 ->disabled()
57 ->group("Sections");
58 }

60 void ConfigSection::required(CLI::Option* opt, bool req) {
61 if (req != opt->get_required()) {
62 if (req) {
63 ++requiredCount;
64 section->needs(opt);
65 opt->default_str("");
66 } else {
67 --requiredCount;
68 section->remove_needs(opt);
69 }
70
71 opt->required(req);
72
73 instance->required(section, requiredCount > 0);
74 }
75 }
