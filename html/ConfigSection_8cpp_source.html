72 ConfigSection::ConfigSection(ConfigInstance* instance, const std::string& name, const std::string& description)
73 : instance(instance) {
74 CLI::App* existingSection = instance->getSection(name);
75 section = existingSection != nullptr ? existingSection
76 : instance //
77 ->addSection(name, description + " for instance '" + instance->getInstanceName() + "'")
78 ->disabled()
79 ->group("Sections");
80 }

82 void ConfigSection::required(CLI::Option* opt, bool req) {
83 if (req != opt->get_required()) {
84 if (req) {
85 ++requiredCount;
86 section->needs(opt);
87 opt->default_str("");
88 } else {
89 --requiredCount;
90 section->remove_needs(opt);
91 }
92
93 opt->required(req);
94
95 instance->required(section, requiredCount > 0);
96 }
97 }

103 CLI::Option* ConfigSection::addOption(const std::string& name, const std::string& description) {
104 if (!instance->getInstanceName().empty() && !section->get_display_name().empty()) {
105 section->disabled(false);
106 }
107
108 CLI::Option* opt = section //
109 ->add_option(name, description);
110
111 if (opt->get_configurable()) {
112 opt->group(section->get_formatter()->get_label("Persistent Options"));
113 }
114
115 return opt;

152 CLI::Option* ConfigSection::addFlagFunction(const std::string& name,
153 const std::function<void()>& callback,
154 const std::string& description,
155 const std::string& typeName,
156 const std::string& defaultValue) {
157 section->disabled(false);
158
159 CLI::Option* opt = section //
160 ->add_flag_function(
161 name,
162 [callback](std::int64_t) {
163 callback();
164 },
165 description)
166 ->default_val(defaultValue)
167 ->type_name(typeName)
168 ->take_last();
169 if (opt->get_configurable()) {
170 opt->group(section->get_formatter()->get_label("Persistent Options"));
171 }
172
173 return opt;
174 }
