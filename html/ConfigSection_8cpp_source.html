61 void ConfigSection::required(CLI::Option* opt, bool req) {
62 if (req && !opt->get_required()) {
63 ++requiredCount;
64 opt //
65 ->default_str("")
66 ->required()
67 ->clear();
68 section //
69 ->needs(opt);
70 } else if (opt->get_required()) {
71 --requiredCount;
72 opt //
73 ->required(false)
74 ->clear();
75 section //
76 ->remove_needs(opt);
77 }
78
79 section //
80 ->required(requiredCount > 0);
81
82 instance->required(section, req);
83 }
