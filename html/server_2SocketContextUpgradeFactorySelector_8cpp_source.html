66 SocketContextUpgradeFactory* SocketContextUpgradeFactorySelector::select(Request& req, Response& res) {
67 SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
68
69 std::string upgradeContextNames = req.get("upgrade");
70
71 while (!upgradeContextNames.empty() && socketContextUpgradeFactory == nullptr) {
72 std::string upgradeContextName;
73 std::string upgradeContextVersion;
74
75 std::tie(upgradeContextName, upgradeContextNames) = httputils::str_split(upgradeContextNames, ',');
76 std::tie(upgradeContextName, upgradeContextVersion) = httputils::str_split(upgradeContextName, '/');
77
78 httputils::str_trimm(upgradeContextName);
79 httputils::to_lower(upgradeContextName);
80
81 socketContextUpgradeFactory = select(upgradeContextName);
82 }
83
84 if (socketContextUpgradeFactory != nullptr) {
85 socketContextUpgradeFactory->prepare(req, res);
86 }
87
88 return socketContextUpgradeFactory;
89 }
