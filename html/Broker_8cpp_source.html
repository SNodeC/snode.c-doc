40 Broker::Broker(uint8_t maxQoS)
41 : sessionStoreFileName((getenv("MQTT_SESSION_STORE") != nullptr) ? getenv("MQTT_SESSION_STORE") : "") // NOLINT
42 , maxQoS(maxQoS)
43 , subscribtionTree(this)
44 , retainTree(this) {
45 if (!sessionStoreFileName.empty()) {
46 std::ifstream sessionStoreFile(sessionStoreFileName);
47
48 if (sessionStoreFile.is_open()) {
49 try {
50 nlohmann::json sessionStoreJson;
51
52 sessionStoreFile >> sessionStoreJson;
53
54 for (auto& [clientId, sessionJson] : sessionStoreJson["session_store"].items()) {
55 sessionStore[clientId].fromJson(sessionJson);
56 }
57 retainTree.fromJson(sessionStoreJson["retain_tree"]);
58 subscribtionTree.fromJson(sessionStoreJson["subscribtion_tree"]);
59
60 LOG(DEBUG) << "Persistent session data loaded successfull";
61 } catch (const nlohmann::json::exception&) {
62 LOG(DEBUG) << "Starting with empty session: Session store '" << sessionStoreFileName << "' empty or corrupted";
63
64 sessionStore.clear();
65 retainTree.clear();
66 subscribtionTree.clear();
67 }
68
69 sessionStoreFile.close();
70 std::remove(sessionStoreFileName.data());
71
72 LOG(INFO) << "Restoring safed session done";
73 } else {
74 PLOG(INFO) << "Could not read session store '" << sessionStoreFileName << "'";
75 }
76 } else {
77 LOG(INFO) << "Session not reloaded: Session store filename empty";
78 }
79 }

81 Broker::~Broker() {
82 if (!sessionStoreFileName.empty()) {
83 nlohmann::json sessionStoreJson;
84
85 for (auto& [clientId, session] : sessionStore) {
86 sessionStoreJson["session_store"][clientId] = session.toJson();
87 }
88 sessionStoreJson["retain_tree"] = retainTree.toJson();
89 sessionStoreJson["subscribtion_tree"] = subscribtionTree.toJson();
90
91 if (sessionStoreJson["session_store"].empty()) {
92 sessionStoreJson.erase("session_store");
93 }
94 if (sessionStoreJson["retain_tree"].empty()) {
95 sessionStoreJson.erase("retain_tree");
96 }
97 if (sessionStoreJson["subscribtion_tree"].empty()) {
98 sessionStoreJson.erase("subscribtion_tree");
99 }
100
101 std::ofstream sessionStoreFile(sessionStoreFileName);
102
103 if (sessionStoreFile.is_open()) {
104 if (!sessionStoreJson.empty()) {
105 sessionStoreFile << sessionStoreJson;
106 }
107
108 sessionStoreFile.close();
109 } else {
110 PLOG(DEBUG) << "Could not write session store '" << sessionStoreFileName << "'";
111 }
112 } else {
113 LOG(INFO) << "Session not saved: Session store filename empty";
114 }
115 }

139 uint8_t Broker::subscribe(const std::string& clientId, const std::string& topic, uint8_t qoS) {
140 qoS = std::min(maxQoS, qoS);
141 uint8_t returnCode = 0;
142
143 if (subscribtionTree.subscribe(topic, clientId, qoS)) {
144 retainTree.appear(clientId, topic, qoS);
145
146 returnCode = SUBSCRIBTION_SUCCESS | qoS;
147 } else {
148 returnCode = SUBSCRIBTION_FAILURE;
149 }
150
151 return returnCode;
152 }
