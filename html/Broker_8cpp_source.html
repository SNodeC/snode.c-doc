41 Broker::Broker(uint8_t maxQoS)
42 : sessionStoreFileName((getenv("MQTT_SESSION_STORE") != nullptr) ? getenv("MQTT_SESSION_STORE") : "") // NOLINT
43 , maxQoS(maxQoS)
44 , subscribtionTree(this)
45 , retainTree(this) {
46 if (!sessionStoreFileName.empty()) {
47 std::ifstream sessionStoreFile(sessionStoreFileName);
48
49 if (sessionStoreFile.is_open()) {
50 try {
51 nlohmann::json sessionStoreJson;
52
53 sessionStoreFile >> sessionStoreJson;
54
55 for (const auto& [clientId, sessionJson] : sessionStoreJson["session_store"].items()) {
56 sessionStore[clientId].fromJson(sessionJson);
57 }
58 retainTree.fromJson(sessionStoreJson["retain_tree"]);
59 subscribtionTree.fromJson(sessionStoreJson["subscribtion_tree"]);
60
61 LOG(DEBUG) << "MQTT Broker: Persistent session data loaded successful";
62 } catch (const nlohmann::json::exception&) {
63 LOG(DEBUG) << "MQTT Broker: Starting with empty session: Session store '" << sessionStoreFileName
64 << "' empty or corrupted";
65
66 sessionStore.clear();
67 retainTree.clear();
68 subscribtionTree.clear();
69 }
70
71 sessionStoreFile.close();
72 std::remove(sessionStoreFileName.data()); // NOLINT
73
74 LOG(INFO) << "MQTT Broker: Restoring saved session done";
75 } else {
76 PLOG(INFO) << "MQTT Broker: Could not read session store '" << sessionStoreFileName << "'";
77 }
78 } else {
79 LOG(INFO) << "MQTT Broker: Session not reloaded: Session store filename empty";
80 }
81 }

83 Broker::~Broker() {
84 if (!sessionStoreFileName.empty()) {
85 nlohmann::json sessionStoreJson;
86
87 for (auto& [clientId, session] : sessionStore) {
88 sessionStoreJson["session_store"][clientId] = session.toJson();
89 }
90 sessionStoreJson["retain_tree"] = retainTree.toJson();
91 sessionStoreJson["subscribtion_tree"] = subscribtionTree.toJson();
92
93 if (sessionStoreJson["session_store"].empty()) {
94 sessionStoreJson.erase("session_store");
95 }
96 if (sessionStoreJson["retain_tree"].empty()) {
97 sessionStoreJson.erase("retain_tree");
98 }
99 if (sessionStoreJson["subscribtion_tree"].empty()) {
100 sessionStoreJson.erase("subscribtion_tree");
101 }
102
103 std::ofstream sessionStoreFile(sessionStoreFileName);
104
105 if (sessionStoreFile.is_open()) {
106 if (!sessionStoreJson.empty()) {
107 sessionStoreFile << sessionStoreJson;
108 }
109
110 sessionStoreFile.close();
111
112 LOG(DEBUG) << "MQTT Broker: Session store written '" << sessionStoreFileName << "'";
113 } else {
114 PLOG(DEBUG) << "MQTT Broker: Could not write session store '" << sessionStoreFileName << "'";
115 }
116 } else {
117 LOG(INFO) << "MQTT Broker: Session not saved: Session store filename empty";
118 }
119 }

144 uint8_t Broker::subscribe(const std::string& clientId, const std::string& topic, uint8_t qoS) {
145 qoS = std::min(maxQoS, qoS);
146 uint8_t returnCode = 0;
147
148 if (subscribtionTree.subscribe(topic, clientId, qoS)) {
149 retainTree.appear(clientId, topic, qoS);
150
151 returnCode = SUBSCRIBTION_SUCCESS | qoS;
152 } else {
153 returnCode = SUBSCRIBTION_FAILURE;
154 }
155
156 return returnCode;
157 }
