62 std::size_t Mqtt::onProcess() {
63 std::size_t consumed = 0;
64
65 switch (state) {
66 case 0:
67 consumed = fixedHeader.deserialize(mqttContext);
68
69 if (!fixedHeader.isComplete()) {
70 break;
71 } else if (fixedHeader.isError()) {
72 mqttContext->close();
73 break;
74 }
75
76 LOG(DEBUG) << "======================================================";
77 LOG(DEBUG) << "Fixed Header: PacketType: 0x" << std::hex << std::setfill('0') << std::setw(2)
78 << static_cast<uint16_t>(fixedHeader.getPacketType());
79 LOG(DEBUG) << " PacketFlags: 0x" << std::hex << std::setfill('0') << std::setw(2)
80 << static_cast<uint16_t>(fixedHeader.getFlags()) << std::dec;
81 LOG(DEBUG) << " RemainingLength: " << fixedHeader.getRemainingLength();
82
83 controlPacketDeserializer = createControlPacketDeserializer(fixedHeader);
84
85 fixedHeader.reset();
86
87 if (controlPacketDeserializer == nullptr) {
88 LOG(DEBUG) << "Received packet-type is unavailable ... closing connection";
89
90 mqttContext->end(true);
91 break;
92 } else if (controlPacketDeserializer->isError()) {
93 LOG(DEBUG) << "Fixed header has error ... closing connection";
94
95 delete controlPacketDeserializer;
96 controlPacketDeserializer = nullptr;
97
98 mqttContext->end(true);
99 break;
100 }
101
102 state++;
103 [[fallthrough]];
104 case 1:
105 consumed += controlPacketDeserializer->deserialize(mqttContext);
106
107 if (controlPacketDeserializer->isError()) {
108 LOG(DEBUG) << "Control packet has error ... closing connection";
109 mqttContext->end(true);
110
111 delete controlPacketDeserializer;
112 controlPacketDeserializer = nullptr;
113
114 state = 0;
115 } else if (controlPacketDeserializer->isComplete()) {
116 deliverPacket(controlPacketDeserializer);
117
118 delete controlPacketDeserializer;
119 controlPacketDeserializer = nullptr;
120
121 state = 0;
122
123 keepAliveTimer.restart();
124 }
125
126 break;
127 }
128
129 return consumed;
130 }

143 void Mqtt::initSession(Session* session, const utils::Timeval& keepAlive) {
144 this->session = session;
145
146 for (auto& [packetIdentifier, publish] : session->publishMap) {
147 LOG(DEBUG) << "Resend PUBLISH";
148 LOG(DEBUG) << "==============";
149
150 send(publish);
151 }
152
153 for (uint16_t packetIdentifier : session->pubrelPacketIdentifierSet) {
154 LOG(DEBUG) << "Resend PUBREL";
155 LOG(DEBUG) << "=============";
156
157 send(iot::mqtt::packets::Pubrel(packetIdentifier));
158 }
159
160 if (keepAlive > 0) {
161 keepAliveTimer = core::timer::Timer::singleshotTimer(
162 [this, keepAlive]() -> void {
163 LOG(DEBUG) << "Keep-alive timer expired. Interval was: " << keepAlive;
164 mqttContext->close();
165 },
166 keepAlive);
167
168 getSocketConnection()->setTimeout(0);
169 }
170 }

182 void Mqtt::sendPublish(const std::string& topic, const std::string& message, uint8_t qoS,
183 bool retain) { // Server & Client
184 LOG(DEBUG) << "Send PUBLISH";
185 LOG(DEBUG) << "============";
186
187 uint16_t pId = qoS != 0 ? getPacketIdentifier() : 0;
188
189 send(iot::mqtt::packets::Publish(pId, topic, message, qoS, false, retain));
190
191 if (qoS == 2) {
192 session->publishMap.emplace(pId, iot::mqtt::packets::Publish(pId, topic, message, qoS, true, retain));
193 }
194 }

239 bool Mqtt::_onPublish(const packets::Publish& publish) {
240 bool deliver = true;
241 printStandardHeader(publish);
242 LOG(DEBUG) << "Topic: " << publish.getTopic();
243 LOG(DEBUG) << "Message: " << publish.getMessage();
244 LOG(DEBUG) << "QoS: " << static_cast<uint16_t>(publish.getQoS());
245 LOG(DEBUG) << "PacketIdentifier: " << publish.getPacketIdentifier();
246 LOG(DEBUG) << "DUP: " << publish.getDup();
247 LOG(DEBUG) << "Retain: " << publish.getRetain();
248
249 if (publish.getQoS() > 2) {
250 LOG(DEBUG) << "Received invalid QoS: " << publish.getQoS();
251 mqttContext->end(true);
252 deliver = false;
253 } else if (publish.getPacketIdentifier() == 0 && publish.getQoS() > 0) {
254 LOG(DEBUG) << "Received QoS > 0 but no PackageIdentifier present";
255 mqttContext->end(true);
256 deliver = false;
257 } else {
258 switch (publish.getQoS()) {
259 case 1:
260 sendPuback(publish.getPacketIdentifier());
261
262 break;
263 case 2:
264 sendPubrec(publish.getPacketIdentifier());
265
266 if (session->publishPacketIdentifierSet.contains(publish.getPacketIdentifier())) {
267 deliver = false;
268 } else {
269 session->publishPacketIdentifierSet.insert(publish.getPacketIdentifier());
270 }
271
272 break;
273 }
274 }
275
276 return deliver;
277 }

355 std::string Mqtt::dataToHexString(const std::vector<char>& data) {
356 std::stringstream ss;
357
358 ss << "Data: ";
359 unsigned long i = 0;
360 for (char ch : data) {
361 if (i != 0 && i % 8 == 0 && i != data.size()) {
362 ss << std::endl;
363 ss << " ";
364 }
365 ++i;
366 ss << "0x" << std::hex << std::setfill('0') << std::setw(2) << static_cast<uint16_t>(static_cast<uint8_t>(ch))
367 << " "; // << " | ";
368 }
369
370 return ss.str();
371 }
