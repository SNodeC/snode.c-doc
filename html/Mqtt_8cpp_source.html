68 std::size_t Mqtt::onReceivedFromPeer() {
69 std::size_t consumed = 0;
70
71 switch (state) {
72 case 0:
73 consumed = fixedHeader.deserialize(mqttContext);
74
75 if (!fixedHeader.isComplete()) {
76 break;
77 }
78 if (fixedHeader.isError()) {
79 mqttContext->close();
80 break;
81 }
82 printFixedHeader(fixedHeader);
83
84 controlPacketDeserializer = createControlPacketDeserializer(fixedHeader);
85
86 fixedHeader.reset();
87
88 if (controlPacketDeserializer == nullptr) {
89 LOG(DEBUG) << "MQTT: Received packet-type is unavailable ... closing connection";
90
91 mqttContext->end(true);
92 break;
93 }
94 if (controlPacketDeserializer->isError()) {
95 LOG(DEBUG) << "MQTT: Fixed header has error ... closing connection";
96
97 delete controlPacketDeserializer;
98 controlPacketDeserializer = nullptr;
99
100 mqttContext->end(true);
101 break;
102 }
103
104 state++;
105
106 [[fallthrough]];
107 case 1:
108 consumed += controlPacketDeserializer->deserialize(mqttContext);
109
110 if (controlPacketDeserializer->isError()) {
111 LOG(DEBUG) << "MQTT: Control packet has error ... closing connection";
112 mqttContext->end(true);
113
114 delete controlPacketDeserializer;
115 controlPacketDeserializer = nullptr;
116
117 state = 0;
118 } else if (controlPacketDeserializer->isComplete()) {
119 deliverPacket(controlPacketDeserializer);
120
121 delete controlPacketDeserializer;
122 controlPacketDeserializer = nullptr;
123
124 state = 0;
125
126 keepAliveTimer.restart();
127 }
128
129 break;
130 }
131
132 return consumed;
133 }

143 void Mqtt::initSession(Session* session, utils::Timeval keepAlive) {
144 this->session = session;
145
146 for (const auto& [packetIdentifier, publish] : session->publishMap) {
147 LOG(DEBUG) << "MQTT: PUBLISH Resend";
148
149 send(publish);
150 }
151
152 for (const uint16_t packetIdentifier : session->pubrelPacketIdentifierSet) {
153 LOG(DEBUG) << "MQTT: PUBREL Resend";
154
155 send(iot::mqtt::packets::Pubrel(packetIdentifier));
156 }
157
158 if (keepAlive > 0) {
159 keepAlive *= 1.5;
160
161 LOG(TRACE) << "MQTT: Keep alive initialized with: " << keepAlive;
162
163 keepAliveTimer = core::timer::Timer::singleshotTimer(
164 [this, keepAlive]() -> void {
165 LOG(DEBUG) << "MQTT: Keep-alive timer expired. Interval was: " << keepAlive;
166 mqttContext->close();
167 },
168 keepAlive);
169
170 getSocketConnection()->setTimeout(0);
171 }
172 }

186 void Mqtt::sendPublish(const std::string& topic, const std::string& message, uint8_t qoS,
187 bool retain) { // Server & Client
188
189 uint16_t packageIdentifier = qoS != 0 ? getPacketIdentifier() : 0;
190
191 send(iot::mqtt::packets::Publish(packageIdentifier, topic, message, qoS, false, retain));
192
193 if (qoS == 2) {
194 session->publishMap.emplace(packageIdentifier,
195 iot::mqtt::packets::Publish(packageIdentifier, topic, message, qoS, true, retain));
196 }

230 bool Mqtt::_onPublish(const packets::Publish& publish) {
231 bool deliver = true;
232
233 LOG(DEBUG) << "MQTT: Topic: " << publish.getTopic();
234 LOG(DEBUG) << "MQTT: Message:\n" << toHexString(publish.getMessage());
235 LOG(DEBUG) << "MQTT: QoS: " << static_cast<uint16_t>(publish.getQoS());
236 LOG(DEBUG) << "MQTT: PacketIdentifier: " << publish.getPacketIdentifier();
237 LOG(DEBUG) << "MQTT: DUP: " << publish.getDup();
238 LOG(DEBUG) << "MQTT: Retain: " << publish.getRetain();
239
240 if (publish.getQoS() > 2) {
241 LOG(DEBUG) << "MQTT: Received invalid QoS: " << publish.getQoS();
242 mqttContext->end(true);
243 deliver = false;
244 } else if (publish.getPacketIdentifier() == 0 && publish.getQoS() > 0) {
245 LOG(DEBUG) << "MQTT: Received QoS > 0 but no PackageIdentifier present";
246 mqttContext->end(true);
247 deliver = false;
248 } else {
249 switch (publish.getQoS()) {
250 case 1:
251 sendPuback(publish.getPacketIdentifier());
252
253 break;
254 case 2:
255 sendPubrec(publish.getPacketIdentifier());
256
257 if (session->publishPacketIdentifierSet.contains(publish.getPacketIdentifier())) {
258 deliver = false;
259 } else {
260 session->publishPacketIdentifierSet.insert(publish.getPacketIdentifier());
261 }
262
263 break;
264 }
265 }
266
267 return deliver;
268 }
