45 template <typename PhysicalSocket>
46 void SocketWriter<PhysicalSocket>::doWrite() {
47 if (!writeBuffer.empty()) {
48 std::size_t writeLen = (writeBuffer.size() < blockSize) ? writeBuffer.size() : blockSize;
49 ssize_t retWrite = write(writeBuffer.data(), writeLen);
50
51 if (retWrite > 0) {
52 writeBuffer.erase(writeBuffer.begin(), writeBuffer.begin() + retWrite);
53
54 if (!isSuspended()) {
55 suspend();
56 }
57 if (!writeBuffer.empty()) {
58 if (writeBuffer.capacity() > writeBuffer.size() * 2) {
59 writeBuffer.shrink_to_fit();
60 }
61 span();
62 } else {
63 writeBuffer.shrink_to_fit();
64
65 if (markShutdown) {
66 shutdown(onShutdown);
67 }
68 }
69 } else if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
70 if (isSuspended()) {
71 resume();
72 }
73 } else {
74 disable();
75 onStatus(errno);
76 }
77 } else if (!isSuspended()) {
78 suspend();
79 }
80 }

87 template <typename PhysicalSocket>
88 void SocketWriter<PhysicalSocket>::sendToPeer(const char* junk, std::size_t junkLen) {
89 if (!shutdownInProgress && !markShutdown && isEnabled()) {
90 if (writeBuffer.empty()) {
91 resume();
92 }
93
94 writeBuffer.insert(writeBuffer.end(), junk, junk + junkLen);
95 }

109 template <typename PhysicalSocket>
110 void SocketWriter<PhysicalSocket>::shutdown(const std::function<void(int)>& onShutdown) {
111 if (!shutdownInProgress) {
112 this->onShutdown = onShutdown;
113 if (writeBuffer.empty()) {
114 shutdownInProgress = true;
115 LOG(TRACE) << "Initiating shutdown process";
116 doWriteShutdown(onShutdown);
117 } else {
118 markShutdown = true;
119 }
120 }
121 }

123 template <typename PhysicalSocket>
124 void SocketWriter<PhysicalSocket>::terminate() {
125 if (!terminateInProgress) {
126 setTimeout(terminateTimeout);
127 shutdown([this]([[maybe_unused]] int errnum) -> void {
128 if (errnum != 0) {
129 PLOG(TRACE) << "SocketWriter::doWriteShutdown";
130 }
131 disable();
132 });
133 terminateInProgress = true;
134 }
