66 void PipeSource::writeEvent() {
67 ssize_t ret = core::system::write(
68 getRegisteredFd(), writeBuffer.data(), (writeBuffer.size() < MAX_SEND_JUNKSIZE) ? writeBuffer.size() : MAX_SEND_JUNKSIZE);
69
70 if (ret > 0) {
71 writeBuffer.erase(writeBuffer.begin(), writeBuffer.begin() + ret);
72
73 if (writeBuffer.empty()) {
74 WriteEventReceiver::suspend();
75 }
76 } else if (errno != EAGAIN && errno != EWOULDBLOCK && errno != EINTR) {
77 WriteEventReceiver::disable();
78
79 if (onError) {
80 onError(errno);
81 }
82 }
83 }
