70 void PipeSource::writeEvent() {
71 const ssize_t ret = core::system::write(
72 getRegisteredFd(), writeBuffer.data(), (writeBuffer.size() < MAX_SEND_JUNKSIZE) ? writeBuffer.size() : MAX_SEND_JUNKSIZE);
73
74 if (ret > 0) {
75 writeBuffer.erase(writeBuffer.begin(), writeBuffer.begin() + ret);
76
77 if (writeBuffer.empty()) {
78 WriteEventReceiver::suspend();
79 }
80 } else if (errno != EAGAIN && errno != EWOULDBLOCK && errno != EINTR) {
81 WriteEventReceiver::disable();
82
83 if (onError) {
84 onError(errno);
85 }
86 }
87 }
