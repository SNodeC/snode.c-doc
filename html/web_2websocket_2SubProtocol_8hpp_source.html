34 template <typename SocketContextUpgradeT>
35 SubProtocol<SocketContextUpgradeT>::SubProtocol(SubProtocolContext* subProtocolContext,
36 const std::string& name,
37 int pingInterval,
38 int maxFlyingPings)
39 : name(name)
40 , subProtocolContext(subProtocolContext) {
41 if (pingInterval > 0) {
42 pingTimer = core::timer::Timer::intervalTimer(
43 [this, maxFlyingPings](const std::function<void()>& stop) -> void {
44 if (flyingPings < maxFlyingPings) {
45 LOG(INFO) << "Ping sent";
46 sendPing();
47 flyingPings++;
48 } else {
49 LOG(INFO) << "MaxFlyingPings exceeded - closing";
50 sendClose();
51 stop();
52 }
53 },
54 pingInterval);
55 }
56
57 subProtocolContext->getSocketConnection()->setTimeout(0);
58 }
