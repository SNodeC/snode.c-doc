59 template <typename SocketContextUpgradeFactory>
60 SocketContextUpgradeFactory*
61 SocketContextUpgradeFactorySelector<SocketContextUpgradeFactory>::load(const std::string& socketContextUpgradeName,
62 const std::string& socketContextUpgradeFactoryLibraryFile,
63 const std::string& socketContextUpgradeFactoryFunctionName) {
64 SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
65
66 void* handle = dlOpen(socketContextUpgradeFactoryLibraryFile, RTLD_LAZY | RTLD_GLOBAL);
67
68 if (handle != nullptr) {
69 SocketContextUpgradeFactory* (*getSocketContextUpgradeFactory)() = reinterpret_cast<SocketContextUpgradeFactory* (*) ()>(
70 core::DynamicLoader::dlSym(handle, socketContextUpgradeFactoryFunctionName));
71
72 if (getSocketContextUpgradeFactory != nullptr) {
73 socketContextUpgradeFactory = getSocketContextUpgradeFactory();
74
75 if (socketContextUpgradeFactory != nullptr) {
76 if (add(socketContextUpgradeFactory, handle)) {
77 VLOG(0) << "SocketContextUpgradeFactory created successfull: " << socketContextUpgradeName;
78 } else {
79 VLOG(0) << "UpgradeSocketContext already existing. Not using: " << socketContextUpgradeName;
80 delete socketContextUpgradeFactory;
81 socketContextUpgradeFactory = nullptr;
82 core::DynamicLoader::dlClose(handle);
83 }
84 } else {
85 VLOG(0) << "SocketContextUpgradeFactory not created: " << socketContextUpgradeName;
86 core::DynamicLoader::dlClose(handle);
87 }
88 } else {
89 VLOG(0) << "Optaining function \"" << socketContextUpgradeFactoryFunctionName
90 << "\" in plugin failed: " << core::DynamicLoader::dlError();
91 core::DynamicLoader::dlClose(handle);
92 }
93 }
94
95 return socketContextUpgradeFactory;
96 }

98 template <typename SocketContextUpgradeFactory>
99 SocketContextUpgradeFactory*
100 SocketContextUpgradeFactorySelector<SocketContextUpgradeFactory>::select(const std::string& socketContextUpgradeName) {
101 SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
102
103 if (socketContextUpgradePlugins.contains(socketContextUpgradeName)) {
104 socketContextUpgradeFactory = socketContextUpgradePlugins[socketContextUpgradeName].socketContextUpgradeFactory;
105 } else if (linkedSocketContextUpgradePlugins.contains(socketContextUpgradeName)) {
106 socketContextUpgradeFactory = linkedSocketContextUpgradePlugins[socketContextUpgradeName]();
107 add(socketContextUpgradeFactory);
108 } else if (!onlyLinked) {
109 socketContextUpgradeFactory = load(socketContextUpgradeName);
110 }
111
112 return socketContextUpgradeFactory;
113 }

129
130 if (socketContextUpgradePlugins.contains(upgradeContextNames)) {
131 SocketContextPlugin& socketContextPlugin = socketContextUpgradePlugins[upgradeContextNames];

137 }
138
139 socketContextUpgradePlugins.erase(upgradeContextNames);
