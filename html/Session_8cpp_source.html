38 nlohmann::json Session::toJson() const {
39 nlohmann::json json;
40
41 std::vector<nlohmann::json> publishJsonVector;
42 for (const auto& [packetIdentifier, publish] : publishMap) {
43 nlohmann::json publishJson;
44
45 publishJson["packet_identifier"] = packetIdentifier;
46 publishJson["topic"] = publish.getTopic();
47 publishJson["message"] = publish.getMessage();
48 publishJson["dup"] = publish.getDup();
49 publishJson["qos"] = publish.getQoS();
50 publishJson["retain"] = publish.getRetain();
51
52 publishJsonVector.emplace_back(publishJson);
53 }
54
55 if (!publishJsonVector.empty()) {
56 json["publish_map"] = publishJsonVector;
57 }
58 if (!pubrelPacketIdentifierSet.empty()) {
59 json["pubrel_packetidentifier_set"] = pubrelPacketIdentifierSet;
60 }
61 if (!publishPacketIdentifierSet.empty()) {
62 json["publish_packetidentifier_set"] = publishPacketIdentifierSet;
63 }
64
65 return json;
66 }

68 void Session::fromJson(const nlohmann::json& json) {
69 if (!json.empty()) {
70 if (json.contains("pubrel_packetidentifier_set")) {
71 pubrelPacketIdentifierSet = static_cast<std::set<uint16_t>>(json["pubrel_packetidentifier_set"]);
72 }
73 if (json.contains("publish_packetidentifier_set")) {
74 publishPacketIdentifierSet = static_cast<std::set<uint16_t>>(json["publish_packetidentifier_set"]);
75 }
76
77 if (json.contains("publish_map")) {
78 for (const auto& publishJson : json["publish_map"]) {
79 if (publishJson.contains("packet_identifer")) {
80 publishMap.emplace(publishJson["packet_identifier"],
81 iot::mqtt::packets::Publish(publishJson["packet_identifier"],
82 publishJson["topic"],
83 publishJson["message"],
84 publishJson["qos"],
85 publishJson["dup"],
86 publishJson["retain"]));
87 }
88 }
89 }
90 }
