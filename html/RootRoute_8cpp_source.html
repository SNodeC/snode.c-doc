74 void RootRoute::dispatch(Controller& controller) {
75 if (!Route::dispatch(controller, "")) {
76 if (strict) {
77 controller.getResponse()->sendStatus(404);
78 } else {
79 if (controller.laxRouting() && !Route::dispatch(controller, "")) {
80 controller.getResponse()->sendStatus(404);
81 }
82 }
