77 void RootRoute::dispatch(Controller& controller) {
78 if (!Route::dispatch(controller, "")) {
79 if (strict) {
80 controller.getResponse()->sendStatus(404);
81 } else if (controller.laxRouting() && !Route::dispatch(controller, "")) {
82 controller.getResponse()->sendStatus(404);
83 }
