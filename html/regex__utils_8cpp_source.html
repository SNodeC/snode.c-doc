56 std::string path_concat(const std::vector<std::string>& stringvec) {
57 std::string s;
58
59 for (std::vector<std::string>::size_type i = 0; i < stringvec.size(); i++) {
60 if (!stringvec[i].empty() && stringvec[i].front() != ' ') {
61 s += "\\/" + stringvec[i];
62 }
63 }
64
65 return s;
66 }

68 std::vector<std::string> explode(const std::string& s, char delim) {
69 std::vector<std::string> result;
70 std::istringstream iss(s);
71
72 for (std::string token; std::getline(iss, token, delim);) {
73 result.push_back(std::move(token));
74 }
75
76 return result;
77 }

100 bool matchFunction(const std::string& cpath, const std::string& reqpath) {
101 std::vector<std::string> explodedString = explode(cpath, '/');
102
103 for (std::vector<std::string>::size_type i = 0; i < explodedString.size(); i++) {
104 if (explodedString[i].front() == ':') {
105 const std::smatch smatch = matchResult(explodedString[i]);
106 std::string regex = "(.*)";
107
108 if (smatch.size() > 1) {
109 if (smatch[1] != "") {
110 regex = smatch[1];
111 }
112 }
113
114 explodedString[i] = regex;
115 }
116 }
117
118 const std::string regexPath = path_concat(explodedString);
119
120 return std::regex_match(reqpath, std::regex(regexPath));
121 }

123 void setParams(const std::string& cpath, Request& req) {
124 std::vector<std::string> explodedString = explode(cpath, '/');
125 std::vector<std::string> explodedReqString = explode(req.url, '/');
126
127 for (std::vector<std::string>::size_type i = 0; i < explodedString.size() && i < explodedReqString.size(); i++) {
128 if (explodedString[i].front() == ':') {
129 const std::smatch smatch = matchResult(explodedString[i]);
130 std::string regex = "(.*)";
131
132 if (smatch.size() > 1) {
133 if (smatch[1] != "") {
134 regex = smatch[1];
135 }
136 }
137
138 if (std::regex_match(explodedReqString[i], std::regex(regex))) {
139 std::string attributeName = smatch[0];
140 attributeName.erase(0, 1);
141 attributeName.erase((attributeName.length() - static_cast<std::size_t>(smatch[1].length())),
142 static_cast<std::size_t>(smatch[1].length()));
143
144 req.params[attributeName] = explodedReqString[i];
145 }
146 }
147 }

150 std::string path_concat(const std::string& first, const std::string& second) {
151 std::string result;
152
153 if (first.back() == '/' && second.front() == '/') {
154 result = first + second.substr(1);
155 } else if (first.back() != '/' && second.front() != '/') {
156 result = first + '/' + second;
157 } else {
158 result = first + second;
159 }
160
161 if (result.length() > 1 && result.back() == '/') {
162 result.pop_back();
163 }
164
165 return result;
166 }
