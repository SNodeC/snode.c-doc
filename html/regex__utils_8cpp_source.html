36 const std::string path_concat(const std::vector<std::string>& stringvec) {
37 std::string s;
38
39 for (std::vector<std::string>::size_type i = 0; i < stringvec.size(); i++) {
40 if (!stringvec[i].empty() && stringvec[i].front() != ' ') {
41 s += "\\/" + stringvec[i];
42 }
43 }
44
45 return s;
46 }

48 const std::vector<std::string> explode(const std::string& s, char delim) {
49 std::vector<std::string> result;

80 bool matchFunction(const std::string& cpath, const std::string& reqpath) {
81 std::vector<std::string> explodedString = explode(cpath, '/');
82
83 for (std::vector<std::string>::size_type i = 0; i < explodedString.size(); i++) {
84 if (explodedString[i].front() == ':') {
85 std::smatch smatch = matchResult(explodedString[i]);
86 std::string regex = "(.*)";
87
88 if (smatch.size() > 1) {
89 if (smatch[1] != "") {
90 regex = smatch[1];
91 }
92 }
93
94 explodedString[i] = regex;
95 }
96 }
97
98 std::string regexPath = path_concat(explodedString);
99
100 return std::regex_match(reqpath, std::regex(regexPath));
101 }

103 void setParams(const std::string& cpath, Request& req) {
104 std::vector<std::string> explodedString = explode(cpath, '/');
105 std::vector<std::string> explodedReqString = explode(req.url, '/');
106
107 for (std::vector<std::string>::size_type i = 0; i < explodedString.size() && i < explodedReqString.size(); i++) {
108 if (explodedString[i].front() == ':') {
109 std::smatch smatch = matchResult(explodedString[i]);
110 std::string regex = "(.*)";
111
112 if (smatch.size() > 1) {
113 if (smatch[1] != "") {
114 regex = smatch[1];
115 }
116 }
117
118 if (std::regex_match(explodedReqString[i], std::regex(regex))) {
119 std::string attributeName = smatch[0];
120 attributeName.erase(0, 1);
121 attributeName.erase((attributeName.length() - static_cast<std::size_t>(smatch[1].length())),
122 static_cast<std::size_t>(smatch[1].length()));
123
124 req.params[attributeName] = explodedReqString[i];
125 }
126 }
127 }

130 std::string path_concat(const std::string& first, const std::string& second) {
131 std::string result;
132
133 if (first.back() == '/' && second.front() == '/') {
134 result = first + second.substr(1);
135 } else if (first.back() != '/' && second.front() != '/') {
136 result = first + '/' + second;
137 } else {
138 result = first + second;
139 }
140
141 if (result.length() > 1 && result.back() == '/') {
142 result.pop_back();
143 }
144
145 return result;
146 }
