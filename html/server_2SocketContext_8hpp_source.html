134 template <typename Request, typename Response>
135 void SocketContext<Request, Response>::requestParsed() {
136 if (!requestInProgress) {
137 currentRequestContext = requestContexts.front();
138 requestContexts.pop_front();
139
140 requestInProgress = true;
141 if (currentRequestContext->status == 0) {
142 if (((currentRequestContext->request.connectionState == ConnectionState::Close) ||
143 (currentRequestContext->request.httpMajor == 0 && currentRequestContext->request.httpMinor == 9) ||
144 (currentRequestContext->request.httpMajor == 1 && currentRequestContext->request.httpMinor == 0 &&
145 currentRequestContext->request.connectionState != ConnectionState::Keep) ||
146 (currentRequestContext->request.httpMajor == 1 && currentRequestContext->request.httpMinor == 1 &&
147 currentRequestContext->request.connectionState == ConnectionState::Close))) {
148 currentRequestContext->response.set("Connection", "close");
149 } else {
150 currentRequestContext->response.set("Connection", "keep-alive");
151 }
152
153 onRequestReady(currentRequestContext->request, currentRequestContext->response);
154 } else {
155 currentRequestContext->response.status(currentRequestContext->status).send(currentRequestContext->reason);
156 reset();
157 shutdownWrite(true);
158 // close();
159 }
160 }
161 }

163 template <typename Request, typename Response>
164 void SocketContext<Request, Response>::sendToPeerCompleted() {
165 // if 0.9 => terminate
166 // if 1.0 && (request != Keep || contentLength = -1) => terminate
167 // if 1.1 && (request == Close || contentLength = -1) => terminate
168 // if (request == Close) => terminate
169
170 if ((currentRequestContext->request.httpMajor == 0 && currentRequestContext->request.httpMinor == 9) ||
171 (currentRequestContext->request.httpMajor == 1 && currentRequestContext->request.httpMinor == 0 &&
172 currentRequestContext->request.connectionState != ConnectionState::Keep) ||
173 (currentRequestContext->request.httpMajor == 1 && currentRequestContext->request.httpMinor == 1 &&
174 currentRequestContext->request.connectionState == ConnectionState::Close) ||
175 (currentRequestContext->response.connectionState == ConnectionState::Close)) {
176 reset();
177 shutdownWrite();
178 } else {
179 reset();
180
181 if (!requestContexts.empty() && requestContexts.front()->ready) {
182 requestParsed();
183 }
184 }
