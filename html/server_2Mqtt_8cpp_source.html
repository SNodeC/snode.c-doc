67 iot::mqtt::ControlPacketDeserializer* Mqtt::createControlPacketDeserializer(iot::mqtt::FixedHeader& fixedHeader) {
68 iot::mqtt::ControlPacketDeserializer* controlPacketDeserializer = nullptr;
69
70 switch (fixedHeader.getPacketType()) {
71 case MQTT_CONNECT:
72 controlPacketDeserializer =
73 new iot::mqtt::server::packets::Connect(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
74 break;
75 case MQTT_PUBLISH:
76 controlPacketDeserializer =
77 new iot::mqtt::server::packets::Publish(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
78 break;
79 case MQTT_PUBACK:
80 controlPacketDeserializer =
81 new iot::mqtt::server::packets::Puback(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
82 break;
83 case MQTT_PUBREC:
84 controlPacketDeserializer =
85 new iot::mqtt::server::packets::Pubrec(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
86 break;
87 case MQTT_PUBREL:
88 controlPacketDeserializer =
89 new iot::mqtt::server::packets::Pubrel(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
90 break;
91 case MQTT_PUBCOMP:
92 controlPacketDeserializer =
93 new iot::mqtt::server::packets::Pubcomp(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
94 break;
95 case MQTT_SUBSCRIBE:
96 controlPacketDeserializer =
97 new iot::mqtt::server::packets::Subscribe(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
98 break;
99 case MQTT_UNSUBSCRIBE:
100 controlPacketDeserializer =
101 new iot::mqtt::server::packets::Unsubscribe(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
102 break;
103 case MQTT_PINGREQ:
104 controlPacketDeserializer =
105 new iot::mqtt::server::packets::Pingreq(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
106 break;
107 case MQTT_DISCONNECT:
108 controlPacketDeserializer =
109 new iot::mqtt::server::packets::Disconnect(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
110 break;
111 default:
112 controlPacketDeserializer = nullptr;
113 break;
114 }
115
116 return controlPacketDeserializer;
117 }

123 void Mqtt::initSession(const utils::Timeval& keepAlive) {
124 if (broker->hasActiveSession(clientId)) {
125 LOG(DEBUG) << "Existing session found for ClientId = `" << clientId << "'";
126 LOG(DEBUG) << " closing";
127
128 sendConnack(MQTT_CONNACK_IDENTIFIERREJECTED, 0);
129
130 mqttContext->end(true);
131
132 willFlag = false;
133 } else if (broker->hasRetainedSession(clientId)) {
134 sendConnack(MQTT_CONNACK_ACCEPT, cleanSession ? MQTT_SESSION_NEW : MQTT_SESSION_PRESENT);
135
136 LOG(DEBUG) << "Retained session found for ClientId = '" << clientId << "'";
137 if (cleanSession) {
138 LOG(DEBUG) << " clean Session = " << this;
139 broker->unsubscribe(clientId);
140 initSession(broker->newSession(clientId, this), keepAlive);
141 } else {
142 LOG(DEBUG) << " renew Session = " << this;
143 initSession(broker->renewSession(clientId, this), keepAlive);
144 broker->restartSession(clientId);
145 }
146 } else {
147 sendConnack(MQTT_CONNACK_ACCEPT, MQTT_SESSION_NEW);
148
149 LOG(DEBUG) << "No session found for ClientId = '" << clientId << "\'";
150 LOG(DEBUG) << " new Session = " << this;
151
152 initSession(broker->newSession(clientId, this), keepAlive);
153 }
154 }

156 void Mqtt::releaseSession() {
157 if (broker->isActiveSession(clientId, this)) {
158 if (cleanSession) {

160 broker->deleteSession(clientId);
161 } else {
162 LOG(DEBUG) << "Retain session: " << clientId;
163 broker->retainSession(clientId);
164 }
165 }

183 void Mqtt::_onConnect(const iot::mqtt::server::packets::Connect& connect) {
184 LOG(DEBUG) << "Received CONNECT: " << clientId;
185 LOG(DEBUG) << "=================";
186 printStandardHeader(connect);
187 LOG(DEBUG) << "Protocol: " << connect.getProtocol();
188 LOG(DEBUG) << "Version: " << static_cast<uint16_t>(connect.getLevel());
189 LOG(DEBUG) << "ConnectFlags: 0x" << std::hex << std::setfill('0') << std::setw(2)
190 << static_cast<uint16_t>(connect.getConnectFlags()) << std::dec << std::setw(0);
191 LOG(DEBUG) << "KeepAlive: " << connect.getKeepAlive();
192 LOG(DEBUG) << "ClientID: " << connect.getClientId();
193 LOG(DEBUG) << "CleanSession: " << connect.getCleanSession();
194
195 if (connect.getWillFlag()) {
196 LOG(DEBUG) << "WillTopic: " << connect.getWillTopic();
197 LOG(DEBUG) << "WillMessage: " << connect.getWillMessage();
198 LOG(DEBUG) << "WillQoS: " << static_cast<uint16_t>(connect.getWillQoS());
199 LOG(DEBUG) << "WillRetain: " << connect.getWillRetain();
200 }
201 if (connect.getUsernameFlag()) {
202 LOG(DEBUG) << "Username: " << connect.getUsername();
203 }
204 if (connect.getPasswordFlag()) {
205 LOG(DEBUG) << "Password: " << connect.getPassword();
206 }
207
208 if (connect.getProtocol() != "MQTT") {
209 LOG(DEBUG) << "Wrong Protocol: " << connect.getProtocol();
210 mqttContext->end(true);
211 } else if (connect.getLevel() != MQTT_VERSION_3_1_1) {
212 LOG(DEBUG) << "Wrong Protocol Level: " << MQTT_VERSION_3_1_1 << " != " << connect.getLevel();
213 sendConnack(MQTT_CONNACK_UNACEPTABLEVERSION, MQTT_SESSION_NEW);
214
215 mqttContext->end(true);
216 } else if (connect.isFakedClientId() && !connect.getCleanSession()) {
217 LOG(DEBUG) << "Resume session but no ClientId present";
218 sendConnack(MQTT_CONNACK_IDENTIFIERREJECTED, MQTT_SESSION_NEW);
219
220 mqttContext->end(true);
221 } else {
222 // V-Header
223 protocol = connect.getProtocol();
224 level = connect.getLevel();
225 connectFlags = connect.getConnectFlags();
226 keepAlive = connect.getKeepAlive();
227
228 // Payload
229 clientId = connect.getClientId();
230 willTopic = connect.getWillTopic();
231 willMessage = connect.getWillMessage();
232 username = connect.getUsername();
233 password = connect.getPassword();
234
235 // Derived from flags
236 usernameFlag = connect.getUsernameFlag();
237 passwordFlag = connect.getPasswordFlag();
238 willRetain = connect.getWillRetain();
239 willQoS = connect.getWillQoS();
240 willFlag = connect.getWillFlag();
241 cleanSession = connect.getCleanSession();
242
243 initSession(1.5 * keepAlive);
244
245 onConnect(connect);
246 }
247 }

260 void Mqtt::_onSubscribe(const iot::mqtt::server::packets::Subscribe& subscribe) {
261 LOG(DEBUG) << "Received SUBSCRIBE: " << clientId;
262 LOG(DEBUG) << "===================";
263 printStandardHeader(subscribe);
264 LOG(DEBUG) << "PacketIdentifier: 0x" << std::hex << std::setfill('0') << std::setw(4) << subscribe.getPacketIdentifier();
265
266 for (const iot::mqtt::Topic& topic : subscribe.getTopics()) {
267 LOG(DEBUG) << " Topic filter: '" << topic.getName() << "', QoS: " << static_cast<uint16_t>(topic.getQoS());
268 }
269
270 if (subscribe.getPacketIdentifier() == 0) {
271 LOG(DEBUG) << "PackageIdentifier missing";
272 mqttContext->end(true);
273 } else {
274 std::list<uint8_t> returnCodes;
275 for (const iot::mqtt::Topic& topic : subscribe.getTopics()) {
276 uint8_t returnCode = broker->subscribe(clientId, topic.getName(), topic.getQoS());
277 returnCodes.push_back(returnCode);
278 }
279
280 sendSuback(subscribe.getPacketIdentifier(), returnCodes);
281
282 onSubscribe(subscribe);
283 }
284 }

286 void Mqtt::_onUnsubscribe(const iot::mqtt::server::packets::Unsubscribe& unsubscribe) {
287 LOG(DEBUG) << "Received UNSUBSCRIBE: " << clientId;
288 LOG(DEBUG) << "=====================";
289 printStandardHeader(unsubscribe);
290 LOG(DEBUG) << "PacketIdentifier: 0x" << std::hex << std::setfill('0') << std::setw(4) << unsubscribe.getPacketIdentifier();
291
292 for (const std::string& topic : unsubscribe.getTopics()) {
293 LOG(DEBUG) << " Topic: " << topic;
294 }
295
296 if (unsubscribe.getPacketIdentifier() == 0) {
297 LOG(DEBUG) << "PackageIdentifier missing";
298 mqttContext->end(true);
299 } else {
300 for (const std::string& topic : unsubscribe.getTopics()) {
301 broker->unsubscribe(clientId, topic);
302 }
303
304 sendUnsuback(unsubscribe.getPacketIdentifier());
305
306 onUnsubscribe(unsubscribe);
307 }
