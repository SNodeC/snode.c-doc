62 void SocketContext::requestPrepared(Request&& request) {
63 const std::string requestLine = std::string(request.method)
64 .append(" ")
65 .append(request.url)
66 .append(" HTTP/")
67 .append(std::to_string(request.httpMajor))
68 .append(".")
69 .append(std::to_string(request.httpMinor));
70
71 if ((flags == Flags::NONE || (flags & Flags::HTTP11) == Flags::HTTP11 || (flags & Flags::KEEPALIVE) == Flags::KEEPALIVE) &&
72 (flags & Flags::CLOSE) != Flags::CLOSE) {
73 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request accepted: " << requestLine;
74
75 flags = (flags & ~Flags::HTTP11) | ((request.httpMajor == 1 && request.httpMinor == 1) ? Flags::HTTP11 : Flags::NONE);
76 flags = (flags & ~Flags::HTTP10) | ((request.httpMajor == 1 && request.httpMinor == 0) ? Flags::HTTP10 : Flags::NONE);
77 flags = (flags & ~Flags::KEEPALIVE) |
78 (web::http::ciContains(request.header("Connection"), "keep-alive") ? Flags::KEEPALIVE : Flags::NONE);
79 flags = (flags & ~Flags::CLOSE) | (web::http::ciContains(request.header("Connection"), "close") ? Flags::CLOSE : Flags::NONE);
80
81 if (!currentRequest) {
82 initiateRequest(request);
83 } else {
84 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request queued: " << requestLine;
85 pendingRequests.emplace_back(std::move(request));
86 }
87 } else {
88 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request rejected: " << requestLine;
89 }
90 }

92 void SocketContext::initiateRequest(Request& request) {
93 currentRequest = std::make_shared<Request>(std::move(request));
94
95 const std::string requestLine = std::string(currentRequest->method)
96 .append(" ")
97 .append(currentRequest->url)
98 .append(" HTTP/")
99 .append(std::to_string(currentRequest->httpMajor))
100 .append(".")
101 .append(std::to_string(currentRequest->httpMinor));
102
103 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request initiating: " << requestLine;
104
105 if (currentRequest->initiate()) {
106 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request initiated: " << requestLine;
107 } else {
108 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request initiating failed: " << requestLine;
109
110 core::EventReceiver::atNextTick([this, masterRequest = static_cast<std::weak_ptr<Request>>(masterRequest)]() -> void {
111 if (!masterRequest.expired()) {
112 if (!pendingRequests.empty()) {
113 initiateRequest(pendingRequests.front());
114 pendingRequests.pop_front();
115 } else {
116 currentRequest = nullptr;
117 }
118 }
119 });
120 }

173 void SocketContext::deliverResponse(Response&& response) {
174 const std::shared_ptr<Request> request = std::make_shared<Request>(std::move(deliveredRequests.front()));
175 deliveredRequests.pop_front();
176
177 const std::string requestLine = std::string(request->method)
178 .append(" ")
179 .append(request->url)
180 .append(" HTTP/")
181 .append(std::to_string(request->httpMajor))
182 .append(".")
183 .append(std::to_string(request->httpMinor));
184
185 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response parsed: " << requestLine;
186
187 const bool httpClose =
188 response.connectionState == ConnectionState::Close ||
189 (response.connectionState == ConnectionState::Default &&
190 ((response.httpMajor == 0 && response.httpMinor == 9) || (response.httpMajor == 1 && response.httpMinor == 0)));
191
192 request->deliverResponse(request, std::make_shared<Response>(std::move(response)));
193
194 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response delivered: " << requestLine;
195
196 responseDelivered(httpClose);
197 }

218 if (httpClose) {

224
225 if (!pipelinedRequests) {
226 core::EventReceiver::atNextTick([this, masterRequest = static_cast<std::weak_ptr<Request>>(masterRequest)]() -> void {
227 if (!masterRequest.expired()) {
228 if (!pendingRequests.empty()) {
229 initiateRequest(pendingRequests.front());
230 pendingRequests.pop_front();

232 currentRequest = nullptr;
233 }
234 }
235 });

256 void SocketContext::onDisconnected() {
257 if (!deliveredRequests.empty()) {
258 const Request& request = deliveredRequests.front();
259 if (request.httpMajor == 1 && request.httpMinor == 0) {
260 deliverResponse(parser.getResponse());
261 }
262 }
263
264 onRequestEnd(masterRequest);
265
266 LOG(INFO) << getSocketConnection()->getInstanceName() << " HTTP: disconnected";
267 }
