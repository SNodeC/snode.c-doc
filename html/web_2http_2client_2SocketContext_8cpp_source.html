60 void SocketContext::requestPrepared(Request& request) {
61 if ((flags == Flags::NONE || (flags & Flags::HTTP11) == Flags::HTTP11 || (flags & Flags::KEEPALIVE) == Flags::KEEPALIVE) &&
62 (flags & Flags::CLOSE) != Flags::CLOSE) {
63 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request accepted: " << request.method << " " << request.url
64 << " HTTP/" << request.httpMajor << "." << request.httpMinor;
65
66 flags = (flags & ~Flags::HTTP11) | ((request.httpMajor == 1 && request.httpMinor == 1) ? Flags::HTTP11 : Flags::NONE);
67 flags = (flags & ~Flags::HTTP10) | ((request.httpMajor == 1 && request.httpMinor == 0) ? Flags::HTTP10 : Flags::NONE);
68 flags = (flags & ~Flags::KEEPALIVE) |
69 (web::http::ciContains(request.header("Connection"), "keep-alive") ? Flags::KEEPALIVE : Flags::NONE);
70 flags = (flags & ~Flags::CLOSE) | (web::http::ciContains(request.header("Connection"), "close") ? Flags::CLOSE : Flags::NONE);
71
72 preparedRequests.emplace_back(std::make_shared<Request>(std::move(request)));
73
74 if (preparedRequests.size() == 1) {
75 dispatchNextRequest();
76 }
77 } else {
78 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request rejected: " << request.method << " " << request.url
79 << " HTTP/" << request.httpMajor << "." << request.httpMinor;
80
81 std::make_shared<Request>(std::move(request)); // Just to prevent memory leak
82 }
83 }

85 void SocketContext::dispatchNextRequest() {
86 if (!preparedRequests.empty()) {
87 Request* request = preparedRequests.front().get();
88 if (request->execute()) {
89 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request executed: " << request->method << " "
90 << request->url << " HTTP/" << request->httpMajor << "." << request->httpMinor;
91 } else {
92 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request execute failed: " << request->method << " "
93 << request->url << " HTTP/" << request->httpMajor << "." << request->httpMinor;
94
95 preparedRequests.pop_front();
96
97 if (!preparedRequests.empty()) {
98 core::EventReceiver::atNextTick([this]() -> void {
99 dispatchNextRequest();
100 });
101 } else {
102 shutdownWrite();
103 }
104 }
105 }
106 }

108 void SocketContext::requestSent(bool success) {
109 sentRequests.emplace_back(preparedRequests.front());
110 preparedRequests.pop_front();
111
112 if (success) {
113 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request sent: " << sentRequests.front()->method << " "
114 << sentRequests.front()->url << " HTTP/" << sentRequests.front()->httpMajor << "."
115 << sentRequests.front()->httpMinor;
116
117 if (((flags & Flags::HTTP11) == Flags::HTTP11 || (flags & Flags::KEEPALIVE) == Flags::KEEPALIVE) &&
118 (flags & Flags::CLOSE) != Flags::CLOSE) {
119 if (!preparedRequests.empty()) {
120 core::EventReceiver::atNextTick([this]() -> void {
121 dispatchNextRequest();
122 });
123 }
124 } else {
125 shutdownWrite();
126 }
127 } else {
128 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request sent failed";
129
130 shutdownWrite();
131 }
132 }

149 void SocketContext::deliverResponse(Response&& response) {
150 currentRequest = std::move(sentRequests.front());
151 sentRequests.pop_front();
152
153 const std::string requestMethod = std::string(currentRequest->method)
154 .append(" ")
155 .append(currentRequest->url)
156 .append(" HTTP/")
157 .append(std::to_string(currentRequest->httpMajor))
158 .append(".")
159 .append(std::to_string(currentRequest->httpMinor));
160
161 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response parsed: " << requestMethod;
162
163 const bool httpClose =
164 response.connectionState == ConnectionState::Close ||
165 (response.connectionState == ConnectionState::Default &&
166 ((response.httpMajor == 0 && response.httpMinor == 9) || (response.httpMajor == 1 && response.httpMinor == 0)));
167
168 currentRequest->deliverResponse(currentRequest, std::make_shared<Response>(std::move(response)));
169
170 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response delivered: " << requestMethod;
171
172 // Starting from here the currentRequest may be invalid, due to reuse in user code during deliverResponse(...). This is
173 // completely fine as the currentRequest is no more used anyway. An optional solution would be to pass the currentRequest as
174 // std::shared_ptr<const Request> to not allow modifications in user code. In user code the masterRequest has to be used instead
175 // in that case. Current decision is to allow reuse of the currentRequest in user code - why not?
176
177 requestCompleted(httpClose);
178 }

180 void SocketContext::responseError(int status, const std::string& reason) {
181 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response parse error: " << status << " : " << reason;
182
183 currentRequest->deliverResponseParseError(currentRequest, reason);
184
185 shutdownWrite(true);
186 }

206 std::size_t SocketContext::onReceivedFromPeer() {
207 std::size_t consumed = 0;
208
209 if (!sentRequests.empty()) {
210 consumed = parser.parse();
211 }
212

216 void SocketContext::onDisconnected() {
217 if (!sentRequests.empty()) {
218 currentRequest = std::move(sentRequests.front());
219 sentRequests.pop_front();
220
221 if (currentRequest->httpMajor == 1 && currentRequest->httpMinor == 0) {
222 deliverResponse(parser.getResponse());
223 }
224 }
225
226 onRequestEnd(masterRequest);
227
228 LOG(INFO) << getSocketConnection()->getInstanceName() << " HTTP: disconnected";
229 }
