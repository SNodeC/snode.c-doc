60 void SocketContext::dispatchNextRequest() {

63
64 preparedRequests.front()->execute();

68 void SocketContext::requestPrepared(Request& request) {
69 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request prepared";
70
71 preparedRequests.emplace_back(std::make_shared<Request>(std::move(request)));
72
73 if (preparedRequests.size() == 1) {

78 void SocketContext::requestSent(bool success) {
79 sentRequests.emplace_back(preparedRequests.front());
80 preparedRequests.pop_front();
81
82 if (success) {
83 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request sent successful";
84
85 if (!preparedRequests.empty()) {
86 core::EventReceiver::atNextTick([this]() -> void {
87 dispatchNextRequest();
88 });
89 }
90 } else {
91 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request sent failed";
92
93 shutdownWrite();
94 }
95 }

117 void SocketContext::responseParsed() {
118 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response parsed";
119
120 const bool close = currentResponse->connectionState == ConnectionState::Close ||
121 (currentResponse->connectionState == ConnectionState::Default &&
122 ((currentResponse->httpMajor == 0 && currentResponse->httpMinor == 9) ||
123 (currentResponse->httpMajor == 1 && currentResponse->httpMinor == 0)));
124
125 currentRequest->deliverResponse(currentRequest, currentResponse);
126
127 requestCompleted(close);
128 }
129
130 void SocketContext::responseError(int status, const std::string& reason) {
131 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Response parse error: " << status << " : " << reason;
132
133 currentRequest->deliverResponseParseError(currentRequest, reason);
134
135 shutdownWrite(true);
136 }

138 void SocketContext::requestCompleted(bool close) {
139 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Request completed successful";
140
141 if (close) {
142 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Connection = Close";
143
144 shutdownWrite();
145 } else {
146 LOG(TRACE) << getSocketConnection()->getInstanceName() << " HTTP: Connection = Keep-Alive";
147 }
148
149 currentRequest = nullptr;
150 currentResponse = nullptr;
151 }
