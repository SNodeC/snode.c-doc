43 void Session::sendPublish(Message& message, uint8_t qoS, bool retain) {
44 std::stringstream messageData;
45 const std::string& mes = message.getMessage();
46
47 unsigned long i = 0;
48 for (char ch : mes) {
49 if (i != 0 && i % 8 == 0 && i != mes.size()) {
50 messageData << std::endl;
51 messageData << " ";
52 }
53 ++i;
54 messageData << "0x" << std::hex << std::setfill('0') << std::setw(2) << static_cast<uint16_t>(static_cast<uint8_t>(ch))
55 << " "; // << " | ";
56 }
57
58 LOG(TRACE) << " TopicName: " << message.getTopic();
59 LOG(TRACE) << " Message: " << messageData.str();
60 LOG(TRACE) << " QoS: " << static_cast<uint16_t>(std::min(qoS, message.getQoS()));
61
62 if (isActive()) {
63 mqtt->sendPublish(message.getTopic(), message.getMessage(), std::min(message.getQoS(), qoS), retain);
64 } else {
65 if (message.getQoS() == 0) {
66 messageQueue.clear();
67 }
68
69 message.setQoS(std::min(message.getQoS(), qoS));
70 messageQueue.emplace_back(message);
71 }
72 }

74 void Session::publishQueued() {
75 LOG(TRACE) << " send queued messages ...";
76 for (iot::mqtt::server::broker::Message& message : messageQueue) {
77 sendPublish(message, message.getQoS(), false);
78 }
79 LOG(TRACE) << " ... done";
80
81 messageQueue.clear();

102 nlohmann::json Session::toJson() const {
103 nlohmann::json json = iot::mqtt::Session::toJson();
104
105 std::vector<nlohmann::json> messageVector;
106 for (const Message& message : messageQueue) {
107 messageVector.emplace_back(message.toJson());
108 }
109
110 json["message_queue"] = messageVector;
111
112 return json;
113 }
