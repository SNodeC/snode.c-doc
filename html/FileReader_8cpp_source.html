78 void FileReader::onEvent([[maybe_unused]] const utils::Timeval& currentTime) {
79 if (running && core::eventLoopState() != core::State::STOPPING) {
80 if (!suspended) {
81 std::vector<char> puffer(pufferSize);
82
83 const ssize_t ret = core::system::read(getFd(), puffer.data(), puffer.capacity());
84 if (ret > 0) {
85 if (this->send(puffer.data(), static_cast<std::size_t>(ret)) < 0) {
86 running = false;
87
88 this->error(errno);
89 }
90 } else {
91 running = false;
92
93 if (ret == 0) {
94 this->eof();
95 } else {
96 this->error(errno);
97 }
98 }
99
100 span();
101 }
102 } else {
103 delete this;
104 }
105 }
