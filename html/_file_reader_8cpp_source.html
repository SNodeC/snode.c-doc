43 errno = 0;

58 void FileReader::onEvent([[maybe_unused]] const utils::Timeval& currentTime) {
59 if (!suspended) {
60 // NOLINTNEXTLINE(cppcoreguidelines-avoid-c-arrays, hicpp-avoid-c-arrays, modernize-avoid-c-arrays)
61 static char junk[MF_READSIZE];
62
63 ssize_t ret = core::system::read(getFd(), junk, MF_READSIZE);
64
65 if (ret > 0) {
66 if (send(junk, static_cast<std::size_t>(ret)) >= 0) {
67 span();
68 } else {
69 delete this;
70 }
71 } else {
72 if (ret == 0) {
73 this->eof();
74 } else {
75 this->error(errno);
76 }
77 delete this;
78 }
79 }
80 }
