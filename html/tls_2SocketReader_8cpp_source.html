37 ssize_t SocketReader::read(char* chunk, std::size_t chunkLen) {
38 const int sslShutdownState = SSL_get_shutdown(ssl);
39
40 chunkLen = chunkLen > std::numeric_limits<int>::max() ? std::numeric_limits<int>::max() : chunkLen;
41 int ret = SSL_read(ssl, chunk, static_cast<int>(chunkLen));
42
43 if (ret <= 0) {
44 const int ssl_err = SSL_get_error(ssl, ret);
45
46 switch (ssl_err) {
47 case SSL_ERROR_NONE:
48 break;
49 case SSL_ERROR_WANT_READ:
50 ret = -1;
51 break;
52 case SSL_ERROR_WANT_WRITE: {
53 const utils::PreserveErrno preserveErrno;
54
55 LOG(TRACE) << getName() << " SSL/TLS: Start renegotiation on read";
56 doSSLHandshake(
57 [this]() -> void {
58 LOG(TRACE) << getName() << " SSL/TLS: Renegotiation on read success";
59 },
60 [this]() -> void {
61 LOG(TRACE) << getName() << " SSL/TLS: Renegotiation on read timed out";
62 },
63 [this](int ssl_err) -> void {
64 ssl_log(getName() + " SSL/TLS: Renegotiation", ssl_err);
65 });
66 }
67 ret = -1;
68 break;
69 case SSL_ERROR_ZERO_RETURN: // received close_notify
70 LOG(TRACE) << getName() << " SSL/TLS: Zero return";
71 SSL_set_shutdown(ssl, SSL_get_shutdown(ssl) | sslShutdownState);
72 doSSLShutdown();
73 errno = 0;
74 ret = 0;
75 break;
76 case SSL_ERROR_SYSCALL: {
77 const utils::PreserveErrno preserveErrno;
78
79 SSL_set_shutdown(ssl, SSL_get_shutdown(ssl) | SSL_RECEIVED_SHUTDOWN);
80 LOG(TRACE) << getName() << " SSL/TLS: TCP-FIN without close_notify. Emulating SSL_RECEIVED_SHUTDOWN";
81 doSSLShutdown();
82 }
83 ret = -1;
84 break;
85 default:
86 ssl_log(getName() + " SSL/TLS: Error read failed", ssl_err);
87 errno = EIO;
88 ret = -1;
89 break;
90 }
91 }
92
93 return ret;
94 }
