52 Request& Request::extend() {
53 originalUrl = url;
54 url = httputils::url_decode(httputils::str_split_last(originalUrl, '?').first);
55 path = httputils::str_split_last(url, '/').first;
56 if (path.empty()) {
57 path = std::string("/");
58 }
59
60 return *this;
61 }
