75 Request& Request::extend() {
76 originalUrl = url;
77 url = httputils::url_decode(httputils::str_split_last(originalUrl, '?').first);
78
79 if (url.length() <= 2 || !url.ends_with("/")) {
80 std::tie(path, file) = httputils::str_split_last(url, '/');
81
82 if (path.empty()) {
83 path = "/";
84 }
85 } else if (url.ends_with("/")) {
86 path = url;
87 path.pop_back();
88 }
89
90 return *this;
91 }
