126 ConfigSocketServer& ConfigSocketServer::setReuseAddress(bool reuseAddress) {
127 if (reuseAddress) {
128 addSocketOption(SOL_SOCKET, SO_REUSEADDR, 1);
129 } else {
130 addSocketOption(SOL_SOCKET, SO_REUSEADDR, 0);
131 }
132
133 reuseAddressOpt //
134 ->default_val(reuseAddress ? "true" : "false")
135 ->clear();
136
137 return *this;
138 }
