85 ConfigSocketServer& ConfigSocketServer::setReusePort(bool reusePort) {
86 if (reusePort) {
87 addSocketOption(SOL_SOCKET, SO_REUSEPORT, 1);
88 } else {
89 removeSocketOption(SO_REUSEPORT);
90 }
91
92 reusePortOpt //
93 ->default_val(reusePort ? "true" : "false")
94 ->take_all()
95 ->clear();
96
97 return *this;
98 }
99
100 bool ConfigSocketServer::getReusePort() const {
101 return reusePortOpt->as<bool>();
102 }
