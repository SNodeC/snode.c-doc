88 TickStatus EventLoop::_tick(const utils::Timeval& tickTimeOut) {
89 TickStatus tickStatus = TickStatus::SUCCESS;
90
91 tickCounter++;
92
93 sigset_t newSet{};
94 sigset_t oldSet{};
95
96 sigaddset(&newSet, SIGINT);
97 sigaddset(&newSet, SIGTERM);
98 sigaddset(&newSet, SIGALRM);
99 sigaddset(&newSet, SIGHUP);
100
101 sigprocmask(SIG_BLOCK, &newSet, &oldSet);
102
103 if (eventLoopState == State::RUNNING || eventLoopState == State::STOPING) {
104 tickStatus = eventMultiplexer.tick(tickTimeOut, oldSet);
105 }
106
107 sigprocmask(SIG_SETMASK, &oldSet, nullptr);
108
109 return tickStatus;
110 }

112 TickStatus EventLoop::tick(const utils::Timeval& timeOut) {
113 TickStatus tickStatus = TickStatus::ERROR;
114
115 if (eventLoopState == State::INITIALIZED) {
116 struct sigaction sact;
117 sigemptyset(&sact.sa_mask);
118 sact.sa_flags = 0;
119 sact.sa_handler = SIG_IGN;
120
121 struct sigaction oldPipeAct {};
122 sigaction(SIGPIPE, &sact, &oldPipeAct);
123
124 tickStatus = EventLoop::instance()._tick(timeOut);
125
126 sigaction(SIGPIPE, &oldPipeAct, nullptr);
127 } else {
128 EventLoop::instance().eventMultiplexer.clear();
129
130 PLOG(ERROR) << "SNodeC not initialized: No events will be processed\nCall SNodeC::init(argc, argv) before SNodeC::tick().";
131 }
132
133 return tickStatus;
134 }

136 int EventLoop::start(const utils::Timeval& timeOut) {
137 core::TickStatus tickStatus = TickStatus::SUCCESS;
138
139 if (eventLoopState == State::INITIALIZED && utils::Config::bootstrap()) {
140 LOG(INFO) << "Starting Eventloop";
141
142 struct sigaction sact;
143 sigemptyset(&sact.sa_mask);
144 sact.sa_flags = 0;
145 sact.sa_handler = SIG_IGN;
146
147 struct sigaction oldPipeAct {};
148 sigaction(SIGPIPE, &sact, &oldPipeAct);
149
150 sact.sa_handler = EventLoop::stoponsig;
151
152 struct sigaction oldIntAct {};
153 sigaction(SIGINT, &sact, &oldIntAct);
154
155 struct sigaction oldTermAct {};
156 sigaction(SIGTERM, &sact, &oldTermAct);
157
158 struct sigaction oldAlarmAct {};
159 sigaction(SIGALRM, &sact, &oldAlarmAct);
160
161 struct sigaction oldHupAct {};
162 sigaction(SIGHUP, &sact, &oldHupAct);
163
164 eventLoopState = State::RUNNING;
165
166 while ((tickStatus == TickStatus::SUCCESS || tickStatus == TickStatus::INTERRUPTED) && eventLoopState == State::RUNNING) {
167 tickStatus = EventLoop::instance()._tick(timeOut);
168 }
169
170 switch (tickStatus) {
171 case TickStatus::SUCCESS:
172 [[fallthrough]];
173 case TickStatus::INTERRUPTED:
174 LOG(INFO) << "EventLoop interrupted";
175 break;
176 case TickStatus::NOOBSERVER:
177 LOG(INFO) << "EventLoop: No Observer";
178 break;
179 case TickStatus::ERROR:
180 PLOG(ERROR) << "EventLoop::instance()._tick()";
181 break;
182 }
183
184 sigaction(SIGPIPE, &oldPipeAct, nullptr);
185 sigaction(SIGINT, &oldIntAct, nullptr);
186 sigaction(SIGTERM, &oldTermAct, nullptr);
187 sigaction(SIGALRM, &oldAlarmAct, nullptr);
188 sigaction(SIGHUP, &oldHupAct, nullptr);
189 } else {
190 LOG(TRACE) << "Eventloop not started SNode.C not initialized or bootstraping failed";
191
192 EventLoop::instance().eventMultiplexer.clear();
193 }
194
195 free(tickStatus);
196
197 return stopsig;
198 }

204 void EventLoop::free(TickStatus tickStatus) {
205 std::string signal = "SIG" + utils::system::sigabbrev_np(stopsig);
206
207 if (signal == "SIGUNKNOWN") {
208 signal = std::to_string(stopsig);
209 }
210
211 LOG(TRACE) << "Sending 'onExit(" << signal << ")' to all DescriptorEventReceivers";
212
213 eventLoopState = State::STOPING;
214
215 if (stopsig != 0) {
216 EventLoop::instance().eventMultiplexer.sigExit(stopsig);
217 }
218
219 utils::Timeval timeout = 2;
220 do {
221 LOG(TRACE) << "Stopping all DescriptorEventReceivers";
222
223 auto t1 = std::chrono::system_clock::now();
224
225 EventLoop::instance().eventMultiplexer.stop();
226
227 utils::Timeval timeoutOp = timeout;
228 tickStatus = EventLoop::instance()._tick(timeoutOp);
229
230 auto t2 = std::chrono::system_clock::now();
231
232 std::chrono::duration<double> seconds = t2 - t1;
233
234 timeout -= seconds.count();
235 } while (timeout > 0 && (tickStatus == TickStatus::SUCCESS || tickStatus == TickStatus::INTERRUPTED));
236
237 LOG(TRACE) << "Closing all libraries opened during runntime";
238
239 DynamicLoader::execDlCloseAll();
240
241 LOG(TRACE) << "Terminating SNode.C";
242
243 utils::Config::terminate();
244
245 LOG(TRACE) << "All resources released ... BYE";
246 }
