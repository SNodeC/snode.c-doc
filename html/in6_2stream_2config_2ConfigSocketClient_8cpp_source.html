101 ConfigSocketClient& ConfigSocketClient::setDisableNagleAlgorithm(bool disableNagleAlgorithm) {
102 if (disableNagleAlgorithm) {
103 addSocketOption(IPPROTO_TCP, TCP_NODELAY, 1);
104 } else {
105 addSocketOption(IPPROTO_TCP, TCP_NODELAY, 0);
106 }
107
108 disableNagleAlgorithmOpt //
109 ->default_val(disableNagleAlgorithm ? "true" : "false")
110 ->clear();
111
112 return *this;
113 }
114
115 bool ConfigSocketClient::getDisableNagleAlgorithm() const {
116 return disableNagleAlgorithmOpt->as<bool>();
117 }
