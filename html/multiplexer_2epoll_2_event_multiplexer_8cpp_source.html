38 EventMultiplexer::EventMultiplexer()
39 : core::EventMultiplexer(
40 new core::epoll::DescriptorEventPublisher(
41 "READ", epfds[core::EventMultiplexer::DISP_TYPE::RD], EPOLLIN, EPOLLIN | EPOLLHUP | EPOLLRDHUP | EPOLLERR),
42 new core::epoll::DescriptorEventPublisher("WRITE", epfds[core::EventMultiplexer::DISP_TYPE::WR], EPOLLOUT, EPOLLOUT),
43 new core::epoll::DescriptorEventPublisher("EXCEPT", epfds[core::EventMultiplexer::DISP_TYPE::EX], EPOLLPRI, EPOLLPRI))
44 , epfd(core::system::epoll_create1(EPOLL_CLOEXEC)) {
45 epoll_event event{};
46 event.events = EPOLLIN;
47
48 event.data.ptr = descriptorEventPublishers[core::EventMultiplexer::DISP_TYPE::RD];
49 core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, epfds[core::EventMultiplexer::DISP_TYPE::RD], &event);
50
51 event.data.ptr = descriptorEventPublishers[core::EventMultiplexer::DISP_TYPE::WR];
52 core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, epfds[core::EventMultiplexer::DISP_TYPE::WR], &event);
53
54 event.data.ptr = descriptorEventPublishers[core::EventMultiplexer::DISP_TYPE::EX];
55 core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, epfds[core::EventMultiplexer::DISP_TYPE::EX], &event);
56
57 LOG(TRACE) << "IO-Multiplexer: epoll";
58 }
