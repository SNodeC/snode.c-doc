46 std::string SocketContextUpgrade::loadSubProtocol(const std::list<std::string>& subProtocolNames) {
47 std::string selectedSubProtocolName;
48
49 for (const std::string& subProtocolName : subProtocolNames) {
50 subProtocolFactory = SubProtocolFactorySelector::instance()->select(subProtocolName, SubProtocolFactorySelector::Role::SERVER);
51
52 if (subProtocolFactory != nullptr) {
53 subProtocol = subProtocolFactory->createSubProtocol(this);
54 selectedSubProtocolName = subProtocol != nullptr ? subProtocolName : "";
55 }
56
57 if (!selectedSubProtocolName.empty()) {
58 break;
59 }
60 }
61
62 return selectedSubProtocolName;
63 }
