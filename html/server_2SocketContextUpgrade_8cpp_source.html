39 SocketContextUpgrade* SocketContextUpgrade::create(
40 web::http::SocketContextUpgradeFactory<web::http::server::Request, web::http::server::Response>* socketContextUpgradeFactory,
41 core::socket::stream::SocketConnection* socketConnection,
42 const std::string& subProtocolName) {
43 SocketContextUpgrade* socketContextUpgrade = nullptr;
44
45 web::websocket::SubProtocolFactory<SubProtocol>* subProtocolFactory =
46 SubProtocolFactorySelector::instance()->select(subProtocolName, SubProtocolFactorySelector::Role::SERVER);
47
48 if (subProtocolFactory != nullptr) {
49 socketContextUpgrade = new SocketContextUpgrade(socketConnection, socketContextUpgradeFactory, subProtocolFactory);
50
51 if (socketContextUpgrade->subProtocol == nullptr) {
52 delete socketContextUpgrade;
53 socketContextUpgrade = nullptr;
54 }
55 }
56
57 return socketContextUpgrade;
58 }
