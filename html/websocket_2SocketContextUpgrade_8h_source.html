116 void sendClose(uint16_t statusCode = 1000, const char* reason = nullptr, std::size_t reasonLength = 0) override {
117 std::size_t closePayloadLength = reasonLength + 2;
118 char* closePayload = new char[closePayloadLength];
119
120 *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode); // cppcheck-suppress uninitdata
121
122 if (reasonLength > 0) {
123 memcpy(closePayload + 2, reason, reasonLength);
124 }
125
126 sendClose(closePayload, closePayloadLength);
127
128 delete[] closePayload;
129 }

131 void sendClose(const char* message, std::size_t messageLength) override {
132 if (!closeSent) {
133 LOG(DEBUG) << "WebSocket: Sending close to peer";
134

136
137 setTimeout(CLOSE_SOCKET_TIMEOUT);

147 /* WSReceiver */
148 void onMessageStart(int opCode) override {
149 receivedOpCode = opCode;
150
151 switch (opCode) {
152 case SubProtocolContext::OpCode::CLOSE:
153 [[fallthrough]];
154 case SubProtocolContext::OpCode::PING:
155 [[fallthrough]];
156 case SubProtocolContext::OpCode::PONG:
157 break;
158 default:
159 subProtocol->onMessageStart(opCode);
160 break;
161 }

164 void onMessageData(const char* chunk, uint64_t chunkLen) override {
165 switch (receivedOpCode) {
166 case SubProtocolContext::OpCode::CLOSE:
167 [[fallthrough]];
168 case SubProtocolContext::OpCode::PING:
169 [[fallthrough]];
170 case SubProtocolContext::OpCode::PONG:
171 pongCloseData += std::string(chunk, static_cast<std::size_t>(chunkLen));
172 break;
173 default:
174 std::size_t chunkOffset = 0;
175
176 do {
177 std::size_t sendChunkLen =
178 (chunkLen - chunkOffset <= SIZE_MAX) ? static_cast<std::size_t>(chunkLen - chunkOffset) : SIZE_MAX;
179 subProtocol->onMessageData(chunk + chunkOffset, sendChunkLen);
180 chunkOffset += sendChunkLen;
181 } while (chunkLen - chunkOffset > 0);
182 break;
183 }
184 }
185
186 void onMessageEnd() override {
187 switch (receivedOpCode) {
188 case SubProtocolContext::OpCode::CLOSE:
189 if (closeSent) { // active close
190 closeSent = false;
191 LOG(DEBUG) << "WebSocket: Close confirmed from peer";
192 } else { // passive close
193 LOG(DEBUG) << "WebSocket: Close request received - replying with close";
194 sendClose(pongCloseData.data(), pongCloseData.length());
195 pongCloseData.clear();
196 shutdownWrite();
197 }
198 break;
199 case SubProtocolContext::OpCode::PING:
200 sendPong(pongCloseData.data(), pongCloseData.length());
201 pongCloseData.clear();
202 break;
203 case SubProtocolContext::OpCode::PONG:
204 subProtocol->onPongReceived();
205 break;
206 default:
207 subProtocol->onMessageEnd();
208 break;
209 }
210 }

292 void dumpFrame(char* frame, uint64_t frameLength) {
293 const unsigned int modul = 4;
294
295 std::stringstream stringStream;
296
297 for (std::size_t i = 0; i < frameLength; i++) {
298 stringStream << std::setfill('0') << std::setw(2) << std::hex
299 << static_cast<unsigned int>(static_cast<unsigned char>(frame[i])) << " ";
300
301 if ((i + 1) % modul == 0 || i == frameLength) { // cppcheck-suppress knownConditionTrueFalse
302 LOG(TRACE) << "WebSocket: Frame = " << stringStream.str();
303 stringStream.str("");
304 }
305 }
306 }
