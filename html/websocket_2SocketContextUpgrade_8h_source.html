117 void sendClose(uint16_t statusCode = 1000, const char* reason = nullptr, std::size_t reasonLength = 0) override {
118 std::size_t closePayloadLength = reasonLength + 2;
119 char* closePayload = new char[closePayloadLength];
120
121 *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
122
123 if (reasonLength > 0) {
124 memcpy(closePayload + 2, reason, reasonLength);
125 }
126
127 sendClose(closePayload, closePayloadLength);
128
129 delete[] closePayload;
130 }

132 void sendClose(const char* message, std::size_t messageLength) override {
133 if (!closeSent) {
134 LOG(DEBUG) << "Sending close to peer";
135
136 sendMessage(8, message, messageLength);
137 shutdownWrite();
138
139 setTimeout(CLOSE_SOCKET_TIMEOUT);
140
141 closeSent = true;

149 /* WSReceiver */
150 void onMessageStart(int opCode) override {
151 receivedOpCode = opCode;
152
153 switch (opCode) {
154 case SubProtocolContext::OpCode::CLOSE:
155 [[fallthrough]];
156 case SubProtocolContext::OpCode::PING:
157 [[fallthrough]];
158 case SubProtocolContext::OpCode::PONG:
159 break;
160 default:
161 subProtocol->onMessageStart(opCode);
162 break;
163 }
164 }
165
166 void onMessageData(const char* junk, uint64_t junkLen) override {
167 switch (receivedOpCode) {
168 case SubProtocolContext::OpCode::CLOSE:
169 [[fallthrough]];
170 case SubProtocolContext::OpCode::PING:
171 [[fallthrough]];
172 case SubProtocolContext::OpCode::PONG:
173 pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
174 break;
175 default:
176 std::size_t junkOffset = 0;
177
178 do {
179 std::size_t sendJunkLen =
180 (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
181 subProtocol->onMessageData(junk + junkOffset, sendJunkLen);
182 junkOffset += sendJunkLen;
183 } while (junkLen - junkOffset > 0);
184 break;
185 }
186 }
187
188 void onMessageEnd() override {
189 switch (receivedOpCode) {
190 case SubProtocolContext::OpCode::CLOSE:
191 if (closeSent) { // active close
192 closeSent = false;
193 LOG(DEBUG) << "Close confirmed from peer";
194 } else { // passive close
195 LOG(DEBUG) << "Close request received - replying with close";
196 sendClose(pongCloseData.data(), pongCloseData.length());
197 pongCloseData.clear();
198 }
199 break;
200 case SubProtocolContext::OpCode::PING:
201 sendPong(pongCloseData.data(), pongCloseData.length());
202 pongCloseData.clear();
203 break;
204 case SubProtocolContext::OpCode::PONG:
205 subProtocol->onPongReceived();
206 break;
207 default:
208 subProtocol->onMessageEnd();
209 break;
210 }
211 }

288 void dumpFrame(char* frame, uint64_t frameLength) {
289 unsigned int modul = 4;
290
291 std::stringstream stringStream;
292
293 for (std::size_t i = 0; i < frameLength; i++) {
294 stringStream << std::setfill('0') << std::setw(2) << std::hex
295 << static_cast<unsigned int>(static_cast<unsigned char>(frame[i])) << " ";
296
297 if ((i + 1) % modul == 0 || i == frameLength) {
298 LOG(TRACE) << "Frame: " << stringStream.str();
299 stringStream.str("");
300 }
301 }
302 }
