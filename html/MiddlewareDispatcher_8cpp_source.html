60 bool MiddlewareDispatcher::dispatch(express::Controller& controller, const std::string& parentMountPath, const MountPoint& mountPoint) {
61 bool requestMatched = false;
62
63 if ((controller.getFlags() & Controller::NEXT) == 0) {
64 const std::string absoluteMountPath = path_concat(parentMountPath, mountPoint.relativeMountPath);
65
66 const std::string requestMethod = controller.getRequest()->method;
67 const std::string requestUrl = controller.getRequest()->url;
68 const std::string requestPath = controller.getRequest()->path;
69
70 // clang-format off
71 requestMatched =
72 (
73 (
74 (
75 mountPoint.method == requestMethod
76 ) || (
77 mountPoint.method == "all"
78 )
79 ) && (
80 (
81 absoluteMountPath == requestUrl
82 ) || (
83 (
84 !controller.getStrictRouting()
85 ) && (
86 absoluteMountPath == requestPath
87 )
88 ) || (
89 checkForUrlMatch(absoluteMountPath, requestUrl)
90 )
91 )
92 ) || (
93 (
94 mountPoint.method == "use"
95 ) && (
96 (
97 requestUrl.starts_with(absoluteMountPath)
98 ) || (
99 (
100 !controller.getStrictRouting()
101 ) && (
102 requestPath.starts_with(absoluteMountPath)
103 )
104 )
105 )
106 );
107 // clang-format on
108
109 if (requestMatched) {
110 if (hasResult(absoluteMountPath)) {
111 setParams(absoluteMountPath, *controller.getRequest());
112 }
113
114 Next next(controller);
115 lambda(controller.getRequest(), controller.getResponse(), next);
116
117 // If next() was called synchronously continue current route-tree traversal
118 if ((next.controller.getFlags() & express::Controller::NEXT) != 0) {
119 requestMatched = false;
120 controller = next.controller;
121 }
122 }
123 }
124
125 return requestMatched;
126 }
