85 class SimpleSocketProtocol : public core::socket::stream::SocketContext {
86 public:

88 : core::socket::stream::SocketContext(socketConnection) {

94 void onConnected() override {

99 }
100

116 shutdownWrite();
117 }

119 private:

121 };
