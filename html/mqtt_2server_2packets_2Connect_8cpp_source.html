37 std::size_t Connect::deserializeVP(iot::mqtt::MqttContext* mqttContext) {
38 std::size_t consumed = 0;
39
40 switch (state) {
41 case 0: // V-Header
42 consumed += protocol.deserialize(mqttContext);
43 if (!protocol.isComplete()) {
44 break;
45 }
46
47 state++;
48 [[fallthrough]];
49 case 1:
50 consumed += level.deserialize(mqttContext);
51 if (!level.isComplete()) {
52 break;
53 }
54 reflect = (level & 0x80) == 0; // msb in level == 1 -> do not reflect message to origin (try_private in mosquitto)
55 level = level & ~0x80;
56
57 state++;
58 [[fallthrough]];
59 case 2:
60 consumed += connectFlags.deserialize(mqttContext);
61 if (!connectFlags.isComplete()) {
62 break;
63 }
64
65 reserved = (connectFlags & 0x01) != 0;
66 cleanSession = (connectFlags & 0x02) != 0;
67 willFlag = (connectFlags & 0x04) != 0;
68 willQoS = (connectFlags & 0x18) >> 3;
69 willRetain = (connectFlags & 0x20) != 0;
70 passwordFlag = (connectFlags & 0x40) != 0;
71 usernameFlag = (connectFlags & 0x80) != 0;
72
73 state++;
74 [[fallthrough]];
75 case 3:
76 consumed += keepAlive.deserialize(mqttContext);
77 if (!keepAlive.isComplete()) {
78 break;
79 }
80
81 state++;
82 [[fallthrough]];
83 case 4: // Payload
84 consumed += clientId.deserialize(mqttContext);
85 if (!clientId.isComplete()) {
86 break;
87 }
88
89 if (clientId.size() == 0) {
90 clientId = utils::Uuid::getUuid();
91 fakedClientId = true;
92 }
93
94 state++;
95 [[fallthrough]];
96 case 5:
97 if (willFlag) {
98 consumed += willTopic.deserialize(mqttContext);
99 if (!willTopic.isComplete()) {
100 break;
101 }
102 }
103
104 state++;
105 [[fallthrough]];
106 case 6:
107 if (willFlag) {
108 consumed += willMessage.deserialize(mqttContext);
109 if (!willMessage.isComplete()) {
110 break;
111 }
112 }
113
114 state++;
115 [[fallthrough]];
116 case 7:
117 if (usernameFlag) {
118 consumed += username.deserialize(mqttContext);
119 if (!username.isComplete()) {
120 break;
121 }
122 }
123
124 state++;
125 [[fallthrough]];
126 case 8:
127 if (passwordFlag) {
128 consumed += password.deserialize(mqttContext);
129 if (!password.isComplete()) {
130 break;
131 }
132 }
133
134 complete = true;
135 break;
136 }
137
138 return consumed;
139 }
