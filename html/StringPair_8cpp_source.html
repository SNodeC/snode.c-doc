63 std::size_t StringPair::deserialize(MqttContext* mqttContext) {
64 std::size_t consumed = 0;
65
66 switch (state) {
67 case 0:
68 consumed += name.deserialize(mqttContext);
69 if (!name.isComplete()) {
70 break;
71 }
72
73 state++;
74 [[fallthrough]];
75 case 1:
76 consumed += value.deserialize(mqttContext);
77
78 complete = value.isComplete();
79 break;
80 }
81
82 return consumed;
83 }
