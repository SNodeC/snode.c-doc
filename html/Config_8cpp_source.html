341 bool Config::bootstrap() {
342 aliases.clear();
343
344 app->final_callback([]() -> void {
345 if (daemonizeOpt->as<bool>() && (*app)["--show-config"]->count() == 0 && (*app)["--write-config"]->count() == 0 &&
346 (*app)["--command-line"]->count() == 0) {
347 std::cout << "Running as daemon (double fork)" << std::endl;
348
349 utils::Daemon::startDaemon(
350 pidDirectory + "/" + applicationName + ".pid", userNameOpt->as<std::string>(), groupNameOpt->as<std::string>());
351
352 logger::Logger::setQuiet();
353
354 const std::string logFile = logFileOpt->as<std::string>();
355 if (!logFile.empty()) {
356 logger::Logger::logToFile(logFile);
357 }
358 } else if ((*app)["--enforce-log-file"]->as<bool>()) {
359 const std::string logFile = logFileOpt->as<std::string>();
360 if (!logFile.empty()) {
361 std::cout << "Writing logs to file " << logFile << std::endl;
362
363 logger::Logger::logToFile(logFile);
364 }
365 }
366 });
367
368 return parse2();
369 }

371 bool Config::parse1() {
372 bool proceed = true;
373
374 try {
375 app->parse(argc, argv);
376 } catch (const CLI::ParseError&) {
377 // Do not process ParseError here but on second parse pass
378 }
379
380 if ((*app)["--kill"]->count() > 0) {
381 try {
382 const pid_t daemonPid = utils::Daemon::stopDaemon(pidDirectory + "/" + applicationName + ".pid");
383 std::cout << "Daemon terminated: Pid = " << daemonPid << std::endl;
384 } catch (const DaemonError& e) {
385 std::cout << "DaemonError: " << e.what() << std::endl;
386 } catch (const DaemonFailure& e) {
387 std::cout << "DaemonFailure: " << e.what() << std::endl;
388 }
389
390 proceed = false;
391 } else {
392 if ((*app)["--show-config"]->count() == 0 && (*app)["--write-config"]->count() == 0 && (*app)["--command-line"]->count() == 0) {
393 app->allow_extras(false);
394 }
395
396 if (!quietOpt->as<bool>()) {
397 logger::Logger::setLogLevel(logLevelOpt->as<int>());
398 logger::Logger::setVerboseLevel(verboseLevelOpt->as<int>());
399 }
400
401 logger::Logger::setQuiet(quietOpt->as<bool>());
402 }
403
404 return proceed;
405 }

407 static void createCommandLineOptions(std::stringstream& out, CLI::App* app, CLI::CallForCommandline::Mode mode) {
408 CLI::Option* disabledOpt = app->get_option_no_throw("--disabled");
409 const bool disabled = disabledOpt != nullptr ? disabledOpt->as<bool>() : false;
410 if (!disabled || mode == CLI::CallForCommandline::Mode::DEFAULT) {
411 for (const CLI::Option* option : app->get_options()) {
412 if (option->get_configurable()) {
413 std::string value;
414
415 switch (mode) {
416 case CLI::CallForCommandline::Mode::STANDARD:
417 if (option->count() > 0) {
418 value = option->as<std::string>();
419 } else if (option->get_required()) {
420 value = "<REQUIRED>";
421 }
422 break;
423 case CLI::CallForCommandline::Mode::REQUIRED:
424 if (option->get_required()) {
425 if (option->count() > 0) {
426 value = option->as<std::string>();
427 } else {
428 value = "<REQUIRED>";
429 }
430 }
431 break;
432 case CLI::CallForCommandline::Mode::FULL:
433 if (option->count() > 0) {
434 value = option->as<std::string>();
435 } else if (!option->get_default_str().empty()) {
436 value = option->get_default_str();
437 } else if (!option->get_required()) {
438 value = "\"\"";
439 } else {
440 value = "<REQUIRED>";
441 }
442 break;
443 case CLI::CallForCommandline::Mode::DEFAULT: {
444 if (!option->get_default_str().empty()) {
445 value = option->get_default_str();
446 } else if (!option->get_required()) {
447 value = "\"\"";
448 } else {
449 value = "<REQUIRED>";
450 }
451 break;
452 }
453 }
454
455 if (!value.empty()) {
456 out << "--" << option->get_single_name() << ((option->get_items_expected_max() == 0) ? "=" : " ") << value << " ";
457 }
458 }
459 }
460 } else if (disabledOpt->get_default_str() == "false") {
461 out << "--disabled=true ";
462 }
463 }

480 static std::string createCommandLineSubcommands(CLI::App* app, CLI::CallForCommandline::Mode mode) {
481 std::stringstream out;
482
483 CLI::Option* disabledOpt = app->get_option_no_throw("--disabled");
484 if (disabledOpt == nullptr || !disabledOpt->as<bool>() || mode == CLI::CallForCommandline::Mode::DEFAULT) {
485 for (CLI::App* subcommand : app->get_subcommands({})) {
486 if (!subcommand->get_name().empty()) {
487 createCommandLineTemplate(out, subcommand, mode);
488 }
489 }
490 }
491
492 return out.str();

495 static void createCommandLineTemplate(std::stringstream& out, CLI::App* app, CLI::CallForCommandline::Mode mode) {
496 std::string outString;
497
498 outString = createCommandLineOptions(app, mode);
499 if (!outString.empty()) {
500 outString += " ";
501 }
502
503 outString += createCommandLineSubcommands(app, mode);
504
505 if (!outString.empty()) {
506 out << app->get_name() << " " << outString;
507 }
508 }

510 static std::string createCommandLineTemplate(CLI::App* app, CLI::CallForCommandline::Mode mode) {
511 std::stringstream out;
512
513 createCommandLineTemplate(out, app, mode);
514
515 std::string outString = out.str();
516 while (app->get_parent() != nullptr) {
517 app = app->get_parent();
518 std::string parentOptions = createCommandLineOptions(app, mode);
519 outString =
520 std::string(app->get_name()).append(" ").append(!parentOptions.empty() ? parentOptions.append(" ") : "").append(outString);
521 }
522
523 if (outString.empty()) {
524 outString = Config::getApplicationName();
525 }
526
527 return outString;
528 }

530 bool Config::parse2() {
531 bool success = false;
532
533 try {
534 try {
535 app->parse(argc, argv);
536
537 if ((*app)["--write-config"]->count() > 0) {
538 throw CLI::CallForWriteConfig((*app)["--write-config"]->as<std::string>());
539 }
540
541 success = true;
542 } catch (const DaemonError& e) {
543 std::cout << "Daemon error: " << e.what() << " ... exiting" << std::endl;
544 } catch (const DaemonFailure& e) {
545 std::cout << "Daemon failure: " << e.what() << " ... exiting" << std::endl;
546 } catch (const DaemonSignaled& e) {
547 std::cout << "Pid: " << getpid() << ", child pid: " << e.getPid() << ": " << e.what() << std::endl;
548 } catch (const CLI::CallForHelp&) {
549 std::cout << app->help() << std::endl;
550 } catch (const CLI::CallForAllHelp&) {
551 std::cout << app->help("", CLI::AppFormatMode::All) << std::endl;
552 } catch (const CLI::CallForVersion&) {
553 std::cout << app->version() << std::endl << std::endl;
554 } catch (const CLI::CallForCommandline& e) {
555 std::cout << e.what() << std::endl;
556 std::cout << std::endl
557 << Color::Code::FG_GREEN << "command@line" << Color::Code::FG_DEFAULT << ":" << Color::Code::FG_BLUE << "~/> "
558 << Color::Code::FG_DEFAULT << createCommandLineTemplate(e.getApp(), e.getMode()) << std::endl
559 << std::endl;
560 } catch (const CLI::CallForShowConfig& e) {
561 try {
562 std::cout << e.getApp()->config_to_str(true, true);
563 } catch (const CLI::ParseError& e1) {
564 std::cout << "Error showing config file: " << e.getApp() << " " << e1.get_name() << " " << e1.what() << std::endl;
565 throw;
566 }
567 } catch (const CLI::CallForWriteConfig& e) {
568 std::cout << e.what() << std::endl;
569 std::ofstream confFile(e.getConfigFile());
570 if (confFile.is_open()) {
571 try {
572 confFile << app->config_to_str(true, true);
573 confFile.close();
574 } catch (const CLI::ParseError& e1) {
575 confFile.close();
576 std::cout << "Error writing config file: " << e1.get_name() << " " << e1.what() << std::endl;
577 throw;
578 }
579 } else {
580 std::cout << "Error writing config file: " << std::strerror(errno) << std::endl;
581 }
582 } catch (const CLI::ConversionError& e) {
583 std::cout << "[" << Color::Code::FG_RED << e.get_name() << Color::Code::FG_DEFAULT << "] " << e.what() << std::endl;
584 throw;
585 } catch (const CLI::ArgumentMismatch& e) {
586 std::cout << "[" << Color::Code::FG_RED << e.get_name() << Color::Code::FG_DEFAULT << "] " << e.what() << std::endl;
587 throw;
588 } catch (const CLI::ConfigError& e) {
589 std::cout << "[" << Color::Code::FG_RED << e.get_name() << Color::Code::FG_DEFAULT << "] " << e.what() << std::endl;
590 std::cout << " Adding '-w' on the command line may solve this problem" << std::endl;
591 throw;
592 } catch (const CLI::ParseError& e) {
593 const std::string what = e.what();
594 if (what.find("[Option Group: ") != std::string::npos) { // If CLI11 throws that error it means for us there are
595 // unconfigured anonymous instances
596 std::cout << Color::Code::FG_RED << "[BootstrapError]" << Color::Code::FG_DEFAULT
597 << " Anonymous instance(s) not configured in source code " << std::endl;
598 } else {
599 std::cout << "[" << Color::Code::FG_RED << e.get_name() << Color::Code::FG_DEFAULT << "] " << what << std::endl;
600 }
601 throw;
602 }
603 } catch (const CLI::ParseError&) {
604 std::cout << std::endl << "Append -h or --help to your command line for more information." << std::endl;
605 } catch (const CLI::Error& e) {
606 std::cout << "Error: " << e.get_name() << " " << e.what() << std::endl;
607 std::cout << "Append -h or --help to your command line for more information." << std::endl;
608 }
609
610 if (!success) {
611 logger::Logger::setQuiet();
612 }
613
614 return success;
615 }

747 CLI::App* Config::addHelp(CLI::App* app) {
748 app //
749 ->set_help_flag();
750
751 app //
752 ->add_flag_function(
753 "-h{standard},--help{standard}",
754 [app]([[maybe_unused]] std::int64_t count) {
755 const std::size_t disabledCount =
756 app->get_subcommands([](CLI::App* app) -> bool {
757 return app->get_group() == "Instance" && app->get_option("--disabled")->as<bool>();
758 })
759 .size();
760 const std::size_t enabledCount =
761 app->get_subcommands([](CLI::App* app) -> bool {
762 return app->get_group() == "Instance" && !app->get_option("--disabled")->as<bool>();
763 })
764 .size();
765
766 for (auto* instance : app->get_subcommands({})) {
767 const std::string& group = instance->get_group();
768 if (group == "Instance") {
769 if (instance->get_option("--disabled")->as<bool>()) {
770 instance->group(std::string("Instance").append((disabledCount > 1) ? "s" : "").append(" (disabled)"));
771 } else {
772 instance->group(std::string("Instance").append((enabledCount > 1) ? "s" : ""));
773 }
774 }
775 }
776
777 const std::string& result = app->get_option("--help")->as<std::string>();
778
779 if (result == "standard") {
780 throw CLI::CallForHelp();
781 }
782 if (result == "expanded") {
783 throw CLI::CallForAllHelp();
784 }
785 },
786 "Print help message")
787 ->take_last()
788 ->configurable(false)
789 ->check(CLI::IsMember({"standard", "expanded"}));
790 // ->trigger_on_parse();
791
792 return app;
793 }
