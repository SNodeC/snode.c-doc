53 ConfigInstance::ConfigInstance(const std::string& instanceName, Role role)
54 : instanceName(instanceName)
55 , role(role) {
56 instanceSc = utils::Config::addInstance(instanceName,
57 std::string("Configuration for ")
58 .append(role == Role::SERVER ? "server" : "client")
59 .append(" instance '")
60 .append(instanceName)
61 .append("'"),
62 "Instance");
63
64 utils::Config::addStandardFlags(instanceSc);
65 utils::Config::addHelp(instanceSc);
66
67 disableOpt = instanceSc
68 ->add_flag_function(
69 "--disabled{true}",
70 [this]([[maybe_unused]] int64_t count) -> void {
71 utils::Config::disabled(instanceSc, disableOpt->as<bool>());
72 },
73 "Disable this instance")
74 ->trigger_on_parse()
75 ->take_last()
76 ->default_val("false")
77 ->type_name("bool")
78 ->check(CLI::IsMember({"true", "false"}))
79 ->group(instanceSc->get_formatter()->get_label("Persistent Options"));
80 }

98 CLI::App* ConfigInstance::addSection(const std::string& name, const std::string& description) {
99 CLI::App* sectionSc = instanceSc //
100 ->add_subcommand(name, description)
101 ->fallthrough()
102 ->configurable(false)
103 ->allow_extras(false)
104 ->group("Sections")
105 ->disabled(this->instanceName.empty() || name.empty());
106
107 sectionSc //
108 ->option_defaults()
109 ->configurable(!sectionSc->get_disabled());
110
111 if (!sectionSc->get_disabled()) {
112 utils::Config::addStandardFlags(sectionSc);
113 utils::Config::addSimpleHelp(sectionSc);
114 }
115
116 return sectionSc;
117 }

119 void ConfigInstance::required(CLI::App* section, bool req) {
120 if (req != section->get_required()) {
121 if (req) {
122 ++requiredCount;
123 instanceSc->needs(section);
124 } else {
125 --requiredCount;
126 instanceSc->remove_needs(section);
127 }
128
129 section->required(req);
130
131 if (!disableOpt->as<bool>()) {
132 utils::Config::required(instanceSc, requiredCount > 0);
133 }
134 }
135 }
