61 ConfigInstance::ConfigInstance(const std::string& instanceName, Role role)
62 : instanceName(instanceName)
63 , role(role) {
64 auto configInstanceApp = net::config::Instance(instanceName,
65 std::string("Configuration for ")
66 .append(role == Role::SERVER ? "server" : "client")
67 .append(" instance '")
68 .append(instanceName)
69 .append("'"),
70 this);
71 instanceSc = utils::Config::newInstance(configInstanceApp, "Instances");
72
73 disableOpt = instanceSc
74 ->add_flag_function(
75 "--disabled{true}",
76 [this](std::size_t) {
77 utils::Config::disabled(instanceSc, disableOpt->as<bool>());
78 },
79 "Disable this instance")
80 ->multi_option_policy(CLI::MultiOptionPolicy::TakeLast)
81 ->trigger_on_parse()
82 ->default_str("false")
83 ->type_name("bool")
84 ->check(CLI::IsMember({"true", "false"}))
85 ->group(instanceSc->get_formatter()->get_label("Persistent Options"));
86 }

104 CLI::App* ConfigInstance::newSection(std::shared_ptr<CLI::App> appWithPtr, const std::string& group) {
105 CLI::App* sectionSc = instanceSc->add_subcommand(appWithPtr)
106 ->fallthrough()
107 ->configurable(false)
108 ->allow_extras(false)
109 ->allow_config_extras()
110 ->group(group)
111 ->ignore_case(false)
112 ->disabled(appWithPtr->get_name().empty())
113 ->formatter(instanceSc->get_formatter())
114 ->config_formatter(instanceSc->get_config_formatter());
115
116 sectionSc //
117 ->option_defaults()
118 ->configurable(!sectionSc->get_disabled());
119
120 if (!sectionSc->get_disabled()) {
121 utils::Config::addStandardFlags(sectionSc);
122 utils::Config::addSimpleHelp(sectionSc);
123 }
124
125 return sectionSc;
126 }

138 void ConfigInstance::required(CLI::App* section, bool req) {
139 if (req != section->get_required()) {
140 if (req) {
141 ++requiredCount;
142 instanceSc //
143 ->needs(section);
144 } else {
145 --requiredCount;
146 instanceSc //
147 ->remove_needs(section);
148 }
149
150 section //
151 ->required(req);
152 section //
153 ->ignore_case(req);
154
155 required(requiredCount > 0);
156 }
157 }
