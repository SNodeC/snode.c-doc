52 ConfigInstance::ConfigInstance(const std::string& name, const std::string& role)
53 : name(name) {
54 instanceSc = utils::Config::add_instance(name, "Configuration for " + role + " instance '" + name + "'");
55
56 disableOpt = instanceSc
57 ->add_flag_function(
58 "--disable",
59 [this]([[maybe_unused]] int64_t count) -> void {
60 instanceSc->disabled(disableOpt->as<bool>());
61 instanceSc->required(!disableOpt->as<bool>());
62 utils::Config::needs_instance(instanceSc);
63 },
64 "Disable this instance")
65 ->trigger_on_parse()
66 ->take_last()
67 ->default_val("false")
68 ->type_name("bool")
69 ->check(CLI::IsMember({"true", "false"}))
70 ->group(instanceSc->get_formatter()->get_label("Persistent Options"));
71
72 instanceSc->final_callback([this]() -> void {
73 (utils::ResetToDefault(disableOpt))(disableOpt->as<std::string>());
74 });
75 }

85 CLI::App* ConfigInstance::add_section(const std::string& name, const std::string& description) {
86 CLI::App* sectionSc = instanceSc //
87 ->add_subcommand(name, description)
88 ->configurable(false)
89 ->group(this->name.empty() ? "" : "Sections")
90 ->disabled(this->name.empty())
91 ->silent(this->name.empty());
92 sectionSc //
93 ->option_defaults()
94 ->configurable(!this->name.empty());
95
96 sectionSc //
97 ->add_flag_callback(
98 "-h,--help",
99 []() {
100 throw CLI::CallForHelp();
101 },
102 "Print this help message and exit")
103 ->configurable(false)
104 ->disable_flag_override()
105 ->trigger_on_parse();
106
107 sectionSc //
108 ->add_flag_callback(
109 "-a,--help-all",
110 []() {
111 throw CLI::CallForAllHelp();
112 },
113 "Print this help message and exit")
114 ->configurable(false)
115 ->disable_flag_override()
116 ->trigger_on_parse();
117
118 return sectionSc;
119 }

121 void ConfigInstance::required(CLI::App* section, bool req) {
122 if (req) {
123 ++requiredCount;
124 instanceSc //
125 ->needs(section);
126 } else {
127 if (requiredCount > 0) {
128 --requiredCount;
129 }
130 instanceSc //
131 ->remove_needs(section);
132 }
133
134 instanceSc //
135 ->required(requiredCount > 0);
136
137 utils::Config::needs_instance(instanceSc);
138 }
