82 template <typename ConfigSocketServerBase>
83 SSL_CTX* ConfigSocketServer<ConfigSocketServerBase>::getSniCtx(const std::string& serverNameIndication) {
84 SSL_CTX* sniCtx = nullptr;
85
86 if (sniCtxMap.empty()) {
87 std::map<std::string, SSL_CTX*> sslSans = core::socket::stream::tls::ssl_get_sans(sslCtx);
88
89 sniCtxMap.insert(sslSans.begin(), sslSans.end());
90
91 for (const auto& [sni, ctx] : sniCtxMap) {
92 LOG(TRACE) << getInstanceName() << " SSL/TLS: SSL_CTX (M) sni for '" << sni << "' from master certificate installed";
93 }
94
95 for (const auto& [domain, sniCertConf] : getSniCerts()) {
96 if (!domain.empty()) {
97 core::socket::stream::tls::SslConfig sslConfig(true);
98
99 sslConfig.instanceName = getInstanceName();
100
101 for (const auto& [key, value] : sniCertConf) {
102 if (key == "Cert") {
103 sslConfig.cert = std::get<std::string>(value);
104 } else if (key == "CertKey") {
105 sslConfig.certKey = std::get<std::string>(value);
106 } else if (key == "CertKeyPassword") {
107 sslConfig.password = std::get<std::string>(value);
108 } else if (key == "CaCert") {
109 sslConfig.caCert = std::get<std::string>(value);
110 } else if (key == "CaCertDir") {
111 sslConfig.caCertDir = std::get<std::string>(value);
112 } else if (key == "CaCertUseDefaultDir") {
113 sslConfig.caCertUseDefaultDir = std::get<bool>(value);
114 } else if (key == "CaCertAcceptUnknown") {
115 sslConfig.caCertAcceptUnknown = std::get<bool>(value);
116 } else if (key == "CipherList") {
117 sslConfig.cipherList = std::get<std::string>(value);
118 } else if (key == "SslOptions") {
119 sslConfig.sslOptions = std::get<ssl_option_t>(value);
120 }
121 }
122
123 SSL_CTX* newCtx = core::socket::stream::tls::ssl_ctx_new(sslConfig);
124
125 if (newCtx != nullptr) {
126 sniCtxs.push_back(newCtx);
127 sniCtxMap.insert({domain, newCtx});
128
129 LOG(TRACE) << getInstanceName() << " SSL/TLS: SSL_CTX (E) sni for '" << domain << "' explicitly installed";
130
131 for (const auto& [san, ctx] : core::socket::stream::tls::ssl_get_sans(newCtx)) {
132 sniCtxMap.insert({san, ctx});
133
134 LOG(TRACE) << getInstanceName() << " SSL/TLS: SSL_CTX (S) sni for '" << san << "' from SAN installed";
135 }
136 } else {
137 LOG(TRACE) << getInstanceName() << " SSL/TLS: Can not create SNI_SSL_CTX for domain '" << domain << "'";
138 }
139 }
140 }
141 LOG(TRACE) << getInstanceName() << " SSL/TLS: SNI list result:";
142 for (const auto& [sni, ctx] : sniCtxMap) {
143 LOG(TRACE) << " SNI: " << sni;
144 }
145 }
146
147 LOG(TRACE) << getInstanceName() << " SSL/TLS: Lookup for sni='" << serverNameIndication << "' in sni certificates";
148
149 std::map<std::string, SSL_CTX*>::iterator sniPairIt = std::find_if(
150 sniCtxMap.begin(), sniCtxMap.end(), [&serverNameIndication, this](const std::pair<std::string, SSL_CTX*>& sniPair) -> bool {
151 LOG(TRACE) << getInstanceName() << " SSL/TLS: .. " << sniPair.first.c_str();
152 return core::socket::stream::tls::match(sniPair.first.c_str(), serverNameIndication.c_str());
153 });
154
155 if (sniPairIt != sniCtxMap.end()) {
156 LOG(TRACE) << getInstanceName() << " SSL/TLS: found '" << sniPairIt->first << "'";
157 sniCtx = sniPairIt->second;
158 } else {
159 LOG(TRACE) << getInstanceName() << " SSL/TLS: not found";
160 }
161
162 return sniCtx;
163 }

165 template <typename ConfigSocketServerBase>
166 int ConfigSocketServer<ConfigSocketServerBase>::clientHelloCallback(SSL* ssl, int* al, void* arg) {
167 int ret = SSL_CLIENT_HELLO_SUCCESS;
168
169 ConfigSocketServer* config = static_cast<ConfigSocketServer*>(arg);
170
171 std::string serverNameIndication = core::socket::stream::tls::ssl_get_servername_from_client_hello(ssl);
172
173 if (!serverNameIndication.empty()) {
174 SSL_CTX* sniSslCtx = config->getSniCtx(serverNameIndication);
175 if (sniSslCtx != nullptr) {
176 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: Setting sni certificate for '" << serverNameIndication << "'";
177 core::socket::stream::tls::ssl_set_ssl_ctx(ssl, sniSslCtx);
178 } else if (config->getForceSni()) {
179 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: No sni certificate found for '" << serverNameIndication
180 << "' but forceSni set - terminating";
181 ret = SSL_CLIENT_HELLO_ERROR;
182 *al = SSL_AD_UNRECOGNIZED_NAME;
183 } else {
184 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: No sni certificate found for '" << serverNameIndication
185 << "'. Still using master certificate";
186 }
187 } else {
188 LOG(TRACE) << config->getInstanceName() << " SSL/TLS: No sni certificate requested from client. Still using master certificate";
189 }
190
191 return ret;
192 }
