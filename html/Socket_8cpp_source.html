47 ssize_t Socket::sendFd(SocketAddress& destAddress, int sendfd) {
48 union {
49 struct cmsghdr cm;
50 char control[CMSG_SPACE(sizeof(int))] = {};
51 } control_un;
52
53 msghdr msg{};
54 msg.msg_name = const_cast<sockaddr*>(&destAddress.getSockAddr());
55 msg.msg_namelen = destAddress.getSockAddrLen();
56
57 msg.msg_control = control_un.control;
58 msg.msg_controllen = sizeof(control_un.control);
59
60 iovec iov[1];
61 msg.msg_iov = iov;
62 msg.msg_iovlen = 1;
63
64 char ptr = 0;
65 msg.msg_iov[0].iov_base = &ptr;
66 msg.msg_iov[0].iov_len = 1;
67
68 cmsghdr* cmptr = CMSG_FIRSTHDR(&msg);
69 cmptr->cmsg_level = SOL_SOCKET;
70 cmptr->cmsg_type = SCM_RIGHTS;
71 *reinterpret_cast<int*>(CMSG_DATA(cmptr)) = sendfd;
72 cmptr->cmsg_len = CMSG_LEN(sizeof(int));
73
74 return sendmsg(core::Descriptor::getFd(), &msg, 0);
75 }

77 ssize_t Socket::recvFd(int* recvfd) {
78 union {
79 struct cmsghdr cm;
80 char control[CMSG_SPACE(sizeof(int))] = {};
81 } control_un;
82
83 msghdr msg{};
84 msg.msg_control = control_un.control;
85 msg.msg_controllen = sizeof(control_un.control);
86
87 msg.msg_name = nullptr;
88 msg.msg_namelen = 0;
89
90 iovec iov[1];
91 msg.msg_iov = iov;
92 msg.msg_iovlen = 1;
93
94 char ptr = 0;
95 msg.msg_iov[0].iov_base = &ptr;
96 msg.msg_iov[0].iov_len = 1;
97
98 ssize_t n = 0;
99
100 if ((n = recvmsg(core::Descriptor::getFd(), &msg, 0)) > 0) {
101 cmsghdr* cmptr = nullptr;
102
103 if ((cmptr = CMSG_FIRSTHDR(&msg)) != nullptr && cmptr->cmsg_len == CMSG_LEN(sizeof(int))) {
104 if (cmptr->cmsg_level != SOL_SOCKET || cmptr->cmsg_type != SCM_RIGHTS) {
105 errno = EBADE;
106 *recvfd = -1;
107 n = -1;
108 } else {
109 *recvfd = *reinterpret_cast<int*>(CMSG_DATA(cmptr));
110 }
111 } else {
112 errno = ENOMSG;
113 *recvfd = -1;
114 n = -1;
115 }
116 }
117
118 return n;
119 }
