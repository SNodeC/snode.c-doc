50 VHost::VHost(const std::string& host)
51 : host(host) {
52 use("/", [&host = this->host] MIDDLEWARE(req, res, next) {
53 if (req->get("Host") == host) {
54 next();
55 } else {
56 next("router");
57 }
58 });
59 }

61 class VHost& VHost::instance(const std::string& host) {
62 // Keep all created vhost middlewares alive
63 static std::map<std::string, std::shared_ptr<class VHost>> vHostMiddlewares;
64
65 if (!vHostMiddlewares.contains(host)) {
66 vHostMiddlewares[host] = std::shared_ptr<VHost>(new VHost(host));
67 }
68
69 return *vHostMiddlewares[host];
70 }

72 // "Constructor" of StaticMiddleware
73 class VHost& VHost(const std::string& host) {
74 return VHost::instance(host);
