63 EventMultiplexer::EventMultiplexer()
64 : core::EventMultiplexer(new core::multiplexer::epoll::DescriptorEventPublisher("READ", //
65 epfds[core::EventMultiplexer::DISP_TYPE::RD],
66 EPOLLIN,
67 EPOLLIN | EPOLLHUP | EPOLLRDHUP | EPOLLERR),
68 new core::multiplexer::epoll::DescriptorEventPublisher("WRITE", //
69 epfds[core::EventMultiplexer::DISP_TYPE::WR],
70 EPOLLOUT,
71 EPOLLOUT),
72 new core::multiplexer::epoll::DescriptorEventPublisher("EXCEPT", //
73 epfds[core::EventMultiplexer::DISP_TYPE::EX],
74 EPOLLPRI,
75 EPOLLPRI))
76 , epfd(core::system::epoll_create1(EPOLL_CLOEXEC)) {
77 epoll_event event{};
78 event.events = EPOLLIN;
79
80 event.data.ptr = descriptorEventPublishers[core::EventMultiplexer::DISP_TYPE::RD];
81 core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, epfds[core::EventMultiplexer::DISP_TYPE::RD], &event);
82
83 event.data.ptr = descriptorEventPublishers[core::EventMultiplexer::DISP_TYPE::WR];
84 core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, epfds[core::EventMultiplexer::DISP_TYPE::WR], &event);
85
86 event.data.ptr = descriptorEventPublishers[core::EventMultiplexer::DISP_TYPE::EX];
87 core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, epfds[core::EventMultiplexer::DISP_TYPE::EX], &event);
88
89 LOG(DEBUG) << "Core::multiplexer: epoll";
90 }

96 void EventMultiplexer::spanActiveEvents(int activeDescriptorCount) {
97 for (int i = 0; i < activeDescriptorCount; i++) {
98 if ((ePollEvents[i].events & EPOLLIN) != 0) {
99 static_cast<core::DescriptorEventPublisher*>(ePollEvents[i].data.ptr)->spanActiveEvents();
100 }
101 }
102 }
