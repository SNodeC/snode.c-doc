31 Connect::Connect(const std::string& clientId,
32 uint16_t keepAlive,
33 bool cleanSession,
34 const std::string& willTopic,
35 const std::string& willMessage,
36 uint8_t willQoS,
37 bool willRetain,
38 const std::string& username,
39 const std::string& password)
40 : Connect() {
41 this->protocol = "MQTT";
42 this->level = MQTT_VERSION_3_1_1;
43 this->keepAlive = keepAlive;
44 this->clientId = clientId;
45 this->willTopic = willTopic;
46 this->willMessage = willMessage;
47 this->willQoS = willQoS;
48 this->willRetain = willRetain;
49 this->willFlag = !willMessage.empty();
50 this->username = username;
51 this->usernameFlag = !username.empty();
52 this->password = password;
53 this->passwordFlag = !password.empty();
54
55 this->connectFlags = static_cast<uint8_t>((!usernameFlag ? 0 : 0x80) | (!passwordFlag ? 0 : 0x40) | (!willRetain ? 0 : 0x20) |
56 ((willQoS << 3) & 0x18) | (!willFlag ? 0 : 0x04) | (!cleanSession ? 0 : 0x02));
57 }
