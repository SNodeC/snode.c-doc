36 template <typename SocketAddress, typename PhysicalSocket, typename Config>
37 SocketAddress getLocalSocketAddress(PhysicalSocket& physicalSocket, Config& config) {
38 typename SocketAddress::SockAddr localSockAddr;
39 typename SocketAddress::SockLen localSockAddrLen = sizeof(typename SocketAddress::SockAddr);
40
41 SocketAddress localPeerAddress;
42
43 if (physicalSocket.getSockName(localSockAddr, localSockAddrLen) == 0) {
44 try {
45 localPeerAddress = config->Local::getSocketAddress(localSockAddr, localSockAddrLen);
46 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << localPeerAddress.toString();
47 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
48 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << badSocketAddress.what();
49 }
50 } else {
51 PLOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local) not retrievable";
52 }
53
54 return localPeerAddress;
55 }

115 template <typename PhysicalSocketClient, typename Config, template <typename PhysicalSocketClientT> typename SocketConnection>
116 void SocketConnector<PhysicalSocketClient, Config, SocketConnection>::init() {
117 if (!config->getDisabled() && core::eventLoopState() == core::State::RUNNING) {
118 try {
119 LOG(TRACE) << config->getInstanceName() << " Starting";
120
121 remoteAddress = config->Remote::getSocketAddress();
122 SocketAddress localAddress = config->Local::getSocketAddress();
123
124 try {
125 if (physicalClientSocket.open(config->getSocketOptions(), PhysicalClientSocket::Flags::NONBLOCK) < 0) {
126 core::socket::State state = core::socket::STATE_OK;
127
128 switch (errno) {
129 case EMFILE:
130 case ENFILE:
131 case ENOBUFS:
132 case ENOMEM:
133 PLOG(TRACE) << config->getInstanceName() << " open: '" << localAddress.toString() << "'";
134
135 state = core::socket::STATE_ERROR;
136 break;
137 default:
138 PLOG(TRACE) << config->getInstanceName() << " open failed: '" << localAddress.toString() << "'";
139
140 state = core::socket::STATE_FATAL;
141 break;
142 }
143
144 onStatus(remoteAddress, state);
145 } else if (physicalClientSocket.bind(localAddress) < 0) {
146 core::socket::State state = core::socket::STATE_OK;
147
148 switch (errno) {
149 case EADDRINUSE:
150 PLOG(TRACE) << config->getInstanceName() << " bind: '" << localAddress.toString() << "'";
151
152 state = core::socket::STATE_ERROR;
153 break;
154 default:
155 PLOG(TRACE) << config->getInstanceName() << " bind failed: '" << localAddress.toString() << "'";
156
157 state = core::socket::STATE_FATAL;
158 break;
159 }
160
161 onStatus(remoteAddress, state);
162 } else if (physicalClientSocket.connect(remoteAddress) < 0 && !PhysicalClientSocket::connectInProgress(errno)) {
163 core::socket::State state = core::socket::STATE_OK;
164
165 switch (errno) {
166 case EADDRINUSE:
167 case EADDRNOTAVAIL:
168 case ECONNREFUSED:
169 case ENETUNREACH:
170 case ENOENT:
171 case EHOSTDOWN:
172 PLOG(TRACE) << config->getInstanceName() << " connect: '" << remoteAddress.toString() << "'";
173
174 state = core::socket::STATE_ERROR;
175 break;
176 default:
177 PLOG(TRACE) << config->getInstanceName() << " connect failed: '" << remoteAddress.toString() << "'";
178
179 state = core::socket::STATE_FATAL;
180 break;
181 }
182
183 if (remoteAddress.useNext()) {
184 onStatus(remoteAddress, state | core::socket::State::NO_RETRY);
185
186 LOG(TRACE) << config->getInstanceName() << " using next SocketAddress: '"
187 << config->Remote::getSocketAddress().toString() << "'";
188
189 useNextSocketAddress();
190 } else {
191 onStatus(remoteAddress, state);
192 }
193 } else if (PhysicalClientSocket::connectInProgress(errno)) {
194 if (enable(physicalClientSocket.getFd())) {
195 LOG(TRACE) << config->getInstanceName() << " connect in progress: '" << remoteAddress.toString() << "'";
196 } else {
197 LOG(TRACE) << config->getInstanceName() << " Error: not monitored by SNode.C";
198 }
199 } else {
200 LOG(TRACE) << config->getInstanceName() << " connect success: '" << remoteAddress.toString() << "'";
201
202 onStatus(remoteAddress, core::socket::STATE_OK);
203
204 SocketConnection* socketConnection =
205 new SocketConnection(config->getInstanceName(),
206 std::move(physicalClientSocket),
207 onDisconnect,
208 remoteAddress.toString(false),
209 getLocalSocketAddress<SocketAddress>(physicalClientSocket, config),
210 getRemoteSocketAddress<SocketAddress>(physicalClientSocket, config),
211 config->getReadTimeout(),
212 config->getWriteTimeout(),
213 config->getReadBlockSize(),
214 config->getWriteBlockSize(),
215 config->getTerminateTimeout());
216
217 onConnect(socketConnection);
218 onConnected(socketConnection);
219 }
220 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
221 LOG(TRACE) << config->getInstanceName() << " " << badSocketAddress.what();
222
223 onStatus({}, core::socket::STATE(badSocketAddress.getState(), badSocketAddress.getErrnum(), badSocketAddress.what()));
224 }
225 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
226 LOG(TRACE) << config->getInstanceName() << " " << badSocketAddress.what();
227
228 onStatus({}, core::socket::STATE(badSocketAddress.getState(), badSocketAddress.getErrnum(), badSocketAddress.what()));
229 }
230 } else {
231 LOG(TRACE) << config->getInstanceName() << " disabled";
232
233 onStatus({}, core::socket::STATE_DISABLED);
234 }
235
236 if (isEnabled()) {
237 core::eventreceiver::ConnectEventReceiver::setTimeout(config->getConnectTimeout());
238 } else {
239 destruct();
240 }
241 }

243 template <typename PhysicalSocketClient, typename Config, template <typename PhysicalSocketClientT> typename SocketConnection>
244 void SocketConnector<PhysicalSocketClient, Config, SocketConnection>::connectEvent() {
245 int cErrno = 0;
246 if (physicalClientSocket.getSockError(cErrno) == 0) { // == 0->return valid : < 0->getsockopt failed
247 const utils::PreserveErrno pe(cErrno); // errno = cErrno
248
249 if (errno == 0) {
250 LOG(TRACE) << config->getInstanceName() << " connect success: '" << remoteAddress.toString() << "'";
251
252 onStatus(remoteAddress, core::socket::STATE_OK);
253
254 SocketConnection* socketConnection =
255 new SocketConnection(config->getInstanceName(),
256 std::move(physicalClientSocket),
257 onDisconnect,
258 remoteAddress.toString(false),
259 getLocalSocketAddress<SocketAddress>(physicalClientSocket, config),
260 getRemoteSocketAddress<SocketAddress>(physicalClientSocket, config),
261 config->getReadTimeout(),
262 config->getWriteTimeout(),
263 config->getReadBlockSize(),
264 config->getWriteBlockSize(),
265 config->getTerminateTimeout());
266
267 onConnect(socketConnection);
268 onConnected(socketConnection);
269
270 disable();
271 } else if (PhysicalClientSocket::connectInProgress(errno)) {
272 LOG(TRACE) << config->getInstanceName() << " connect still in progress: '" << remoteAddress.toString() << "'";
273 } else if (remoteAddress.useNext()) {
274 core::socket::State state = core::socket::STATE_OK;
275
276 switch (errno) {
277 case EADDRINUSE:
278 case EADDRNOTAVAIL:
279 case ECONNREFUSED:
280 case ENETUNREACH:
281 case ENOENT:
282 case EHOSTDOWN:
283 PLOG(TRACE) << config->getInstanceName() << " connect failed: '" << remoteAddress.toString() << "'";
284
285 state = core::socket::STATE_ERROR;
286 break;
287 default:
288 PLOG(TRACE) << config->getInstanceName() << ": connect failed '" << remoteAddress.toString() << "'";
289
290 state = core::socket::STATE_FATAL;
291 break;
292 }
293
294 onStatus(remoteAddress, (state | core::socket::State::NO_RETRY));
295
296 LOG(TRACE) << config->getInstanceName() << " using next SocketAddress: '" << config->Remote::getSocketAddress().toString()
297 << "'";
298
299 useNextSocketAddress();
300
301 disable();
302 } else {
303 core::socket::State state = core::socket::STATE_OK;
304
305 switch (errno) {
306 case EADDRINUSE:
307 case EADDRNOTAVAIL:
308 case ECONNREFUSED:
309 case ENETUNREACH:
310 case ENOENT:
311 case EHOSTDOWN:
312 PLOG(TRACE) << config->getInstanceName() << " connect failed: '" << remoteAddress.toString() << "'";
313
314 state = core::socket::STATE_ERROR;
315 break;
316 default:
317 PLOG(TRACE) << config->getInstanceName() << " connect failed: '" << remoteAddress.toString() << "'";
318
319 state = core::socket::STATE_FATAL;
320 break;
321 }
322
323 onStatus(remoteAddress, state);
324
325 disable();
326 }
327 } else {
328 PLOG(TRACE) << config->getInstanceName() << " getsockopt syscall error: '" << remoteAddress.toString() << "'";
329
330 onStatus(remoteAddress, core::socket::STATE_FATAL);
331 disable();
332 }
333 }
