36 template <typename SocketAddress, typename PhysicalSocket, typename Config>
37 SocketAddress getLocalSocketAddress(PhysicalSocket& physicalSocket, Config& config) {
38 typename SocketAddress::SockAddr localSockAddr;
39 typename SocketAddress::SockLen localSockAddrLen = sizeof(typename SocketAddress::SockAddr);
40
41 SocketAddress localPeerAddress;
42
43 if (physicalSocket.getSockName(localSockAddr, localSockAddrLen) == 0) {
44 try {
45 localPeerAddress = config->Local::getSocketAddress(localSockAddr, localSockAddrLen);
46 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << localPeerAddress.toString();
47 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
48 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << badSocketAddress.what();
49
50 try {
51 localPeerAddress = config->Local::Super::getSocketAddress(localSockAddr, localSockAddrLen);
52 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << localPeerAddress.toString();
53 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) { // cppcheck-suppress shadowVariable
54 LOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local): " << badSocketAddress.what();
55 }
56 }
57 } else {
58 PLOG(TRACE) << config->getInstanceName() << std::setw(24) << " PeerAddress (local) not retrievable";
59 }
60
61 return localPeerAddress;
62 }

129 template <typename PhysicalSocketClient, typename Config, template <typename PhysicalSocketClientT> typename SocketConnection>
130 void SocketConnector<PhysicalSocketClient, Config, SocketConnection>::init() {
131 if (!config->getDisabled() && core::eventLoopState() == core::State::RUNNING) {
132 try {
133 LOG(TRACE) << config->getInstanceName() << ": Starting";
134
135 remoteAddress = config->Remote::getSocketAddress();
136 SocketAddress localAddress = config->Local::getSocketAddress();
137
138 try {
139 if (physicalClientSocket.open(config->getSocketOptions(), PhysicalClientSocket::Flags::NONBLOCK) < 0) {
140 core::socket::State state = core::socket::STATE_OK;
141
142 switch (errno) {
143 case EMFILE:
144 case ENFILE:
145 case ENOBUFS:
146 case ENOMEM:
147 PLOG(TRACE) << config->getInstanceName() << ": open '" << localAddress.toString() << "'";
148
149 state = core::socket::STATE_ERROR;
150 break;
151 default:
152 PLOG(TRACE) << config->getInstanceName() << ": open failed '" << localAddress.toString() << "'";
153
154 state = core::socket::STATE_FATAL;
155 break;
156 }
157
158 onStatus(remoteAddress, state);
159 } else if (physicalClientSocket.bind(localAddress) < 0) {
160 core::socket::State state = core::socket::STATE_OK;
161
162 switch (errno) {
163 case EADDRINUSE:
164 PLOG(TRACE) << config->getInstanceName() << ": bind '" << localAddress.toString() << "'";
165
166 state = core::socket::STATE_ERROR;
167 break;
168 default:
169 PLOG(TRACE) << config->getInstanceName() << ": bind failed '" << localAddress.toString() << "'";
170
171 state = core::socket::STATE_FATAL;
172 break;
173 }
174
175 onStatus(remoteAddress, state);
176 } else if (physicalClientSocket.connect(remoteAddress) < 0 && !PhysicalClientSocket::connectInProgress(errno)) {
177 core::socket::State state = core::socket::STATE_OK;
178
179 switch (errno) {
180 case EADDRINUSE:
181 case EADDRNOTAVAIL:
182 case ECONNREFUSED:
183 case ENETUNREACH:
184 case ENOENT:
185 case EHOSTDOWN:
186 PLOG(TRACE) << config->getInstanceName() << ": connect '" << remoteAddress.toString() << "'";
187
188 state = core::socket::STATE_ERROR;
189 break;
190 default:
191 PLOG(TRACE) << config->getInstanceName() << ": connect failed '" << remoteAddress.toString() << "'";
192
193 state = core::socket::STATE_FATAL;
194 break;
195 }
196
197 if (remoteAddress.useNext()) {
198 onStatus(remoteAddress, state | core::socket::State::NO_RETRY);
199
200 LOG(TRACE) << config->getInstanceName() << ": using next SocketAddress '"
201 << config->Remote::getSocketAddress().toString() << "'";
202
203 useNextSocketAddress();
204 } else {
205 onStatus(remoteAddress, state);
206 }
207 } else if (PhysicalClientSocket::connectInProgress(errno)) {
208 if (enable(physicalClientSocket.getFd())) {
209 LOG(TRACE) << config->getInstanceName() << ": connect in progress '" << remoteAddress.toString() << "'";
210 } else {
211 LOG(TRACE) << config->getInstanceName() << ": Error: not monitored by SNode.C";
212 }
213 } else {
214 LOG(TRACE) << config->getInstanceName() << ": connect success '" << remoteAddress.toString() << "'";
215
216 onStatus(remoteAddress, core::socket::STATE_OK);
217
218 SocketConnection* socketConnection =
219 new SocketConnection(config->getInstanceName(),
220 std::move(physicalClientSocket),
221 onDisconnect,
222 remoteAddress.toString(false),
223 getLocalSocketAddress<SocketAddress>(physicalClientSocket, config),
224 getRemoteSocketAddress<SocketAddress>(physicalClientSocket, config),
225 config->getReadTimeout(),
226 config->getWriteTimeout(),
227 config->getReadBlockSize(),
228 config->getWriteBlockSize(),
229 config->getTerminateTimeout());
230
231 onConnect(socketConnection);
232 onConnected(socketConnection);
233 }
234 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
235 LOG(TRACE) << config->getInstanceName() << ": " << badSocketAddress.what();
236
237 onStatus({}, core::socket::STATE(badSocketAddress.getState(), badSocketAddress.getErrnum(), badSocketAddress.what()));
238 }
239 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
240 LOG(TRACE) << config->getInstanceName() << ": " << badSocketAddress.what();
241
242 onStatus({}, core::socket::STATE(badSocketAddress.getState(), badSocketAddress.getErrnum(), badSocketAddress.what()));
243 }
244 } else {
245 LOG(TRACE) << config->getInstanceName() << ": disabled";
246
247 onStatus({}, core::socket::STATE_DISABLED);
248 }
249
250 if (isEnabled()) {
251 core::eventreceiver::ConnectEventReceiver::setTimeout(config->getConnectTimeout());
252 } else {
253 destruct();
254 }
255 }

257 template <typename PhysicalSocketClient, typename Config, template <typename PhysicalSocketClientT> typename SocketConnection>
258 void SocketConnector<PhysicalSocketClient, Config, SocketConnection>::connectEvent() {
259 int cErrno = 0;
260 if (physicalClientSocket.getSockError(cErrno) == 0) { // == 0->return valid : < 0->getsockopt failed
261 const utils::PreserveErrno pe(cErrno); // errno = cErrno
262
263 if (errno == 0) {
264 LOG(TRACE) << config->getInstanceName() << ": connect success '" << remoteAddress.toString() << "'";
265
266 onStatus(remoteAddress, core::socket::STATE_OK);
267
268 SocketConnection* socketConnection =
269 new SocketConnection(config->getInstanceName(),
270 std::move(physicalClientSocket),
271 onDisconnect,
272 remoteAddress.toString(false),
273 getLocalSocketAddress<SocketAddress>(physicalClientSocket, config),
274 getRemoteSocketAddress<SocketAddress>(physicalClientSocket, config),
275 config->getReadTimeout(),
276 config->getWriteTimeout(),
277 config->getReadBlockSize(),
278 config->getWriteBlockSize(),
279 config->getTerminateTimeout());
280
281 onConnect(socketConnection);
282 onConnected(socketConnection);
283
284 disable();
285 } else if (PhysicalClientSocket::connectInProgress(errno)) {
286 LOG(TRACE) << config->getInstanceName() << ": connect still in progress '" << remoteAddress.toString() << "'";
287 } else if (remoteAddress.useNext()) {
288 core::socket::State state = core::socket::STATE_OK;
289
290 switch (errno) {
291 case EADDRINUSE:
292 case EADDRNOTAVAIL:
293 case ECONNREFUSED:
294 case ENETUNREACH:
295 case ENOENT:
296 case EHOSTDOWN:
297 PLOG(TRACE) << config->getInstanceName() << ": connect failed '" << remoteAddress.toString() << "'";
298
299 state = core::socket::STATE_ERROR;
300 break;
301 default:
302 PLOG(TRACE) << config->getInstanceName() << ": connect failed '" << remoteAddress.toString() << "'";
303
304 state = core::socket::STATE_FATAL;
305 break;
306 }
307
308 onStatus(remoteAddress, (state | core::socket::State::NO_RETRY));
309
310 LOG(TRACE) << config->getInstanceName() << ": using next SocketAddress '" << config->Remote::getSocketAddress().toString()
311 << "'";
312
313 useNextSocketAddress();
314
315 disable();
316 } else {
317 core::socket::State state = core::socket::STATE_OK;
318
319 switch (errno) {
320 case EADDRINUSE:
321 case EADDRNOTAVAIL:
322 case ECONNREFUSED:
323 case ENETUNREACH:
324 case ENOENT:
325 case EHOSTDOWN:
326 PLOG(TRACE) << config->getInstanceName() << ": connect failed '" << remoteAddress.toString() << "'";
327
328 state = core::socket::STATE_ERROR;
329 break;
330 default:
331 PLOG(TRACE) << config->getInstanceName() << ": connect failed '" << remoteAddress.toString() << "'";
332
333 state = core::socket::STATE_FATAL;
334 break;
335 }
336
337 onStatus(remoteAddress, state);
338
339 disable();
340 }
341 } else {
342 PLOG(TRACE) << config->getInstanceName() << ": getsockopt syscall error '" << remoteAddress.toString() << "'";
343
344 onStatus(remoteAddress, core::socket::STATE_FATAL);
345 disable();
346 }
347 }
