60 template <typename PhysicalClientSocket, typename Config, template <typename PhysicalClientSocketT> typename SocketConnection>
61 void SocketConnector<PhysicalClientSocket, Config, SocketConnection>::initConnectEvent() {
62 if (!config->getDisabled()) {
63 LOG(INFO) << "Instance '" << this->config->getInstanceName() << "' enabled";
64
65 core::eventreceiver::ConnectEventReceiver::setTimeout(config->getConnectTimeout());
66
67 SocketAddress localAddress;
68
69 try {
70 localAddress = config->Local::getSocketAddress();
71
72 try {
73 remoteAddress = config->Remote::getSocketAddress();
74
75 physicalSocket = new PhysicalSocket();
76
77 if (physicalSocket->open(config->getSocketOptions(), PhysicalSocket::Flags::NONBLOCK) < 0) {
78 progressLog->addEntry(0)
79 << this->config->getInstanceName() << ": SocketConnector::open '" << remoteAddress.toString() << "'";
80 } else if (physicalSocket->bind(localAddress) < 0) {
81 progressLog->addEntry(0)
82 << this->config->getInstanceName() << ": SocketConnector::bind '" << remoteAddress.toString() << "'";
83 } else if (physicalSocket->connect(remoteAddress) != 0 && !physicalSocket->connectInProgress(errno)) {
84 progressLog->addEntry(0)
85 << this->config->getInstanceName() << ": SocketConnector::initial connect '" << remoteAddress.toString() << "'";
86 } else if (physicalSocket->connectInProgress(errno)) {
87 utils::PreserveErrno pe(0);
88 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketConnector::initial connect '"
89 << remoteAddress.toString() << "' connecting in progress";
90
91 enable(physicalSocket->getFd());
92 } else {
93 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketConnector::initial connect '"
94 << remoteAddress.toString() << "' connection estableshed";
95 onError(*progressLog);
96
97 SocketConnectionFactory(socketContextFactory, onConnect, onConnected, onDisconnect).create(*physicalSocket, config);
98 }
99
100 if (!isEnabled()) {
101 if (remoteAddress.useNext()) {
102 new SocketConnector(socketContextFactory, onConnect, onConnected, onDisconnect, onError, config, progressLog);
103 } else {
104 onError(*progressLog);
105 }
106
107 destruct();
108 }
109 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
110 utils::PreserveErrno pe(0);
111 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketConnector::getRemoteAddress '"
112 << remoteAddress.toString() << "': BadSocketAddress: " << badSocketAddress.what();
113 onError(*progressLog);
114
115 destruct();
116 }
117 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
118 utils::PreserveErrno pe(0);
119 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketConnector::getLocalAddress '"
120 << localAddress.toString() << "': BadSocketAddress: " << badSocketAddress.what();
121 onError(*progressLog);
122
123 destruct();
124 }
125 } else {
126 LOG(INFO) << "Instance '" << this->config->getInstanceName() << "' disabled";
127
128 destruct();
129 }
130 }

132 template <typename PhysicalClientSocket, typename Config, template <typename PhysicalClientSocketT> typename SocketConnection>
133 void SocketConnector<PhysicalClientSocket, Config, SocketConnection>::connectEvent() {
134 int cErrno;
135
136 if (physicalSocket->getSockError(cErrno) == 0) { // == 0->return valid : < 0->getsockopt failed errno = cErrno;
137 errno = cErrno;
138
139 if (errno == 0) {
140 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketConnector::connectEvent '"
141 << remoteAddress.toString() << "'";
142 onError(*progressLog.get());
143
144 disable();
145
146 SocketConnectionFactory(socketContextFactory, onConnect, onConnected, onDisconnect).create(*physicalSocket, config);
147 } else if (!physicalSocket->connectInProgress(errno)) {
148 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketConnector::connectEvent::error '"
149 << remoteAddress.toString() << "'";
150 if (remoteAddress.useNext()) {
151 new SocketConnector(socketContextFactory, onConnect, onConnected, onDisconnect, onError, config, progressLog);
152 } else {
153 onError(*progressLog);
154 }
155
156 disable();
157 }
158 } else { // syscall error
159 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketConnector::connectEvent '" << remoteAddress.toString()
160 << "'";
161 onError(*progressLog);
162
163 disable();
164 }
165 }
