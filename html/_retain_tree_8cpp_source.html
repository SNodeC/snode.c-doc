54 RetainTree::TopicLevel& RetainTree::TopicLevel::fromJson(const nlohmann::json& json) {
55 subTopicLevels.clear();

82 bool RetainTree::TopicLevel::retain(const Message& message, std::string topic, bool appeared) {
83 if (appeared) {
84 if (!message.getTopic().empty()) {
85 LOG(TRACE) << "Retaining: " << message.getTopic() << " - " << message.getMessage();
86 this->message = message;
87 }
88 } else {
89 std::string::size_type slashPosition = topic.find('/');
90
91 std::string topicLevel = topic.substr(0, slashPosition);
92 bool appeared = slashPosition == std::string::npos;
93
94 topic.erase(0, topicLevel.size() + 1);
95
96 if (subTopicLevels.insert({topicLevel, TopicLevel(broker)}).first->second.retain(message, topic, appeared)) {
97 subTopicLevels.erase(topicLevel);
98 }
99 }
100
101 return this->message.getMessage().empty() && subTopicLevels.empty();
102 }

108 void RetainTree::TopicLevel::appear(const std::string& clientId, std::string topic, uint8_t qoS, bool appeared) {
109 if (appeared) {
110 if (!message.getMessage().empty()) {
111 LOG(TRACE) << "Retained message found: " << message.getTopic() << " - " << message.getMessage() << " - "
112 << static_cast<uint16_t>(message.getQoS());
113 LOG(TRACE) << " distribute message ...";
114 broker->sendPublish(clientId, message, qoS, true);
115 LOG(TRACE) << " ... completed!";
116 }
117 } else {
118 std::string::size_type slashPosition = topic.find('/');
119
120 std::string topicLevel = topic.substr(0, slashPosition);
121 bool leafFound = slashPosition == std::string::npos;
122
123 topic.erase(0, topicLevel.size() + 1);
124
125 auto foundNode = subTopicLevels.find(topicLevel);
126 if (foundNode != subTopicLevels.end()) {
127 foundNode->second.appear(clientId, topic, qoS, leafFound);
128 } else if (topicLevel == "+") {
129 for (auto& [notUsed, topicTree] : subTopicLevels) {
130 topicTree.appear(clientId, topic, qoS, leafFound);
131 }
132 } else if (topicLevel == "#") {
133 appear(clientId, qoS);
134 }
135 }
136 }

138 void RetainTree::TopicLevel::appear(const std::string& clientId, uint8_t clientQoS) {
139 if (!message.getTopic().empty()) {
140 LOG(TRACE) << "Retained message found: " << message.getTopic() << " - " << message.getMessage() << " - "
141 << static_cast<uint16_t>(message.getQoS());
142 LOG(TRACE) << " distribute message ...";
143 broker->sendPublish(clientId, message, clientQoS, true);
144 LOG(TRACE) << " ... completed!";
145 }
146
147 for (auto& [topicLevel, topicTree] : subTopicLevels) {
148 topicTree.appear(clientId, clientQoS);
149 }
