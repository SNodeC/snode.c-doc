37 CLI11_INLINE std::string
38 ConfigFormatter::to_config(const App* app, bool default_also, bool write_description, std::string prefix) const {
39 std::stringstream out;
40 std::string commentLead;
41 commentLead.push_back(commentChar);
42 commentLead.push_back(' ');
43
44 std::vector<std::string> groups = app->get_groups();
45 bool defaultUsed = false;
46 groups.insert(groups.begin(), std::string("Options"));
47 for (const auto& group : groups) {
48 if (group == "Options" || group.empty()) {
49 if (defaultUsed) {
50 continue;
51 }
52 defaultUsed = true;
53 }
54 if (write_description && group != "Options" && !group.empty() &&
55 !app->get_options([group](const Option* opt) -> bool {
56 return opt->get_group() == group && opt->get_configurable();
57 })
58 .empty()) {
59 out << '\n' << commentChar << commentLead << group << "\n";
60 }
61 for (const Option* opt : app->get_options({})) {
62 // Only process options that are configurable
63 if (opt->get_configurable()) {
64 if (opt->get_group() != group) {
65 if (!(group == "Options" && opt->get_group().empty())) {
66 continue;
67 }
68 }
69 const std::string name = prefix + opt->get_single_name();
70 std::string value =
71 detail::ini_join(opt->reduced_results(), arraySeparator, arrayStart, arrayEnd, stringQuote, characterQuote);
72
73 std::string defaultValue{};
74 if (default_also) {
75 static_assert(std::string::npos + static_cast<std::string::size_type>(1) == 0,
76 "std::string::npos + static_cast<std::string::size_type>(1) != 0");
77 if (!value.empty() &&
78 opt->get_default_str() ==
79 value.substr(0, value.find_last_of('"')).substr(value.find('"') + static_cast<std::string::size_type>(1))) {
80 value.clear();
81 }
82 if (!opt->get_default_str().empty()) {
83 defaultValue = detail::convert_arg_for_ini(opt->get_default_str(), stringQuote, characterQuote);
84 } else if (opt->get_expected_min() == 0) {
85 defaultValue = "false";
86 } else if (opt->get_run_callback_for_default()) {
87 defaultValue = "\"\""; // empty string default value
88 } else if (opt->get_required()) {
89 defaultValue = "\"<REQUIRED>\"";
90 } else {
91 defaultValue = "\"\"";
92 }
93 }
94 if (write_description && opt->has_description() && (default_also || !value.empty())) {
95 out << commentLead << detail::fix_newlines(commentLead, opt->get_description()) << '\n';
96 }
97 if (default_also) {
98 out << commentChar << name << valueDelimiter << defaultValue << "\n";
99 }
100 if (!value.empty()) {
101 if (!opt->get_fnames().empty()) {
102 value = opt->get_flag_value(name, value);
103 }
104 out << name << valueDelimiter << value << "\n\n";
105 } else {
106 out << '\n';
107 }
108 }
109 }
110 }
111 auto subcommands = app->get_subcommands({});
112 for (const App* subcom : subcommands) {
113 if (subcom->get_name().empty()) {
114 if (write_description && !subcom->get_group().empty()) {
115 out << '\n' << commentLead << subcom->get_group() << " Options\n";
116 }
117 out << to_config(subcom, default_also, write_description, prefix);
118 }
119 }
120
121 for (const App* subcom : subcommands) {
122 if (!subcom->get_name().empty()) {
123 if (subcom->get_configurable() && app->got_subcommand(subcom)) {
124 if (!prefix.empty() || app->get_parent() == nullptr) {
125 out << '[' << prefix << subcom->get_name() << "]\n";
126 } else {
127 std::string subname = app->get_name() + parentSeparatorChar + subcom->get_name();
128 const auto* p = app->get_parent();
129 while (p->get_parent() != nullptr) {
130 subname = p->get_name() + parentSeparatorChar + subname;
131 p = p->get_parent();
132 }
133 out << '[' << subname << "]\n";
134 }
135 out << to_config(subcom, default_also, write_description, "");
136 } else {
137 out << to_config(subcom, default_also, write_description, prefix + subcom->get_name() + parentSeparatorChar);
138 }
139 }
140 }
141
142 std::string outString;
143 if (write_description && !out.str().empty()) {
144 outString =
145 commentChar + std::string("#") + commentLead + detail::fix_newlines(commentChar + commentLead, app->get_description());
146 }
147
148 return outString + out.str();
149 }

166 CLI11_INLINE std::string HelpFormatter::make_description(const App* app) const {
167 std::string desc = app->get_description();
168 auto min_options = app->get_require_option_min();
169 auto max_options = app->get_require_option_max();
170 // ############## Next line changed (if statement removed)
171 if ((max_options == min_options) && (min_options > 0)) {
172 if (min_options == 1) {
173 desc += " \n[Exactly 1 of the following options is required]";
174 } else {
175 desc += " \n[Exactly " + std::to_string(min_options) + " options from the following list are required]";
176 }
177 } else if (max_options > 0) {
178 if (min_options > 0) {
179 desc += " \n[Between " + std::to_string(min_options) + " and " + std::to_string(max_options) +
180 " of the follow options are required]";
181 } else {
182 desc += " \n[At most " + std::to_string(max_options) + " of the following options are allowed]";
183 }
184 } else if (min_options > 0) {
185 desc += " \n[At least " + std::to_string(min_options) + " of the following options are required]";
186 }
187 return (!desc.empty()) ? desc + "\n" : std::string{};
188 }

190 CLI11_INLINE std::string HelpFormatter::make_usage(const App* app, std::string name) const {
191 const std::string usage = app->get_usage();
192 if (!usage.empty()) {
193 return usage + "\n";
194 }
195
196 std::stringstream out;
197
198 out << get_label("Usage") << ":" << (name.empty() ? "" : " ") << name;
199
200 const std::vector<std::string> groups = app->get_groups();
201
202 // Print an Options badge if any options exist
203 const std::vector<const Option*> non_pos_options = app->get_options([](const Option* opt) {
204 return opt->nonpositional();
205 });
206 if (!non_pos_options.empty()) {
207 out << " [" << get_label("OPTIONS") << "]";
208 }
209
210 // Positionals need to be listed here
211 std::vector<const Option*> positionals = app->get_options([](const Option* opt) {
212 return opt->get_positional();
213 });
214
215 // Print out positionals if any are left
216 if (!positionals.empty()) {
217 // Convert to help names
218 std::vector<std::string> positional_names(positionals.size());
219 std::transform(positionals.begin(), positionals.end(), positional_names.begin(), [this](const Option* opt) {
220 return make_option_usage(opt);
221 });
222
223 out << " " << detail::join(positional_names, " ");
224 }
225
226 // Add a marker if subcommands are expected or optional
227 if (!app->get_subcommands([](const App* subc) {
228 return !subc->get_disabled() && !subc->get_name().empty();
229 })
230 .empty()) {
231 // ############## Next line changed
232 out << " ["
233 << get_label(app->get_subcommands([](const CLI::App* subc) {
234 return ((!subc->get_disabled()) && (!subc->get_name().empty()) /*&& subc->get_required()*/);
235 }).size() <= 1
236 ? "SUBCOMMAND"
237 : "SUBCOMMANDS")
238 << " [--help]"
239 << "]";
240 }
241
242 const Option* disabledOpt = app->get_option_no_throw("--disabled");
243 if (disabledOpt != nullptr ? disabledOpt->as<bool>() : false) {
244 out << " " << get_label("DISABLED");
245 } else if (app->get_required()) {
246 out << " " << get_label("REQUIRED");
247 }
248
249 out << std::endl << std::endl;
250
251 return out.str();
252 }

254 CLI11_INLINE std::string HelpFormatter::make_subcommands(const App* app, AppFormatMode mode) const {
255 std::stringstream out;
256
257 const std::vector<const App*> subcommands = app->get_subcommands([](const App* subc) {
258 return !subc->get_disabled() && !subc->get_name().empty();
259 });
260
261 // Make a list in alphabetic order of the groups seen
262 std::set<std::string> subcmd_groups_seen;
263 for (const App* com : subcommands) {
264 if (com->get_name().empty()) {
265 if (!com->get_group().empty()) {
266 out << make_expanded(com);
267 }
268 continue;
269 }
270 std::string group_key = com->get_group();
271 if (!group_key.empty() &&
272 std::find_if(subcmd_groups_seen.begin(), subcmd_groups_seen.end(), [&group_key](const std::string& a) {
273 return detail::to_lower(a) == detail::to_lower(group_key);
274 }) == subcmd_groups_seen.end()) {
275 subcmd_groups_seen.insert(group_key);
276 }
277 }
278
279 // For each group, filter out and print subcommands
280 for (const std::string& group : subcmd_groups_seen) {
281 out << "\n" << group << ":\n";
282 const std::vector<const App*> subcommands_group = app->get_subcommands([&group](const App* sub_app) {
283 return detail::to_lower(sub_app->get_group()) == detail::to_lower(group) && !sub_app->get_disabled() &&
284 !sub_app->get_name().empty();
285 });
286 for (const App* new_com : subcommands_group) {
287 if (new_com->get_name().empty()) {
288 continue;
289 }
290 if (mode != AppFormatMode::All) {
291 out << make_subcommand(new_com);
292 } else {
293 out << new_com->help(new_com->get_name(), AppFormatMode::Sub);
294 out << "\n";
295 }
296 }
297 }
298
299 // ########## Next line(s) changed
300
301 std::string tmp = out.str();
302 if (mode == AppFormatMode::All && !tmp.empty()) {
303 tmp.pop_back();
304 }
305
306 return tmp;
307 }

322 CLI11_INLINE std::string HelpFormatter::make_expanded(const App* sub) const {
323 std::stringstream out;
324 // ########## Next lines changed
325 const Option* disabledOpt = sub->get_option_no_throw("--disabled");
326 out << sub->get_display_name(true) + " [OPTIONS]" + (!sub->get_subcommands({}).empty() ? " [SECTIONS]" : "") +
327 ((disabledOpt != nullptr ? disabledOpt->as<bool>() : false) ? ""
328 : (sub->get_required() ? " " + get_label("REQUIRED") : ""))
329 << "\n";
330
331 out << make_description(sub);
332 if (sub->get_name().empty() && !sub->get_aliases().empty()) {
333 detail::format_aliases(out, sub->get_aliases(), column_width_ + 2);
334 }
335 out << make_positionals(sub);
336 out << make_groups(sub, AppFormatMode::Sub);
337 out << make_subcommands(sub, AppFormatMode::Sub);
338
339 // Drop blank spaces
340 std::string tmp = out.str();
341 tmp.pop_back();
342
343 // Indent all but the first line (the name)
344 return detail::find_and_replace(tmp, "\n", "\n ") + "\n";
345 }

347 CLI11_INLINE std::string HelpFormatter::make_option_opts(const Option* opt) const {
348 std::stringstream out;
349
350 if (!opt->get_option_text().empty()) {
351 out << " " << opt->get_option_text();
352 } else {
353 if (opt->get_type_size() != 0) {
354 if (!opt->get_type_name().empty()) {
355 // ########## Next line changed
356 out << ((opt->get_items_expected_max() == 0) ? "=" : " ") << get_label(opt->get_type_name());
357 }
358 if (!opt->get_default_str().empty()) {
359 out << " [" << opt->get_default_str() << "]";
360 }
361 if (opt->get_expected_max() == detail::expected_max_vector_size) {
362 out << " ... ";
363 } else if (opt->get_expected_min() > 1) {
364 out << " x " << opt->get_expected();
365 }
366 if (opt->get_required()) {
367 out << " " << get_label("REQUIRED");
368 }
369 if (opt->get_configurable()) {
370 out << " " << get_label("PERSISTENT");
371 }
372 }
373 if (!opt->get_envname().empty()) {
374 out << " (" << get_label("Env") << ":" << opt->get_envname() << ") ";
375 }
376 if (!opt->get_needs().empty()) {
377 out << " " << get_label("Needs") << ":";
378 for (const Option* op : opt->get_needs()) {
379 out << " " << op->get_name();
380 }
381 out << " ";
382 }
383 if (!opt->get_excludes().empty()) {
384 out << " " << get_label("Excludes") << ":";
385 for (const Option* op : opt->get_excludes()) {
386 out << " " << op->get_name();
387 }
388 out << " ";
389 }
390 }
391 return out.str();
392 }
