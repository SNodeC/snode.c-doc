40 MariaDBConnection::MariaDBConnection(MariaDBClient* mariaDBClient, const MariaDBConnectionDetails& connectionDetails)
41 : ReadEventReceiver("MariaDBConnectionRead", core::DescriptorEventReceiver::TIMEOUT::DISABLE)
42 , WriteEventReceiver("MariaDBConnectionWrite", core::DescriptorEventReceiver::TIMEOUT::DISABLE)
43 , ExceptionalConditionEventReceiver("MariaDBConnectionExceptional", core::DescriptorEventReceiver::TIMEOUT::DISABLE)
44 , mariaDBClient(mariaDBClient)
45 , mysql(mysql_init(nullptr))
46 , commandStartEvent("MariaDBCommandStartEvent", this) {
47 mysql_options(mysql, MYSQL_OPT_NONBLOCK, nullptr);
48
49 execute_async(std::move(MariaDBCommandSequence().execute_async(new database::mariadb::commands::async::MariaDBConnectCommand(
50 connectionDetails,
51 [this]() -> void {
52 if (mysql_errno(mysql) == 0) {
53 int fd = mysql_get_socket(mysql);
54
55 LOG(DEBUG) << "Got valid descriptor: " << fd;
56
57 ReadEventReceiver::enable(fd);
58 WriteEventReceiver::enable(fd);
59 ExceptionalConditionEventReceiver::enable(fd);
60
61 ReadEventReceiver::suspend();
62 WriteEventReceiver::suspend();
63 ExceptionalConditionEventReceiver::suspend();
64
65 connected = true;
66 } else {
67 LOG(WARNING) << "Got no valid descriptor: " << mysql_error(mysql) << ", " << mysql_errno(mysql);
68 }
69 },
70 []() -> void {
71 LOG(DEBUG) << "Connect success";
72 },
73 [](const std::string& errorString, unsigned int errorNumber) -> void {
74 LOG(WARNING) << "Connect error: " << errorString << " : " << errorNumber;
75 }))));
76 }

78 MariaDBConnection::~MariaDBConnection() {
79 for (MariaDBCommandSequence& mariaDBCommandSequence : commandSequenceQueue) {
80 for (MariaDBCommand* mariaDBCommand : mariaDBCommandSequence.sequence()) {
81 if (core::SNodeC::state() == core::State::RUNNING && connected) {
82 mariaDBCommand->commandError(mysql_error(mysql), mysql_errno(mysql));
83 }
84
85 delete mariaDBCommand;
86 }
87 }
88
89 if (mariaDBClient != nullptr) {
90 mariaDBClient->connectionVanished();
91 }
92
93 mysql_close(mysql);
94 mysql_library_end();
95 }

120 void MariaDBConnection::commandStart(const utils::Timeval& currentTime) {
121 if (!commandSequenceQueue.empty()) {
122 currentCommand = commandSequenceQueue.front().nextCommand();
123
124 currentCommand->setMariaDBConnection(this);
125 checkStatus(currentCommand->commandStart(mysql, currentTime));
126 } else if (mariaDBClient != nullptr) {
127 if (ReadEventReceiver::isSuspended() && ReadEventReceiver::isEnabled()) {
128 ReadEventReceiver::resume();
129 }
130 } else {
131 ReadEventReceiver::disable();
132 WriteEventReceiver::disable();
133 ExceptionalConditionEventReceiver::disable();
134 }

163 void MariaDBConnection::checkStatus(int status) {
164 if (connected) {
165 if ((status & MYSQL_WAIT_READ) != 0) {
166 if (ReadEventReceiver::isSuspended() && ReadEventReceiver::isEnabled()) {
167 ReadEventReceiver::resume();
168 }
169 } else if (!ReadEventReceiver::isSuspended()) {
170 ReadEventReceiver::suspend();
171 }
172
173 if ((status & MYSQL_WAIT_WRITE) != 0) {
174 if (WriteEventReceiver::isSuspended() && WriteEventReceiver::isEnabled()) {
175 WriteEventReceiver::resume();
176 }
177 } else if (!WriteEventReceiver::isSuspended()) {
178 WriteEventReceiver::suspend();
179 }
180
181 if ((status & MYSQL_WAIT_EXCEPT) != 0) {
182 if (ExceptionalConditionEventReceiver::isSuspended() && ExceptionalConditionEventReceiver::isEnabled()) {
183 ExceptionalConditionEventReceiver::resume();
184 }
185 } else if (!ExceptionalConditionEventReceiver::isSuspended()) {
186 ExceptionalConditionEventReceiver::suspend();
187 }
188
189 if ((status & MYSQL_WAIT_TIMEOUT) != 0) {
190 ReadEventReceiver::setTimeout(mysql_get_timeout_value(mysql));
191 // WriteEventReceiver::setTimeout(mysql_get_timeout_value(mysql));
192 // ExceptionalConditionEventReceiver::setTimeout(mysql_get_timeout_value(mysql));
193 } else {
194 ReadEventReceiver::setTimeout(core::DescriptorEventReceiver::TIMEOUT::DEFAULT);
195 // WriteEventReceiver::setTimeout(core::DescriptorEventReceiver::TIMEOUT::DEFAULT);
196 // ExceptionalConditionEventReceiver::setTimeout(core::DescriptorEventReceiver::TIMEOUT::DEFAULT);
197 }
198
199 if (status == 0) {
200 if (mysql_errno(mysql) == 0) {
201 if (currentCommand->commandCompleted()) {
202 commandCompleted();
203 }
204 } else {
205 currentCommand->commandError(mysql_error(mysql), mysql_errno(mysql));
206 commandCompleted();
207 }
208 commandStartEvent.span();
209 }
210 } else {
211 currentCommand->commandError(mysql_error(mysql), mysql_errno(mysql));
212 commandCompleted();
213 delete this;
214 }
215 }
