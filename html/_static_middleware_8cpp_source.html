32 StaticMiddleware::StaticMiddleware(const std::string& root)
33 : root(root) {
34 use(
35 [&stdHeaders = this->stdHeaders, &stdCookies = this->stdCookies, &forceClose = this->forceClose] MIDDLEWARE(req, res, next) {
36 if (req.method == "GET") {
37 if (forceClose) {
38 res.set("Connection", "Close");
39 } else {
40 res.set("Connection", "Keep-Alive");
41 }
42 res.set(stdHeaders);
43 for (auto& [value, options] : stdCookies) {
44 res.cookie(value, options.getValue(), options.getOptions());
45 }
46 next();
47 } else {
48 LOG(DEBUG) << "Wrong method " << req.method;
49 res.set("Connection", "Close");
50 res.sendStatus(400);
51 }
52 },
53 [] MIDDLEWARE(req, res, next) {
54 if (req.url == "/") {
55 LOG(INFO) << "REDIRECT " + req.url + " -> " + "/index.html";
56 res.redirect(308, "/index.html");
57 } else {
58 next();
59 }
60 },
61 [&root = this->root] APPLICATION(req, res) {
62 LOG(INFO) << "GET " + req.url + " -> " + root + req.url;
63 res.sendFile(root + req.url, [&req, &res](int ret) -> void {
64 if (ret != 0) {
65 PLOG(ERROR) << req.url;
66 res.status(404).end();
67 }
68 });
69 });
70 }

110 class StaticMiddleware& StaticMiddleware::instance(const std::string& root) {
111 // Keep all created static middlewares alive
112 static std::map<const std::string, std::shared_ptr<class StaticMiddleware>> staticMiddlewares;
113
114 if (!staticMiddlewares.contains(root)) {
115 staticMiddlewares[root] = std::shared_ptr<StaticMiddleware>(new StaticMiddleware(root));
116 }
117
118 return *staticMiddlewares[root];
