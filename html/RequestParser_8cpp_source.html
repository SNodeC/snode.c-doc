69 enum Parser::ParserState RequestParser::parseStartLine(const std::string& line) {
70 enum Parser::ParserState parserState = Parser::ParserState::HEADER;
71
72 if (!line.empty()) {
73 std::string remaining;
74
75 std::tie(method, remaining) = httputils::str_split(line, ' ');
76 std::tie(url, httpVersion) = httputils::str_split(remaining, ' ');
77
78 std::string queriesLine;
79 std::tie(std::ignore, queriesLine) = httputils::str_split(url, '?');
80
81 if (!methodSupported(method)) {
82 parserState = parsingError(400, "Bad request method");
83 } else if (url.empty() || url.front() != '/') {
84 parserState = parsingError(400, "Malformed request");
85 } else {
86 std::smatch httpVersionMatch;
87 if (!std::regex_match(httpVersion, httpVersionMatch, httpVersionRegex)) {
88 parserState = parsingError(400, "Wrong protocol-version");
89 } else {
90 httpMajor = std::stoi(httpVersionMatch.str(1));
91 httpMinor = std::stoi(httpVersionMatch.str(2));
92
93 while (!queriesLine.empty()) {
94 std::string query;
95
96 std::tie(query, queriesLine) = httputils::str_split(queriesLine, '&');
97 queries.insert(httputils::str_split(query, '='));
98 }
99
100 onRequest(method, url, httpVersion, httpMajor, httpMinor, queries);
101 }
102 }
103 } else {
104 parserState = parsingError(400, "Request-line empty");
105 }
106
107 return parserState;
108 }

110 enum Parser::ParserState RequestParser::parseHeader() {
111 for (auto& [headerFieldName, headerFieldValue] : Parser::headers) {
112 if (headerFieldName != "cookie") {
113 if (headerFieldName == "content-length") {
114 Parser::contentLength = std::stoul(headerFieldValue);
115 }
116 } else {
117 std::string cookiesLine = headerFieldValue;
118
119 while (!cookiesLine.empty()) {
120 std::string cookieLine;
121 std::tie(cookieLine, cookiesLine) = httputils::str_split(cookiesLine, ',');
122
123 while (!cookieLine.empty()) {
124 std::string cookie;
125 std::tie(cookie, cookieLine) = httputils::str_split(cookieLine, ';');
126
127 std::string cookieName;
128 std::string cookieValue;
129 std::tie(cookieName, cookieValue) = httputils::str_split(cookie, '=');
130
131 httputils::str_trimm(cookieName);
132 httputils::str_trimm(cookieValue);
133
134 cookies.insert({cookieName, cookieValue});
135 }
136 }
137 }
138 }
139
140 Parser::headers.erase("cookie");
141
142 onHeader(Parser::headers, cookies);
143
144 enum Parser::ParserState parserState = Parser::ParserState::BODY;
145 if (contentLength == 0 && httpMinor == 1) {
146 parsingFinished();
147 parserState = ParserState::BEGIN;
148 }
149
150 return parserState;
151 }
