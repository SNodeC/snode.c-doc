32 template <unsigned N>
33 struct fixed_string {
34 char buf[N + 1]{};
35 constexpr fixed_string(char const* s) { // cppcheck-suppress noExplicitConstructor
36 for (unsigned i = 0; i != N; ++i)
37 buf[i] = s[i];
38 }
39
40 constexpr operator char const*() const {
41 return buf;
42 }
43
44 auto operator<=>(const fixed_string&) const = default;
45 };
46 template <unsigned N>
47 fixed_string(char const (&)[N]) -> fixed_string<N - 1>;
