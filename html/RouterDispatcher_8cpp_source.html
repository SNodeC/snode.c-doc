39 bool
40 RouterDispatcher::dispatch(express::Controller& controller, const std::string& parentMountPath, const express::MountPoint& mountPoint) {
41 bool dispatched = false;
42
43 const std::string absoluteMountPath = path_concat(parentMountPath, mountPoint.relativeMountPath);
44
45 if ((controller.getRequest()->path.rfind(absoluteMountPath, 0) == 0 &&
46 (mountPoint.method == "use" || controller.getRequest()->method == mountPoint.method || mountPoint.method == "all"))) {
47 for (Route& route : routes) {
48 dispatched = route.dispatch(controller, absoluteMountPath);
49
50 if (dispatched || controller.nextRouter()) {
51 break;
52 }
53 }
54 }
55
56 return dispatched;
57 }
