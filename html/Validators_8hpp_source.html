44 class IsValue : public Validator {
45 public:
46 template <typename T>
47 explicit IsValue(T val) {
48 std::stringstream out;
49 out << detail::type_name<T>() << " is '" << val << "'";
50 description(out.str());
51
52 func_ = [val](std::string& input) {
53 using CLI::detail::lexical_cast;
54 T inputVal;
55 bool converted = lexical_cast(input, inputVal);
56 if (!converted) {
57 return std::string("Value ") + input + " could not be convertedto " + detail::type_name<T>();
58 }
59 if (val != inputVal) {
60 return input + " is not \"" + val + "\"";
61 }
62 return std::string{};
63 };
64 }
65 };
