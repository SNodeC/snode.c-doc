55 int DescriptorEventPublisher::spanActiveEvents() {
56 int count = 0;
57
58 pollfd* pollfds = pollFds.getEvents();
59
60 const std::unordered_map<int, PollFdsManager::PollFdIndex>& pollFdsIndices = pollFds.getPollFdIndices();
61
62 for (auto& [fd, eventReceivers] : observedEventReceivers) {
63 pollfd& pollFd = pollfds[pollFdsIndices.find(fd)->second.index];
64
65 if ((pollFd.events & events) != 0 && (pollFd.revents & revents) != 0) {
66 core::DescriptorEventReceiver* eventReceiver = eventReceivers.front();
67 eventCounter++;
68 eventReceiver->span();
69 count++;
70 }
71 }
72
73 return count;
74 }
