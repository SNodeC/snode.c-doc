113 uint32_t Binary::getIntV() {
114 uint32_t value = 0;
115 uint32_t multiplier = 1;
116
117 do {
118 if (!error && pointer + sizeof(uint8_t) <= length) {
119 uint8_t byte = *reinterpret_cast<uint8_t*>(binary.data() + pointer);
120 value += (byte & 0x7F) * multiplier;
121
122 if (multiplier > 0x80 * 0x80 * 0x80) {
123 error = true;
124 } else {
125 multiplier *= 0x80;
126 completed = (byte & 0x80) == 0;
127 }
128
129 pointer += sizeof(uint8_t);
130 } else {
131 error = true;
132 }
133 } while (!completed && !error);
134
135 return value;
136 }

220 void Binary::putIntV(uint32_t value) {
221 do {
222 uint8_t encodedByte = static_cast<uint8_t>(value % 0x80);
223
224 value /= 0x80;
225 if (value > 0) {
226 encodedByte |= 0x80;
227 }
228
229 putInt8(encodedByte);
230 } while (value > 0);
231 }
