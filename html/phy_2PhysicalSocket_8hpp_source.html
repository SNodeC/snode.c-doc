62 template <typename SocketAddress>
63 int PhysicalSocket<SocketAddress>::open(const std::map<int, const net::phy::PhysicalSocketOption>& socketOptions, Flags flags) {
64 int ret = Super::operator=(core::system::socket(domain, type | flags, protocol)).getFd();
65
66 if (ret >= 0) {
67 for (const auto& [optName, socketOption] : socketOptions) {
68 int setSockoptRet =
69 setSockopt(socketOption.getOptLevel(), socketOption.getOptName(), socketOption.getOptValue(), socketOption.getOptLen());
70
71 ret = (ret >= 0 && setSockoptRet < 0) ? setSockoptRet : ret;
72
73 if (ret < 0) {
74 break;
75 }
76 }
77 }
78
79 return ret;
80 }
