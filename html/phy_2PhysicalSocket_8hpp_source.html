79 template <typename SocketAddress>
80 int PhysicalSocket<SocketAddress>::open(const std::map<int, const net::phy::PhysicalSocketOption>& socketOptions, Flags flags) {
81 int ret = Super::operator=(core::system::socket(domain, type | flags, protocol)).getFd();
82
83 if (ret >= 0) {
84 for (const auto& [optName, socketOption] : socketOptions) {
85 int setSockoptRet =
86 setSockopt(socketOption.getOptLevel(), socketOption.getOptName(), socketOption.getOptValue(), socketOption.getOptLen());
87
88 ret = (ret >= 0 && setSockoptRet < 0) ? setSockoptRet : ret;
89
90 if (ret < 0) {
91 break;
92 }
93 }
94 }
95
96 return ret;
97 }

99 template <typename SocketAddress>
100 int PhysicalSocket<SocketAddress>::bind(SocketAddress& bindAddress) {
101 int ret = core::system::bind(core::Descriptor::getFd(), &bindAddress.getSockAddr(), bindAddress.getSockAddrLen());
102
103 if (ret == 0) {
104 this->bindAddress = bindAddress;
105 }
106
107 return ret;
108 }
