35 template <typename PhysicalServerSocket, typename Config>
36 core::socket::stream::tls::SocketAcceptor<PhysicalServerSocket, Config>::SocketAcceptor(
37 const std::shared_ptr<SocketContextFactory>& socketContextFactory,
38 const std::function<void(SocketConnection*)>& onConnect,
39 const std::function<void(SocketConnection*)>& onConnected,
40 const std::function<void(SocketConnection*)>& onDisconnect,
41 const std::function<void(const core::ProgressLog&)>& onError,
42 const std::shared_ptr<Config>& config)
43 : Super(
44 socketContextFactory,
45 [onConnect, this](SocketConnection* socketConnection) -> void { // onConnect
46 socketConnection->startSSL(this->masterSslCtx, this->config->getInitTimeout(), this->config->getShutdownTimeout());
47
48 onConnect(socketConnection);
49 },
50 [onConnected](SocketConnection* socketConnection) -> void {
51 SSL* ssl = socketConnection->getSSL();
52
53 if (ssl != nullptr) {
54 SSL_set_accept_state(ssl);
55
56 socketConnection->doSSLHandshake(
57 [onConnected, socketConnection]() -> void { // onSuccess
58 LOG(INFO) << "SSL/TLS initial handshake success";
59 onConnected(socketConnection);
60 socketConnection->onConnected();
61 },
62 []() -> void { // onTimeout
63 LOG(WARNING) << "SSL/TLS initial handshake timed out";
64 },
65 [](int sslErr) -> void { // onError
66 ssl_log("SSL/TLS initial handshake failed", sslErr);
67 });
68 } else {
69 socketConnection->close();
70 ssl_log_error("SSL/TLS initialization failed");
71 }
72 },
73 [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
74 socketConnection->stopSSL();
75 onDisconnect(socketConnection);
76 },
77 onError,
78 config) {
79 }

100 template <typename PhysicalServerSocket, typename Config>
101 void SocketAcceptor<PhysicalServerSocket, Config>::initAcceptEvent() {
102 if (!config->getDisabled()) {
103 masterSslCtx = ssl_ctx_new(config);
104
105 if (masterSslCtx != nullptr) {
106 masterSslCtxSans = ssl_get_sans(masterSslCtx);
107
108 for (const std::string& san : masterSslCtxSans) {
109 LOG(INFO) << "SSL_CTX for (san)'" << san << "' as master installed";
110 }
111
112 for (const auto& [domain, sniCertConf] : config->getSniCerts()) {
113 if (!sniCertConf.empty()) {
114 SSL_CTX* sniSslCtx = ssl_ctx_new(sniCertConf);
115
116 if (sniSslCtx != nullptr) {
117 if (!domain.empty()) {
118 if (sniSslCtxs.contains(domain)) {
119 ssl_ctx_free(sniSslCtxs[domain]);
120 }
121 sniSslCtxs.insert_or_assign(domain, sniSslCtx);
122
123 LOG(INFO) << "SSL_CTX for (dom)'" << domain << "' as server name indication (sni) installed";
124 }
125
126 for (const std::string& san : ssl_get_sans(sniSslCtx)) {
127 sniSslCtxs.insert({san, sniSslCtx});
128
129 LOG(INFO) << "SSL_CTX for (san)'" << san << "' as server name indication (sni) installed";
130 }
131 } else {
132 LOG(INFO) << "Can not create SNI_SSL_CTX for domain '" << domain << "'";
133 }
134 }
135 }
136 forceSni = config->getForceSni();
137
138 SSL_CTX_set_client_hello_cb(masterSslCtx, clientHelloCallback, this);
139
140 Super::initAcceptEvent();
141 } else {
142 LOG(ERROR) << "Master SSL/TLS certificate activation failed";
143
144 Super::onError(*Super::progressLog.get());
145 Super::destruct();
146 }
147 } else {
148 Super::initAcceptEvent();
149 }
150 }

152 template <typename PhysicalServerSocket, typename Config>
153 SSL_CTX* SocketAcceptor<PhysicalServerSocket, Config>::getMasterSniCtx(const std::string& serverNameIndication) {
154 SSL_CTX* sniSslCtx = nullptr;
155
156 LOG(INFO) << "Search for sni = '" << serverNameIndication << "' in master certificate";
157
158 std::set<std::string>::iterator masterSniIt =
159 std::find_if(masterSslCtxSans.begin(), masterSslCtxSans.end(), [&serverNameIndication](const std::string& sni) -> bool {
160 LOG(TRACE) << " .. " << sni.c_str();
161 return match(sni.c_str(), serverNameIndication.c_str());
162 });
163 if (masterSniIt != masterSslCtxSans.end()) {
164 LOG(INFO) << "found: " << *masterSniIt;
165 sniSslCtx = masterSslCtx;
166 } else {
167 LOG(INFO) << "not found";
168 }
169
170 return sniSslCtx;
171 }

173 template <typename PhysicalServerSocket, typename Config>
174 SSL_CTX* SocketAcceptor<PhysicalServerSocket, Config>::getPoolSniCtx(const std::string& serverNameIndication) {
175 SSL_CTX* sniCtx = nullptr;
176
177 LOG(INFO) << "Search for sni = '" << serverNameIndication << "' in sni certificates";
178
179 std::map<std::string, SSL_CTX*>::iterator sniPairIt = std::find_if(
180 sniSslCtxs.begin(), sniSslCtxs.end(), [&serverNameIndication](const std::pair<std::string, SSL_CTX*>& sniPair) -> bool {
181 LOG(TRACE) << " .. " << sniPair.first.c_str();
182 return match(sniPair.first.c_str(), serverNameIndication.c_str());
183 });
184
185 if (sniPairIt != sniSslCtxs.end()) {
186 LOG(INFO) << "found: " << sniPairIt->first;
187 sniCtx = sniPairIt->second;
188 } else {
189 LOG(INFO) << "not found";
190 }
191
192 return sniCtx;
193 }

206 template <typename PhysicalServerSocket, typename Config>
207 int SocketAcceptor<PhysicalServerSocket, Config>::clientHelloCallback(SSL* ssl, int* al, void* arg) {
208 int ret = SSL_CLIENT_HELLO_SUCCESS;
209
210 SocketAcceptor* socketAcceptor = static_cast<SocketAcceptor*>(arg);
211
212 std::string serverNameIndication = ssl_get_servername_from_client_hello(ssl);
213
214 if (!serverNameIndication.empty()) {
215 SSL_CTX* sniSslCtx = socketAcceptor->getSniCtx(serverNameIndication);
216 if (sniSslCtx != nullptr) {
217 LOG(INFO) << "Setting sni certificate for " << serverNameIndication;
218 ssl_set_ssl_ctx(ssl, sniSslCtx);
219 } else if (socketAcceptor->forceSni) {
220 LOG(WARNING) << "No sni certificate found but forceSni set - terminating";
221 ret = SSL_CLIENT_HELLO_ERROR;
222 *al = SSL_AD_UNRECOGNIZED_NAME;
223 } else {
224 LOG(INFO) << "No sni certificate found - still using master certificate";
225 }
226 } else {
227 LOG(INFO) << "No sni certificate set - the client did not request one";
228 }
229
230 return ret;
231 }

233 template <typename PhysicalServerSocket, typename Config>
234 bool SocketAcceptor<PhysicalServerSocket, Config>::match(const char* first, const char* second) {
235 // If we reach at the end of both strings, we are done
236 if (*first == '\0' && *second == '\0')
237 return true;
238
239 // Make sure to eliminate consecutive '*'
240 if (*first == '*') {
241 while (*(first + 1) == '*')
242 first++;
243 }
244
245 // Make sure that the characters after '*' are present
246 // in second string. This function assumes that the
247 // first string will not contain two consecutive '*'
248 if (*first == '*' && *(first + 1) != '\0' && *second == '\0')
249 return false;
250
251 // If the first string contains '?', or current
252 // characters of both strings match
253 if (*first == '?' || *first == *second)
254 return match(first + 1, second + 1);
255
256 // If there is *, then there are two possibilities
257 // a) We consider current character of second string
258 // b) We ignore current character of second string.
259 if (*first == '*')
260 return match(first + 1, second) || match(first, second + 1);
261 return false;
262 }
