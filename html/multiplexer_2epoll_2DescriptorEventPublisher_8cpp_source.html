41 void DescriptorEventPublisher::EPollEvents::muxAdd(core::DescriptorEventReceiver* eventReceiver) {
42 const utils::PreserveErrno preserveErrno;
43
44 epoll_event ePollEvent{};
45
46 ePollEvent.data.ptr = eventReceiver;
47 ePollEvent.events = events;
48
49 if (core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, eventReceiver->getRegisteredFd(), &ePollEvent) == 0) {
50 interestCount++;
51
52 if (interestCount >= ePollEvents.size()) {
53 ePollEvents.resize(ePollEvents.size() * 2);
54 }
55 } else if (errno == EEXIST) {
56 muxMod(eventReceiver->getRegisteredFd(), events, eventReceiver);
57 }
58 }

60 void DescriptorEventPublisher::EPollEvents::muxDel(int fd) {
61 const utils::PreserveErrno preserveErrno;
62
63 if (core::system::epoll_ctl(epfd, EPOLL_CTL_DEL, fd, nullptr) == 0 || errno == EBADF) {
64 interestCount--;
65
66 if (ePollEvents.size() > (interestCount * 2) + 1) {
67 ePollEvents.resize(ePollEvents.size() / 2);
68 ePollEvents.shrink_to_fit();
69 }
70 }
71 }

126 void DescriptorEventPublisher::spanActiveEvents() {
127 const int count = core::system::epoll_wait(ePollEvents.getEPFd(), ePollEvents.getEvents(), ePollEvents.getInterestCount(), 0);
128
129 for (int i = 0; i < count; i++) {
130 const epoll_event& ev = ePollEvents.getEvents()[i];
131 core::DescriptorEventReceiver* eventReceiver = static_cast<core::DescriptorEventReceiver*>(ev.data.ptr);
132 if (eventReceiver != nullptr && (ev.events & revents) != 0) {
133 eventCounter++;
134 eventReceiver->span();
135 }
136 }
137 }
