60 template <typename PhysicalServerSocket, typename Config, template <typename PhysicalServerSocketT> typename SocketConnection>
61 void SocketAcceptor<PhysicalServerSocket, Config, SocketConnection>::initAcceptEvent() {
62 if (!config->getDisabled()) {
63 LOG(INFO) << "Instance '" << this->config->getInstanceName() << "' enabled";
64
65 core::eventreceiver::AcceptEventReceiver::setTimeout(config->getAcceptTimeout());
66
67 try {
68 localAddress = config->Local::getSocketAddress();
69 physicalSocket = new PhysicalSocket();
70
71 if (physicalSocket->open(config->getSocketOptions(), PhysicalSocket::Flags::NONBLOCK) < 0) {
72 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketAcceptor::open '" << localAddress.toString()
73 << "'";
74 } else if (physicalSocket->bind(localAddress) < 0) {
75 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketAcceptor::bind '" << localAddress.toString()
76 << "'";
77 } else if (physicalSocket->listen(config->getBacklog()) < 0) {
78 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketAcceptor::listen '" << localAddress.toString()
79 << "'";
80 } else {
81 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketAcceptor::listen '" << localAddress.toString()
82 << "'";
83 enable(physicalSocket->getFd());
84 }
85
86 if (localAddress.useNext()) {
87 new SocketAcceptor(socketContextFactory, onConnect, onConnected, onDisconnect, onError, config, progressLog);
88 } else {
89 onError(*progressLog);
90 }
91
92 if (!isEnabled()) {
93 destruct();
94 }
95 } catch (const typename SocketAddress::BadSocketAddress& badSocketAddress) {
96 utils::PreserveErrno pe(0);
97 progressLog->addEntry(0) << this->config->getInstanceName() << ": SocketAcceptor::getLocalAddress '"
98 << localAddress.toString() << "': BadSocketAddress: " << badSocketAddress.what();
99 onError(*progressLog);
100
101 destruct();
102 }
103 } else {
104 LOG(INFO) << "Instance '" << this->config->getInstanceName() << "' disabled";
105
106 destruct();
107 }
108 }

110 template <typename PhysicalServerSocket, typename Config, template <typename PhysicalServerSocketT> typename SocketConnection>
111 void SocketAcceptor<PhysicalServerSocket, Config, SocketConnection>::acceptEvent() {
112 int acceptsPerTick = config->getAcceptsPerTick();
113
114 do {
115 PhysicalSocket physicalClientSocket(physicalSocket->accept4(PhysicalSocket::Flags::NONBLOCK));
116
117 if (physicalClientSocket.isValid()) {
118 LOG(TRACE) << this->config->getInstanceName() << ": SocketAcceptor::accept4 success '" << localAddress.toString() << "'";
119
120 SocketConnectionFactory(socketContextFactory, onConnect, onConnected, onDisconnect).create(physicalClientSocket, config);
121 } else if (errno != EINTR && errno != EAGAIN && errno != EWOULDBLOCK) {
122 PLOG(ERROR) << this->config->getInstanceName() << ": SocketAcceptor::accept4 failed '" << localAddress.toString() << "'";
123 }
124 } while (--acceptsPerTick > 0);
125 }
