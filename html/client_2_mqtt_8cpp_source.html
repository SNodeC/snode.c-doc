53 Mqtt::Mqtt()
54 : sessionStoreFileName((getenv("MQTT_SESSION_STORE") != nullptr) ? getenv("MQTT_SESSION_STORE") : "") { // NOLINT
55 if (!sessionStoreFileName.empty()) {
56 LOG(INFO) << "Restoring safed session ...";
57
58 std::ifstream sessionStoreFile(sessionStoreFileName);
59
60 if (sessionStoreFile.is_open()) {
61 try {
62 nlohmann::json sessionStoreJson;
63
64 sessionStoreFile >> sessionStoreJson;
65
66 session.fromJson(sessionStoreJson);
67
68 LOG(TRACE) << " ... Persistent session data loaded successfull";
69 } catch (const nlohmann::json::exception&) {
70 LOG(TRACE) << " ... Starting with empty session: Session store '" << sessionStoreFileName << "' empty or corrupted";
71
72 session.clear();
73 }
74 sessionStoreFile.close();
75 std::remove(sessionStoreFileName.data());
76 } else {
77 PLOG(TRACE) << " ... Could not read session store '" << sessionStoreFileName << "'";
78 }
79 LOG(INFO) << "Restoring safed session done";
80 } else {
81 LOG(INFO) << "Session not reloaded: Session store filename empty";
82 }
83 }

85 Mqtt::~Mqtt() {
86 if (!sessionStoreFileName.empty()) {
87 nlohmann::json sessionJson = session.toJson();
88
89 std::ofstream sessionStoreFile(sessionStoreFileName);
90
91 if (sessionStoreFile.is_open()) {
92 if (!sessionJson.empty()) {
93 sessionStoreFile << sessionJson;
94 }
95
96 sessionStoreFile.close();
97 } else {
98 PLOG(TRACE) << "Could not write session store '" << sessionStoreFileName << "'";
99 }
100 } else {
101 LOG(INFO) << "Session not saved: Session store filename empty";
102 }
103
104 pingTimer.cancel();
105 }

107 iot::mqtt::ControlPacketDeserializer* Mqtt::createControlPacketDeserializer(iot::mqtt::FixedHeader& fixedHeader) {
108 iot::mqtt::ControlPacketDeserializer* currentPacket = nullptr;
109
110 switch (fixedHeader.getPacketType()) {
111 case MQTT_CONNACK:
112 currentPacket = new iot::mqtt::client::packets::Connack(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
113 break;
114 case MQTT_PUBLISH:
115 currentPacket = new iot::mqtt::client::packets::Publish(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
116 break;
117 case MQTT_PUBACK:
118 currentPacket = new iot::mqtt::client::packets::Puback(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
119 break;
120 case MQTT_PUBREC:
121 currentPacket = new iot::mqtt::client::packets::Pubrec(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
122 break;
123 case MQTT_PUBREL:
124 currentPacket = new iot::mqtt::client::packets::Pubrel(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
125 break;
126 case MQTT_PUBCOMP:
127 currentPacket = new iot::mqtt::client::packets::Pubcomp(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
128 break;
129 case MQTT_SUBACK:
130 currentPacket = new iot::mqtt::client::packets::Suback(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
131 break;
132 case MQTT_UNSUBACK:
133 currentPacket = new iot::mqtt::client::packets::Unsuback(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
134 break;
135 case MQTT_PINGRESP:
136 currentPacket = new iot::mqtt::client::packets::Pingresp(fixedHeader.getRemainingLength(), fixedHeader.getFlags());
137 break;
138 default:
139 currentPacket = nullptr;
140 break;
141 }
142
143 return currentPacket;
144 }

162 void Mqtt::_onConnack(const iot::mqtt::client::packets::Connack& connack) {
163 LOG(DEBUG) << "Received Connack:";
164 LOG(DEBUG) << "=================";
165 printStandardHeader(connack);
166 LOG(DEBUG) << "Acknowledge Flag: " << static_cast<int>(connack.getAcknowledgeFlags());
167 LOG(DEBUG) << "Return code: " << static_cast<int>(connack.getReturnCode());
168 LOG(DEBUG) << "Session present: " << connack.getSessionPresent();
169
170 if (connack.getReturnCode() != MQTT_CONNACK_ACCEPT) {
171 LOG(TRACE) << "Negative ack received";
172 mqttContext->end(true);
173 } else {
174 initSession(&session, keepAlive * 2);
175
176 pingTimer = core::timer::Timer::intervalTimer(
177 [this](void) -> void {
178 sendPingreq();
179 },
180 keepAlive);
181
182 onConnack(connack);
183 }
184 }

195 void Mqtt::_onSuback(const iot::mqtt::client::packets::Suback& suback) {
196 LOG(DEBUG) << "Received SUBACK:";
197 LOG(DEBUG) << "================";
198 printStandardHeader(suback);
199 LOG(DEBUG) << "PacketIdentifier: 0x" << std::hex << std::setfill('0') << std::setw(4) << suback.getPacketIdentifier();
200
201 std::stringstream ss;
202 std::list<uint8_t>::size_type i = 0;
203
204 for (uint8_t returnCode : suback.getReturnCodes()) {
205 if (i != 0 && i % 8 == 0 && i != suback.getReturnCodes().size()) {
206 ss << std::endl;
207 ss << " ";
208 }
209 ++i;
210 ss << "0x" << std::hex << std::setfill('0') << std::setw(2) << static_cast<uint16_t>(returnCode) << " "; // << " | ";
211 }
212
213 LOG(DEBUG) << "Return codes: " << ss.str();
214
215 if (suback.getPacketIdentifier() == 0) {
216 LOG(TRACE) << "PackageIdentifier missing";
217 mqttContext->end(true);
218 } else {
219 onSuback(suback);
220 }
221 }
