88 void DescriptorEventReceiver::suspend() {

104 void DescriptorEventReceiver::resume() {
105 if (enabled) {
106 if (suspended) {

131 void DescriptorEventReceiver::setTimeout(const utils::Timeval& timeout) {
132 if (timeout == TIMEOUT::DEFAULT) {
133 this->maxInactivity = initialTimeout;
134 } else {

138 triggered(utils::Timeval::currentTime());

141 utils::Timeval DescriptorEventReceiver::getTimeout(const utils::Timeval& currentTime) const {
142 return maxInactivity > 0 ? currentTime > lastTriggered ? maxInactivity - (currentTime - lastTriggered) : 0 : TIMEOUT::MAX;
143 }
