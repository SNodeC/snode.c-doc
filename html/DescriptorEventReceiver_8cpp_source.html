67 bool DescriptorEventReceiver::enable(int fd) {
68 if (core::eventLoopState() != core::State::STOPPING) {
69 if (!enabled) {
70 observedFd = fd;
71
72 enabled = true;
73 descriptorEventPublisher.enable(this);
74 } else {
75 LOG(TRACE) << getName() << " (" << observedFd << "): Double enable";
76 }
77 } else {
78 LOG(TRACE) << getName() << " (" << observedFd << "): Enable after signal";
79 }
80
81 return enabled;
82 }

139 void DescriptorEventReceiver::setTimeout(const utils::Timeval& timeout) {
140 if (timeout == TIMEOUT::DEFAULT) {
141 this->maxInactivity = initialTimeout;
142 } else {
143 this->maxInactivity = timeout;
144 }
145
146 triggered(utils::Timeval::currentTime());
147 }

149 utils::Timeval DescriptorEventReceiver::getTimeout(const utils::Timeval& currentTime) const {
150 return maxInactivity > 0 ? currentTime > lastTriggered ? maxInactivity - (currentTime - lastTriggered) : 0 : TIMEOUT::MAX;
151 }
