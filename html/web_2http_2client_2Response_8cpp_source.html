45 const std::string& Response::header(const std::string& key, int i) const {
46 std::string tmpKey = key;
47 httputils::to_lower(tmpKey);
48
49 if (headers.find(tmpKey) != headers.end()) {
50 std::pair<std::multimap<std::string, std::string>::const_iterator, std::multimap<std::string, std::string>::const_iterator>
51 range = headers.equal_range(tmpKey);
52
53 if (std::distance(range.first, range.second) >= i) {
54 std::advance(range.first, i);
55 return (*(range.first)).second;
56 }
57
58 return nullstr;
59 }
60
61 return nullstr;
62 }

74 void Response::upgrade(Request& request) {
75 if (httputils::ci_contains(this->header("connection"), "Upgrade")) {
76 web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
77 web::http::client::SocketContextUpgradeFactorySelector::instance()->select(request, *this);
78
79 if (socketContextUpgradeFactory != nullptr) {
80 core::socket::stream::SocketContext* newSocketContext =
81 socketContextUpgradeFactory->create(socketContext->getSocketConnection());
82
83 if (newSocketContext != nullptr) {
84 socketContext->switchSocketContext(newSocketContext);
85 } else {
86 LOG(DEBUG) << "HTTP: SocketContextUpgrade not created";
87 socketContext->close();
88 }
89 } else {
90 LOG(DEBUG) << "HTTP: SocketContextUpgradeFactory not existing";
91 socketContext->close();
92 }
93 } else {
94 LOG(DEBUG) << "HTTP: Response did not contain upgrade";
95 socketContext->close();
96 }
97 }
