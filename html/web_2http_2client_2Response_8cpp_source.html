40 const std::string& Response::header(const std::string& key, int i) const {
41 std::string tmpKey = key;
42 httputils::to_lower(tmpKey);
43
44 if (headers.find(tmpKey) != headers.end()) {
45 std::pair<std::multimap<std::string, std::string>::const_iterator, std::multimap<std::string, std::string>::const_iterator>
46 range = headers.equal_range(tmpKey);
47
48 if (std::distance(range.first, range.second) >= i) {
49 std::advance(range.first, i);
50 return (*(range.first)).second;
51 }
52
53 return nullstr;
54 }
55
56 return nullstr;
57 }

69 void Response::upgrade(Request& request) {
70 if (httputils::ci_contains(this->header("connection"), "Upgrade")) {
71 web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
72 web::http::client::SocketContextUpgradeFactorySelector::instance()->select(request, *this);
73
74 if (socketContextUpgradeFactory != nullptr) {
75 if (socketContext->switchSocketContext(socketContextUpgradeFactory) == nullptr) {
76 LOG(DEBUG) << "SocketContextUpgrade not created";
77 socketContext->close();
78 }
79 } else {
80 LOG(DEBUG) << "SocketContextUpgradeFactory not existing";
81 socketContext->close();
82 }
83 } else {
84 LOG(DEBUG) << "Response did not contain upgrade";
85 socketContext->close();
86 }
87 }
