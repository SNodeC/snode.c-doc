53 std::size_t EchoSocketContext::onReceivedFromPeer() {
54 char chunk[4096];
55
56 const std::size_t chunklen = readFromPeer(chunk, 4096);
57
58 if (chunklen > 0) {
59 VLOG(1) << "Data to reflect: " << std::string(chunk, chunklen);
60 sendToPeer(chunk, chunklen);
61 }
62
63 return chunklen;
64 }
