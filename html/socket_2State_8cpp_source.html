92 std::string State::what() const {
93 std::string stateString;
94
95 switch (state & ~NO_RETRY) {
96 case OK:
97 stateString = "OK";
98 break;
99 case DISABLED:
100 stateString = "DISABLED";
101 break;
102 case ERROR:
103 [[fallthrough]];
104 case FATAL:
105 stateString = errstr;
106 break;
107 }
108
109 return stateString.append(" [").append(std::to_string(errnum)).append("]");
110 }
