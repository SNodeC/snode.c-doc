65 std::size_t Receiver::receive() {
66 std::size_t ret = 0;
67 std::size_t consumed = 0;
68
69 // dumpFrame(chunk, chunkLen);
70
71 do {
72 switch (parserState) {
73 case ParserState::BEGIN:
74 parserState = ParserState::OPCODE;
75 [[fallthrough]];
76 case ParserState::OPCODE:
77 ret = readOpcode();
78 break;
79 case ParserState::LENGTH:
80 ret = readLength();
81 break;
82 case ParserState::ELENGTH:
83 ret = readELength();
84 break;
85 case ParserState::MASKINGKEY:
86 ret = readMaskingKey();
87 break;
88 case ParserState::PAYLOAD:
89 ret = readPayload();
90 break;
91 case ParserState::ERROR:
92 onMessageError(errorState);
93 reset();
94 break;
95 }
96 consumed += ret;
97 } while (ret > 0 && parserState != ParserState::BEGIN && parserState != ParserState::ERROR);
98
99 return consumed;
100 }

102 std::size_t Receiver::readOpcode() {
103 char byte = 0;
104 const std::size_t ret = readFrameData(&byte, 1);
105
106 if (ret > 0) {
107 const uint8_t opCodeByte = static_cast<uint8_t>(byte);
108
109 fin = (opCodeByte & 0b10000000) != 0;
110 opCode = opCodeByte & 0b00001111;
111
112 if (!continuation) {
113 onMessageStart(opCode);
114 parserState = ParserState::LENGTH;
115 } else if (opCode == 0) {
116 parserState = ParserState::LENGTH;
117 } else {
118 parserState = ParserState::ERROR;
119 errorState = 1002;
120 LOG(ERROR) << "WebSocket: Error opcode in continuation frame";
121 }
122 continuation = !fin;
123 }
124
125 return ret;
126 }

128 std::size_t Receiver::readLength() {
129 char byte = 0;
130 const std::size_t ret = readFrameData(&byte, 1);
131
132 if (ret > 0) {
133 const uint8_t lengthByte = static_cast<uint8_t>(byte);
134
135 masked = (lengthByte & 0b10000000) != 0;
136 payLoadNumBytes = payLoadNumBytesLeft = lengthByte & 0b01111111;
137
138 if (payLoadNumBytes > 125) {
139 switch (payLoadNumBytes) {
140 case 126:
141 elengthNumBytes = elengthNumBytesLeft = 2;
142 break;
143 case 127:
144 elengthNumBytes = elengthNumBytesLeft = 8;
145 break;
146 }
147 parserState = ParserState::ELENGTH;
148 payLoadNumBytes = payLoadNumBytesLeft = 0;
149 } else {
150 if (masked) {
151 parserState = ParserState::MASKINGKEY;
152 } else if (payLoadNumBytes > 0) {
153 parserState = ParserState::PAYLOAD;
154 } else {
155 if (fin) {
156 onMessageEnd();
157 }
158 reset();
159 }
160 }
161 }
162
163 return ret;
164 }

166 std::size_t Receiver::readELength() {
167 const std::size_t ret = readFrameData(elengthChunk, elengthNumBytesLeft);
168
169 for (std::size_t i = 0; i < ret; i++) {
170 payLoadNumBytes |= *reinterpret_cast<uint64_t*>(elengthChunk + i) << (elengthNumBytes - elengthNumBytesLeft) * 8;
171
172 elengthNumBytesLeft--;
173 }
174
175 if (elengthNumBytesLeft == 0) {
176 switch (elengthNumBytes) {
177 case 2:
178 payLoadNumBytes = payLoadNumBytesLeft = be16toh(static_cast<uint16_t>(payLoadNumBytes));
179 break;
180 case 8:
181 payLoadNumBytes = payLoadNumBytesLeft = be64toh(payLoadNumBytes);
182 break;
183 }
184
185 if ((payLoadNumBytes & static_cast<uint64_t>(0x01) << 63) != 0) {
186 parserState = ParserState::ERROR;
187 errorState = 1004;
188 } else if (masked) {
189 parserState = ParserState::MASKINGKEY;
190 } else {
191 parserState = ParserState::PAYLOAD;
192 }
193 }
194
195 return ret;
196 }

198 std::size_t Receiver::readMaskingKey() {
199 const std::size_t ret = readFrameData(maskingKeyChunk, maskingKeyNumBytesLeft);
200
201 for (std::size_t i = 0; i < ret; i++) {
202 maskingKey |= static_cast<uint32_t>(*reinterpret_cast<unsigned char*>(maskingKeyChunk + i))
203 << (maskingKeyNumBytes - maskingKeyNumBytesLeft) * 8;
204 maskingKeyNumBytesLeft--;
205 }
206
207 if (maskingKeyNumBytesLeft == 0) {
208 maskingKeyAsArray = {.key = maskingKey};
209
210 if (payLoadNumBytes > 0) {
211 parserState = ParserState::PAYLOAD;
212 } else {
213 if (fin) {
214 onMessageEnd();
215 }
216 reset();
217 }
218 }
219
220 return ret;
221 }

223 std::size_t Receiver::readPayload() {
224 const std::size_t payloadChunkLeft = (MAX_PAYLOAD_JUNK_LEN <= payLoadNumBytesLeft) ? static_cast<std::size_t>(MAX_PAYLOAD_JUNK_LEN)
225 : static_cast<std::size_t>(payLoadNumBytesLeft);
226
227 const std::size_t ret = readFrameData(payloadChunk, payloadChunkLeft);
228
229 if (ret > 0) {
230 const std::size_t payloadChunkLen = static_cast<std::size_t>(ret);
231
232 if (masked) {
233 for (std::size_t i = 0; i < payloadChunkLen; i++) {
234 *(payloadChunk + i) =
235 *(payloadChunk + i) ^ *(maskingKeyAsArray.keyAsArray + (i + payLoadNumBytes - payLoadNumBytesLeft) % 4);
236 }
237 }
238
239 LOG(TRACE) << "WebSocket receive: Frame data\n" << utils::hexDump(payloadChunk, payloadChunkLen, 32, true);
240
241 onMessageData(payloadChunk, payloadChunkLen);
242
243 payLoadNumBytesLeft -= payloadChunkLen;
244 }
245
246 if (payLoadNumBytesLeft == 0) {
247 if (fin) {
248 onMessageEnd();
249 }
250 reset();
251 }
252
253 return ret;
254 }

256 void Receiver::reset() {

258
259 fin = false;
260 continuation = false;
261 masked = false;
262
263 opCode = 0;
264
265 elengthNumBytes = 0;
266 elengthNumBytesLeft = 0;
267
268 payLoadNumBytes = 0;
269 payLoadNumBytesLeft = 0;
270
271 maskingKey = 0;
272 maskingKeyAsArray = {.key = 0};
273 maskingKeyNumBytes = 4;
274 maskingKeyNumBytesLeft = 4;
275
276 errorState = 0;
277 }
