72 std::size_t Receiver::readOpcode() {
73 char byte = 0;
74 const std::size_t ret = readFrameData(&byte, 1);
75
76 if (ret > 0) {
77 const uint8_t opCodeByte = static_cast<uint8_t>(byte);
78
79 fin = (opCodeByte & 0b10000000) != 0;
80 opCode = opCodeByte & 0b00001111;
81
82 if (!continuation) {
83 onMessageStart(opCode);
84 parserState = ParserState::LENGTH;
85 } else if (opCode == 0) {
86 parserState = ParserState::LENGTH;
87 } else {
88 parserState = ParserState::ERROR;
89 errorState = 1002;
90 LOG(DEBUG) << "WebSocket: Error opcode in continuation frame";
91 }
92 continuation = !fin;
93 }
94
95 return ret;
96 }

98 std::size_t Receiver::readLength() {
99 char byte = 0;
100 const std::size_t ret = readFrameData(&byte, 1);
101
102 if (ret > 0) {
103 const uint8_t lengthByte = static_cast<uint8_t>(byte);
104
105 masked = (lengthByte & 0b10000000) != 0;
106 payLoadNumBytes = payLoadNumBytesLeft = lengthByte & 0b01111111;
107
108 if (payLoadNumBytes > 125) {
109 switch (payLoadNumBytes) {
110 case 126:
111 elengthNumBytes = elengthNumBytesLeft = 2;
112 break;
113 case 127:
114 elengthNumBytes = elengthNumBytesLeft = 8;
115 break;
116 }
117 parserState = ParserState::ELENGTH;
118 payLoadNumBytes = payLoadNumBytesLeft = 0;
119 } else {
120 if (masked) {
121 parserState = ParserState::MASKINGKEY;
122 } else if (payLoadNumBytes > 0) {
123 parserState = ParserState::PAYLOAD;
124 } else {
125 if (fin) {
126 onMessageEnd();
127 }
128 reset();
129 }
130 }
131 }
132
133 return ret;
134 }

136 std::size_t Receiver::readELength() {
137 const std::size_t ret = readFrameData(elengthChunk, elengthNumBytesLeft);
138
139 for (std::size_t i = 0; i < ret; i++) {
140 payLoadNumBytes |= *reinterpret_cast<uint64_t*>(elengthChunk + i) << (elengthNumBytes - elengthNumBytesLeft) * 8;
141
142 elengthNumBytesLeft--;
143 }
144
145 if (elengthNumBytesLeft == 0) {
146 switch (elengthNumBytes) {
147 case 2:
148 payLoadNumBytes = payLoadNumBytesLeft = be16toh(static_cast<uint16_t>(payLoadNumBytes));
149 break;
150 case 8:
151 payLoadNumBytes = payLoadNumBytesLeft = be64toh(payLoadNumBytes);
152 break;
153 }
154
155 if ((payLoadNumBytes & static_cast<uint64_t>(0x01) << 63) != 0) {
156 parserState = ParserState::ERROR;
157 errorState = 1004;
158 } else if (masked) {
159 parserState = ParserState::MASKINGKEY;
160 } else {
161 parserState = ParserState::PAYLOAD;
162 }
163 }
164
165 return ret;
166 }

168 std::size_t Receiver::readMaskingKey() {
169 const std::size_t ret = readFrameData(maskingKeyChunk, maskingKeyNumBytesLeft);
170
171 for (std::size_t i = 0; i < ret; i++) {
172 maskingKey |= static_cast<uint32_t>(*reinterpret_cast<unsigned char*>(maskingKeyChunk + i))
173 << (maskingKeyNumBytes - maskingKeyNumBytesLeft) * 8;
174 maskingKeyNumBytesLeft--;
175 }
176
177 if (maskingKeyNumBytesLeft == 0) {
178 maskingKeyAsArray = {.key = maskingKey};
179
180 if (payLoadNumBytes > 0) {
181 parserState = ParserState::PAYLOAD;
182 } else {
183 if (fin) {
184 onMessageEnd();
185 }
186 reset();
187 }
188 }
189

193 std::size_t Receiver::readPayload() {
194 const std::size_t payloadChunkLeft = (MAX_PAYLOAD_JUNK_LEN <= payLoadNumBytesLeft) ? static_cast<std::size_t>(MAX_PAYLOAD_JUNK_LEN)
195 : static_cast<std::size_t>(payLoadNumBytesLeft);
196
197 const std::size_t ret = readFrameData(payloadChunk, payloadChunkLeft);
198
199 if (ret > 0) {
200 const std::size_t payloadChunkLen = static_cast<std::size_t>(ret);
201
202 if (masked) {
203 for (std::size_t i = 0; i < payloadChunkLen; i++) {
204 *(payloadChunk + i) =
205 *(payloadChunk + i) ^ *(maskingKeyAsArray.keyAsArray + (i + payLoadNumBytes - payLoadNumBytesLeft) % 4);
206 }
207 }
208
209 onMessageData(payloadChunk, payloadChunkLen);
210
211 payLoadNumBytesLeft -= payloadChunkLen;
212 }
213
214 if (payLoadNumBytesLeft == 0) {
215 if (fin) {
216 onMessageEnd();
217 }
218 reset();
219 }
220
221 return ret;
222 }
