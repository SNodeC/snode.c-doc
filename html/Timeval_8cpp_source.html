169 const Timeval& Timeval::normalize() {
170 if (timeVal.tv_usec > 999'999) {
171 timeVal.tv_usec -= 1'000'000;
172 timeVal.tv_sec++;
173 } else if (timeVal.tv_usec < 0) {
174 timeVal.tv_usec += 1'000'000;
175 timeVal.tv_sec--;
176 }
177
178 return *this;
