102 void realListen(const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
103 unsigned int tries,
104 double retryTimeoutScale) const {
105 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
106 new SocketAcceptor(
107 socketContextFactory,
108 onConnect,
109 onConnected,
110 onDisconnect,
111 [server = *this, onStatus, tries, retryTimeoutScale](const SocketAddress& socketAddress,
112 core::socket::State state) mutable -> void {
113 const bool retry = (state & core::socket::State::NO_RETRY) == 0 &&
114 (server.getConfig().getRetryTries() == 0 || tries < server.getConfig().getRetryTries());
115
116 state &= ~core::socket::State::NO_RETRY;
117 onStatus(socketAddress, state);
118
119 if (retry) {
120 switch (state) {
121 case core::socket::State::OK:
122 server.currentOk++;
123 [[fallthrough]];
124 case core::socket::State::DISABLED:
125 break;
126 case core::socket::State::ERROR:
127 server.currentError++;
128 break;
129 case core::socket::State::FATAL:
130 server.currentFatal++;
131 break;
132 }
133
134 if ((server.currentError > 0 && server.getConfig().getRetry()) ||
135 (server.currentFatal > 0 && server.getConfig().getRetry() && server.getConfig().getRetryOnFatal())) {
136 if (server.totalOk < server.currentError + server.currentFatal) {
137 server.totalOk += server.currentOk;
138
139 double relativeRetryTimeout =
140 server.getConfig().getRetryLimit() > 0
141 ? std::min<double>(server.getConfig().getRetryTimeout() * retryTimeoutScale,
142 server.getConfig().getRetryLimit())
143 : server.getConfig().getRetryTimeout() * retryTimeoutScale;
144 relativeRetryTimeout -= utils::Random::getInRange(-server.getConfig().getRetryJitter(),
145 server.getConfig().getRetryJitter()) *
146 relativeRetryTimeout / 100.;
147
148 LOG(INFO) << "Server OnStatus: " << server.getConfig().getInstanceName();
149 LOG(INFO) << " retrying in " << relativeRetryTimeout << " seconds";
150
151 core::timer::Timer::singleshotTimer(
152 [server, onStatus, tries, retryTimeoutScale]() mutable -> void {
153 server.getConfig().Local::renew();
154 server.currentOk = 0;
155 server.currentError = 0;
156 server.currentFatal = 0;
157
158 server.realListen(onStatus, tries + 1, retryTimeoutScale * server.getConfig().getRetryBase());
159 },
160 relativeRetryTimeout);
161 }
162 }
163 }
164 },
165 Super::config);
166 }
