147 private:
148 void realListen(const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
149 unsigned int tries,
150 double retryTimeoutScale) const {
151 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
152 new SocketAcceptor(
153 socketContextFactory,
154 onConnect,
155 onConnected,
156 onDisconnect,
157 [server = *this, onStatus, tries, retryTimeoutScale](const SocketAddress& socketAddress,
158 core::socket::State state) -> void {
159 onStatus(socketAddress, state);
160
161 switch (state) {
162 case core::socket::State::ERROR:
163 if (server.getConfig().getRetry() &&
164 (server.getConfig().getRetryTries() == 0 || tries < server.getConfig().getRetryTries())) {
165 double relativeRetryTimeout =
166 server.getConfig().getRetryLimit() > 0
167 ? std::min<double>(server.getConfig().getRetryTimeout() * retryTimeoutScale,
168 server.getConfig().getRetryLimit())
169 : server.getConfig().getRetryTimeout() * retryTimeoutScale;
170 relativeRetryTimeout -= utils::Random::getInRange(-server.getConfig().getRetryJitter(),
171 server.getConfig().getRetryJitter()) *
172 relativeRetryTimeout / 100.;
173
174 LOG(INFO) << "Retrying in " << relativeRetryTimeout << " seconds";
175
176 core::timer::Timer::singleshotTimer(
177 [server, onStatus, tries, retryTimeoutScale]() mutable -> void {
178 server.getConfig().Local::reset();
179 server.realListen(onStatus, tries + 1, retryTimeoutScale * server.getConfig().getRetryBase());
180 },
181 relativeRetryTimeout);
182 }
183 break;
184 case core::socket::State::OK:
185 case core::socket::State::DISABLED:
186 case core::socket::State::FATAL:
187 break;
188 }
189 },
190 Super::config);
191 }
192 }
