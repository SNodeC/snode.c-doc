152 void realListen(const std::function<void(const SocketAddress&, core::socket::State)>& onStatus,
153 unsigned int tries,
154 double retryTimeoutScale) const {
155 if (core::SNodeC::state() == core::State::RUNNING || core::SNodeC::state() == core::State::INITIALIZED) {
156 new SocketAcceptor(
157 socketContextFactory,
158 onConnect,
159 onConnected,
160 onDisconnect,
161 [server = *this, onStatus, tries, retryTimeoutScale](const SocketAddress& socketAddress,
162 core::socket::State state) mutable -> void {
163 const bool retry = (state & core::socket::State::NO_RETRY) == 0 &&
164 (server.getConfig().getRetryTries() == 0 || tries < server.getConfig().getRetryTries());
165
166 state &= ~core::socket::State::NO_RETRY;
167 onStatus(socketAddress, state);
168
169 if (retry) {
170 switch (state) {
171 case core::socket::State::OK:
172 server.currentOk++;
173 [[fallthrough]];
174 case core::socket::State::DISABLED:
175 break;
176 case core::socket::State::ERROR:
177 server.currentError++;
178 break;
179 case core::socket::State::FATAL:
180 server.currentFatal++;
181 break;
182 }
183
184 if ((server.currentError > 0 && server.getConfig().getRetry()) ||
185 (server.currentFatal > 0 && server.getConfig().getRetry() && server.getConfig().getRetryOnFatal())) {
186 if (server.totalOk < server.currentError + server.currentFatal) {
187 server.totalOk += server.currentOk;
188
189 double relativeRetryTimeout =
190 server.getConfig().getRetryLimit() > 0
191 ? std::min<double>(server.getConfig().getRetryTimeout() * retryTimeoutScale,
192 server.getConfig().getRetryLimit())
193 : server.getConfig().getRetryTimeout() * retryTimeoutScale;
194 relativeRetryTimeout -= utils::Random::getInRange(-server.getConfig().getRetryJitter(),
195 server.getConfig().getRetryJitter()) *
196 relativeRetryTimeout / 100.;
197
198 LOG(INFO) << "Server OnStatus: " << server.getConfig().getInstanceName();
199 LOG(INFO) << " retrying in " << relativeRetryTimeout << " seconds";
200
201 core::timer::Timer::singleshotTimer(
202 [server, onStatus, tries, retryTimeoutScale]() mutable -> void {
203 server.getConfig().Local::renew();
204 server.currentOk = 0;
205 server.currentError = 0;
206 server.currentFatal = 0;
207
208 server.realListen(onStatus, tries + 1, retryTimeoutScale * server.getConfig().getRetryBase());
209 },
210 relativeRetryTimeout);
211 }
212 }
213 }
214 },
215 Super::config);
216 }
217 }
